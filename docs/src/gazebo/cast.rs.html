11//! Cast between types with no conversion.
12//!
13//! Most of these operations are inherently unsafe, but provided as strongly-typed wrappers
14//! to reduce the chance of typos ending up with even more unsafe functions. If you use the
15//! result in incorrect ways, it will cause undefined behaviour.
16
17// These are inherently unsafe in fairly obvious ways. Safety is left entirely to the user.
18// So doc's wouldn't really help.
19#![allow(clippy::missing_safety_doc)]
20
21use std::mem;
22use std::ptr;
23
24#[inline(always)]
25pub fn ptr_to_usize<T: ?Sized>(x: &T) -> usize {
26    x as *const T as *const () as usize
27}
28
29/// Undefined behaviour if the argument is zero, or does not satisfy the alignment
30/// of type `T`.
31#[inline(always)]
32pub unsafe fn usize_to_ptr<'a, T>(x: usize) -> &'a T {
33    unsafe { &*(x as *const T) }
34}
35
36/// Undefined behaviour if the argument does not satisfy the alignment of type `To`.
37#[inline(always)]
38pub unsafe fn ptr<From, To>(x: &From) -> &To {
39    unsafe { &*(x as *const From as *const To) }
40}
41
42/// Undefined behaviour if the argument does not satisfy the alignment of type `To`.
43#[inline(always)]
44pub unsafe fn ptr_mut<From, To>(x: &mut From) -> &mut To {
45    unsafe { &mut *(x as *mut From as *mut To) }
46}
47
48#[inline(always)]
49pub unsafe fn ptr_lifetime<'b, T: ?Sized>(x: &T) -> &'b T {
50    unsafe { &*(x as *const T) }
51}

53/// Like normal [`transmute`](std::mem::transmute), but without the compile-time
54/// check that the sizes of the input and output are the same. Despite the removal
55/// of the compile-time check, this property _must still hold_, and there is
56/// a [`debug_assert_eq`] which checks that. All the rules and warnings from
57/// `transmute` still apply.
58///
59/// This function is mostly useful when you have types that are polymorphic,
60/// e.g. `Vec<T>`, that `transmute` cannot be applied to.
61#[inline]
62pub unsafe fn transmute_unchecked<A, B>(x: A) -> B {
63    unsafe {
64        assert_eq!(mem::size_of::<A>(), mem::size_of::<B>());
65        debug_assert_eq!(0, (&x as *const A).align_offset(mem::align_of::<B>()));
66        let b = ptr::read(&x as *const A as *const B);
67        mem::forget(x);
68        b
69    }
70}

72#[macro_export]
73/// `transmute!(from-type, to-type, value)` will do a [`transmute`](std::mem::transmute),
74/// but the original and result types must be specified.
75macro_rules! transmute {
76    ($from:ty, $to:ty, $e:expr_2021) => {
77        std::mem::transmute::<$from, $to>($e)
78    };
79}

81#[cfg(test)]

87    #[test]
88    fn test_transmute() {
89        #[allow(clippy::useless_transmute)]
90        unsafe fn downcast_string<'a>(x: &'a str) -> &'static str {
91            unsafe { transmute!(&'a str, &'static str, x) }
92        }
93        assert_eq!(unsafe { downcast_string("test") }, "test");
94    }

96    struct Foo<'a, T>(&'a Cell<isize>, &'a T);
97
98    impl<T> Drop for Foo<'_, T> {
99        fn drop(&mut self) {
100            self.0.set(self.0.get() - 1)
101        }
102    }
103
104    #[test]
105    fn test_transmute_unchecked() {
106        fn magic<'a, T>(x: T) -> Foo<'a, Vec<u8>> {
107            unsafe { transmute_unchecked(x) }
108        }
109
110        let s = "magic".to_owned();
111        let c = Cell::new(8);
112        let input = Foo(&c, &s);
113        let output = magic(input);
114        assert_eq!(c.get(), 8);
115        assert_eq!(output.1, b"magic");
116        mem::drop(output);
117        assert_eq!(c.get(), 7);
118    }
119
120    #[test]
121    fn test_transmute_unchecked_bytes() {
122        fn to_bytes<T>(x: T) -> [u8; 8] {
123            unsafe { transmute_unchecked(x) }
124        }
125
126        fn from_bytes<T>(x: [u8; 8]) -> T {
127            unsafe { transmute_unchecked(x) }
128        }
129
130        let x: u64 = 8932489732;
131        let y: u64 = from_bytes(to_bytes(x));
132        assert_eq!(x, y)
133    }
