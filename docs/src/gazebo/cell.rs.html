11//! Additions to the [`Ref`] mechanism.
12
13// We used to implement `ARef` as an enum of `{Ptr(&'a T), Ref(Ref<'a, T>)}`.
14// That works, but consumes 3 words and requires a branch on every access of the underlying
15// pointer. We can optimise that by relying on the underlying details of `Ref`, which is
16// (currently) defined as (after a bit of inlining):
17//
18// ```
19// pub struct Ref<'a, T: ?Sized + 'a> {
20//    value: &'a T,
21//    borrow: &'a Cell<isize>,
22// }
23// ```
24//
25// Because the pointer must always be non-null, we can switch that out for:
26//
27// ```
28// pub struct ARef<'a, T: ?Sized + 'a> {
29//    value: &'a T,
30//    borrow: Option<&'a Cell<isize>>,
31// }
32// ```
33//
34// And use `None` to represent the `Ptr` case. We do that with transmute trickery,
35// but write some good tests that will break if the representation changes,
36// and if necessary we can always switch to the enum representation.
37
38use std::cell::BorrowError;
39use std::cell::Ref;
40use std::cell::RefCell;
41use std::cmp::Ordering;
42use std::fmt;
43use std::fmt::Display;
44use std::hash::Hash;
45use std::hash::Hasher;
46use std::ops::Deref;
47
48#[derive(Debug)]
49enum ARefImpl<'a, T: ?Sized + 'a> {
50    Ptr(&'a T),
51    Ref(Ref<'a, T>),
52}
53
54/// A [`Ref`] that might not actually be borrowed.
55/// Either a `Ptr` (a normal & style reference), or a `Ref` (like from
56/// [`RefCell`]), but exposes all the methods available on [`Ref`].
57#[derive(Debug)]
58pub struct ARef<'a, T: ?Sized + 'a>(ARefImpl<'a, T>);
59
60impl<T: ?Sized> Deref for ARef<'_, T> {
61    type Target = T;
62
63    fn deref(&self) -> &T {
64        match &self.0 {
65            ARefImpl::Ptr(p) => p,
66            ARefImpl::Ref(r) => r.deref(),
67        }
68    }
69}
70
71impl<'a, T: ?Sized + 'a> ARef<'a, T> {
72    /// Create a new [`ARef`] from a pointer.
73    pub fn new_ptr(x: &'a T) -> Self {
74        ARef(ARefImpl::Ptr(x))
75    }
76
77    /// Create a new [`ARef`] from a reference.
78    pub fn new_ref(x: Ref<'a, T>) -> Self {
79        ARef(ARefImpl::Ref(x))
80    }
81
82    /// See [`Ref.clone`](Ref::clone). Not a self method since that interferes with the [`Deref`].
83    #[allow(clippy::should_implement_trait)]
84    pub fn clone(orig: &Self) -> Self {
85        match &orig.0 {
86            ARefImpl::Ptr(p) => ARef(ARefImpl::Ptr(p)),
87            ARefImpl::Ref(r) => ARef(ARefImpl::Ref(Ref::clone(r))),
88        }
89    }
90
91    /// See [`Ref.map`](Ref::map). Not a self method since that interferes with the [`Deref`].
92    pub fn map<U: ?Sized, F>(orig: ARef<'a, T>, f: F) -> ARef<'a, U>
93    where
94        F: FnOnce(&T) -> &U,
95    {
96        match orig.0 {
97            ARefImpl::Ptr(p) => ARef(ARefImpl::Ptr(f(p))),
98            ARefImpl::Ref(r) => ARef(ARefImpl::Ref(Ref::map(r, f))),
99        }
100    }
101
102    /// See [`Ref.map_split`](Ref::map_split). Not a self method since that interferes with the
103    /// [`Deref`].
104    pub fn map_split<U: ?Sized, V: ?Sized, F>(orig: ARef<'a, T>, f: F) -> (ARef<'a, U>, ARef<'a, V>)
105    where
106        F: FnOnce(&T) -> (&U, &V),
107    {
108        match orig.0 {
109            ARefImpl::Ptr(p) => {
110                let (u, v) = f(p);
111                (ARef(ARefImpl::Ptr(u)), ARef(ARefImpl::Ptr(v)))
112            }
113            ARefImpl::Ref(r) => {
114                let (u, v) = Ref::map_split(r, f);
115                (ARef(ARefImpl::Ref(u)), ARef(ARefImpl::Ref(v)))
116            }
117        }
118    }
119
120    /// See [`Ref.filter_map`](Ref::filter_map). Not a self method since that interferes with the
121    /// [`Deref`].
122    pub fn filter_map<U: ?Sized, F>(orig: ARef<'a, T>, f: F) -> Result<ARef<'a, U>, Self>
123    where
124        F: FnOnce(&T) -> Option<&U>,
125    {
126        match orig.0 {
127            ARefImpl::Ptr(p) => match f(p) {
128                Some(u) => Ok(ARef(ARefImpl::Ptr(u))),
129                None => Err(ARef(ARefImpl::Ptr(p))),
130            },
131            ARefImpl::Ref(r) => match Ref::filter_map(r, f) {
132                Ok(u) => Ok(ARef(ARefImpl::Ref(u))),
133                Err(r) => Err(ARef(ARefImpl::Ref(r))),
134            },
135        }
136    }
137}

139// `Ref` doesn't have many traits on it. I don't really know why - I think that's an oversight.
140// & references do have many traits on them. Therefore, when being "either" we choose to do as many
141// implementations as we can.
142
143impl<T: Display + ?Sized> Display for ARef<'_, T> {
144    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
145        ARef::deref(self).fmt(f)
146    }
147}
148
149impl<T: Hash + ?Sized> Hash for ARef<'_, T> {
150    fn hash<H: Hasher>(&self, state: &mut H) {
151        ARef::deref(self).hash(state)
152    }
153}
154
155impl<A: PartialEq<B> + ?Sized, B: ?Sized> PartialEq<ARef<'_, B>> for ARef<'_, A> {
156    fn eq(&self, other: &ARef<'_, B>) -> bool {
157        ARef::deref(self).eq(ARef::deref(other))
158    }
159}
160
161impl<A: Eq + ?Sized> Eq for ARef<'_, A> {}
162
163impl<A: PartialOrd<B> + ?Sized, B: ?Sized> PartialOrd<ARef<'_, B>> for ARef<'_, A> {
164    fn partial_cmp(&self, other: &ARef<'_, B>) -> Option<Ordering> {
165        ARef::deref(self).partial_cmp(ARef::deref(other))
166    }
167}
168
169impl<A: Ord + ?Sized> Ord for ARef<'_, A> {
170    fn cmp(&self, other: &Self) -> Ordering {
171        ARef::deref(self).cmp(ARef::deref(other))
172    }
173}

175/// Obtain an [`ARef`] from either a normal pointer or a [`RefCell`].
176pub trait AsARef<T: ?Sized> {
177    /// Get an [`ARef`] pointing at this type.
178    fn as_aref(this: &Self) -> ARef<T>;
179    /// Try and get an [`ARef`] pointing at this type. Returns an [`Err`] if
180    /// the type `Self` is a [`RefCell`] which is already mutably borrowed.
181    fn try_as_aref(this: &Self) -> Result<ARef<T>, BorrowError>;
182    /// Return the underlying [`RefCell`] if `Self` is one, otherwise [`None`].
183    fn as_ref_cell(this: &Self) -> Option<&RefCell<T>>;
184}
185
186impl<T: ?Sized> AsARef<T> for T {
187    fn as_aref(this: &Self) -> ARef<T> {
188        ARef::new_ptr(this)
189    }
190    fn try_as_aref(this: &Self) -> Result<ARef<T>, BorrowError> {
191        Ok(ARef::new_ptr(this))
192    }
193    fn as_ref_cell(_this: &Self) -> Option<&RefCell<T>> {
194        None
195    }
196}
197
198impl<T: ?Sized> AsARef<T> for RefCell<T> {
199    fn as_aref(this: &Self) -> ARef<T> {
200        ARef::new_ref(this.borrow())
201    }
202    fn try_as_aref(this: &Self) -> Result<ARef<T>, BorrowError> {
203        Ok(ARef::new_ref(this.try_borrow()?))
204    }
205    fn as_ref_cell(this: &Self) -> Option<&RefCell<T>> {
206        Some(this)
207    }
208}

216    #[test]
217    fn test_from_ref_docs() {
218        let c = RefCell::new((5, 'b'));
219        let b1: ARef<(u32, char)> = ARef::new_ref(c.borrow());
220        let b2: ARef<u32> = ARef::map(b1, |t| &t.0);
221        assert_eq!(*b2, 5);
222
223        let cell = RefCell::new([1, 2, 3, 4]);
224        let borrow = ARef::new_ref(cell.borrow());
225        let (begin, end) = ARef::map_split(borrow, |slice| slice.split_at(2));
226        assert_eq!(*begin, [1, 2]);
227        assert_eq!(*end, [3, 4]);
228    }

230    #[test]
231    fn test_borrow_guards() {
232        let c = RefCell::new(5);
233        assert!(c.try_borrow_mut().is_ok());
234        let r1 = ARef::new_ref(c.borrow());
235        assert!(c.try_borrow_mut().is_err());
236        let r2 = c.borrow();
237        assert!(c.try_borrow_mut().is_err());
238        mem::drop(r1);
239        assert!(c.try_borrow_mut().is_err());
240        mem::drop(r2);
241        assert!(c.try_borrow_mut().is_ok());
242    }

247        let p = ARef::new_ptr(&c);
248        let p2 = ARef::map(p, |x| &x[1..3]);
249        assert_eq!(&*p2, "es");

272    #[test]
273    /// Test that the representation of ARef is what we expect
274    fn test_ref_as_expected() {
275        let orig = RefCell::new("test".to_owned());
276        let p = orig.borrow();
277        let p2 = Ref::clone(&p);
278        let (pointer, cell): (*const String, *const ()) = unsafe { mem::transmute(p) };
279        // We expect the first to be a pointer to the underlying string
280        assert_eq!(pointer, Ref::deref(&p2) as *const String);
281        // We want to make sure the second is never zero
282        assert!(!cell.is_null());
283
284        // Put it back as it was, to make sure our test doesn't leak memory
285        let _ignore: Ref<String> = unsafe { mem::transmute((pointer, cell)) };
286    }
