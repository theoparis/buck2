18use std::fmt::Debug;
19use std::fmt::Display;
20use std::hash::Hash;
21use std::hash::Hasher;
22use std::ops::Deref;
23
24use allocative::Allocative;
25use dupe::Dupe;
26use equivalent::Equivalent;
27use strong_hash::StrongHash;
28
29use crate::hash_value::StarlarkHashValue;
30
31/// A key and its hash.
32#[derive(PartialEq, Eq, Debug, Clone, Copy, Dupe, Allocative)]
33pub struct Hashed<K> {
34    hash: StarlarkHashValue,
35    key: K,
36}
37
38// We deliberately know that this is a hash and value, so our Eq/Hash are fine
39#[allow(clippy::derived_hash_with_manual_eq)]
40impl<K> Hash for Hashed<K> {
41    fn hash<S: Hasher>(&self, state: &mut S) {
42        // Only hash the hash, not the key.
43        self.hash.hash(state)
44    }
45}
46
47impl<K: StrongHash> StrongHash for Hashed<K> {
48    fn strong_hash<S: Hasher>(&self, state: &mut S) {
49        // Only hash the key, not the (weak) hash.
50        self.key.strong_hash(state);
51    }
52}
53
54impl<K> Deref for Hashed<K> {
55    type Target = K;
56
57    fn deref(&self) -> &Self::Target {
58        &self.key
59    }
60}
61
62impl<K: Display> Display for Hashed<K> {
63    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
64        self.key.fmt(f)
65    }
66}
67
68impl<K: PartialOrd> PartialOrd for Hashed<K> {
69    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
70        self.key.partial_cmp(&other.key)
71    }
72}
73
74impl<K: Ord> Ord for Hashed<K> {
75    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
76        self.key.cmp(&other.key)
77    }
78}
79
80impl<K> Hashed<K> {
81    /// Create a new [`Hashed`] value using the [`Hash`] of the key.
82    pub fn new(key: K) -> Self
83    where
84        K: Hash,
85    {
86        Hashed::new_unchecked(StarlarkHashValue::new(&key), key)
87    }
88
89    /// Directly create a new [`Hashed`] using a given hash value.
90    /// If the hash does not correspond to the key, its will cause issues.
91    pub fn new_unchecked(hash: StarlarkHashValue, key: K) -> Self {
92        Hashed { hash, key }
93    }
94
95    /// Get the underlying key.
96    pub fn key(&self) -> &K {
97        &self.key
98    }
99
100    /// Get the underlying key, as mutable.
101    pub fn key_mut(&mut self) -> &mut K {
102        &mut self.key
103    }
104
105    /// Get the underlying key taking ownership.
106    pub fn into_key(self) -> K {
107        self.key
108    }
109
110    /// Get the underlying hash.
111    pub fn hash(&self) -> StarlarkHashValue {
112        self.hash
113    }
114
115    /// Convert `Hashed<K>` to `Hashed<&K>`.
116    pub fn as_ref(&self) -> Hashed<&K> {
117        Hashed::new_unchecked(self.hash, &self.key)
118    }
119}
120
121impl<K> Hashed<&K> {
122    /// Make `Hashed<K>` from `Hashed<&K>`.
123    pub fn copied(self) -> Hashed<K>
124    where
125        K: Copy,
126    {
127        Hashed::new_unchecked(self.hash, *self.key)
128    }
129
130    /// Make `Hashed<K>` from `Hashed<&K>`, where `K` is `Clone`.
131    pub fn cloned(self) -> Hashed<K>
132    where
133        K: Clone,
134    {
135        Hashed::new_unchecked(self.hash, self.key.clone())
136    }
137}
138
139impl<K: ?Sized> Hashed<&K> {
140    /// Make `Hashed<K>` from `Hashed<&K>`, where `T` is the owned form of `K`.
141    pub fn owned<T>(self) -> Hashed<T>
142    where
143        K: Equivalent<T>,
144        K: ToOwned<Owned = T>,
145    {
146        Hashed::new_unchecked(self.hash, self.key.to_owned())
147    }
148}
