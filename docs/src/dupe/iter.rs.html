11use std::iter::Cloned;
12
13use crate::Dupe;
14
15pub trait IterDupedExt: Sized {
16    /// Like `duped()`, but only works for types that implement `Dupe`.
17    /// Note that the return type is deliberately `Cloned`, as that behaves
18    /// the same as a `Duped` would be, but can take advantage of standard library
19    /// optimisations.
20    ///
21    /// ```
22    /// use std::rc::Rc;
23    ///
24    /// use dupe::IterDupedExt;
25    /// let inputs = vec![Rc::new("Hello"), Rc::new("World")];
26    /// let outputs = inputs.iter().duped().collect::<Vec<_>>();
27    /// assert_eq!(inputs, outputs);
28    /// ```
29    /// use gazebo::prelude::*;
30    /// use std::cmp::Ordering;
31    fn duped(self) -> Cloned<Self>;
32}
33
34impl<'a, I, T> IterDupedExt for I
35where
36    I: Sized,
37    I: Iterator<Item = &'a T>,
38    T: 'a + Dupe,
39{
40    fn duped(self) -> Cloned<Self> {
41        self.cloned()
42    }
43}
