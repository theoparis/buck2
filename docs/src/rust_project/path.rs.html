11use std::io;
12use std::path::Path;
13use std::path::PathBuf;

15/// Try to canonicalize `path`, but return `path` as-is if we couldn't
16/// (e.g. the path didn't exist, broken symlinks, etc).
17///
18/// This is useful when being invoked by the IDE on unfinished code: we
19/// might see all sorts of broken paths whilst the user is working.
20pub(crate) fn safe_canonicalize(path: &Path) -> PathBuf {
21    match canonicalize(path) {
22        Ok(path) => path,
23        Err(_) => path.to_owned(),
24    }
25}

27/// If the canonical version of `path` is in a source code checkout, return
28/// the canonical path, otherwise return the path as-is.
29pub(crate) fn canonicalize_to_vcs_path(path: &Path, project_root: &Path) -> PathBuf {
30    let canonical_path = safe_canonicalize(path);
31
32    // Buck builds Rust code by creating symlinks in buck-out/ to the
33    // files in VCS (e.g. hg, git). This is what rustc sees, but we
34    // don't want rust-analyzer to see the files in buck-out. We want
35    // rust-analyzer to see the files in their original location, so
36    // go-to-def takes the user to `my-project/src/foo.rs` not
37    // `buck-out/abc123/my-project/foo.rs`.
38    //
39    // For generated files, there is no file in VCS. The symlink in
40    // buck-out just points to the build output that generated the
41    // file. The build output directory may not be a valid crate
42    // directory.
43    //
44    // ```text
45    // $ ls -l ~/fbsource/buck-out/.rust-analyzer/gen/abc123/my_project/__foo_type_defs-rust__/__srcs/
46    // consts.rs -> ../../__foo_type_defs-rust-foo_type_defs.thrift__/out/gen-rust/consts.rs
47    // docs.md -> ../../__foo_type_defs-rust-crate__/out/types.md
48    // lib.rs -> ../../__foo_type_defs-rust-foo_type_defs.thrift__/out/gen-rust/types.rs
49    // services.rs -> ../../__foo_type_defs-rust-foo_type_defs.thrift__/out/gen-rust/services.rs
50    // ```
51    //
52    // In this example, expanding the symlink to the root module
53    // (lib.rs) would mean that we'd use
54    // ~/fbsource/buck-out/.rust-analyzer/gen/fbcode/abc123/my_project/__foo_type_defs-rust-foo_type_defs.thrift__/out/gen-rust/
55    // as the crate root, which does not contain a lib.rs file.
56    if let Ok(rel) = canonical_path.strip_prefix(project_root) {
57        // The file was generated, use the original path, don't expand
58        // symlinks.
59        if rel.starts_with("buck-out") {
60            return path.to_owned();
61        }
62    }
63
64    canonical_path
65}

67fn canonicalize<P: AsRef<Path>>(path: P) -> io::Result<PathBuf> {
68    let canonical_path = dunce::canonicalize(&path)?;
69
70    if cfg!(windows) && path.as_ref().starts_with("\\\\?\\") {
71        tracing::warn!(
72            path = path.as_ref().display().to_string(),
73            "Couldn't strip UNC prefix from path. Using it as-is."
74        );
75    }
76
77    Ok(canonical_path)
78}
