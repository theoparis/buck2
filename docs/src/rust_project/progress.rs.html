11use std::io::Write;

13use rustc_hash::FxHashMap;
14use serde::Serialize;
15use tracing::span;
16use tracing_subscriber::Layer;
17use tracing_subscriber::fmt::MakeWriter;

19pub(crate) struct ProgressLayer<S, W> {
20    _s: std::marker::PhantomData<S>,
21    writer: W,
22}
23
24impl<S, W> ProgressLayer<S, W> {
25    pub(crate) fn new(writer: W) -> Self {
26        Self {
27            _s: std::marker::PhantomData,
28            writer,
29        }
30    }
31}
32
33#[derive(Serialize, Debug, Clone, PartialEq)]
34struct Out<'a> {
35    #[serde(flatten)]
36    event_fields: &'a FxHashMap<String, serde_json::Value>,
37    #[serde(flatten)]
38    span_fields: &'a FxHashMap<String, serde_json::Value>,
39}
40
41#[derive(Debug, Clone, PartialEq)]
42struct ProgressStorage {
43    data: FxHashMap<String, serde_json::Value>,
44}
45
46impl<S, W> Layer<S> for ProgressLayer<S, W>
47where
48    S: tracing::Subscriber + for<'a> tracing_subscriber::registry::LookupSpan<'a>,
49    W: for<'a> MakeWriter<'a> + 'static,
50{
51    fn on_new_span(
52        &self,
53        attrs: &span::Attributes<'_>,
54        id: &span::Id,
55        ctx: tracing_subscriber::layer::Context<'_, S>,
56    ) {
57        let span = ctx.span(id).unwrap();
58        let mut extensions = span.extensions_mut();
59
60        let mut fields = FxHashMap::default();
61        let mut visitor = JsonVisitor(&mut fields);
62        attrs.record(&mut visitor);
63
64        extensions.insert(ProgressStorage { data: fields });
65    }
66
67    fn on_event(&self, event: &tracing::Event<'_>, ctx: tracing_subscriber::layer::Context<'_, S>) {
68        let mut event_fields = FxHashMap::default();
69        let mut visitor = JsonVisitor(&mut event_fields);
70        event.record(&mut visitor);
71
72        if !event_fields.contains_key("kind") {
73            return;
74        }
75
76        let span_fields = match ctx.lookup_current() {
77            Some(span) => {
78                let ext = span.extensions();
79                if let Some(storage) = ext.get::<ProgressStorage>() {
80                    storage.data.clone()
81                } else {
82                    FxHashMap::default()
83                }
84            }
85            _ => FxHashMap::default(),
86        };
87
88        let out = Out {
89            event_fields: &event_fields,
90            span_fields: &span_fields,
91        };
92        let out = serde_json::to_string(&out).unwrap();
93        let mut writer = self.writer.make_writer();
94        writeln!(writer, "{out}").expect("unable to write");
95    }
96}
97
98struct JsonVisitor<'a>(&'a mut FxHashMap<String, serde_json::Value>);
99
100impl tracing::field::Visit for JsonVisitor<'_> {
101    fn record_str(&mut self, field: &tracing::field::Field, value: &str) {
102        let value: String = if field.name() == "project" {
103            serde_json::from_str(value).unwrap()
104        } else {
105            String::from(value)
106        };
107        self.0
108            .insert(field.name().to_owned(), serde_json::Value::from(value));
109    }
110
111    fn record_debug(&mut self, field: &tracing::field::Field, value: &dyn std::fmt::Debug) {
112        self.0.insert(
113            field.name().to_owned(),
114            serde_json::Value::from(format!("{value:?}")),
115        );
116    }
117}
