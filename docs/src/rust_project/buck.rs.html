53pub(crate) fn infer_xplat_platforms(targets: &[Target]) -> Vec<&'static str> {
54    let mut xplat_platforms = vec!["CXX"];
55    for target in targets {
56        if target.contains("fbcode") {
57            xplat_platforms.push("FBCODE");
58            break;
59        }
60    }
61
62    // If the user is running rust-project on macOS, assume they want
63    // APPLE as an available platform. This is only a heuristic: users'
64    // projects may not necessarily match the platform they're currently
65    // developing on, but it's often true in practice.
66    if cfg!(target_os = "macos") {
67        xplat_platforms.push("APPLE");
68    }
69    xplat_platforms
70}

72pub(crate) fn to_project_json(
73    sysroot: Sysroot,
74    expanded_and_resolved: ExpandedAndResolved,
75    aliases: FxHashMap<Target, AliasedTargetInfo>,
76    check_cycles: bool,
77    include_all_buildfiles: bool,
78    extra_cfgs: &[String],
79    buck: &Buck,
80) -> Result<ProjectJson, anyhow::Error> {
81    let project_root = buck.resolve_project_root()?;
82
83    let ExpandedAndResolved {
84        expanded_targets: _,
85        queried_proc_macros: proc_macros,
86        resolved_deps: target_map,
87    } = expanded_and_resolved;
88
89    let target_index = merge_unit_test_targets(target_map);
90
91    // A rust-project.json uses file indexes to associate dependencies with the
92    // relevant crate.
93    //
94    // "crates": [
95    //   {
96    //     "display_name": "my-project",
97    //     "deps": [
98    //       {
99    //         "crate": 1,
100    //         "name": "my-lib"
101    //       }
102    //     ]
103    //   },
104    //   {
105    //     "display_name": "my-lib",
106    //     "deps": []
107    //   }
108    // ]
109    //
110    // This means that we must iterate over targets in a consistent order, so
111    // the indexes are correct. Build an ordered Vec and a corresponding HashMap
112    // from target name to index.
113    let targets_vec = target_index.keys().cloned().collect::<Vec<Target>>();
114
115    let mut targets_to_ids: FxHashMap<&Target, usize> = FxHashMap::default();
116    for (index, target) in targets_vec.iter().enumerate() {
117        targets_to_ids.insert(target, index);
118    }
119
120    let mut crates: Vec<Crate> = Vec::with_capacity(targets_vec.len());
121    for target in &targets_vec {
122        let info = target_index.get(&target).unwrap();
123
124        let dep_targets = resolve_aliases(&info.deps, &aliases, &proc_macros);
125        let deps = as_deps(&dep_targets, info, &targets_to_ids, &target_index);
126
127        let edition = match &info.edition {
128            Some(edition) => edition.clone(),
129            None => Edition::Edition2021,
130        };
131
132        // the mapping here is inverted, which means we need to search through the keys for the Target.
133        // thankfully, most projects don't have to many proc macros, which means the size of this list
134        // remains in the two digit space.
135        let proc_macro_dylib_path = proc_macros
136            .values()
137            .find(|macro_output| macro_output.actual == *target)
138            .map(|macro_output| macro_output.dylib.clone());
139        if let Some(ref dylib) = proc_macro_dylib_path {
140            trace!(?target, ?dylib, "target is a proc macro");
141        }
142
143        // corresponds to the BUCK/TARGETS file of a target.
144        let build_file = project_root.join(info.project_relative_buildfile.clone());
145
146        // We don't need to push the source folder as rust-analyzer by default will use the root-module parent().
147        // info.root_module() will output either the fbcode source file or the symlinked one based on if it's a mapped source or not
148        let root_module = info.root_module(&project_root);
149
150        let mut env = FxHashMap::default();
151
152        // Populate the environment variables the target configuration's environment variables,
153        // but ignore OUT_DIR as we handle that later.
154        env.extend(info.env.clone().into_iter());
155
156        // If $CARGO_MANIFEST_DIR is set, resolve it to an absolute path.
157        if let Some(rel_cargo_manifest_dir) = info.env.get("CARGO_MANIFEST_DIR") {
158            let cargo_manifest_dir = info.source_folder.join(rel_cargo_manifest_dir);
159            env.insert(
160                "CARGO_MANIFEST_DIR".to_owned(),
161                cargo_manifest_dir.to_string_lossy().into_owned(),
162            );
163        }
164
165        let mut include_dirs = vec![];
166        if let Some(out_dir) = info.env.get("OUT_DIR") {
167            // to ensure that the `OUT_DIR` is included as part of the `PackageRoot` in rust-analyzer,
168            // manually insert the parent of the `out_dir` into `include_dirs`.
169            if let Some(parent) = Path::new(out_dir).parent() {
170                include_dirs.push(parent.to_owned());
171            }
172        }
173
174        // We want to include all the directories from srcs, because sometimes the source files on
175        // disk aren't in the same layout as buck-out. In this situation, the root_module directory
176        // isn't sufficient to find all files in the module tree.
177        //
178        // You could construct a pathological case using mapped_srcs that means we miss
179        // directories here, but this covers all the cases I've tested.
180        let mut src_dirs: Vec<PathBuf> = info
181            .srcs
182            .iter()
183            .filter_map(|p| p.parent())
184            .map(|p| p.to_owned())
185            .collect();
186        src_dirs.sort();
187        include_dirs.extend(src_dirs.into_iter());
188
189        // We want to include the directory that contains the root module. Typically this file
190        // will be listed the srcs declaration, but it may be a generated file listed in mapped_srcs.
191        if let Some(parent) = root_module.parent() {
192            include_dirs.push(parent.to_owned());
193        }
194
195        include_dirs = remove_duplicates_preserve_order(include_dirs);
196
197        let build = if include_all_buildfiles || info.in_workspace {
198            let build = Build {
199                label: target.clone(),
200                build_file: build_file.to_owned(),
201                target_kind: info.kind.clone().into(),
202            };
203            Some(build)
204        } else {
205            None
206        };
207
208        let crate_info = Crate {
209            display_name: Some(info.display_name()),
210            root_module,
211            edition,
212            deps,
213            is_workspace_member: info.in_workspace,
214            source: Some(Source {
215                include_dirs,
216                exclude_dirs: vec![],
217            }),
218            cfg: info
219                .cfg()
220                .into_iter()
221                .chain(extra_cfgs.iter().cloned())
222                .collect(),
223            env,
224            build,
225            is_proc_macro: info.proc_macro.unwrap_or(false),
226            proc_macro_dylib_path,
227            target: None,
228        };
229        crates.push(crate_info);
230    }
231
232    if check_cycles {
233        check_cycles_in_crate_graph(&crates);
234    }
235
236    let jp = ProjectJson {
237        sysroot: Box::new(sysroot),
238        crates,
239        runnables: vec![Runnable {
240            program: "buck".to_owned(),
241            args: vec![
242                "test".to_owned(),
243                CLIENT_METADATA_RUST_PROJECT.to_owned(),
244                "{label}".to_owned(),
245                "--".to_owned(),
246                "{test_id}".to_owned(),
247                "--print-passing-details".to_owned(),
248            ],
249            cwd: project_root.to_owned(),
250            kind: RunnableKind::TestOne,
251        }],
252        // needed to ignore the generated `rust-project.json` in diffs, but including the actual
253        // string will mark this file as generated
254        generated: String::from("\x40generated"),
255    };
256
257    Ok(jp)
258}

260/// Check that there are no cycles in the crate dependency graph: a
261/// crate should never transitively depend on itself.
262///
263/// If a cycle is found, print the offending crate and terminate.
264fn check_cycles_in_crate_graph(crates: &[Crate]) {
265    // From a start crate ID, each ID we can reach, along with an example route.
266    let mut reachable: FxHashMap<usize, FxHashMap<usize, Vec<usize>>> = FxHashMap::default();
267
268    // Initialize the reachable crates from immediate dependencies.
269    for (idx, krate) in crates.iter().enumerate() {
270        let mut routes: FxHashMap<usize, Vec<usize>> = FxHashMap::default();
271        for dep in &krate.deps {
272            routes.insert(dep.crate_index, vec![idx, dep.crate_index]);
273        }
274
275        reachable.insert(idx, routes);
276    }
277
278    let mut changed = true;
279    while changed {
280        changed = false;
281
282        let mut new_reachable = reachable.clone();
283
284        // Iterate all the dependencies, and add any transitive
285        // dependencies that weren't already in `reachable`.
286        for (crate_idx, deps_idxs) in reachable.iter() {
287            for (dep_idx, route) in deps_idxs.iter() {
288                for transitive_dep_idx in reachable[dep_idx].keys() {
289                    if transitive_dep_idx == crate_idx {
290                        let mut cycle_route = route.clone();
291                        cycle_route.push(*transitive_dep_idx);
292
293                        tracing::error!(
294                            crate = crates[*crate_idx].display_name,
295                            route = format_route(&cycle_route, crates),
296                            "Found a cycle",
297                        );
298                        std::process::exit(2);
299                    }
300
301                    if !deps_idxs.contains_key(transitive_dep_idx) {
302                        let mut new_route = route.clone();
303                        new_route.push(*transitive_dep_idx);
304
305                        new_reachable
306                            .get_mut(crate_idx)
307                            .expect("We should always have initialized the dependencies for each crate.")
308                            .insert(*transitive_dep_idx, new_route);
309                        changed = true;
310                    }
311                }
312            }
313        }
314
315        reachable = new_reachable;
316    }
317}

319fn format_route(route: &[usize], crates: &[Crate]) -> String {
320    let mut formatted_crates = vec![];
321    for idx in route {
322        formatted_crates.push(format!(
323            "{} ({idx})",
324            crates[*idx]
325                .display_name
326                .clone()
327                .unwrap_or("<unnamed>".to_owned()),
328        ));
329    }
330
331    formatted_crates.join(" -> ")
332}

334/// If any target in `targets` is an alias, resolve it to the actual target.
335fn resolve_aliases(
336    targets: &[Target],
337    aliases: &FxHashMap<Target, AliasedTargetInfo>,
338    proc_macros: &FxHashMap<Target, MacroOutput>,
339) -> Vec<Target> {
340    let mut seen = FxHashSet::default();
341    let mut resolved_targets = vec![];
342
343    for target in targets {
344        let destination_target = match aliases.get(target) {
345            Some(actual) => &actual.actual,
346            None => {
347                // we fall back to check the proc macros for aliases
348                // (these should exist in the aliases map, but they don't. yolo.)
349                match proc_macros.get(target) {
350                    Some(MacroOutput { actual, .. }) => actual,
351                    None => target,
352                }
353            }
354        };
355
356        if !seen.contains(destination_target) {
357            resolved_targets.push(destination_target.to_owned());
358            seen.insert(destination_target);
359        }
360    }
361
362    resolved_targets
363}

365/// Convert `dep_targets` to `Dep` values.
366fn as_deps(
367    dep_targets: &[Target],
368    info: &TargetInfo,
369    target_to_ids: &FxHashMap<&Target, usize>,
370    target_index: &FxHashMap<Target, TargetInfo>,
371) -> Vec<Dep> {
372    let overridden_names = info.overridden_dep_names();
373
374    let mut seen_targets = FxHashSet::default();
375
376    let mut deps = vec![];
377    for dep_target in dep_targets {
378        seen_targets.insert(dep_target);
379
380        let Some(info) = target_index.get(dep_target) else {
381            trace!(?dep_target, "not present in target_index");
382            continue;
383        };
384
385        let crate_index = *target_to_ids.get(dep_target).unwrap();
386        let name = match overridden_names.get(dep_target) {
387            Some(n) => n.replace('-', "_"),
388            None => info.crate_name(),
389        };
390
391        deps.push(Dep { crate_index, name });
392    }
393
394    for (target, name) in overridden_names.into_iter() {
395        if !seen_targets.contains(&target) {
396            let Some(crate_index) = target_to_ids.get(&target) else {
397                continue;
398            };
399            deps.push(Dep {
400                crate_index: *crate_index,
401                name: name.replace('-', "_"),
402            });
403        }
404    }
405
406    deps
407}

409/// For every test target, drop it from `target_map` and include test
410/// target's dependencies in the target that references the test
411/// target.
412fn merge_unit_test_targets(
413    target_map: FxHashMap<Target, TargetInfo>,
414) -> FxHashMap<Target, TargetInfo> {
415    let mut target_index = FxHashMap::default();
416
417    let (tests, mut targets): (FxHashMap<Target, TargetInfo>, FxHashMap<Target, TargetInfo>) =
418        target_map
419            .into_iter()
420            .partition(|(_, info)| info.kind == Kind::Test);
421
422    let (generated_unit_tests, standalone_tests): (
423        FxHashMap<Target, TargetInfo>,
424        FxHashMap<Target, TargetInfo>,
425    ) = tests.into_iter().partition(|(test_target, _)| {
426        test_target.ends_with("-unittest")
427            && targets
428                .iter()
429                .any(|(_, value)| value.test_deps.contains(test_target))
430    });
431
432    targets.extend(standalone_tests);
433
434    for (index, (target, mut info)) in targets.into_iter().enumerate() {
435        trace!(?target, ?info, index, "adding dependency");
436
437        // Merge the `-unittest` target with the parent target.
438        let unittest_target = Target::new(format!("{target}-unittest"));
439        if info.test_deps.contains(&unittest_target) {
440            if let Some(test_info) = generated_unit_tests.get(&unittest_target) {
441                for test_dep in &test_info.deps {
442                    if !info.deps.contains(test_dep) && *test_dep != target {
443                        info.deps.push(test_dep.clone())
444                    }
445                }
446
447                info.in_workspace |= test_info.in_workspace;
448            }
449        }
450
451        target_index.insert(target.to_owned(), info);
452    }
453    target_index
454}

470    /// Invoke `buck2` with the given subcommands.
471    ///
472    /// Care should be taken to ensure that buck is invoked with the same set
473    /// options and configuration to avoid invalidating caches.
474    fn command<I, S>(&self, subcommands: I, xplat_platforms: &[&str]) -> Command
475    where
476        I: IntoIterator<Item = S>,
477        S: AsRef<OsStr>,
478    {
479        let mut cmd = self.command_without_config(subcommands);
480        cmd.args([
481            "-c=client.id=rust-project",
482            "-c=rust.rust_project_build=true",
483        ]);
484
485        if !xplat_platforms.is_empty() {
486            cmd.arg(format!(
487                "-c=xplat.available_platforms={}",
488                xplat_platforms.join(",")
489            ));
490        }
491
492        cmd
493    }

495    /// Invoke `buck` with the given subcommands.
496    ///
497    /// This method should only be used with with buck commands that do not accept
498    /// configuration options, such as `root`. [`Buck::command`] should be preferred.
499    fn command_without_config<I, S>(&self, subcommands: I) -> Command
500    where
501        I: IntoIterator<Item = S>,
502        S: AsRef<OsStr>,
503    {
504        let mut cmd = Command::new(&self.command);
505
506        // rust-analyzer invokes the check-on-save command with `RUST_BACKTRACE=short`
507        // set. Unfortunately, buck2 doesn't handle that well and becomes extremely
508        // slow when the daemon is started with backtrace variables set. Until that is
509        // fixed, just unset them here.
510        cmd.env_remove("RUST_BACKTRACE")
511            .env_remove("RUST_LIB_BACKTRACE");
512
513        cmd.args(["--isolation-dir", ".rust-analyzer"]);
514        cmd.args(subcommands);
515        cmd.args(["--oncall", "rust_devx"]);
516
517        cmd
518    }

520    /// Return the absolute path of the current Buck project root.
521    pub(crate) fn resolve_project_root(&self) -> Result<PathBuf, anyhow::Error> {
522        let mut command = self.command_without_config(["root"]);
523        command.arg("--kind=project");
524
525        let mut stdout = utf8_output(command.output(), &command)?;
526        truncate_line_ending(&mut stdout);
527
528        if enabled!(Level::TRACE) {
529            trace!(%stdout, "got root from buck");
530        }
531
532        Ok(stdout.into())
533    }

535    pub(crate) fn resolve_sysroot_src(&self) -> Result<PathBuf, anyhow::Error> {
536        let mut command = self.command(["audit", "config"], &DEFAULT_XPLAT_PLATFORMS);
537        command.args(["--json", "--", "rust.sysroot_src_path"]);
538        command
539            .stderr(Stdio::null())
540            .stdout(Stdio::piped())
541            .stderr(Stdio::piped());
542
543        // the `spawn()` here is load-bearing to allow spawning
544        // this command concurrently with rustc.
545        let child = command.spawn().context("Unable to spawn command")?;
546
547        #[derive(Deserialize)]
548        struct BuckConfig {
549            #[serde(rename = "rust.sysroot_src_path")]
550            sysroot_src_path: PathBuf,
551        }
552        let cfg: BuckConfig = deserialize_output(child.wait_with_output(), &command)?;
553        Ok(cfg.sysroot_src_path)
554    }

556    /// Determines the owning target(s) of the saved file and builds them.
557    #[instrument]
558    pub(crate) fn check_saved_file(
559        &self,
560        use_clippy: bool,
561        saved_file: &Path,
562    ) -> Result<CheckOutput, anyhow::Error> {
563        let mut command = self.command(["bxl"], &DEFAULT_XPLAT_PLATFORMS);
564
565        if let Some(mode) = &self.mode {
566            command.arg(mode);
567        }
568
569        command.arg("prelude//rust/rust-analyzer/check.bxl:check");
570
571        let mut file_path = saved_file.to_owned();
572        if !file_path.is_absolute() {
573            if let Ok(cwd) = std::env::current_dir() {
574                file_path = cwd.join(saved_file);
575            }
576        }
577
578        // apply BXL scripts-specific arguments:
579        command.args(["--", "--file"]);
580        command.arg(file_path.as_os_str());
581
582        command.args(["--use-clippy", &use_clippy.to_string()]);
583
584        // Set working directory to the containing directory of the target file.
585        // This fixes cases where the working directory happens to be an
586        // unrelated buck project (e.g. www).
587        if let Some(parent_dir) = saved_file.parent() {
588            command.current_dir(parent_dir);
589        }
590
591        let output = command.output();
592
593        let files = deserialize_output(output, &command)?;
594        Ok(files)
595    }

597    #[instrument(skip_all)]
598    pub(crate) fn expand_and_resolve(
599        &self,
600        targets: &[Target],
601        exclude_workspaces: bool,
602    ) -> anyhow::Result<ExpandedAndResolved> {
603        if targets.is_empty() {
604            return Ok(ExpandedAndResolved::default());
605        }
606
607        let xplat_platforms = infer_xplat_platforms(targets);
608        let mut command = self.command(["bxl"], &xplat_platforms);
609        if let Some(mode) = &self.mode {
610            command.arg(mode);
611        }
612        command.args([
613            "prelude//rust/rust-analyzer/resolve_deps.bxl:resolve_targets",
614            "--",
615            "--exclude_workspaces",
616            exclude_workspaces.to_string().as_str(),
617            "--targets",
618        ]);
619        command.args(targets);
620        deserialize_file_output(command.output(), &command)
621    }

623    #[instrument(skip_all)]
624    pub(crate) fn query_aliased_libraries(
625        &self,
626        targets: &[Target],
627    ) -> Result<FxHashMap<Target, AliasedTargetInfo>, anyhow::Error> {
628        // FIXME: Do this in bxl as well instead of manually writing a separate query
629
630        let xplat_platforms = infer_xplat_platforms(targets);
631        let mut command = self.command(["cquery"], &xplat_platforms);
632
633        // Fetch all aliases used by transitive deps. This is so we
634        // can translate an apparent dependency of e.g.
635        // fbsource//third-party/rust:once_cell to the actual target
636        // name of fbsource//third-party/rust:once_cell-1.15. This
637        // query also fetches non-Rust aliases, but they shouldn't
638        // hurt anything.
639        if let Some(mode) = &self.mode {
640            command.arg(mode);
641        }
642        command.args(["--output-all-attributes", "kind('^alias$', deps(%Ss))"]);
643        command.args(targets);
644
645        info!("resolving aliased libraries");
646        let raw: FxHashMap<Target, AliasedTargetInfo> =
647            deserialize_output(command.output(), &command)?;
648
649        if enabled!(Level::TRACE) {
650            for (target, info) in &raw {
651                trace!(%target, ?info, "parsed target from buck");
652            }
653        }
654        Ok(raw)
655    }

657    /// Find the buildfile that owns each file specified, and return the path to
658    /// each buildfile along with all the targets it contains.
659    #[instrument(skip_all)]
660    pub(crate) fn query_owners(
661        &self,
662        input: Input,
663        max_extra_targets: usize,
664    ) -> Result<FxHashMap<PathBuf, Vec<Target>>, anyhow::Error> {
665        let mut command = self.command(["bxl"], &DEFAULT_XPLAT_PLATFORMS);
666
667        command.args([
668            "prelude//rust/rust-analyzer/resolve_deps.bxl:resolve_owning_buildfile",
669            "--",
670        ]);
671
672        info!(
673            kind = "progress",
674            ?input,
675            "querying buck to determine owning buildfile and its targets"
676        );
677
678        match input {
679            Input::Targets(targets) => {
680                command.arg("--targets");
681                command.args(targets);
682            }
683            Input::Files(files) => {
684                command.arg("--files");
685                command.args(files);
686            }
687            Input::Buildfile(files) => {
688                command.arg("--buildfiles");
689                command.args(files);
690            }
691        };
692
693        command.arg("--max_extra_targets");
694        command.arg(max_extra_targets.to_string());
695
696        let out: FxHashMap<PathBuf, Vec<Target>> = deserialize_output(command.output(), &command)?;
697
698        for (k, v) in out.iter() {
699            info!("Found {} with {} targets", k.display(), v.len());
700        }
701
702        Ok(out)
703    }
704}

712pub(crate) fn utf8_output(
713    output: io::Result<Output>,
714    command: &Command,
715) -> Result<String, anyhow::Error> {
716    match output {
717        Ok(Output {
718            stdout,
719            stderr,
720            status,
721        }) if status.success() => String::from_utf8(stdout)
722            .or_else(|err| {
723                let context = cmd_err(command, status, &stderr);
724                Err(err).context(context)
725            })
726            .context("command returned non-utf8 output"),
727        Ok(Output {
728            stdout: _,
729            stderr,
730            status,
731        }) => Err(cmd_err(command, status, &stderr))
732            .with_context(|| format!("command ended with {status}")),
733        Err(err) => Err(err)
734            .with_context(|| format!("command `{command:?}`"))
735            .context("failed to execute command"),
736    }
737}
738
739fn deserialize_output<T>(output: io::Result<Output>, command: &Command) -> Result<T, anyhow::Error>
740where
741    T: for<'a> Deserialize<'a>,
742{
743    match output {
744        Ok(Output {
745            stdout,
746            stderr,
747            status,
748        }) => {
749            // If we have a non-zero exit code due to a compiler crash (a rustc
750            // ICE), then we should exit with a non-zero exit code. If the
751            // compiler is crashing, we may not be showing all diagnostics, so
752            // the user should know something is wrong.
753            //
754            // Ignore non-zero exit codes otherwise. It's possible to configure
755            // a build to fail on warnings, such that we get well-formed JSON of
756            // the rustc diagnostics but the exit code is non-zero (D46666035).
757            if !status.success() {
758                let stderr_str = String::from_utf8_lossy(&stderr);
759                if stderr_str.contains("error: the compiler unexpectedly panicked") {
760                    return Err(anyhow::anyhow!("{}", stderr_str));
761                }
762            }
763
764            tracing::debug!(?command, "parsing command output");
765            serde_json::from_slice(&stdout)
766                .with_context(|| cmd_err(command, status, &stderr))
767                .context("failed to deserialize command output")
768        }
769        Err(err) => Err(err)
770            .with_context(|| format!("command `{command:?}`"))
771            .context("failed to execute command"),
772    }
773}
774
775fn deserialize_file_output<T>(
776    output: io::Result<Output>,
777    command: &Command,
778) -> Result<T, anyhow::Error>
779where
780    T: for<'a> Deserialize<'a>,
781{
782    match output {
783        Ok(Output {
784            stdout,
785            stderr,
786            status,
787        }) => {
788            tracing::debug!(?command, "parsing file output");
789            serde_json_from_stdout_path(&stdout)
790                .with_context(|| cmd_err(command, status, &stderr))
791                .context("failed to deserialize command output")
792        }
793        Err(err) => Err(err)
794            .with_context(|| format!("command `{command:?}`"))
795            .context("failed to execute command"),
796    }
797}

799fn serde_json_from_stdout_path<T>(stdout: &[u8]) -> Result<T, anyhow::Error>
800where
801    T: for<'a> Deserialize<'a>,
802{
803    let file_path = std::str::from_utf8(stdout)?;
804    let file_path = Path::new(file_path.lines().next().context("no file path in output")?);
805    let contents =
806        fs::read_to_string(file_path).with_context(|| format!("failed to read {file_path:?}"))?;
807    serde_json::from_str(&contents).context("failed to deserialize file")
808}

810fn cmd_err(command: &Command, status: ExitStatus, stderr: &[u8]) -> anyhow::Error {
811    anyhow::anyhow!(
812        "command `{:?}` (exit code: {})\nstderr:\n{}",
813        command,
814        status,
815        String::from_utf8_lossy(stderr),
816    )
817}

819/// Trim a trailing new line from `String`.
820/// Useful when trimming command output.
821pub(crate) fn truncate_line_ending(s: &mut String) {
822    if let Some(x) = s.strip_suffix("\r\n").or_else(|| s.strip_suffix('\n')) {
823        s.truncate(x.len());
824    }

827pub(crate) fn select_mode(mode: Option<&str>) -> Option<String> {
828    if let Some(mode) = mode {
829        Some(mode.to_owned())
830    } else if cfg!(all(fbcode_build, target_os = "windows")) {
831        Some("@fbcode//mode/win".to_owned())
832    } else {
833        // fallback to the platform default mode. This is likely slower than optimal, but
834        // `rust-project check` will work.
835        None
836    }
837}

839fn remove_duplicates_preserve_order(paths: Vec<PathBuf>) -> Vec<PathBuf> {
840    let mut uniq_paths = Vec::with_capacity(paths.len());
841    let mut seen = FxHashSet::default();
842
843    for path in paths {
844        if seen.insert(path.clone()) {
845            uniq_paths.push(path);
846        }
847    }
848
849    uniq_paths
850}
