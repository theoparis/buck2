33/// Choose sysroot and sysroot_src based on platform.
34///
35/// `sysroot` is the directory that contains std crates:
36/// <https://doc.rust-lang.org/rustc/command-line-arguments.html#--sysroot-override-the-system-root>
37/// and also contains libexec helpers such as rust-analyzer-proc-macro-srv.
38///
39/// `sysroot_src` is the directory that contains the source to std crates:
40/// <https://rust-analyzer.github.io/manual.html#non-cargo-based-projects>
41#[instrument]
42pub(crate) fn resolve_buckconfig_sysroot(
43    buck: &Buck,
44    project_root: &Path,
45) -> Result<Sysroot, anyhow::Error> {
46    let sysroot: PathBuf = {
47        // TODO(diliopoulos): remove hardcoded path to toolchain sysroot and replace with something
48        // derived from buck, e.g.
49        //
50        // $ buck cquery -u fbcode//buck2/integrations/rust-project:rust-project -a compiler fbcode//buck2/platform/rust:rust_bootstrap
51        // ...
52        //     "compiler": "fbcode//tools/build/buck/wrappers:rust-platform010-clang-17-nosan-compiler (fbcode//buck2/platform/execution:linux-x86_64#54c5d1cbad5316cb)"
53        // $ buck cquery -u fbcode//buck2/integrations/rust-project:rust-project -a exe fbcode//tools/build/buck/wrappers:rust-platform010-clang-17-nosan-compiler
54        // ...
55        //     "exe": "fbcode//third-party-buck/platform010/build/rust/llvm-fb-17:bin/rustc (fbcode//buck2/platform/execution:linux-x86_64#54c5d1cbad5316cb)",
56        let fbsource_rustc = project_root.join("xplat/rust/toolchain/current/basic/bin/rustc");
57        let mut sysroot_cmd = Command::new(fbsource_rustc);
58        sysroot_cmd
59            .arg("--print=sysroot")
60            .stdin(Stdio::null())
61            .stdout(Stdio::piped())
62            .stderr(Stdio::piped());
63        let sysroot_child = sysroot_cmd.spawn()?;
64
65        let mut sysroot = utf8_output(sysroot_child.wait_with_output(), &sysroot_cmd)
66            .context("error asking rustc for sysroot")?;
67        truncate_line_ending(&mut sysroot);
68        sysroot.into()
69    };
70
71    let sysroot_src = buck.resolve_sysroot_src()?;
72    let sysroot_targets = Target::new(format!("fbsource//{}:", sysroot_src.to_string_lossy()));
73    // the `library` path component needs to be appended to the `sysroot_src_path`
74    // so that rust-analyzer will be able to find standard library sources.
75    let sysroot_src = project_root.join(sysroot_src).join("library");
76
77    let mut sysroot_project = develop_with_sysroot(
78        buck,
79        vec![sysroot_targets],
80        Sysroot {
81            sysroot: sysroot.clone(),
82            sysroot_src: Some(sysroot_src.clone()),
83            sysroot_project: None,
84        },
85        true,
86        false,
87        false,
88        &[], // sysroot doesn't get any extra cfgs
89    )?;
90    for krate in &mut sysroot_project.crates {
91        if let Some(display_name) = &mut krate.display_name {
92            *display_name = display_name
93                .strip_suffix("-0.0.0") // rust-analyzer identifies lang crates by name, so we need `core-0.0.0` to be `core`
94                .unwrap_or(display_name)
95                .to_owned();
96        }
97    }
98
99    Ok(Sysroot {
100        sysroot,
101        sysroot_src: Some(sysroot_src),
102        sysroot_project: Some(sysroot_project),
103    })
104}

106#[instrument(ret)]
107pub(crate) fn resolve_rustup_sysroot() -> Result<Sysroot, anyhow::Error> {
108    let mut cmd = Command::new("rustc");
109    cmd.arg("--print=sysroot")
110        .stdin(Stdio::null())
111        .stdout(Stdio::piped())
112        .stderr(Stdio::piped());
113
114    let mut output = utf8_output(cmd.output(), &cmd)?;
115    truncate_line_ending(&mut output);
116    let sysroot = PathBuf::from(output);
117    let sysroot_src = sysroot
118        .join("lib")
119        .join("rustlib")
120        .join("src")
121        .join("rust")
122        .join("library");
123
124    let sysroot = Sysroot {
125        sysroot,
126        sysroot_src: Some(sysroot_src),
127        sysroot_project: None, // rustup sysroot is not buckified
128    };
129    Ok(sysroot)
130}
