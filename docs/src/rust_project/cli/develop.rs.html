44pub(crate) struct OutputCfg {
45    out: Output,
46    pretty: bool,
47}

49#[derive(Debug)]
50pub(crate) enum Output {
51    Path(PathBuf),
52    Stdout,
53}
54
55impl Develop {
56    pub(crate) fn from_command(command: Command) -> (Develop, Input, OutputCfg) {
57        if let crate::Command::Develop {
58            files,
59            targets,
60            out,
61            stdout,
62            prefer_rustup_managed_toolchain,
63            sysroot,
64            pretty,
65            mode,
66            check_cycles,
67            buck2_command,
68            include_all_buildfiles,
69            ..
70        } = command
71        {
72            let out = if stdout {
73                Output::Stdout
74            } else {
75                Output::Path(out)
76            };
77
78            let sysroot = if prefer_rustup_managed_toolchain {
79                SysrootConfig::Rustup
80            } else if let Some(sysroot) = sysroot {
81                SysrootConfig::Sysroot(sysroot)
82            } else {
83                SysrootConfig::BuckConfig
84            };
85
86            let mode = select_mode(mode.as_deref());
87            let buck = buck::Buck::new(buck2_command.clone(), mode);
88
89            let develop = Develop {
90                sysroot,
91                buck,
92                check_cycles,
93                invoked_by_ra: false,
94                include_all_buildfiles,
95            };
96            let out = OutputCfg { out, pretty };
97
98            let input = if !targets.is_empty() {
99                let targets = targets.into_iter().map(Target::new).collect();
100                Input::Targets(targets)
101            } else {
102                Input::Files(files)
103            };
104
105            return (develop, input, out);
106        }
107
108        if let crate::Command::DevelopJson {
109            sysroot_mode,
110            args,
111            buck2_command,
112            ..
113        } = command
114        {
115            let out = Output::Stdout;
116
117            let sysroot = match sysroot_mode {
118                crate::SysrootMode::BuckConfig => SysrootConfig::BuckConfig,
119                crate::SysrootMode::Rustc => SysrootConfig::Rustup,
120                crate::SysrootMode::FullPath(path) => SysrootConfig::Sysroot(path),
121                crate::SysrootMode::Command(cmd_args) => {
122                    let cmd = cmd_args[0].clone();
123                    let args = cmd_args[1..].to_vec();
124                    let output = std::process::Command::new(cmd).args(args).output().unwrap();
125                    let path = String::from_utf8(output.stdout).unwrap();
126                    SysrootConfig::Sysroot(PathBuf::from(path.trim()))
127                }
128            };
129
130            let buck = buck::Buck::new(buck2_command, None);
131
132            let develop = Develop {
133                sysroot,
134                buck,
135                check_cycles: false,
136                invoked_by_ra: true,
137                include_all_buildfiles: false,
138            };
139            let out = OutputCfg { out, pretty: false };
140
141            let input = match args {
142                crate::JsonArguments::Path(path) => Input::Files(vec![path]),
143                crate::JsonArguments::Buildfile(path) => Input::Buildfile(vec![path]),
144                crate::JsonArguments::Label(target) => Input::Targets(vec![Target::new(target)]),
145            };
146
147            return (develop, input, out);
148        }
149
150        unreachable!("No other subcommand is supported.")
151    }
152}

154const DEFAULT_EXTRA_TARGETS: usize = 50;

156#[derive(Serialize, Deserialize)]
157pub(crate) struct OutputData {
158    pub(crate) buildfile: PathBuf,
159    pub(crate) project: ProjectJson,
160    pub(crate) kind: String,
161}
162
163impl Develop {
164    pub(crate) fn run(self, input: Input, cfg: OutputCfg) -> Result<(), anyhow::Error> {
165        let start = std::time::Instant::now();
166        let input = match input {
167            Input::Targets(targets) => Input::Targets(targets),
168            Input::Files(files) => {
169                let canonical_files = files
170                    .into_iter()
171                    .map(|p| safe_canonicalize(&p))
172                    .collect::<Vec<_>>();
173
174                Input::Files(canonical_files)
175            }
176            Input::Buildfile(buildfiles) => Input::Buildfile(buildfiles),
177        };
178        let mut writer: BufWriter<Box<dyn Write>> = match cfg.out {
179            Output::Path(ref p) => {
180                let out = std::fs::File::create(p)?;
181                BufWriter::new(Box::new(out))
182            }
183            Output::Stdout => BufWriter::new(Box::new(std::io::stdout())),
184        };
185
186        let targets = self.related_targets(input.clone())?;
187        if targets.is_empty() {
188            let err = anyhow::anyhow!("No owning target found")
189                .context(format!("Could not find owning target for {input:?}"));
190            return Err(err);
191        }
192
193        if self.invoked_by_ra {
194            for (buildfile, targets) in targets {
195                let project = self.run_inner(targets)?;
196
197                // we have to log before we write the output, because rust-analyzer will kill us after the write
198                crate::scuba::log_develop(start.elapsed(), input.clone(), self.invoked_by_ra);
199
200                let out = OutputData {
201                    buildfile,
202                    project,
203                    kind: "finished".to_owned(),
204                };
205                let out = serde_json::to_string(&out)?;
206                println!("{out}");
207            }
208        } else {
209            let mut targets = targets.into_values().flatten().collect::<Vec<_>>();
210            targets.sort();
211            targets.dedup();
212
213            let project = self.run_inner(targets)?;
214            crate::scuba::log_develop(start.elapsed(), input, self.invoked_by_ra);
215
216            if cfg.pretty {
217                serde_json::to_writer_pretty(&mut writer, &project)?;
218            } else {
219                serde_json::to_writer(&mut writer, &project)?;
220            }
221            writeln!(writer)?;
222            match &cfg.out {
223                Output::Path(p) => info!(file = ?p, "wrote rust-project.json"),
224                Output::Stdout => info!("wrote rust-project.json to stdout"),
225            }
226        }
227
228        Ok(())
229    }

231    pub(crate) fn run_inner(&self, targets: Vec<Target>) -> Result<ProjectJson, anyhow::Error> {
232        let Develop {
233            sysroot,
234            buck,
235            check_cycles,
236            include_all_buildfiles,
237            ..
238        } = self;
239
240        info!(kind = "progress", "fetching sysroot");
241        let sysroot = match &sysroot {
242            SysrootConfig::Sysroot(path) => Sysroot {
243                sysroot: safe_canonicalize(&expand_tilde(path)?),
244                sysroot_src: None,
245                sysroot_project: None,
246            },
247            SysrootConfig::BuckConfig => {
248                let project_root = buck.resolve_project_root()?;
249                resolve_buckconfig_sysroot(&buck, &project_root)?
250            }
251            SysrootConfig::Rustup => resolve_rustup_sysroot()?,
252        };
253
254        let exclude_workspaces =
255            std::env::var("RUST_PROJECT_EXCLUDE_WORKSPACES").is_ok_and(|it| it != "0");
256
257        // FIXME(JakobDegen): This should be set via a configuration mechanism of some kind.
258        #[cfg(not(fbcode_build))]
259        let extra_cfgs = &["test".to_owned()];
260        #[cfg(fbcode_build)]
261        let extra_cfgs = &["test".to_owned(), "fbcode_build".to_owned()];
262
263        develop_with_sysroot(
264            buck,
265            targets,
266            sysroot,
267            exclude_workspaces,
268            *check_cycles,
269            *include_all_buildfiles,
270            extra_cfgs,
271        )
272    }

274    /// For every Rust file, return the relevant buck targets that should be used to configure rust-analyzer.
275    pub(crate) fn related_targets(
276        &self,
277        input: Input,
278    ) -> Result<FxHashMap<PathBuf, Vec<Target>>, anyhow::Error> {
279        // We want to load additional targets from the enclosing buildfile, to help users
280        // who have a bunch of small targets in their buildfile. However, we want to set a limit
281        // so we don't try to load everything in very large generated buildfiles.
282        let max_extra_targets: usize = match std::env::var("RUST_PROJECT_EXTRA_TARGETS") {
283            Ok(s) => s.parse::<usize>().unwrap_or(DEFAULT_EXTRA_TARGETS),
284            Err(_) => DEFAULT_EXTRA_TARGETS,
285        };
286
287        // We always want the targets that directly own these Rust files.
288        self.buck.query_owners(input, max_extra_targets)
289    }
290}

292fn expand_tilde(path: &Path) -> Result<PathBuf, anyhow::Error> {
293    if path.starts_with("~") {
294        let path = path.strip_prefix("~")?;
295        let home = std::env::var("HOME")?;
296        let home = PathBuf::from(home);
297        Ok(home.join(path))
298    } else {
299        Ok(path.to_path_buf())
300    }
301}

303pub(crate) fn develop_with_sysroot(
304    buck: &Buck,
305    targets: Vec<Target>,
306    sysroot: Sysroot,
307    exclude_workspaces: bool,
308    check_cycles: bool,
309    include_all_buildfiles: bool,
310    extra_cfgs: &[String],
311) -> Result<ProjectJson, anyhow::Error> {
312    info!(kind = "progress", "building generated code");
313    let expanded_and_resolved = buck.expand_and_resolve(&targets, exclude_workspaces)?;
314
315    info!(kind = "progress", "resolving aliased libraries");
316    let aliased_libraries =
317        buck.query_aliased_libraries(&expanded_and_resolved.expanded_targets)?;
318
319    info!(
320        kind = "progress",
321        "converting buck info to rust-project.json"
322    );
323    let rust_project = to_project_json(
324        sysroot,
325        expanded_and_resolved,
326        aliased_libraries,
327        check_cycles,
328        include_all_buildfiles,
329        extra_cfgs,
330        buck,
331    )?;
332
333    Ok(rust_project)
334}
