16#[cfg(fbcode_build)]
17pub(crate) fn log_develop(duration: Duration, input: Input, invoked_by_ra: bool) {
18    if !is_ci() {
19        let mut sample = new_sample("develop");
20        sample.add("duration_ms", duration.as_millis() as i64);
21        sample.add("input", format!("{input:?}"));
22        sample.add("revision", get_sl_revision());
23        sample.add("invoked_by_ra", invoked_by_ra);
24        sample.log();
25        sample.flush(Duration::from_millis(500));
26    }
27}
28
29#[cfg(not(fbcode_build))]
30pub(crate) fn log_develop(_duration: Duration, _input: Input, _invoked_by_ra: bool) {}

48#[cfg(fbcode_build)]
49fn get_sl_revision() -> String {
50    std::process::Command::new("sl")
51        .arg("id")
52        .output()
53        .ok()
54        .and_then(|output| String::from_utf8(output.stdout).ok())
55        .unwrap_or("unknown".to_owned())
56}

86#[cfg(fbcode_build)]
87fn new_sample(kind: &str) -> scuba::ScubaSampleBuilder {
88    let fb = fbinit::expect_init();
89    let mut sample = scuba::ScubaSampleBuilder::new(fb, "rust_project");
90    sample.add("root_span", kind);
91    sample.add("unixname", whoami::username());
92    sample.add(
93        "hostname",
94        whoami::fallible::hostname()
95            .unwrap_or("unknown hostname".to_owned())
96            .to_ascii_lowercase(),
97    );
98
99    // RA_PROXY_SESSION_ID is an environment variable set by the VS Code extension when it starts
100    // rust-analyzer-proxy. rust-analyzer-proxy then starts rust-analyzer with the same
101    // environment, and rust-analyzer invokes rust-project with the inherited environment.
102    if let Ok(session_id) = std::env::var("RA_PROXY_SESSION_ID") {
103        sample.add("session_id", session_id);
104    }
105    sample
106}
