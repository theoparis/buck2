24/// Includes all user related computation-specific data.
25#[derive(Allocative)]
26pub struct UserComputationData {
27    /// The DiceData provides a spot for users to attach whatever extra things they want.
28    ///
29    /// This can contain arbitrary data from users that will not be part of the dice graph.
30    /// As an example, users may want to inject some form of event dispatcher to send events from their computations.
31    pub data: DiceData,
32    pub tracker: Arc<dyn DiceEventListener>,
33    #[allocative(skip)]
34    pub spawner: Arc<dyn Spawner<Self>>,
35
36    #[allocative(skip)]
37    pub cycle_detector: Option<Arc<dyn UserCycleDetector>>,
38
39    #[allocative(skip)]
40    pub activation_tracker: Option<Arc<dyn ActivationTracker>>,
41
42    /// We require that UserComputationData always be constructed with `..Default::default()`
43    pub _requires_default: RequireDefault,
44}
45
46/// A UserCycleDetector can be used for custom cycle detection in the DICE computation.
47pub trait UserCycleDetector: Send + Sync + 'static {
48    /// Called by DICE when it starts computing a key. `key` will be a user Key type (and so user can reliably downcast it to known types).
49    fn start_computing_key(&self, key: &DynKey) -> Option<Arc<dyn UserCycleDetectorGuard>>;
50
51    /// Called by DICE when the key finished computing.
52    fn finished_computing_key(&self, key: &DynKey);
53}
54
55/// A UserCycleDetectorGuard is used to track the currently computing key. User code can access this through
56/// ComputationData::cycle_guard() (and then downcast it with as_any to potentially access custom cycle behavior).
57pub trait UserCycleDetectorGuard: AsAnyArc + Send + Sync + 'static {
58    /// Called by dice when a dependency edge is encountered.
59    fn add_edge(&self, key: &DynKey);
60
61    /// Used in error messages.
62    fn type_name(&self) -> &'static str;
63}
64
65pub trait AsAnyArc {
66    fn as_any_arc(self: Arc<Self>) -> Arc<dyn Any + Send + Sync>;
67}
68
69impl<T: Send + Sync + 'static> AsAnyArc for T {
70    fn as_any_arc(self: Arc<Self>) -> Arc<dyn Any + Send + Sync> {
71        self
72    }
73}

75#[derive(Allocative)]
76pub struct RequireDefault(());
77
78impl UserComputationData {
79    pub fn new() -> Self {
80        Self::default()
81    }
82}
83
84impl Default for UserComputationData {
85    fn default() -> Self {
86        Self {
87            data: DiceData::new(),
88            tracker: Arc::new(NoOpTracker),
89            spawner: Arc::new(TokioSpawner),
90            cycle_detector: None,
91            activation_tracker: None,
92            _requires_default: RequireDefault(()),
93        }
94    }
95}
