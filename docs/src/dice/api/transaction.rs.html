26/// The struct for which we build transactions. This is where changes are recorded, and committed
27/// to DICE, which returns the Transaction where we spawn computations.
28#[derive(Allocative)]
29#[repr(transparent)]
30pub struct DiceTransactionUpdater(pub(crate) DiceTransactionUpdaterImpl);
31
32impl DiceTransactionUpdater {
33    pub fn existing_state(&self) -> impl Future<Output = DiceTransaction> {
34        self.0.existing_state()
35    }
36
37    /// Records a set of `Key`s as changed so that they, and any dependents will
38    /// be recomputed on the next set of requests at the next version.
39    pub fn changed<K, I>(&mut self, changed: I) -> DiceResult<()>
40    where
41        K: Key,
42        I: IntoIterator<Item = K> + Send + Sync + 'static,
43    {
44        self.0.changed(changed)
45    }
46
47    /// Records a set of `Key`s as changed to a particular value so that any
48    /// dependents will be recomputed on the next set of requests. The
49    /// `Key`s themselves will be update to the new value such that they
50    /// will not need to be recomputed as long as they aren't recorded to be
51    /// `changed` again (or invalidated by other means). Calling this method
52    /// does not in anyway alter the types of the key such that they
53    /// permanently becomes a special "inject value only" key.
54    pub fn changed_to<K, I>(&mut self, changed: I) -> DiceResult<()>
55    where
56        K: Key,
57        I: IntoIterator<Item = (K, K::Value)> + Send + Sync + 'static,
58    {
59        self.0.changed_to(changed)
60    }
61
62    /// Commit the changes registered via 'changed' and 'changed_to' to the current newest version.
63    pub fn commit(self) -> impl Future<Output = DiceTransaction> {
64        self.0.commit()
65    }
66
67    /// Commit the changes registered via 'changed' and 'changed_to' to the current newest version,
68    /// replacing the user data with the given set
69    pub fn commit_with_data(
70        self,
71        extra: UserComputationData,
72    ) -> impl Future<Output = DiceTransaction> {
73        self.0.commit_with_data(extra)
74    }
75
76    pub fn unstable_take(self) -> Self {
77        Self(self.0.unstable_take())
78    }
79}

81/// The base struct for which all computations start. This is clonable, and dupe, and can be
82/// moved to different runtimes to start computations.
83/// All computations on this transaction will see only changes at the most-up-to-date version at
84/// the time of creation of this transaction.
85///
86/// This SHOULD NOT be ever stored by computations, or any results of computations.
87#[derive(Allocative, Clone, Dupe)]
88pub struct DiceTransaction(pub(crate) DiceTransactionImpl);
89
90impl DiceTransaction {
91    /// Returns whether the `DiceTransaction` is equivalent. Equivalent is defined as whether the
92    /// two Transactions are based off the same underlying set of key states. That is, all
93    /// injected keys are the same, and the same compute keys are dirtied, and that any computations
94    /// that occur between the two transactions can be shared.
95    pub fn equivalent<E>(&self, other: &E) -> bool
96    where
97        E: DiceEquivalent,
98    {
99        self.version_for_equivalence() == other.version_for_equivalence()
100    }
101
102    pub fn equality_token(&self) -> DiceEquality {
103        DiceEquality(self.0.get_version())
104    }
105
106    /// Creates an Updater to record changes to DICE that upon committing, creates a new transaction
107    /// that keeps the same set of user data. This is equivalent to `Dice::updater_with_user_data(data)`
108    /// where the `data` is taken from the current Transaction.
109    pub fn into_updater(self) -> DiceTransactionUpdater {
110        self.0.into_updater()
111    }
112}

128pub trait DiceEquivalent: private::Sealed {

132impl DiceEquivalent for DiceTransaction {

134        self.equality_token()
135    }

138impl DiceEquivalent for DiceEquality {
139    fn version_for_equivalence(&self) -> DiceEquality {
140        *self
141    }
