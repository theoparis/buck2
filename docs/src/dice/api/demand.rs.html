11use std::any::TypeId;
12use std::marker::PhantomData;
13
14pub(crate) trait DemandImpl {
15    fn tag() -> TypeId;
16}
17
18pub(crate) struct DemandValue<T: 'static> {
19    pub(crate) value: Option<T>,
20}
21
22pub(crate) struct DemandRef<'a, T: ?Sized + 'static> {
23    pub(crate) value: Option<&'a T>,
24}
25
26impl<T: 'static> DemandImpl for DemandValue<T> {
27    fn tag() -> TypeId {
28        TypeId::of::<DemandValue<T>>()
29    }
30}
31
32impl<T: ?Sized + 'static> DemandImpl for DemandRef<'_, T> {
33    fn tag() -> TypeId {
34        TypeId::of::<DemandRef<'static, T>>()
35    }
36}
37
38/// Supplied value can be passed using this object.
39pub struct Demand<'a> {
40    /// What we want?
41    demand_impl_tag: TypeId,
42    // SAFETY: Must be a `&'a mut` to a `DemandRef<T>` or `DemandValue<T>`, matching the `TypeId` above
43    demand_impl: *mut (),
44    _phantom: PhantomData<&'a mut ()>,
45}
46
47impl<'a> Demand<'a> {
48    pub(crate) fn new<I>(demand_impl: &'a mut I) -> Demand<'a>
49    where
50        I: DemandImpl,
51    {
52        Demand {
53            demand_impl_tag: I::tag(),
54            demand_impl: demand_impl as *mut _ as *mut (),
55            _phantom: PhantomData,
56        }
57    }
58
59    /// Provide a value. Discard the value if the type is not matched.
60    pub fn provide_value<T: 'static>(&mut self, value: T) {
61        self.provide_value_with(|| value);
62    }
63
64    /// Provide a value. Discard the value if the type is not matched.
65    pub fn provide_value_with<T: 'static>(&mut self, f: impl FnOnce() -> T) {
66        if self.demand_impl_tag == DemandValue::<T>::tag() {
67            // SAFETY: We've checked that this is a pointer to a `DemandValue<T>`
68            let demand_impl = unsafe { &mut *(self.demand_impl as *mut DemandValue<T>) };
69            demand_impl.value = Some(f());
70        }
71    }
72
73    /// Provide a reference. Discard the value if the type is not matched.
74    pub fn provide_ref<T: ?Sized + 'static>(&mut self, value: &'a T) {
75        self.provide_ref_with(|| value);
76    }
77
78    /// Provide a reference. Discard the value if the type is not matched.
79    pub fn provide_ref_with<T: ?Sized + 'static>(&mut self, f: impl FnOnce() -> &'a T) {
80        if self.demand_impl_tag == DemandRef::<T>::tag() {
81            // SAFETY: We've checked that this is a pointer to a `DemandRef<T>`
82            let demand_impl = unsafe { &mut *(self.demand_impl as *mut DemandRef<T>) };
83            demand_impl.value = Some(f());
84        }
85    }
86}

88#[cfg(test)]
89mod tests {
90    use allocative::Allocative;
91    use async_trait::async_trait;
92    use buck2_futures::cancellation::CancellationContext;
93
94    use crate::Demand;
95    use crate::DiceComputations;
96    use crate::DynKey;
97    use crate::Key;
98
99    #[test]
100    fn test_request() {
101        #[derive(derive_more::Display, Eq, PartialEq, Hash, Allocative, Clone, Debug)]
102        struct MKey(u32);
103
104        #[async_trait]
105        impl Key for MKey {
106            type Value = ();
107
108            async fn compute(
109                &self,
110                _ctx: &mut DiceComputations,
111                _cancellations: &CancellationContext,
112            ) -> Self::Value {
113                panic!("not needed in tests")
114            }
115
116            fn equality(_x: &Self::Value, _y: &Self::Value) -> bool {
117                panic!("not needed in tests")
118            }
119
120            fn provide<'a>(&'a self, demand: &mut Demand<'a>) {
121                demand.provide_value(self.0);
122                demand.provide_ref::<dyn Foo>(self);
123            }
124        }
125
126        trait Foo {
127            fn foo(&self) -> u32;
128        }
129
130        impl Foo for MKey {
131            fn foo(&self) -> u32 {
132                self.0
133            }
134        }
135
136        let key = MKey(17);
137        let key = DynKey::from_key(key);
138        assert_eq!(Some(17), key.request_value::<u32>());
139        assert_eq!(None, key.request_value::<i32>());
140        assert_eq!(17, key.request_ref::<dyn Foo>().unwrap().foo());
141        assert!(key.request_ref::<dyn std::io::Read>().is_none());
142    }
