11//! Dice computations track "invalidation paths" for every node in the computation.
12//!
13//! Invalidation path tracking is a data flow analysis that tracks the flow of changed
14//! state through the computation graph. Directly invalidated or injected nodes
15//! are the root "invalidation sources". For every computed node, its invalidation paths
16//! will each either be "clean" (if all of its dependencies' invalidation paths are clean)
17//! or will point to one of its non-clean dependencies . When multiple deps
18//! are not clean, it will point to the one with the most recently changed invalidation
19//! source. In this way, we record an invalidation path through the graph to the invalidation
20//! source for every affected node.
21//!
22//! We track invalidations of two priorities: "normal" and "high". Each key type specifies
23//! its own invalidation source priority that is used when that node is directly invalidated or
24//! injected. Each node will record the most recent invalidation source for each priority.
25//! A high priority invalidation will also count for normal priority tracking (but not the
26//! reverse). So if a node has two deps A and B, and A has a normal priority invalidation
27//! at v1 and B has a high priority invalidation at v2, B would be recorded for both normal
28//! and high priority source. If those versions were reversed (A at v2, B at v1), we'd
29//! record A for the normal priority invalidation path (because its invalidation is more recent) and would
30//! record B for the high priority invalidation path (because A is normal priority).

43/// The invalidation paths for a key. This is accessible from [`crate::DiceComputations::get_invalidation_paths()`]
44#[derive(Debug)]
45pub struct DiceKeyTrackedInvalidationPaths {
46    pub normal_priority_path: DiceTrackedInvalidationPath,
47    pub high_priority_path: DiceTrackedInvalidationPath,
48}

50/// The invalidation path state for a computation node.
51#[derive(Debug, VariantName)]
52pub enum DiceTrackedInvalidationPath {
53    /// Indicates no (non-Ignored) invalidated source data flowed into this computation.
54    Clean,
55    /// The invalidation path state is unknown. We only track the most recent invalidation source for a node, and if we compute the value
56    /// at an older version we won't know the invalidation source.
57    Unknown,
58    /// Invalidated data has flowed into this computation. [DiceInvalidationPath] holds the invalidation path.
59    Invalidated(DiceInvalidationPath),
60}
61
62pub struct DiceInvalidationPath {
63    dice: Arc<DiceModern>,
64    data: crate::arc::Arc<InvalidationPathNode>,
65}
66
67impl std::fmt::Debug for DiceInvalidationPath {
68    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
69        f.debug_struct("DiceInvalidationPath")
70            .field("data", &self.data)
71            .finish()
72    }
73}
74
75/// A node in the invalidation path.
76pub struct InvalidationPathEntry {
77    pub key: DynKey,
78    pub version: VersionNumber,
79}
80
81impl DiceInvalidationPath {
82    /// Returns the "invalidation path".
83    ///
84    /// This will contain all nodes from the invalidation source to this node indicating how the
85    /// data has flowed to this node. The first entry of the result will be the invalidation source.
86    pub fn get_invalidation_path(&self) -> Vec<InvalidationPathEntry> {
87        let mut res = Vec::new();
88
89        let mut data = &self.data;
90        loop {
91            let key = self.dice.key_index.get(data.key);
92            res.push(InvalidationPathEntry {
93                key: DynKey { erased: key.dupe() },
94                version: data.version,
95            });
96            match &data.cause {
97                InvalidationPath::Invalidated(cause) => {
98                    data = cause;
99                }
100                _ => {
101                    break;
102                }
103            }
104        }
105
106        res.reverse();
107        res
108    }
109}
110
111impl DiceKeyTrackedInvalidationPaths {
112    pub(crate) fn new(
113        dice: Arc<DiceModern>,
114        normal_priority_path: InvalidationPath,
115        high_priority_path: InvalidationPath,
116    ) -> Self {
117        Self {
118            normal_priority_path: DiceTrackedInvalidationPath::new(
119                dice.dupe(),
120                normal_priority_path,
121            ),
122            high_priority_path: DiceTrackedInvalidationPath::new(dice, high_priority_path),
123        }
124    }
125}
126
127impl DiceTrackedInvalidationPath {
128    pub(crate) fn new(dice: Arc<DiceModern>, path: InvalidationPath) -> Self {
129        match path {
130            InvalidationPath::Clean => DiceTrackedInvalidationPath::Clean,
131            InvalidationPath::Unknown => DiceTrackedInvalidationPath::Unknown,
132            InvalidationPath::Invalidated(data) => {
133                DiceTrackedInvalidationPath::Invalidated(DiceInvalidationPath { dice, data })
134            }
135        }
136    }
