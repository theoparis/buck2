11use std::any::Any;
12
13use crate::DynKey;
14
15/// An ActivationTracker can be used to identify which keys were either reused or computed during a
16/// transaction.
17pub trait ActivationTracker: Send + Sync + 'static {
18    /// Receives when a key was activated (computed, or reused). The caller will want to downcast
19    /// the key and deps to types they care about. The caller also receives whatever the key passed
20    /// to `store_evaluation_data` (if any).
21    fn key_activated(
22        &self,
23        key: &DynKey,
24        deps: &mut dyn Iterator<Item = &DynKey>,
25        activation_data: ActivationData,
26    );
27}
28
29/// Describes the kind of activation, and possibly carries data passed by the key's evaluation.
30pub enum ActivationData {
31    /// This key was evaluated. Evaluation data will be passed if the key's evaluation set any.
32    Evaluated(Option<Box<dyn Any + Send + Sync + 'static>>),
33
34    /// This key was reused. No data is passed.
35    Reused,
36}
