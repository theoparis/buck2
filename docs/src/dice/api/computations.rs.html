30/// The context for computations to register themselves, and request for additional dependencies.
31/// The dependencies accessed are tracked for caching via the `DiceCtx`.
32///
33/// The computations are registered onto `DiceComputations` via implementing traits for the
34/// `DiceComputation`.
35///
36/// The context is valid only for the duration of the computation of a single key, and cannot be
37/// owned.
38#[derive(Allocative)]
39pub struct DiceComputations<'a>(pub(crate) DiceComputationsImpl<'a>);

89    /// DiceComputations &mut-based api can make some computations much more complex to express, but without it
90    /// the data dependencies between different compute requests are impossible to track. with_linear_recompute()
91    /// is an escape hatch in the case where we are willing to sacrifice recompute performance for easier expression
92    /// of a computation. It should not be used lightly, it can be difficult to attribute performance regressions to its use.
93    ///
94    /// Withing the with_linear_recompute(), all deps will be recorded as accessed sequentially and so recomputation
95    /// will not trigger them in parallel. This will apply to calls on DiceComputations returned by .ctx() and by the ones received
96    /// in closures from compute_many and friends. It will not apply to the other deps from which the linear recompute was created.
97    ///
98    /// For example:
99    ///
100    /// ```ignore
101    /// let mut ctx = something();
102    /// let keys1 = vec![Key(10), Key(11)];
103    /// let keys2 = vec![Key(20), Key(21)];
104    /// let keys3 = vec![Key(30), Key(31)];
105    /// let keys4 = vec![Key(40), Key(41)]
106    /// ctx.compute_join(keys1, |ctx, key1| ctx.compute(key1).boxed()).await;
107    /// ctx.with_linear_recompute(|mut linear| {
108    ///     linear.ctx().compute_join(keys2, |ctx, key2| async move {
109    ///         ctx.compute2(
110    ///             |ctx| ctx.compute(key2).boxed(),
111    ///             |ctx| ctx.compute_join(keys3, |ctx, key3| ctx.compute(key3).boxed()).boxed()
112    ///         ).await;
113    ///     }.boxed()
114    /// ).await
115    /// ctx.compute_join(keys4, |ctx, key4| ctx.compute(key4).boxed()).await;
116    /// });
117    /// ```
118    ///
119    /// In this example, the recomputation of all of keys2 and keys3 would be done linearly, but keys1 and keys4 would be recomputed in parallel.
120    pub fn with_linear_recompute<'a, Func, Fut, T>(
121        &'a mut self,
122        func: Func,
123    ) -> impl Future<Output = T> + use<'a, Func, Fut, T>
124    where
125        Func: FnOnce(LinearRecomputeDiceComputations<'a>) -> Fut,
126        Fut: Future<Output = T>,
127    {
128        self.0.with_linear_recompute(func)
129    }

131    /// Creates computation Futures for all the given tasks.
132    ///
133    /// ```ignore
134    /// let mut ctx: &'a DiceComputations = ctx();
135    /// let data: String = data();
136    /// let keys : Vec<Key> = keys();
137    /// let futs = ctx.compute_many(keys.into_iter().map(|k|
138    ///   DiceComputations::declare_closure(
139    ///     |dice: &mut DiceComputations| -> BoxFuture<String> {
140    ///       async move {
141    ///         dice.compute(k).await + data
142    ///       }.boxed()
143    ///     }
144    ///   )
145    /// ));
146    /// futures::future::join_all(futs).await;
147    /// ```
148    pub fn compute_many<'a, Computes, F, T>(
149        &'a mut self,
150        computes: Computes,
151    ) -> Vec<impl Future<Output = T> + use<'a, Computes, F, T>>
152    where
153        Computes: IntoIterator<Item = F>,
154        F: for<'x> FnOnce(&'x mut DiceComputations<'a>) -> BoxFuture<'x, T> + Send,
155    {
156        self.0.compute_many(computes)
157    }

159    /// Maps the items into computation futures and joins on them.
160    ///
161    /// ```ignore
162    /// let mut ctx: &'a DiceComputations = ctx();
163    /// let data: String = data();
164    /// let keys : Vec<Key> = keys();
165    /// // When defined inplance, there's no need to use a declare helper.
166    /// ctx.compute_join(keys, |dice: &mut DiceComputations, k: &Key| {
167    ///     async move {
168    ///       dice.compute(k).await + data
169    ///     }
170    ///   }).await;
171    ///
172    /// // If the closure is going to be declared outside the compute_many itself, you need to use
173    /// // declare_join_closure for it to get the right lifetime bounds.
174    /// let compute_one = DiceComputations::declare_join_closure(
175    ///   |dice: &mut DiceComputations, k: &Key| {
176    ///     async move {
177    ///       dice.compute(k).await + data
178    ///     }
179    ///   }
180    /// );
181    /// ctx.compute_join(keys, compute_one).await;
182    /// ````
183    pub fn compute_join<'a, Items, Mapper, T, R>(
184        &'a mut self,
185        items: Items,
186        mapper: Mapper,
187    ) -> impl Future<Output = Vec<R>> + use<'a, Items, Mapper, T, R>
188    where
189        Items: IntoIterator<Item = T>,
190        Mapper: for<'x> FnOnce(&'x mut DiceComputations<'a>, T) -> BoxFuture<'x, R>
191            + Send
192            + Sync
193            + Copy,
194        T: Send,
195    {
196        let futs = self.compute_many(items.into_iter().map(move |v| {
197            DiceComputations::declare_closure(move |ctx: &mut DiceComputations| -> BoxFuture<R> {
198                mapper(ctx, v)
199            })
200        }));
201        futures::future::join_all(futs)
202    }
203
204    /// Maps the items into computations futures and then returns a future which represents either a
205    /// collection of the results or an error.
206    pub fn try_compute_join<'a, Items, Mapper, T, R, E>(
207        &'a mut self,
208        items: Items,
209        mapper: Mapper,
210    ) -> impl Future<Output = Result<Vec<R>, E>> + use<'a, Items, Mapper, T, R, E>
211    where
212        Items: IntoIterator<Item = T>,
213        Mapper: for<'x> FnOnce(&'x mut DiceComputations<'a>, T) -> BoxFuture<'x, Result<R, E>>
214            + Send
215            + Sync
216            + Copy,
217        T: Send,
218    {
219        let futs = self.compute_many(items.into_iter().map(move |v| {
220            DiceComputations::declare_closure(
221                move |ctx: &mut DiceComputations| -> BoxFuture<Result<R, E>> { mapper(ctx, v) },
222            )
223        }));
224        crate::future::try_join_all(futs)
225    }
