11//!
12//! A cache that deals with versions
13//!
14//! This is responsible for performing incremental caching and invalidations
15//! with multiple versions in-flight at the same time.
16//!
17//! The 'VersionedCache' will track dependency edges and use computed version
18//! number for each cache entry and a global version counter to determine
19//! up-to-date-ness of cache entries.

50/// Actual entries as seen when querying the VersionedGraph.
51///
52/// This is responsible for tracking the information related to a single
53/// node (i.e. a DiceKey) required for incremental computations.
54#[derive(UnpackVariants, Allocative, Debug)]
55pub(crate) enum VersionedGraphNode {
56    Occupied(OccupiedGraphNode),
57    Injected(InjectedGraphNode),
58    Vacant(VacantGraphNode),
59}
60
61impl VersionedGraphNode {
62    pub(crate) fn force_dirty(
63        &mut self,
64        v: VersionNumber,
65        invalidation_priority: InvalidationSourcePriority,
66    ) -> InvalidateResult {
67        match self {
68            VersionedGraphNode::Occupied(e) => e.force_dirty(v, invalidation_priority),
69            VersionedGraphNode::Vacant(e) => {
70                if e.force_dirty(v, invalidation_priority) {
71                    InvalidateResult::Changed(None)
72                } else {
73                    InvalidateResult::NoChange
74                }
75            }
76            VersionedGraphNode::Injected(e) => {
77                panic!("injected keys don't get invalidated (`{e:?}`)")
78            }
79        }
80    }
81
82    pub(crate) fn mark_invalidated(
83        &mut self,
84        v: VersionNumber,
85        invalidation_priority: Option<InvalidationSourcePriority>,
86    ) -> InvalidateResult {
87        match self {
88            VersionedGraphNode::Occupied(e) => {
89                if e.mark_invalidated(v, invalidation_priority) {
90                    InvalidateResult::Changed(Some(e.metadata.rdeps.drain()))
91                } else {
92                    InvalidateResult::NoChange
93                }
94            }
95            VersionedGraphNode::Vacant(e) => {
96                panic!("vacant nodes shouldn't get invalidated (`{e:?}`)")
97            }
98            VersionedGraphNode::Injected(e) => {
99                panic!("injected keys don't get invalidated (`{e:?}`)")
100            }
101        }
102    }
103
104    /// Returns the VersionedGraphResult for the entry at the provided version.
105    pub(crate) fn at_version(&self, v: VersionNumber) -> VersionedGraphResult {
106        match self {
107            VersionedGraphNode::Occupied(entry) => entry.at_version(v),
108            VersionedGraphNode::Vacant(_) => VersionedGraphResult::Compute,
109            VersionedGraphNode::Injected(entry) => entry.at_version(v),
110        }
111    }
112
113    pub(crate) fn add_rdep_at(&mut self, v: VersionNumber, k: DiceKey) {
114        match self {
115            VersionedGraphNode::Occupied(occ) => occ.add_rdep_at(v, k),
116            VersionedGraphNode::Injected(inj) => inj.add_rdep_at(v, k),
117            VersionedGraphNode::Vacant(_) => {
118                unreachable!("we can't have an rdep on something that has never seen a value")
119            }
120        }
121    }

123    pub(crate) fn on_injected(
124        &mut self,
125        version: VersionNumber,
126        value: DiceValidValue,
127        invalidation_priority: InvalidationSourcePriority,
128    ) -> InvalidateResult {
129        match self {
130            VersionedGraphNode::Occupied(occ) => {
131                occ.on_injected(version, value, invalidation_priority)
132            }
133            VersionedGraphNode::Injected(inj) => {
134                inj.on_injected(version, value, invalidation_priority)
135            }
136            VersionedGraphNode::Vacant(vac) => {
137                let entry = OccupiedGraphNode::new(
138                    vac.key,
139                    value,
140                    Arc::new(SeriesParallelDeps::None),
141                    VersionRange::begins_with(version).into_ranges(),
142                    vac.dirtied_history.clone(),
143                    TrackedInvalidationPaths::new(invalidation_priority, vac.key, version),
144                );
145                *self = Self::Occupied(entry);
146                InvalidateResult::Changed(None)
147            }
148        }
149    }

151    /// Returns the newly updated value for the key, and whether or not any state changed.
152    #[cfg_attr(debug_assertions, instrument(level = "debug", skip(self, value, deps, reusable), fields(key = ?key, valid_deps_versions = ?valid_deps_versions)))]
153    pub(crate) fn on_computed(
154        &mut self,
155        key: super::types::VersionedGraphKey,
156        value: DiceValidValue,
157        mut valid_deps_versions: VersionRanges,
158        reusable: super::storage::ValueReusable,
159        deps: Arc<SeriesParallelDeps>,
160        mut invalidation_paths: TrackedInvalidationPaths,
161    ) -> (DiceComputedValue, bool) {
162        let (dirtied_history, overwrite_entry) = match self {
163            VersionedGraphNode::Occupied(entry) if reusable.is_reusable(&value, &deps, entry) => {
164                debug!("marking graph entry as unchanged");
165                entry.mark_unchanged(key.v, valid_deps_versions, invalidation_paths);
166                let ret = entry.computed_val(key.v);
167                return (ret, false);
168            }
169            VersionedGraphNode::Occupied(entry) => {
170                // TODO(cjhopman): Should this consider the max version in valid_deps_version rather than just key.v?
171                (
172                    &entry.metadata.dirtied_history,
173                    !entry.metadata.ever_valid_after(key.v),
174                )
175            }
176            VersionedGraphNode::Vacant(entry) => (&entry.dirtied_history, true),
177            _ => unreachable!("injected nodes are never computed"),
178        };
179
180        let (force_dirty_restricted_range, invalidation_priority) = dirtied_history.get_x(key.v);
181        if force_dirty_restricted_range.begin() > VersionNumber(0) {
182            invalidation_paths.update(TrackedInvalidationPaths::new(
183                invalidation_priority,
184                key.k,
185                force_dirty_restricted_range.begin(),
186            ))
187        }
188
189        valid_deps_versions.intersect_range(force_dirty_restricted_range);
190        let computed_version = VersionRange::bounded(key.v, VersionNumber::new(key.v.0 + 1));
191        valid_deps_versions.insert(computed_version);
192
193        if !overwrite_entry {
194            // TODO(cjhopman): Returning `true` here matches previous behavior, but it seems odd
195            // that we claim something changed when we don't change anything. It's likely that the
196            // the return value actually is used to mean something different than that we changed
197            // something.
198            debug!("skipping new graph entry because value is older than current entry");
199            return (
200                DiceComputedValue::new(
201                    MaybeValidDiceValue::valid(value),
202                    Arc::new(valid_deps_versions),
203                    invalidation_paths,
204                ),
205                true,
206            );
207        }
208
209        debug!("making new graph entry because value not reusable");
210        let new = OccupiedGraphNode::new(
211            key.k,
212            value,
213            deps,
214            valid_deps_versions,
215            dirtied_history.clone(),
216            invalidation_paths,
217        );
218        let ret = new.computed_val(key.v);
219        *self = VersionedGraphNode::Occupied(new);
220
221        (ret, true)
222    }

224    pub(crate) fn to_introspectable(&self) -> Option<SerializedGraphNode> {
225        fn visit_deps<'a>(deps: impl Iterator<Item = DiceKey> + 'a) -> HashSet<KeyID> {
226            deps.map(|d| d.introspect()).collect()
227        }
228
229        fn visit_rdeps(rdeps: impl Iterator<Item = DiceKey>) -> Vec<NodeID> {
230            rdeps.unique().map(|d| NodeID(d.index as usize)).collect()
231        }
232
233        match self {
234            VersionedGraphNode::Occupied(o) => Some(SerializedGraphNode {
235                node_id: NodeID(o.key.index as usize),
236                kind: GraphNodeKind::Occupied,
237                history: crate::introspection::graph::CellHistory {
238                    valid_ranges: o.metadata.verified_ranges.to_introspectable(),
239                    force_dirtied_at: o.metadata.dirtied_history.to_introspectable(),
240                },
241                deps: Some(visit_deps(o.deps().iter_keys())),
242                rdeps: Some(visit_rdeps(o.rdeps())),
243            }),
244            VersionedGraphNode::Vacant(_) => {
245                // TODO(bobyf) should probably write the metadata of vacant
246                None
247            }
248            VersionedGraphNode::Injected(inj) => {
249                let latest = inj.latest();
250                Some(SerializedGraphNode {
251                    node_id: NodeID(inj.key.index as usize),
252                    kind: GraphNodeKind::Occupied,
253                    history: crate::introspection::graph::CellHistory {
254                        valid_ranges: latest.valid_versions.to_introspectable(),
255                        force_dirtied_at: Vec::new(),
256                    },
257                    deps: None,
258                    rdeps: Some(visit_rdeps(inj.rdeps.iter())),
259                })
260            }
261        }
262    }
263
264    pub(crate) fn intersect_valid_versions_at(
265        &self,
266        v: VersionNumber,
267        deps_verified_ranges: &mut VersionRanges,
268    ) {
269        match self.valid_versions_at(v) {
270            None => {
271                deps_verified_ranges.clear();
272            }
273            Some(valid_ranges) => {
274                deps_verified_ranges.intersect_in_place(valid_ranges);
275            }
276        }
277    }
278
279    fn valid_versions_at(&self, v: VersionNumber) -> Option<&VersionRanges> {
280        match self {
281            VersionedGraphNode::Occupied(occ) => {
282                if occ.metadata.verified_ranges.contains(v) {
283                    Some(&occ.metadata.verified_ranges)
284                } else {
285                    None
286                }
287            }
288            VersionedGraphNode::Injected(inj) => Some(&inj.data_at(v).unwrap().1.valid_versions),
289            VersionedGraphNode::Vacant(_) => {
290                unreachable!()
291            }
292        }
293    }

311/// Meta data about a DICE node, which are its edges and history information
312#[derive(Allocative, Debug)]
313pub(crate) struct NodeMetadata {
314    deps: Arc<SeriesParallelDeps>,
315    rdeps: LazyDepsSet,
316    verified_ranges: Arc<VersionRanges>,
317    dirtied_history: ForceDirtyHistory,
318}
319
320impl NodeMetadata {
321    fn should_add_rdep_at(&self, v: VersionNumber) -> bool {
322        match self.verified_ranges.last() {
323            Some(last) => match (last.begin(), last.end()) {
324                (begin, _) if begin > v => false,
325                (_, Some(_end)) => false,
326                _ => true,
327            },
328            None => true,
329        }
330    }
331
332    fn ever_valid_after(&self, v: VersionNumber) -> bool {
333        match self.verified_ranges.last() {
334            Some(last) => match last.end() {
335                Some(end) => end > v,
336                _ => true,
337            },
338            _ => false,
339        }
340    }
341}

343/// For a node, keeps a history of every version that that node has been force-dirtied at.
344///
345/// Across a force-dirtied version, we cannot ever reuse a node's value based on its deps' values not changing.
346#[derive(Allocative, Clone, Debug)]
347pub(crate) struct ForceDirtyHistory {
348    #[allow(clippy::box_collection)]
349    versions: Option<Box<(Vec<VersionNumber>, InvalidationSourcePriority)>>,
350}
351
352// the vast majority of nodes are never force-dirtied, so we want to make sure that we optimize for that.
353static_assertions::assert_eq_size!(ForceDirtyHistory, [usize; 1]);
354
355impl ForceDirtyHistory {
356    pub(crate) fn new() -> Self {
357        Self { versions: None }
358    }
359
360    /// Marks a version as force-dirtied. Returns true if the version was not already marked.
361    ///
362    /// Should only ever be called at increasing version numbers.
363    pub(crate) fn force_dirty(
364        &mut self,
365        v: VersionNumber,
366        invalidation_priority: InvalidationSourcePriority,
367    ) -> bool {
368        match &mut self.versions {
369            Some(data) => {
370                if *data.0.last().unwrap() == v {
371                    false
372                } else {
373                    data.0.push(v);
374                    true
375                }
376            }
377            None => {
378                self.versions = Some(Box::new((vec![v], invalidation_priority)));
379                true
380            }
381        }
382    }
383
384    /// Returns the force-dirtied bounds around the provided version.
385    fn restricted_range(&self, version: VersionNumber) -> VersionRange {
386        self.get_x(version).0
387    }
388
389    fn get_x(&self, version: VersionNumber) -> (VersionRange, InvalidationSourcePriority) {
390        match &self.versions {
391            None => (
392                VersionRange::begins_with(VersionNumber::ZERO),
393                InvalidationSourcePriority::Normal,
394            ),
395            Some(data) => {
396                let (dirties, invalidation_priority) = &**data;
397                let mut end = None;
398                let mut begin = None;
399                for dirty_v in dirties.iter().rev() {
400                    if *dirty_v <= version {
401                        begin = Some(*dirty_v);
402                        break;
403                    } else {
404                        end = Some(*dirty_v);
405                    }
406                }
407
408                (
409                    match (begin, end) {
410                        (Some(begin), Some(end)) => VersionRange::bounded(begin, end),
411                        (Some(begin), None) => VersionRange::begins_with(begin),
412                        (None, Some(end)) => VersionRange::bounded(VersionNumber::new(0), end),
413                        (None, None) => VersionRange::begins_with(VersionNumber::ZERO),
414                    },
415                    *invalidation_priority,
416                )
417            }
418        }
419    }
420
421    pub(crate) fn to_introspectable(&self) -> Vec<crate::introspection::graph::VersionNumber> {
422        match &self.versions {
423            Some(data) => data.0.iter().map(|v| v.to_introspectable()).collect(),
424            None => Vec::new(),
425        }
426    }
427}

429impl OccupiedGraphNode {
430    pub(crate) fn new(
431        key: DiceKey,
432        res: DiceValidValue,
433        deps: Arc<SeriesParallelDeps>,
434        verified_ranges: VersionRanges,
435        dirtied_history: ForceDirtyHistory,
436        invalidation_paths: TrackedInvalidationPaths,
437    ) -> Self {
438        Self {
439            key,
440            res,
441            metadata: NodeMetadata {
442                deps,
443                rdeps: LazyDepsSet::new(),
444                verified_ranges: Arc::new(verified_ranges),
445                dirtied_history,
446            },
447            invalidation_paths,
448        }
449    }
450
451    pub(crate) fn mark_unchanged(
452        &mut self,
453        version: VersionNumber,
454        mut valid_deps_versions: VersionRanges,
455        new_invalidation_paths: TrackedInvalidationPaths,
456    ) {
457        valid_deps_versions
458            .intersect_range(self.metadata.dirtied_history.restricted_range(version));
459        valid_deps_versions.insert(VersionRange::bounded(
460            version,
461            VersionNumber::new(version.0 + 1),
462        ));
463
464        if !valid_deps_versions.is_empty() {
465            Arc::make_mut(&mut self.metadata.verified_ranges).union_in_place(&valid_deps_versions);
466        }
467
468        self.invalidation_paths.update(new_invalidation_paths)
469    }
470
471    pub(crate) fn val(&self) -> &DiceValidValue {
472        &self.res
473    }
474
475    pub(crate) fn computed_val(&self, for_version: VersionNumber) -> DiceComputedValue {
476        DiceComputedValue::new(
477            MaybeValidDiceValue::valid(self.res.dupe()),
478            self.metadata.verified_ranges.dupe(),
479            self.invalidation_paths.at_version(for_version),
480        )
481    }
482
483    pub(crate) fn on_injected(
484        &mut self,
485        version: VersionNumber,
486        value: DiceValidValue,
487        invalidation_priority: InvalidationSourcePriority,
488    ) -> InvalidateResult {
489        // TODO(cjhopman): accepting injections only for InjectedKey would make the VersionedGraph simpler. Currently, this is used
490        // for "mocking" dice keys in tests via DiceBuilder::mock_and_return().
491        if self.val().equality(&value) {
492            // TODO(cjhopman): This is wrong. The node could currently be in a dirtied state and we
493            // aren't recording that the value is verified at this version.
494            return InvalidateResult::NoChange;
495        }
496
497        self.res = value;
498        self.metadata.deps = Arc::new(SeriesParallelDeps::None);
499        self.metadata.verified_ranges = Arc::new(VersionRange::begins_with(version).into_ranges());
500        self.invalidation_paths
501            .update(TrackedInvalidationPaths::new(
502                invalidation_priority,
503                self.key,
504                version,
505            ));
506
507        InvalidateResult::Changed(Some(self.metadata.rdeps.drain()))
508    }
509
510    fn at_version(&self, v: VersionNumber) -> VersionedGraphResult {
511        match self.metadata.verified_ranges.find_value_upper_bound(v) {
512            Some(found) if found == v => VersionedGraphResult::Match(self.computed_val(v)),
513            Some(prev_verified_version) => {
514                if self
515                    .metadata
516                    .dirtied_history
517                    .restricted_range(v)
518                    .contains(&prev_verified_version)
519                {
520                    VersionedGraphResult::CheckDeps(VersionedGraphResultMismatch {
521                        entry: self.val().dupe(),
522                        prev_verified_version,
523                        deps_to_validate: self.metadata.deps.dupe(),
524                    })
525                } else {
526                    VersionedGraphResult::Compute
527                }
528            }
529            None => VersionedGraphResult::Compute,
530        }
531    }
532
533    fn add_rdep_at(&mut self, v: VersionNumber, k: DiceKey) {
534        if self.metadata.should_add_rdep_at(v) {
535            self.metadata.rdeps.insert(v, k);
536        }
537    }
538
539    fn force_dirty(
540        &mut self,
541        v: VersionNumber,
542        invalidation_priority: InvalidationSourcePriority,
543    ) -> InvalidateResult {
544        self.mark_invalidated(v, Some(invalidation_priority));
545        if self
546            .metadata
547            .dirtied_history
548            .force_dirty(v, invalidation_priority)
549        {
550            InvalidateResult::Changed(Some(self.metadata.rdeps.drain()))
551        } else {
552            InvalidateResult::NoChange
553        }
554    }
555
556    fn mark_invalidated(
557        &mut self,
558        v: VersionNumber,
559        invalidation_priority: Option<InvalidationSourcePriority>,
560    ) -> bool {
561        if let Some(invalidation_priority) = invalidation_priority {
562            self.invalidation_paths
563                .update(TrackedInvalidationPaths::new(
564                    invalidation_priority,
565                    self.key,
566                    v,
567                ));
568        }
569        Arc::make_mut(&mut self.metadata.verified_ranges)
570            .intersect_range(VersionRange::bounded(VersionNumber::ZERO, v))
571    }
572
573    fn rdeps(&self) -> impl Iterator<Item = DiceKey> {
574        self.metadata.rdeps.iter()
575    }
576
577    pub(crate) fn deps(&self) -> &Arc<SeriesParallelDeps> {
578        &self.metadata.deps
579    }
580
581    pub(crate) fn is_verified_at(&self, version: VersionNumber) -> bool {
582        self.metadata.verified_ranges.contains(version)
583    }
584}

586/// An entry in the graph that has no computation value associated. This is used to store the
587/// history information that is known.
588/// This will be replaced by `OccupiedGraphNode` when a computed value is associated with
589/// this node. There is no guarantees of when, or even if that will occur since users may never
590/// need the associated value at this node.
591#[derive(Allocative, Debug)]
592pub(crate) struct VacantGraphNode {
593    key: DiceKey,
594    dirtied_history: ForceDirtyHistory,
595    invalidation_priority: InvalidationSourcePriority,
596}
597impl VacantGraphNode {
598    pub(crate) fn new(key: DiceKey, invalidation_priority: InvalidationSourcePriority) -> Self {
599        Self {
600            key,
601            dirtied_history: ForceDirtyHistory::new(),
602            invalidation_priority,
603        }
604    }
605
606    pub(crate) fn force_dirty(
607        &mut self,
608        v: VersionNumber,
609        invalidation_priority: InvalidationSourcePriority,
610    ) -> bool {
611        assert!(self.invalidation_priority == invalidation_priority);
612        self.dirtied_history.force_dirty(v, invalidation_priority)
613    }
614}

616/// An entry in the graph for an InjectedKey. This will store all injected values it ever sees because
617/// we cannot recompute them if they are dropped.
618#[derive(Allocative, Debug)]
619pub(crate) struct InjectedGraphNode {
620    key: DiceKey,
621    values: SortedVectorMap<VersionNumber, InjectedNodeData>,
622    rdeps: LazyDepsSet,
623    invalidation_paths: TrackedInvalidationPaths,
624}
625
626#[derive(Allocative, Debug)]
627pub(crate) struct InjectedNodeData {
628    value: DiceValidValue,
629    first_valid_version: VersionNumber,
630    // Used to cache the version ranges for `at_version`. This is a single VersionRange.
631    valid_versions: Arc<VersionRanges>,
632}
633
634impl InjectedGraphNode {
635    /// Returns a list of rdeps to invalidate and a bool indicating if the value changed. Should only ever be called at increasing version numbers.
636    pub(crate) fn on_injected(
637        &mut self,
638        version: VersionNumber,
639        value: DiceValidValue,
640        invalidation_priority: InvalidationSourcePriority,
641    ) -> InvalidateResult {
642        match self.values.values_mut().next_back() {
643            Some(v) if v.value.equality(&value) => {
644                return InvalidateResult::NoChange;
645            }
646            Some(v) => {
647                v.valid_versions =
648                    Arc::new(VersionRange::bounded(v.first_valid_version, version).into_ranges());
649            }
650            None => {}
651        };
652
653        self.values
654            .insert(version, Self::new_node_data(value, version));
655        self.invalidation_paths
656            .update(TrackedInvalidationPaths::new(
657                invalidation_priority,
658                self.key,
659                version,
660            ));
661
662        InvalidateResult::Changed(Some(self.rdeps.drain()))
663    }
664
665    pub(crate) fn at_version(&self, v: VersionNumber) -> VersionedGraphResult {
666        match self.data_at(v) {
667            Some((_, data)) => VersionedGraphResult::Match(DiceComputedValue::new(
668                MaybeValidDiceValue::valid(data.value.dupe()),
669                data.valid_versions.dupe(),
670                self.invalidation_paths.at_version(v),
671            )),
672            None => VersionedGraphResult::Compute,
673        }
674    }

676    pub(crate) fn add_rdep_at(&mut self, v: VersionNumber, k: DiceKey) {
677        for version in self.values.keys().rev() {
678            if *version <= v {
679                self.rdeps.insert(v, k);
680                return;
681            }
682        }
683        unreachable!()
684    }
685
686    pub(crate) fn new(
687        k: DiceKey,
688        v: VersionNumber,
689        value: DiceValidValue,
690        invalidation_priority: InvalidationSourcePriority,
691    ) -> InjectedGraphNode {
692        InjectedGraphNode {
693            key: k,
694            values: [(v, Self::new_node_data(value, v))].into_iter().collect(),
695            rdeps: LazyDepsSet::new(),
696            invalidation_paths: TrackedInvalidationPaths::new(invalidation_priority, k, v),
697        }
698    }
699
700    pub(crate) fn latest(&self) -> &InjectedNodeData {
701        // We don't ever create an empty values map
702        self.values.values().next_back().unwrap()
703    }
704
705    fn new_node_data(value: DiceValidValue, version: VersionNumber) -> InjectedNodeData {
706        InjectedNodeData {
707            value,
708            first_valid_version: version,
709            valid_versions: Arc::new(VersionRange::begins_with(version).into_ranges()),
710        }
711    }
712
713    fn data_at(&self, v: VersionNumber) -> Option<(&VersionNumber, &InjectedNodeData)> {
714        self.values
715            .range((Bound::Unbounded, Bound::Included(v)))
716            .next_back()
717    }
