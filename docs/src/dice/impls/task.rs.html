34pub(crate) fn spawn_dice_task<S>(
35    key: DiceKey,
36    spawner: &dyn Spawner<S>,
37    ctx: &S,
38    f: impl for<'a, 'b> FnOnce(&'a mut DiceTaskHandle<'b>) -> BoxFuture<'a, Box<dyn Any + Send>> + Send,
39) -> DiceTask {
40    let internal = DiceTaskInternal::new(key);
41
42    // detach the task, we'll cancel it explicitly if we want it canceled.
43    // we don't observe the result via the future so we can just drop that.
44    let (_fut, cancellation_handle) = spawn_dropcancel(
45        {
46            let internal = internal.dupe();
47            |cancellations| {
48                let handle = DiceTaskHandle::new(internal, cancellations);
49
50                OwningFuture::new(handle, f).boxed()
51            }
52        },
53        spawner,
54        ctx,
55    )
56    .detach();
57
58    DiceTask {
59        internal,
60        cancellations: Cancellations::new(cancellation_handle),
61    }
62}

64/// Unsafe as this creates a Task that must be completed explicitly otherwise polling will never
65/// complete.
66pub(crate) unsafe fn sync_dice_task(key: DiceKey) -> DiceTask {
67    let internal = DiceTaskInternal::new(key);
68
69    DiceTask {
70        internal,
71        cancellations: Cancellations::not_cancellable(),
72    }
73}
