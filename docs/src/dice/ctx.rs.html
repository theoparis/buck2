32/// This is a wrapper around ModernComputeCtx.
33///
34/// It forwards calls to the underlying ModernComputeCtx implementation.
35#[derive(Allocative)]
36pub(crate) struct DiceComputationsImpl<'a>(pub(crate) ModernComputeCtx<'a>);
37
38impl DiceComputationsImpl<'_> {
39    /// Gets all the result of of the given computation key.
40    /// recorded as dependencies of the current computation for which this
41    /// context is for.
42    pub(crate) fn compute<'a, K>(
43        &'a mut self,
44        key: &K,
45    ) -> impl Future<Output = DiceResult<<K as Key>::Value>> + use<'a, K>
46    where
47        K: Key,
48    {
49        self.0.compute(key)
50    }
51
52    /// Compute "opaque" value where the value is only accessible via projections.
53    /// Projections allow accessing derived results from the "opaque" value,
54    /// where the dependency of reading a projection is the projection value rather
55    /// than the entire opaque value.
56    pub(crate) fn compute_opaque<'a, K>(
57        &'a self,
58        key: &K,
59    ) -> impl Future<Output = DiceResult<OpaqueValue<K>>> + use<'a, K>
60    where
61        K: Key,
62    {
63        self.0
64            .compute_opaque(key)
65            .map(|r| r.map(|x| OpaqueValue::new(x)))
66    }
67
68    pub fn projection<K: Key, P: ProjectionKey<DeriveFromKey = K>>(
69        &mut self,
70        derive_from: &OpaqueValue<K>,
71        projection_key: &P,
72    ) -> DiceResult<P::Value> {
73        self.0
74            .projection(&derive_from.implementation, projection_key)
75    }
76
77    pub fn opaque_into_value<K: Key>(
78        &mut self,
79        derive_from: OpaqueValue<K>,
80    ) -> DiceResult<K::Value> {
81        Ok(self.0.opaque_into_value(derive_from.implementation))
82    }
83
84    /// Computes all the given tasks in parallel, returning an unordered Stream
85    pub(crate) fn compute_many<'a, Computes, F, T>(
86        &'a mut self,
87        computes: Computes,
88    ) -> Vec<impl Future<Output = T> + use<'a, Computes, F, T>>
89    where
90        Computes: IntoIterator<Item = F>,
91        F: for<'x> FnOnce(&'x mut DiceComputations<'a>) -> BoxFuture<'x, T> + Send,
92    {
93        self.0.compute_many(computes)
94    }
95
96    pub(crate) fn compute2<'a, Compute1, T, Compute2, U>(
97        &'a mut self,
98        compute1: Compute1,
99        compute2: Compute2,
100    ) -> (
101        impl Future<Output = T> + use<'a, Compute1, T, Compute2, U>,
102        impl Future<Output = U> + use<'a, Compute1, T, Compute2, U>,
103    )
104    where
105        Compute1: for<'x> FnOnce(&'x mut DiceComputations<'a>) -> BoxFuture<'x, T> + Send,
106        Compute2: for<'x> FnOnce(&'x mut DiceComputations<'a>) -> BoxFuture<'x, U> + Send,
107    {
108        self.0.compute2(compute1, compute2)
109    }
110
111    pub(crate) fn compute3<'a, Compute1, T, Compute2, U, Compute3, V>(
112        &'a mut self,
113        compute1: Compute1,
114        compute2: Compute2,
115        compute3: Compute3,
116    ) -> (
117        impl Future<Output = T> + use<'a, Compute1, T, Compute2, U, Compute3, V>,
118        impl Future<Output = U> + use<'a, Compute1, T, Compute2, U, Compute3, V>,
119        impl Future<Output = V> + use<'a, Compute1, T, Compute2, U, Compute3, V>,
120    )
121    where
122        Compute1: for<'x> FnOnce(&'x mut DiceComputations<'a>) -> BoxFuture<'x, T> + Send,
123        Compute2: for<'x> FnOnce(&'x mut DiceComputations<'a>) -> BoxFuture<'x, U> + Send,
124        Compute3: for<'x> FnOnce(&'x mut DiceComputations<'a>) -> BoxFuture<'x, V> + Send,
125    {
126        self.0.compute3(compute1, compute2, compute3)
127    }
128
129    pub(crate) fn with_linear_recompute<'a, Func, Fut, T>(
130        &'a mut self,
131        func: Func,
132    ) -> impl Future<Output = T> + use<'a, Func, Fut, T>
133    where
134        Func: FnOnce(LinearRecomputeDiceComputations<'a>) -> Fut,
135        Fut: Future<Output = T>,
136    {
137        self.0.with_linear_recompute(func)
138    }
139
140    /// Data that is static per the entire lifetime of Dice. These data are initialized at the
141    /// time that Dice is initialized via the constructor.
142    pub(crate) fn global_data(&self) -> &DiceData {
143        self.0.global_data()
144    }
145
146    /// Data that is static for the lifetime of the current request context. This lifetime is
147    /// the lifetime of the top-level `DiceComputation` used for all requests.
148    /// The data is also specific to each request context, so multiple concurrent requests can
149    /// each have their own individual data.
150    pub(crate) fn per_transaction_data(&self) -> &UserComputationData {
151        self.0.per_transaction_data()
152    }
153
154    pub(crate) fn cycle_guard<T: UserCycleDetectorGuard>(&self) -> DiceResult<Option<Arc<T>>> {
155        self.0.cycle_guard()
156    }
157
158    pub fn store_evaluation_data<T: Send + Sync + 'static>(&self, value: T) -> DiceResult<()> {
159        self.0.store_evaluation_data(value)
160    }
161
162    pub(crate) fn get_version(&self) -> VersionNumber {
163        self.0.get_version()
164    }
165
166    pub fn get_invalidation_paths(&mut self) -> DiceKeyTrackedInvalidationPaths {
167        self.0.get_invalidation_paths()
168    }
169}
