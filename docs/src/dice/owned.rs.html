11//! Provides a pointer that is owned uniquely by a holder, which can handout an un-lifetimed reference.
12//! The reference can only be accessed via a lambda providing an optional access based on whether
13//! the reference is still valid.
14
15// Intend to use this again when parallel computations have correct dep recording.
16#![allow(unused)]
17
18use std::ops::Deref;
19use std::sync::Arc;
20use std::sync::Weak;
21
22use allocative::Allocative;
23use dupe::Clone_;
24use dupe::Dupe_;
25
26/// Owns 'T' uniquely.
27/// Note that it is up to the user the ensure there is no race between dropping the 'Owned' value
28/// and any 'Ref' accesses.
29#[derive(Allocative)]
30pub struct Owned<T>(Arc<T>);
31
32impl<T> Deref for Owned<T> {
33    type Target = T;
34
35    fn deref(&self) -> &Self::Target {
36        &self.0
37    }
38}
39
40/// Reference that has no compile time lifetimes and isn't guaranteed to be alive
41#[derive(Allocative, Clone_, Dupe_)]
42pub struct Ref<T>(Weak<T>);
43
44impl<T> Owned<T> {
45    pub fn new(t: T) -> Self {
46        Owned(Arc::new(t))
47    }
48
49    pub fn as_ref(&self) -> Ref<T> {
50        Ref(Arc::downgrade(&self.0))
51    }
52
53    /// Effectively drops the 'Owned' and gets a unique ownership to the underlying data.
54    /// Note that it is up to the user the ensure there is no race between calling `into_inner`
55    /// and any 'Ref' accesses.
56    pub fn unwrap_inner(self) -> anyhow::Result<T> {
57        Arc::into_inner(self.0).ok_or_else(|| anyhow::anyhow!("Some reference is using it"))
58    }
59}
60
61impl<T> Ref<T> {
62    pub fn maybe_access<U>(&self, f: impl FnOnce(&T) -> U) -> Option<U> {
63        self.0.upgrade().map(|t| f(&*t))
64    }
65
66    /// gets the reference. Ensures that the reference is alive at time of access, but does not
67    /// prevent the `Owned` from being dropped while the returned reference is alive
68    pub unsafe fn deref(&self) -> Option<&T> {
69        if Weak::strong_count(&self.0) > 0 {
70            Some(unsafe { &*Weak::as_ptr(&self.0) })
71        } else {
72            None
73        }
74    }
75}

77#[cfg(test)]
78mod tests {
79    use crate::owned::Owned;
80
81    #[test]
82    fn valid_references() {
83        let owner = Owned::new(());
84
85        let reference = owner.as_ref();
86        assert!(reference.maybe_access(|_x| ()).is_some());
87
88        owner.unwrap_inner().unwrap();
89
90        assert!(
91            reference
92                .maybe_access(|_x| panic!("no reference"))
93                .is_none()
94        );
95    }
96}
