28pub enum Tier {
29    // Expected errors in inputs explicitly tracked by buck.
30    Input,
31    // Errors that may be triggered by issues with the host,
32    // resource limits, non-explicit dependencies or potentially
33    // ambiguous input errors.
34    // These can be tracked but not eliminated.
35    Environment,
36    // Unexpected errors in buck2 or core dependencies.
37    // It should be possible to eliminate these, in theory.
38    Tier0,
39}
40
41struct TagMetadata {
42    category: Option<Tier>,
43    rank: u32,
44    // If true and an error includes non-generic tags,
45    // generic tags will be excluded from category key.
46    generic: bool,
47    // Hidden tags are only used to determine category,
48    // they are excluded from both category keys and error_tags
49    // reported externally.
50    // These should be removed/avoided if possible.
51    hidden: bool,
52    // Allows overriding the exit code derived from the error tier.
53    exit_code: ExitCode,
54}
55
56impl TagMetadata {
57    fn generic(self, generic: bool) -> Self {
58        Self { generic, ..self }
59    }
60
61    fn hidden(self) -> Self {
62        Self {
63            hidden: true,
64            ..self
65        }
66    }
67
68    fn exit_code(self, exit_code: ExitCode) -> Self {
69        Self { exit_code, ..self }
70    }
71}

73macro_rules! rank {
74    ( $tier:ident ) => {
75        match stringify!($tier) {
76            "environment" => TagMetadata {
77                category: Some(Tier::Environment),
78                rank: line!(),
79                generic: false,
80                hidden: false,
81                exit_code: ExitCode::InfraError,
82            },
83            "tier0" => TagMetadata {
84                category: Some(Tier::Tier0),
85                rank: line!(),
86                generic: false,
87                hidden: false,
88                exit_code: ExitCode::InfraError,
89            },
90            "input" => TagMetadata {
91                category: Some(Tier::Input),
92                rank: line!(),
93                generic: false,
94                hidden: false,
95                exit_code: ExitCode::UserError,
96            },
97            "unspecified" => TagMetadata {
98                category: None,
99                rank: line!(),
100                generic: true,
101                hidden: false,
102                exit_code: ExitCode::UnknownFailure,
103            },
104            _ => unreachable!(),
105        }
106    };
107}

119pub trait ErrorTagExtra {
120    fn source_area(&self) -> ErrorSourceArea;
121    fn exit_code(&self) -> ExitCode;
122}

124impl ErrorTagExtra for ErrorTag {
125    fn source_area(&self) -> ErrorSourceArea {
126        let tag_name = self.as_str_name();
127        if tag_name.starts_with("IO_EDEN") {
128            ErrorSourceArea::Eden
129        } else if tag_name.starts_with("RE") {
130            ErrorSourceArea::Re
131        } else if tag_name.starts_with("WATCHMAN") {
132            ErrorSourceArea::Watchman
133        } else if *self == crate::ErrorTag::Tpx || *self == crate::ErrorTag::TestExecutor {
134            ErrorSourceArea::TestExecutor
135        } else if tag_name.starts_with("INSTALLER") {
136            ErrorSourceArea::Installer
137        } else {
138            ErrorSourceArea::Buck2
139        }
140    }
141
142    fn exit_code(&self) -> ExitCode {
143        tag_metadata(*self).exit_code
144    }
145}

425/// Errors can be categorized by tags only if they have any non-generic tags.
426pub fn tag_is_generic(tag: &ErrorTag) -> bool {
427    let metadata = tag_metadata(*tag);
428    metadata.generic || metadata.hidden
429}
430
431/// Hidden tags only used internally, for categorization.
432pub fn tag_is_hidden(tag: &ErrorTag) -> bool {
433    tag_metadata(*tag).hidden
434}

436pub trait ErrorLike {
437    fn best_tag(&self) -> Option<ErrorTag>;
438
439    fn error_rank(&self) -> u32;
440
441    fn category(&self) -> Tier;
442}
443
444impl ErrorLike for buck2_data::ErrorReport {
445    fn best_tag(&self) -> Option<ErrorTag> {
446        best_tag(self.tags.iter().filter_map(|t| {
447            // This should never be `None`, but with weak prost types,
448            // it is safer to just ignore incorrect integers.
449            ErrorTag::try_from(*t).ok()
450        }))
451    }
452
453    fn error_rank(self: &buck2_data::ErrorReport) -> u32 {
454        self.best_tag().map(tag_rank).unwrap_or(u32::MAX)
455    }
456
457    fn category(&self) -> Tier {
458        self.best_tag()
459            .map(|t| tag_metadata(t).category)
460            .flatten()
461            .unwrap_or(Tier::Tier0)
462    }
463}

465/// Pick the most interesting error by best tag.
466pub fn best_error<'a>(
467    tags: impl IntoIterator<Item = &'a buck2_data::ErrorReport>,
468) -> Option<&'a buck2_data::ErrorReport> {
469    tags.into_iter().min_by_key(|e| e.error_rank())
470}
471
472/// Pick the most interesting tag from a list of tags.
473pub fn best_tag(tags: impl IntoIterator<Item = ErrorTag>) -> Option<ErrorTag> {
474    tags.into_iter().min_by_key(|t| tag_rank(*t))
475}
476
477/// Tag rank: smaller is more interesting.
478fn tag_rank(tag: ErrorTag) -> u32 {
479    tag_metadata(tag).rank
480}
481
482/// Some tags are known to be either infrastructure or user errors.
483pub(crate) fn error_tag_category(tag: ErrorTag) -> Option<Tier> {
484    tag_metadata(tag).category
485}
486
487// Buck2 is the fallback/default source area, use the first non-buck2 source area.
488pub fn source_area(tags: impl IntoIterator<Item = ErrorTag>) -> ErrorSourceArea {
489    tags.into_iter()
490        .find_map(|tag| {
491            let area = tag.source_area();
492            if area != ErrorSourceArea::Buck2 {
493                Some(area)
494            } else {
495                None
496            }
497        })
498        .unwrap_or(ErrorSourceArea::Buck2)
499}
