39// Helper function that can be explicited called to convert `std::error::Error` into `buck2_error`.
40// Common types should have a proper From implemented in this file, but this function is useful for
41// one-off error types in the codebase
42#[cold]
43#[track_caller]
44pub fn from_any_with_tag<T>(e: T, tag: ErrorTag) -> crate::Error
45where
46    T: Into<anyhow::Error>,
47    // This bound prevent this function from being called on an error that's
48    // already a `buck2_error` which prevents unnecessary conversions
49    Result<(), T>: anyhow::Context<(), T>,
50{
51    let anyhow: anyhow::Error = e.into();
52    let source_location =
53        crate::source_location::SourceLocation::new(std::panic::Location::caller().file());
54    recover_crate_error(anyhow.as_ref(), source_location, tag)
55}
