11/// Common exit codes for buck with stronger semantic meanings
12///
13/// The exit code is u8 integer and has the following meanings for common exit codes.
14/// - Success             : 0
15/// - Uncategorized Error : 1
16/// - Infra Error         : 2
17/// - User Error          : 3
18/// - Signal Interruption : 129-192 (128 + signal number)
19#[derive(Clone, Copy, Debug)]
20pub enum ExitCode {
21    Success,
22    UnknownFailure,
23    InfraError,
24    UserError,
25    DaemonIsBusy,
26    DaemonPreempted,
27    Timeout,
28    ConnectError,
29    SignalInterrupt,
30    BrokenPipe,
31    /// Test runner explicitly requested that this exit code be returned
32    TestRunner(u8),
33}
34
35impl ExitCode {
36    pub const fn exit_code(self) -> u32 {
37        use ExitCode::*;
38        match self {
39            Success => 0,
40            UnknownFailure => 1,
41            InfraError => 2,
42            UserError => 3,
43            DaemonIsBusy => 4,
44            DaemonPreempted => 5,
45            Timeout => 6,
46            ConnectError => 11,
47            // Exit codes after 129 should be reserved for exits caused by signals,
48            // which should be 128 + signal number.
49            // However, SIGINT and SIGPIPE are reversed..
50            BrokenPipe => 130,      // 128 + SIGINT(2)
51            SignalInterrupt => 141, // 128 + SIGPIPE(13)
52            TestRunner(code) => code as u32,
53        }
54    }
55
56    pub fn name(self) -> &'static str {
57        use ExitCode::*;
58        match self {
59            Success => "SUCCESS",
60            UnknownFailure => "UNKNOWN_FAILURE",
61            InfraError => "INFRA_ERROR",
62            UserError => "USER_ERROR",
63            DaemonIsBusy => "DAEMON_IS_BUSY",
64            DaemonPreempted => "DAEMON_PREEMPTED",
65            Timeout => "TIMEOUT",
66            ConnectError => "CONNECT_ERROR",
67            BrokenPipe => "BROKEN_PIPE",
68            SignalInterrupt => "SIGNAL_INTERRUPT",
69            TestRunner(_) => "TEST_RUNNER",
70        }
71    }
72}
