18#[derive(allocative::Allocative, Clone)]
19pub enum ContextValue {
20    Dyn(Arc<str>),
21    Tags(SmallVec<[crate::ErrorTag; 1]>),
22    Typed(Arc<dyn TypedContext>),
23    StarlarkError(StarlarkContext),
24    StringTag(StringTag),
25}
26
27#[derive(allocative::Allocative, Debug, Clone, Eq, PartialEq)]
28pub struct StringTag {
29    pub tag: String,
30}
31
32impl ContextValue {
33    /// Returns whether the context should be included in the error message
34    pub(crate) fn should_display(&self) -> bool {
35        match self {
36            Self::Dyn(..) => true,
37            Self::Typed(e) => e.should_display(),
38            Self::Tags(_) => false,
39            Self::StringTag(..) => false,
40            Self::StarlarkError(..) => false,
41        }
42    }
43
44    #[cfg(test)]
45    pub(crate) fn assert_eq(&self, other: &Self) {
46        match (self, other) {
47            (ContextValue::Dyn(a), ContextValue::Dyn(b)) => {
48                assert_eq!(a, b);
49            }
50            (ContextValue::Tags(a), ContextValue::Tags(b)) => {
51                assert_eq!(a, b);
52            }
53            (ContextValue::Typed(left), ContextValue::Typed(right)) => {
54                assert!(left.eq(&**right))
55            }
56            (ContextValue::StringTag(a), ContextValue::StringTag(b)) => {
57                assert_eq!(a, b);
58            }
59            (ContextValue::StarlarkError(a), ContextValue::StarlarkError(b)) => {
60                assert_eq!(a, b);
61            }
62            (_, _) => panic!("context variants don't match!"),
63        }
64    }
65}
66
67impl std::fmt::Display for ContextValue {
68    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
69        match self {
70            Self::Dyn(v) => f.write_str(v),
71            Self::Tags(tags) => write!(f, "{tags:?}"),
72            Self::Typed(v) => std::fmt::Display::fmt(v, f),
73            Self::StringTag(v) => f.write_str(&v.tag),
74            Self::StarlarkError(v) => write!(f, "{v}"),
75        }
76    }
77}
78
79impl From<String> for ContextValue {
80    fn from(value: String) -> Self {
81        ContextValue::Dyn(value.into())
82    }
83}
84
85impl From<&str> for ContextValue {
86    fn from(value: &str) -> Self {
87        ContextValue::Dyn(value.into())
88    }
89}

91pub trait TypedContext:
92    allocative::Allocative + Send + Sync + std::fmt::Display + std::any::Any + 'static
93{
94    fn eq(&self, other: &dyn TypedContext) -> bool;
95
96    fn should_display(&self) -> bool {
97        true
98    }
99}
100
101impl<T: TypedContext> From<T> for ContextValue {
102    fn from(value: T) -> Self {
103        ContextValue::Typed(Arc::new(value))
104    }
105}

107#[derive(Clone, allocative::Allocative, Debug, PartialEq, Eq, Hash)]
108pub struct StarlarkContext {
109    // TODO(minglunli): We could take in the CallStack type and do some magic to make it look nicer
110    // but I think just concatenating the string form and it's accurate and look good enough
111    pub call_stack: String,
112    pub error_msg: String,
113    pub span: Option<FileSpan>,
114}
115
116impl StarlarkContext {
117    pub fn concat(&self, other: Option<Self>) -> Self {
118        if let Some(ctx) = other {
119            let trimmed = ctx
120                .call_stack
121                .trim_start_matches(starlark_syntax::call_stack::CALL_STACK_TRACEBACK_PREFIX);
122            let call_stack = format!("{}{}", self.call_stack, trimmed);
123            Self {
124                call_stack,
125                error_msg: ctx.error_msg.clone(),
126                span: ctx.span.clone(),
127            }
128        } else {
129            self.clone()
130        }
131    }
132}
133
134impl std::fmt::Display for StarlarkContext {
135    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
136        let span = span_display(
137            self.span.as_ref().map(|s| s.as_ref()),
138            self.error_msg.as_str(),
139            false,
140        );
141        write!(f, "{}\n{}", self.call_stack, span)
142    }
143}
