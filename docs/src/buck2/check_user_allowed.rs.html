11#[cfg(windows)]
12pub(crate) fn check_user_allowed() -> buck2_error::Result<()> {
13    use std::io;
14    use std::mem;
15    use std::mem::MaybeUninit;
16    use std::ptr;
17
18    use buck2_core::ci::is_ci;
19    use buck2_error::BuckErrorContext;
20    use buck2_wrapper_common::win::winapi_handle::WinapiHandle;
21    use winapi::ctypes::c_void;
22    use winapi::shared::minwindef::DWORD;
23    use winapi::um::processthreadsapi::GetCurrentProcess;
24    use winapi::um::processthreadsapi::OpenProcessToken;
25    use winapi::um::securitybaseapi::GetTokenInformation;
26    use winapi::um::winnt::TOKEN_ELEVATION;
27    use winapi::um::winnt::TOKEN_QUERY;
28    use winapi::um::winnt::TokenElevation;
29
30    let mut handle = ptr::null_mut();
31    let token_ok = unsafe { OpenProcessToken(GetCurrentProcess(), TOKEN_QUERY, &mut handle) };
32    if token_ok == 0 {
33        return Err(io::Error::last_os_error()).buck_error_context("OpenProcessToken failed");
34    }
35
36    let handle = unsafe {
37        WinapiHandle::new_check_last_os_error(handle).buck_error_context("OpenProcessToken")?
38    };
39    let size = mem::size_of::<TOKEN_ELEVATION>();
40    let elevation: MaybeUninit<TOKEN_ELEVATION> = MaybeUninit::zeroed();
41    let mut ret_size = 0;
42
43    let success_get = unsafe {
44        GetTokenInformation(
45            handle.handle(),
46            TokenElevation,
47            elevation.as_ptr() as *mut c_void,
48            size as DWORD,
49            &mut ret_size,
50        )
51    };
52    if success_get == 0 {
53        return Err(io::Error::last_os_error()).buck_error_context("GetTokenInformation failed");
54    }
55
56    let elevation_struct: TOKEN_ELEVATION = unsafe { elevation.assume_init() };
57    if elevation_struct.TokenIsElevated == 1 {
58        // In CI, if buck2 got run from an admin shell, we need not worry that a
59        // subsequent invocation might come from a non-admin shell. It almost
60        // certainly will not.
61        if !is_ci()? {
62            tracing::warn!(
63                "You're running buck2 from an admin shell. Invocations from non-admin shells will likely fail going forward. To remediate, run `buck2 clean` in this admin shell, then switch to a non-admin shell."
64            );
65        }
66    }
67    Ok(())
68}

70#[cfg(not(windows))]
71pub(crate) fn check_user_allowed() -> buck2_error::Result<()> {
72    use std::os::unix::fs::MetadataExt;
73
74    use buck2_core::fs::fs_util;
75    use buck2_core::fs::paths::abs_path::AbsPath;
76    use buck2_core::soft_error;
77    use buck2_error::BuckErrorContext;
78
79    #[derive(Debug, buck2_error::Error)]
80    #[error("buck2 is not allowed to run as root (unless home dir is owned by root)")]
81    #[buck2(tag = Input)]
82    struct RootError;
83
84    if nix::unistd::geteuid().is_root() {
85        let home_dir = dirs::home_dir().buck_error_context("home dir not found")?;
86        if let Ok(home_dir) = AbsPath::new(&home_dir) {
87            let home_dir_metadata = fs_util::metadata(home_dir)?;
88            if home_dir_metadata.uid() != 0 {
89                soft_error!("root_not_allowed", RootError.into())?;
90            }
91        }
92    }
93    Ok(())
94}
