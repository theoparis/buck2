11//! The panic hook, shared by the buck2 CLI and daemon.
12//!
13//! The Buck2 CLI unwinds the stack on a panic, while the Buck2 daemon terminates without unwinding.
14//! This module sets up a shared panic hook to run before unwinding/termination for crash reporting.
15
16use std::panic;
17use std::panic::PanicHookInfo;
18
19use buck2_error::BuckErrorContext;
20use fbinit::FacebookInit;
21
22/// Initializes the panic hook.
23pub fn initialize() -> anyhow::Result<()> {
24    let hook = panic::take_hook();
25    panic::set_hook(Box::new(move |info| {
26        let fb = buck2_common::fbinit::get_or_init_fbcode_globals();
27        the_panic_hook(fb, info);
28        hook(info);
29    }));
30    buck2_core::error::initialize(Box::new(move |category, err, loc, options| {
31        let fb = buck2_common::fbinit::get_or_init_fbcode_globals();
32        imp::write_soft_error(
33            fb,
34            category,
35            err,
36            buck2_data::Location {
37                file: loc.0.to_owned(),
38                line: loc.1,
39                column: loc.2,
40            },
41            options,
42        );
43    }))
44    .buck_error_context_anyhow("Error initializing soft errors")?;
45    Ok(())
46}
47
48/// The panic hook, initialized during `initialize`. Invoked immediately on a panic, but prior to the process being
49/// aborted.
50///
51/// The panic hook is called on the same thread that the panic occurred. It is possible to perform a backtrace here
52/// to collect additional information.
53fn the_panic_hook(fb: FacebookInit, info: &PanicHookInfo) {
54    imp::write_panic_to_scribe(fb, info);
55}

72    fn get_stack() -> Vec<buck2_data::structured_error::StackFrame> {
73        fn ptr_to_string<T>(ptr: *mut T) -> String {
74            format!("0x{:x}", ptr as usize)
75        }
76
77        let trace = Backtrace::new();
78        trace
79            .frames()
80            .iter()
81            .map(|frame| {
82                let symbols = frame
83                    .symbols()
84                    .iter()
85                    .map(|symbol| buck2_data::structured_error::Symbol {
86                        name: symbol
87                            .name()
88                            .map_or_else(|| "".to_owned(), |s| s.to_string()),
89                        address: symbol.addr().map_or_else(|| "".to_owned(), ptr_to_string),
90                        file: symbol
91                            .filename()
92                            .map_or_else(|| "".to_owned(), |p| p.display().to_string()),
93                        line: symbol.lineno().unwrap_or(0),
94                        column: symbol.colno().unwrap_or(0),
95                    })
96                    .collect::<Vec<_>>();
97
98                buck2_data::structured_error::StackFrame {
99                    instruction_pointer: ptr_to_string(frame.ip()),
100                    symbol_address: ptr_to_string(frame.symbol_address()),
101                    module_base_address: frame
102                        .module_base_address()
103                        .map_or_else(|| "".to_owned(), ptr_to_string),
104                    symbols,
105                }
106            })
107            .collect()
108    }

110    /// Extracts a stringly-formatted payload from the given PanicHookInfo - usually the argument to `panic!`.
111    fn get_message_for_panic(info: &PanicHookInfo) -> String {
112        // Rust panic payloads can be anything, but they generally take one of two forms:
113        //  1. &'static str, for panics with a constant string parameter,
114        //  2. String, for panics that use the format `{}` syntax to construct a message.
115        //
116        // Since PanicHookInfo's Display implementation is implemented in libcore, it can't cover the String case (which is)
117        // a liballoc exclusive), so this code here checks for formatted messages and uses that as the message if present.
118        if let Some(literal_msg) = info.payload().downcast_ref::<&str>() {
119            (*literal_msg).to_owned()
120        } else if let Some(format_msg) = info.payload().downcast_ref::<String>() {
121            format_msg.clone()
122        } else {
123            "explicit panic with no message".to_owned()
124        }
125    }

127    /// Collects metadata from the current environment for use in LogView.
128    fn get_metadata_for_panic(options: &StructuredErrorOptions) -> HashMap<String, String> {
129        #[cfg_attr(client_only, allow(unused_mut))]
130        let mut map = metadata::collect();
131        #[cfg(not(client_only))]
132        if let Some(commands) = buck2_server::active_commands::try_active_commands() {
133            let commands = commands.keys().map(|id| id.to_string()).collect::<Vec<_>>();
134            map.insert("active_commands".to_owned(), commands.join(","));
135        }
136        if let Some(logview_key) = options
137            .low_cardinality_key_for_additional_logview_samples
138            .as_ref()
139        {
140            map.insert(
141                "low_cardinality_key_for_additional_logview_samples".to_owned(),
142                logview_key.to_string(),
143            );
144        }
145        map
146    }

148    /// Writes a representation of the given `PanicHookInfo` to Scribe, via the `StructuredError` event.
149    pub(crate) fn write_panic_to_scribe(fb: FacebookInit, info: &PanicHookInfo) {
150        let message = get_message_for_panic(info);
151        let location = info.location().map(|loc| Location {
152            file: loc.file().to_owned(),
153            line: loc.line(),
154            column: loc.column(),
155        });
156        write_to_scribe(
157            fb,
158            panic_payload(location, message, get_stack(), &Default::default(), None),
159        );
160    }

162    pub(crate) fn write_soft_error(
163        fb: FacebookInit,
164        category: &str,
165        err: &buck2_error::Error,
166        location: Location,
167        options: StructuredErrorOptions,
168    ) {
169        let event = panic_payload(
170            Some(location),
171            format!("Soft Error: {category}: {err:#}"),
172            Vec::new(),
173            &options,
174            Some(buck2_data::SoftError {
175                category: category.to_owned(),
176                is_quiet: options.quiet,
177            }),
178        );
179
180        // If the soft error was fired in a context with an ambient dispatcher, then we only send
181        // it there, but some contexts don't have one, and in that case, we notify all running
182        // commands.
183        match buck2_events::dispatch::get_dispatcher_opt() {
184            Some(dispatcher) => {
185                dispatcher.instant_event(event.clone());
186            }
187            None => {
188                #[cfg(client_only)]
189                let warn = !options.quiet;
190                #[cfg(not(client_only))]
191                let warn = !buck2_server::active_commands::broadcast_instant_event(&event)
192                    && !options.quiet;
193                if warn {
194                    tracing::warn!("Warning \"{}\": {:#}", category, err);
195                }
196            }
197        }
198
199        write_to_scribe(fb, event);
200    }

202    fn panic_payload(
203        location: Option<Location>,
204        message: String,
205        backtrace: Vec<buck2_data::structured_error::StackFrame>,
206        options: &StructuredErrorOptions,
207        soft_error_category: Option<buck2_data::SoftError>,
208    ) -> buck2_data::StructuredError {
209        let metadata = get_metadata_for_panic(options);
210        buck2_data::StructuredError {
211            location,
212            payload: message,
213            metadata,
214            backtrace,
215            quiet: options.quiet,
216            task: Some(options.task),
217            soft_error_category: soft_error_category
218                .map(|arg0: buck2_data::SoftError| ToOwned::to_owned(&arg0)),
219            daemon_in_memory_state_is_corrupted: options.daemon_in_memory_state_is_corrupted,
220            daemon_materializer_state_is_corrupted: options.daemon_materializer_state_is_corrupted,
221            action_cache_is_corrupted: options.action_cache_is_corrupted,
222            deprecation: options.deprecation,
223        }
224    }

226    /// Writes a representation of the given error (hard or soft) to Scribe
227    fn write_to_scribe(fb: FacebookInit, data: buck2_data::StructuredError) {
228        use std::time::SystemTime;
229
230        use buck2_core::facebook_only;
231        use buck2_data::InstantEvent;
232        use buck2_events::sink::remote;
233        use buck2_wrapper_common::invocation_id::TraceId;
234
235        facebook_only();
236        if !remote::is_enabled() {
237            return;
238        }
239
240        let sink = match new_remote_event_sink_if_enabled(fb, ScribeConfig::default()) {
241            #[allow(unreachable_patterns)]
242            Ok(Some(sink)) => sink,
243            _ => {
244                // We're already panicking and we can't connect to the scribe daemon? Things are bad and we're SOL.
245                return;
246            }
247        };
248
249        // There are some dubious ways of acquiring a handle to the current Tokio runtime, if one exists, such as this
250        // one: https://docs.rs/tokio/latest/tokio/runtime/struct.Handle.html#method.current. However, there doesn't
251        // appear to be a good way to figure out if the thread we're running on is a tokio thread at all.
252        //
253        // Since this is the panic handler, and the panic handler runs in the context of the thread that panicked, we
254        // can make no assumptions about whether the thread we're on is a tokio thread. To flush the Scribe client,
255        // which is normally an async operation, we spawn a new thread, spawn a runtime on that, and await the flush
256        // on that thread.
257        //
258        // Note that if we fail to spawn a writer thread, then we just won't log.
259        let _err = thread_spawn("buck2-write-panic-to-scribe", move || {
260            let runtime = Builder::new_current_thread().enable_all().build().unwrap();
261            let _res = runtime.block_on(
262                sink.send_now(BuckEvent::new(
263                    SystemTime::now(),
264                    TraceId::new(),
265                    None,
266                    None,
267                    InstantEvent {
268                        data: Some(data.into()),
269                    }
270                    .into(),
271                )),
272            );
273        })
274        .map_err(|_| ())
275        .and_then(|t| t.join().map_err(|_| ()));
276    }
