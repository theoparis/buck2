26#[derive(Debug, buck2_error::Error)]
27#[buck2(tag = Environment)]
28enum TraceIdError {
29    #[error("`{}` environment variable is not UTF-8", BUCK_WRAPPER_UUID_ENV_VAR)]
30    EnvVarNotUtf8,
31}
32
33/// A TraceId is a unique identifier for a trace. Trace IDs are globally unique; their textual form is a v4 UUID.
34///
35/// TraceIds generally correspond to commands, but they do not have to, e.g. in the case of a Buck daemon producing
36/// events even when a command is not running.
37#[derive(Debug, Clone, PartialEq, Eq, Hash, Allocative)]
38pub struct TraceId(
39    #[allocative(skip)] // `Uuid` is inline.
40    Uuid,
41);
42
43// UUID is 16 bytes.
44impl Dupe for TraceId {}
45
46impl serde::ser::Serialize for TraceId {
47    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
48    where
49        S: serde::Serializer,
50    {
51        serializer.serialize_str(&self.to_string())
52    }
53}
54
55impl<'de> serde::de::Deserialize<'de> for TraceId {
56    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
57    where
58        D: serde::Deserializer<'de>,
59    {
60        use serde::de::Error;
61
62        <&str>::deserialize(deserializer)?
63            .parse()
64            .map_err(D::Error::custom)
65    }
66}
67
68impl Display for TraceId {
69    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
70        let mut buf = Uuid::encode_buffer();
71        f.write_str(self.0.hyphenated().encode_lower(&mut buf))
72    }
73}
74
75impl FromStr for TraceId {
76    type Err = uuid::Error;
77    fn from_str(s: &str) -> Result<TraceId, Self::Err> {
78        Ok(TraceId(Uuid::parse_str(s)?))
79    }
80}
81
82impl TraceId {
83    /// Generates a new TraceId, suitable for identifying a particular trace.
84    pub fn new() -> TraceId {
85        TraceId(Uuid::new_v4())
86    }
87
88    pub fn null() -> TraceId {
89        TraceId(Uuid::nil())
90    }
91
92    /// Fetch `TraceId` from environment variable or generate a new one.
93    pub fn from_env_or_new() -> buck2_error::Result<TraceId> {
94        match env::var(BUCK_WRAPPER_UUID_ENV_VAR) {
95            Ok(s) => Ok(TraceId::from_str(&s).with_buck_error_context(|| {
96                format!("Parsing buck2 invocation id from env variable {BUCK_WRAPPER_UUID_ENV_VAR}")
97            })?),
98            Err(env::VarError::NotPresent) => Ok(TraceId::new()),
99            Err(env::VarError::NotUnicode(_)) => Err(TraceIdError::EnvVarNotUtf8.into()),
100        }
101    }
102
103    /// Generate short hash to be used as a message key for a Scribe client.
104    pub fn hash(&self) -> i64 {
105        let mut hasher = DefaultHasher::new();
106        Hash::hash(self, &mut hasher);
107        hasher.finish() as i64
108    }
109}
