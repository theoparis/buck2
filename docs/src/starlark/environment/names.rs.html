18use std::cell::RefCell;
19
20use allocative::Allocative;
21use starlark_syntax::syntax::ast::Visibility;
22
23use crate::collections::Hashed;
24use crate::collections::SmallMap;
25use crate::environment::slots::ModuleSlotId;
26use crate::values::FrozenStringValue;
27
28/// MutableNames are how we allocate slots (index-based) to variables
29/// (name-based). The slots field is the current active mapping of names to
30/// index.
31///
32/// In a statement context there are things that define variables, e.g. x=...,
33/// for x in ... Importantly, the expression x can refer to either a global x,
34/// or a local x that hasn't yet been defined, based on the future
35/// presence/absence of a statement defining x. Therefore, we first capture all
36/// the definitions with collect_defines_lvalue, allocate them slots,
37/// then replace variables with slot numbers when compiling.
38///
39/// Comprehensions are a bit different. Given [x for x in y] that defines x, but
40/// in a way that shadows any existing x, and the definition immediately binds
41/// x. We do that with add_scoped()/unscope(). On an add_scope, we allocate
42/// fresh slots at the end, and bind them to the names in the comprehension.
43/// On an unscope, we do the reverse, putting things back to how they were
44/// before (apart from the total) number of slots required.
45#[derive(Debug)]
46pub(crate) struct MutableNames(RefCell<SmallMap<FrozenStringValue, (ModuleSlotId, Visibility)>>);
47
48#[derive(Debug, Allocative)]
49pub(crate) struct FrozenNames(SmallMap<FrozenStringValue, (ModuleSlotId, Visibility)>);
50
51impl MutableNames {
52    pub(crate) fn new() -> Self {
53        Self(RefCell::new(SmallMap::new()))
54    }
55
56    pub(crate) fn slot_count(&self) -> u32 {
57        self.0.borrow().len().try_into().unwrap()
58    }
59
60    /// Try and go back from a slot to a name.
61    /// Inefficient - only use in error paths.
62    pub(crate) fn get_slot(&self, slot: ModuleSlotId) -> Option<FrozenStringValue> {
63        for (s, (i, _vis)) in &*self.0.borrow() {
64            if *i == slot {
65                return Some(*s);
66            }
67        }
68        None
69    }
70
71    pub(crate) fn get_name(&self, name: Hashed<&str>) -> Option<(ModuleSlotId, Visibility)> {
72        self.0.borrow().get_hashed(name).copied()
73    }
74
75    /// Add a name with explicit visibility to the module.
76    pub(crate) fn add_name_visibility(
77        &self,
78        name: FrozenStringValue,
79        vis: Visibility,
80    ) -> ModuleSlotId {
81        let mut x = self.0.borrow_mut();
82        match x.get_mut_hashed(name.get_hashed().as_ref()) {
83            Some((slot, stored_vis)) => {
84                // Public visibility wins.
85                if *stored_vis == Visibility::Private {
86                    *stored_vis = vis;
87                }
88                *slot
89            }
90            None => {
91                let slot = ModuleSlotId::new(x.len().try_into().unwrap());
92                x.insert_hashed(name.get_hashed(), (slot, vis));
93                slot
94            }
95        }
96    }
97
98    // Add an exported name, or if it's already there, return the existing name
99    pub(crate) fn add_name(&self, name: FrozenStringValue) -> ModuleSlotId {
100        self.add_name_visibility(name, Visibility::Public)
101    }
102
103    pub(crate) fn hide_name(&self, name: &str) {
104        self.0.borrow_mut().shift_remove(name);
105    }
106
107    pub(crate) fn all_names_and_slots(&self) -> Vec<(FrozenStringValue, ModuleSlotId)> {
108        self.0
109            .borrow()
110            .iter()
111            .map(|(name, (slot, _vis))| (*name, *slot))
112            .collect()
113    }
114
115    pub(crate) fn all_names_and_visibilities(&self) -> Vec<(FrozenStringValue, Visibility)> {
116        self.0
117            .borrow()
118            .iter()
119            .map(|(name, (_slot, vis))| (*name, *vis))
120            .collect()
121    }
122
123    pub(crate) fn all_names_slots_and_visibilities(
124        &self,
125    ) -> Vec<(FrozenStringValue, ModuleSlotId, Visibility)> {
126        self.0
127            .borrow()
128            .iter()
129            .map(|(name, (slot, vis))| (*name, *slot, *vis))
130            .collect()
131    }
132
133    pub(crate) fn freeze(self) -> FrozenNames {
134        FrozenNames(self.0.into_inner())
135    }
136}
137
138impl FrozenNames {
139    pub(crate) fn get_name(&self, name: &str) -> Option<(ModuleSlotId, Visibility)> {
140        self.0.get(name).copied()
141    }
142
143    /// Symbols including private.
144    pub(crate) fn all_symbols(
145        &self,
146    ) -> impl Iterator<Item = (FrozenStringValue, ModuleSlotId)> + '_ {
147        self.0.iter().map(|(name, (slot, _vis))| (*name, *slot))
148    }
149
150    /// Exported symbols.
151    pub(crate) fn symbols(&self) -> impl Iterator<Item = (FrozenStringValue, ModuleSlotId)> + '_ {
152        self.0.iter().filter_map(|(name, (slot, vis))| match vis {
153            Visibility::Private => None,
154            Visibility::Public => Some((*name, *slot)),
155        })
156    }
157}
