18//! Run conformance tests, which are used by the Go starlark.
19//! e.g. <https://github.com/google/skylark/tree/master/testdata>
20
21// `if_then_panic` is only in newer clippy, delete this in future.
22#![allow(unknown_lints)]
23// We want to carefully control the panic message.
24#![allow(clippy::if_then_panic)]
25
26use itertools::Itertools;
27
28use crate::assert::assert::Assert;
29
30impl<'a> Assert<'a> {
31    /// Run a conformance test, e.g. the Go Starlark tests
32    pub fn conformance(&self, code: &str) {
33        self.conformance_except(code, &[])
34    }
35
36    /// Run a conformance test, but where some test cases are allowed to fail.
37    /// The `except` argument represents a list of these permissible failures,
38    /// in the order they occur in the conformance test set,
39    /// identified by a substring that occurs in the test.
40    pub fn conformance_except(&self, code: &str, except: &[&str]) {
41        let mut except = except.iter().peekable();
42
43        'next: for x in ConformanceTest::parse(code) {
44            if let Some(e) = except.peek() {
45                if x.code.contains(**e) {
46                    except.next();
47                    continue 'next;
48                }
49            }
50
51            x.test(self)
52        }
53
54        if let Some(missed) = except.next() {
55            panic!("Exception given but not used, `{missed}`");
56        }
57    }
58}
59
60/// Describe a conformance test
61struct ConformanceTest {
62    /// The code of the test
63    code: String,
64    /// If this might throw an error, what is it
65    error: Option<(usize, String)>,
66}
67
68impl ConformanceTest {
69    fn parse(code: &str) -> Vec<Self> {
70        // First split on "---"
71        code.lines()
72            .collect::<Vec<_>>()
73            .split(|x| *x == "---")
74            .map(|xs| Self {
75                code: xs.join("\n"),
76                error: xs
77                    .iter()
78                    .find_position(|x| x.contains("###"))
79                    .map(|(i, x)| {
80                        (
81                            i + 1,
82                            (**x).split_once("###").unwrap().1.trim_start().to_owned(),
83                        )
84                    }),
85            })
86            .collect()
87    }
88
89    fn test(&self, assert: &Assert) {
90        fn get_line(err: &crate::Error) -> Option<usize> {
91            err.span().map(|span| span.resolve_span().begin.line + 1)
92        }
93
94        match &self.error {
95            None => {
96                assert.pass(&self.code);
97            }
98            Some((line, _msg)) => {
99                // We don't actually check error messages, since these tests were taken from upstream
100                // and our error messages are different
101                let err = assert.fail(&self.code, "");
102                let got = get_line(&err);
103                if got != Some(*line) {
104                    panic!(
105                        "starlark::assert::conformance, failed at wrong line!\nCode:\n{}\nError:\n{}\nExpected: {}\nGot: {:?}\n",
106                        &self.code, &err, line, got
107                    );
108                }
109            }
110        }
111    }
112}
