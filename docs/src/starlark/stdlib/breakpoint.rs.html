33// A breakpoint takes over the console UI, so having two going at once confuses everything.
34// Have a global mutex to ensure one at a time.
35static BREAKPOINT_MUTEX: Lazy<Mutex<State>> = Lazy::new(|| Mutex::new(State::Allow));
36
37/// `breakpoint` function uses this trait to perform console IO.
38pub(crate) trait BreakpointConsole {
39    /// Return `None` on EOF.
40    fn read_line(&mut self) -> anyhow::Result<Option<String>>;
41    fn println(&mut self, line: &str);
42}
43
44/// Breakpoint handler implemented with `rustyline`.
45pub(crate) struct RealBreakpointConsole {
46    read_line: ReadLine,
47}
48
49impl BreakpointConsole for RealBreakpointConsole {
50    fn read_line(&mut self) -> anyhow::Result<Option<String>> {
51        self.read_line.read_line("$> ")
52    }
53
54    fn println(&mut self, line: &str) {
55        eprintln!("{line}");
56    }
57}
58
59impl RealBreakpointConsole {
60    pub(crate) fn factory() -> Box<dyn Fn() -> anyhow::Result<Box<dyn BreakpointConsole>>> {
61        Box::new(|| {
62            Ok(Box::new(RealBreakpointConsole {
63                read_line: ReadLine::new("STARLARK_RUST_DEBUGGER_HISTFILE")?,
64            }))
65        })
66    }
67}

69/// Is debugging allowed or not? After the user hits Ctrl-C they probably
70/// just want to stop hard, so don't keep dropping them into breakpoints.
71#[derive(PartialEq, Eq)]
72enum State {
73    Allow, // More breakpoints are fine
74    Stop,  // No more breakpoints
75}
76
77/// We've run a breakpoint command, what should we do.
78enum Next {
79    Again,  // Accept another breakpoint command
80    Resume, // Continue running
81    Fail,   // Stop running
82}
83
84fn cmd_help(_eval: &mut Evaluator, rl: &mut dyn BreakpointConsole) -> anyhow::Result<Next> {
85    for (name, msg, _) in COMMANDS {
86        rl.println(&format!("* :{}, {}", name[0], msg))
87    }
88    Ok(Next::Again)
89}
90
91fn cmd_variables(eval: &mut Evaluator, rl: &mut dyn BreakpointConsole) -> anyhow::Result<Next> {
92    fn truncate(mut s: String, n: usize) -> String {
93        if s.len() > n {
94            s.truncate(n);
95            s.push_str("...");
96        }
97        s
98    }
99
100    for (name, value) in eval.local_variables() {
101        rl.println(&format!("* {} = {}", name, truncate(value.to_string(), 80)))
102    }
103    Ok(Next::Again)
104}
105
106fn cmd_stack(eval: &mut Evaluator, rl: &mut dyn BreakpointConsole) -> anyhow::Result<Next> {
107    for line in eval.call_stack().to_string().lines() {
108        rl.println(line)
109    }
110    Ok(Next::Again)
111}
112
113fn cmd_resume(_eval: &mut Evaluator, _rl: &mut dyn BreakpointConsole) -> anyhow::Result<Next> {
114    Ok(Next::Resume)
115}
116
117fn cmd_fail(_eval: &mut Evaluator, _rl: &mut dyn BreakpointConsole) -> anyhow::Result<Next> {
118    Ok(Next::Fail)
119}
120
121const COMMANDS: &[(
122    &[&str], // Possible names
123    &str,    // Help text
124    fn(eval: &mut Evaluator, &mut dyn BreakpointConsole) -> anyhow::Result<Next>,
125)] = &[
126    (&["help", "?"], "Show this help message", cmd_help),
127    (&["vars"], "Show all local variables", cmd_variables),
128    (&["stack"], "Show the stack trace", cmd_stack),
129    (&["resume", "quit", "exit"], "Resume execution", cmd_resume),
130    (&["fail"], "Abort with a failure message", cmd_fail),
131];
132
133fn pick_command(
134    x: &str,
135    rl: &mut dyn BreakpointConsole,
136) -> Option<fn(eval: &mut Evaluator, &mut dyn BreakpointConsole) -> anyhow::Result<Next>> {
137    // If we can find a command that matches perfectly, do that
138    // Otherwise return the longest match, but if they are multiple, show a warning
139    let mut poss = Vec::new();
140    for (names, _, cmd) in COMMANDS {
141        for n in *names {
142            if *n == x {
143                return Some(*cmd);
144            }
145            if n.starts_with(x) {
146                poss.push((n, cmd));
147                break;
148            }
149        }
150    }
151    match poss.as_slice() {
152        [] => rl.println("Unrecognised command, type :help for all commands"),
153        [x] => return Some(*x.1),
154        xs => rl.println(&format!(
155            "Ambiguous command, could have been any of: {}",
156            xs.iter().map(|x| x.0).join(" ")
157        )),
158    }
159    None
160}
161
162fn breakpoint_loop(
163    eval: &mut Evaluator,
164    mut rl: Box<dyn BreakpointConsole>,
165) -> anyhow::Result<State> {
166    loop {
167        let readline = rl.read_line()?;
168        match readline {
169            Some(line) => {
170                if let Some(line) = line.strip_prefix(':') {
171                    if let Some(cmd) = pick_command(line.trim_end(), &mut *rl) {
172                        match cmd(eval, &mut *rl)? {
173                            Next::Again => {}
174                            Next::Resume => return Ok(State::Allow),
175                            Next::Fail => {
176                                return Err(anyhow::anyhow!("Selected :fail at breakpoint()"));
177                            }
178                        }
179                    }
180                } else {
181                    let ast = AstModule::parse("interactive", line, &Dialect::AllOptionsInternal);
182                    let res = ast.and_then(|ast| eval.eval_statements(ast));
183                    match res {
184                        Err(e) => {
185                            rl.println(&format!("{e:#}"));
186                        }
187                        Ok(v) => {
188                            if !v.is_none() {
189                                rl.println(&v.to_string())
190                            }
191                        }
192                    }
193                }
194            }
195            None => return Ok(State::Stop),
196        }
197    }
198}

200#[derive(Error, Debug)]
201enum BreakpointError {
202    #[error("Breakpoint handler is not enabled for current Evaluator")]
203    NoHandler,
204}
205
206const BREAKPOINT_HIT_MESSAGE: &str = "BREAKPOINT HIT! :resume to continue, :help for all options";
207
208#[starlark_module]
209pub fn global(builder: &mut GlobalsBuilder) {
210    /// When a debugger is available, breaks into the debugger.
211    fn breakpoint(eval: &mut Evaluator) -> anyhow::Result<NoneType> {
212        {
213            let mut guard = BREAKPOINT_MUTEX.lock().unwrap();
214            if *guard == State::Allow {
215                let mut rl = match &mut eval.breakpoint_handler {
216                    Some(rl) => rl()?,
217                    None => return Err(BreakpointError::NoHandler.into()),
218                };
219                rl.println(BREAKPOINT_HIT_MESSAGE);
220                *guard = breakpoint_loop(eval, rl)?;
221            }
222        }
223        Ok(NoneType)
224    }
225}

238    // Breakpoint tests should not be executed concurrently
239    // to avoid interfering with `BREAKPOINT_MUTEX`.
240    static TEST_MUTEX: Lazy<Mutex<()>> = Lazy::new(|| Mutex::new(()));
241
242    fn reset_global_state() {
243        // `breakpoint()` function modifies the global state.
244        *BREAKPOINT_MUTEX.lock().unwrap() = State::Allow;
245    }
246
247    #[test]
248    // Test with: BREAKPOINT=1 cargo test -p starlark breakpoint -- --nocapture
249    fn test_breakpoint_real() {
250        let _g = TEST_MUTEX.lock();
251        reset_global_state();
252
253        if env::var("BREAKPOINT") != Ok("1".to_owned()) {
254            return;
255        }
256
257        let mut a = Assert::new();
258        a.setup_eval(|e| e.enable_terminal_breakpoint_console());
259        a.globals_add(global);
260        a.pass("x = [1,2,3]; breakpoint(); print(x)");
261    }
262
263    #[test]
264    fn test_breakpoint_mock() {
265        let _g = TEST_MUTEX.lock();
266        reset_global_state();
267
268        let printed_lines = Rc::new(RefCell::new(Vec::new()));
269        let printed_lines_copy = printed_lines.dupe();
270
271        let mut a = Assert::new();
272        a.globals_add(global);
273        a.setup_eval(move |eval| {
274            let printed_lines = printed_lines.dupe();
275            eval.breakpoint_handler = Some(Box::new(move || {
276                // `Assert` runs tests several times, take only lines from the last iteration.
277                printed_lines.borrow_mut().clear();
278
279                struct Handler {
280                    printed_lines: Rc<RefCell<Vec<String>>>,
281                    called: bool,
282                }
283
284                impl BreakpointConsole for Handler {
285                    fn read_line(&mut self) -> anyhow::Result<Option<String>> {
286                        let called = self.called;
287                        self.called = true;
288                        if !called {
289                            Ok(Some("x".to_owned()))
290                        } else {
291                            Ok(None)
292                        }
293                    }
294
295                    fn println(&mut self, line: &str) {
296                        self.printed_lines.borrow_mut().push(line.to_owned());
297                    }
298                }
299
300                Ok(Box::new(Handler {
301                    printed_lines: printed_lines.dupe(),
302                    called: false,
303                }))
304            }));
305        });
306        a.pass("x = [1,2,3]; breakpoint()");
307
308        assert_eq!(
309            vec![BREAKPOINT_HIT_MESSAGE, "[1, 2, 3]"],
310            *printed_lines_copy.borrow()
311        );
312    }
313
314    #[test]

318
