57#[starlark_module]
58pub fn partial(builder: &mut GlobalsBuilder) {
59    /// Construct a partial application. In almost all cases it is simpler to use a `lamdba`.
60    fn partial<'v>(
61        #[starlark(require = pos)] func: Value<'v>,
62        #[starlark(args)] args: Value<'v>,
63        #[starlark(kwargs)] kwargs: DictRef<'v>,
64    ) -> anyhow::Result<Partial<'v>> {
65        debug_assert!(Tuple::from_value(args).is_some());
66        let names: Vec<_> = kwargs
67            .keys()
68            .map(|x| {
69                let x = StringValue::new(x).unwrap();
70                (
71                    // We duplicate string here.
72                    // If this becomes hot, we should do better.
73                    Symbol::new_hashed(x.as_str_hashed()),
74                    x,
75                )
76            })
77            .collect();
78        let mut names_index = HashTable::with_capacity(names.len());
79        for (i, (k, _)) in names.iter().enumerate() {
80            names_index.insert_unique(k.hash(), i, |i| names[*i].0.hash());
81        }
82        Ok(Partial {
83            func,
84            pos: args,
85            named: kwargs.values().collect(),
86            names,
87            names_index,
88        })
89    }
90}

92#[derive(Debug, Coerce, Trace, NoSerialize, ProvidesStaticType, Allocative)]
93#[repr(C)]
94struct PartialGen<V, S> {
95    func: V,
96    // Always references a tuple.
97    pos: V,
98    named: Vec<V>,
99    names: Vec<(Symbol, S)>,
100    names_index: HashTable<usize>,
101}
102
103impl<'v, V: ValueLike<'v>, S> PartialGen<V, S> {
104    fn pos_content(&self) -> &'v [Value<'v>] {
105        Tuple::from_value(self.pos.to_value()).unwrap().content()
106    }
107}
108
109impl<'v, V: ValueLike<'v>, S> Display for PartialGen<V, S> {
110    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
111        write!(f, "partial({}, *[", self.func)?;
112        for (i, v) in self.pos_content().iter().enumerate() {
113            if i != 0 {
114                write!(f, ",")?;
115            }
116            v.fmt(f)?;
117        }
118        write!(f, "], **{{")?;
119        for (i, (k, v)) in self.names.iter().zip(self.named.iter()).enumerate() {
120            if i != 0 {
121                write!(f, ",")?;
122            }
123            write!(f, "{}:", k.0.as_str())?;
124            v.to_value().fmt(f)?;
125        }
126        write!(f, "}})")
127    }
128}
129
130type Partial<'v> = PartialGen<Value<'v>, StringValue<'v>>;
131type FrozenPartial = PartialGen<FrozenValue, FrozenStringValue>;
132starlark_complex_values!(Partial);
133
134impl<'v> Freeze for Partial<'v> {
135    type Frozen = FrozenPartial;
136    fn freeze(self, freezer: &Freezer) -> FreezeResult<Self::Frozen> {
137        Ok(FrozenPartial {
138            func: self.func.freeze(freezer)?,
139            pos: freezer.freeze(self.pos)?,
140            named: self.named.try_map(|x| x.freeze(freezer))?,
141            names: self
142                .names
143                .into_try_map(|(s, x)| Ok((s, x.freeze(freezer)?)))?,
144            names_index: self.names_index,
145        })
146    }
147}

149#[starlark_value(type = FUNCTION_TYPE)]
150impl<'v, V: ValueLike<'v>, S: StringValueLike<'v>> StarlarkValue<'v> for PartialGen<V, S>
151where
152    Self: ProvidesStaticType<'v>,
153{
154    type Canonical = PartialGen<Value<'v>, StringValue<'v>>;
155
156    fn name_for_call_stack(&self, _me: Value<'v>) -> String {
157        "partial".to_owned()
158    }
159
160    fn invoke(
161        &self,
162        _me: Value<'v>,
163        args: &Arguments<'v, '_>,
164        eval: &mut Evaluator<'v, '_, '_>,
165    ) -> crate::Result<Value<'v>> {
166        // apply the partial arguments first, then the remaining arguments I was given
167
168        let self_pos = self.pos_content();
169        let self_named = coerce(&self.named);
170        let self_names = coerce(&self.names);
171
172        for (symbol, _) in args.0.names.names() {
173            if self
174                .names_index
175                .find(symbol.hash(), |i| &self.names[*i].0 == symbol)
176                .is_some()
177            {
178                return Err(value_error!(
179                    "partial() got multiple values for argument `{}`",
180                    symbol.as_str(),
181                ));
182            }
183        }
184
185        eval.alloca_concat(self_pos, args.0.pos, |pos, eval| {
186            eval.alloca_concat(self_named, args.0.named, |named, eval| {
187                eval.alloca_concat(self_names, args.0.names.names(), |names, eval| {
188                    let params = Arguments(ArgumentsFull {
189                        pos,
190                        named,
191                        names: ArgNames::new_unique(names),
192                        args: args.0.args,
193                        kwargs: args.0.kwargs,
194                    });
195                    self.func
196                        .to_value()
197                        .invoke_with_loc(Some(rust_loc!()), &params, eval)
198                })
199            })
200        })
201    }
202}

204#[cfg(test)]
205mod tests {
206    use crate::assert;
207
208    fn eq(expected: &str, expr: &str) {
209        let sum = r#"
210def sum(a, b, *args, **kwargs):
211    # print("a=%s b=%s args=%s kwargs=%s" % (a, b, args, kwargs))
212    # TODO(nga): fix typecheck.
213    args = noop((a, b)) + args
214    return [args, kwargs]
215"#;
216
217        assert::eq(expected, &format!("{sum}{expr}"));
218    }
219
220    #[test]
221    fn test_simple() {
222        eq(
223            "[(1, 2, 3), {'other': True, 'third': None}]",
224            "(partial(sum, 1, other=True))(2, 3, third=None)",
225        );
226    }
227
228    #[test]
229    fn test_star_to_partial() {
230        eq(
231            "[(1, 2, 3), {'other': True, 'third': None}]",
232            "(partial(sum, *[1], **{'other': True}))(2, 3, third=None)",
233        );
234    }
235
236    #[test]
237    fn test_start_to_returned_func() {
238        eq(
239            "[(1, 2, 3), {'other': True, 'third': None}]",
240            "(partial(sum, other=True))(*[1, 2, 3], **{'third': None})",
241        );
242    }
