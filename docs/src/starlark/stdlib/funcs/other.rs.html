43    /// fail: fail the execution
44    ///
45    /// ```
46    /// # starlark::assert::fail(r#"
47    /// fail("this is an error")  # fail: this is an error
48    /// # "#, "this is an error");
49    /// # starlark::assert::fail(r#"
50    /// fail("oops", 1, False)  # fail: oops 1 False
51    /// # "#, "oops 1 False");
52    /// ```
53    fn fail(#[starlark(args)] args: UnpackTuple<Value>) -> starlark::Result<StarlarkNever> {
54        let mut s = String::new();
55        for x in args.items {
56            s.push(' ');
57            match x.unpack_str() {
58                Some(x) => s.push_str(x),
59                None => x.collect_repr(&mut s),
60            }
61        }
62        Err(starlark::Error::new_kind(starlark::ErrorKind::Fail(
63            anyhow::Error::msg(s),
64        )))
65    }

67    /// [any](
68    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#any
69    /// ): returns true if any value in the iterable object have a truth value
70    /// of true.
71    ///
72    /// ```
73    /// # starlark::assert::all_true(r#"
74    /// any([0, True]) == True
75    /// any([0, 1]) == True
76    /// any([0, 1, True]) == True
77    /// any([0, 0]) == False
78    /// any([0, False]) == False
79    /// # "#);
80    /// ```
81    #[starlark(speculative_exec_safe)]
82    fn any<'v>(
83        #[starlark(require = pos)] x: ValueOfUnchecked<'v, StarlarkIter<Value<'v>>>,
84        heap: &'v Heap,
85    ) -> starlark::Result<bool> {
86        for i in x.get().iterate(heap)? {
87            if i.to_bool() {
88                return Ok(true);
89            }
90        }
91        Ok(false)
92    }

95    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#all
96    /// ): returns true if all values in the iterable object have a truth value
97    /// of true.
98    ///
99    /// ```
100    /// # starlark::assert::all_true(r#"
101    /// all([1, True]) == True
102    /// all([1, 1]) == True
103    /// all([0, 1, True]) == False
104    /// all([True, 1, True]) == True
105    /// all([0, 0]) == False
106    /// all([0, False]) == False
107    /// all([True, 0]) == False
108    /// all([1, False]) == False
109    /// # "#);
110    /// ```
111    #[starlark(speculative_exec_safe)]
112    fn all<'v>(
113        #[starlark(require = pos)] x: ValueOfUnchecked<'v, StarlarkIter<Value<'v>>>,
114        heap: &'v Heap,
115    ) -> starlark::Result<bool> {
116        for i in x.get().iterate(heap)? {
117            if !i.to_bool() {
118                return Ok(false);
119            }
120        }
121        Ok(true)
122    }

142    /// [enumerate](
143    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#enumerate
144    /// ): return a list of (index, element) from an iterable.
145    ///
146    /// `enumerate(x)` returns a list of `(index, value)` pairs, each containing
147    /// successive values of the iterable sequence and the index of the
148    /// value within the sequence.
149    ///
150    /// The optional second parameter, `start`, specifies an integer value to
151    /// add to each index.
152    ///
153    /// ```
154    /// # starlark::assert::all_true(r#"
155    /// enumerate(["zero", "one", "two"]) == [(0, "zero"), (1, "one"), (2, "two")]
156    /// enumerate(["one", "two"], 1) == [(1, "one"), (2, "two")]
157    /// # "#);
158    /// ```
159    #[starlark(speculative_exec_safe)]
160    fn enumerate<'v>(
161        #[starlark(require = pos)] it: ValueOfUnchecked<'v, StarlarkIter<Value<'v>>>,
162        #[starlark(default = 0)] start: i32,
163        heap: &'v Heap,
164    ) -> starlark::Result<impl AllocValue<'v>> {
165        let v = it
166            .get()
167            .iterate(heap)?
168            .enumerate()
169            .map(move |(k, v)| (k as i32 + start, v));
170        Ok(AllocList(v))
171    }

173    /// [getattr](
174    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#getattr
175    /// ): returns the value of an attribute
176    ///
177    /// `getattr(x, name)` returns the value of the attribute (field or method)
178    /// of x named `name`. It is a dynamic error if x has no such attribute.
179    ///
180    /// `getattr(x, "f")` is equivalent to `x.f`.
181    ///
182    /// `getattr(x, "f", d)` is equivalent to `x.f if hasattr(x, "f") else d`
183    /// and will never raise an error.
184    ///
185    /// ```
186    /// # starlark::assert::all_true(r#"
187    /// getattr("banana", "split")("a") == ["b", "n", "n", ""] # equivalent to "banana".split("a")
188    /// # "#);
189    /// ```
190    #[starlark(speculative_exec_safe)]
191    fn getattr<'v>(
192        #[starlark(require = pos)] a: Value<'v>,
193        #[starlark(require = pos)] attr: &str,
194        #[starlark(require = pos)] default: Option<Value<'v>>,
195        heap: &'v Heap,
196    ) -> starlark::Result<Value<'v>> {
197        // TODO(nga): this doesn't cache string hash, so it is suboptimal.
198        match a.get_attr(attr, heap)? {
199            Some(v) => Ok(v),
200            None => match default {
201                Some(x) => Ok(x),
202                None => ValueError::unsupported_owned(a.get_type(), &format!(".{attr}"), None),
203            },
204        }
205    }

222    /// [hash](
223    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#hash
224    /// ): returns the hash number of a value.
225    ///
226    /// `hash(x)` returns an integer hash value for x such that `x == y`
227    /// implies `hash(x) == hash(y)`.
228    ///
229    /// `hash` fails if x, or any value upon which its hash depends, is
230    /// unhashable.
231    ///
232    /// ```
233    /// # starlark::assert::all_true(r#"
234    /// hash("hello") != hash("world")
235    /// # "#);
236    /// ```
237    #[starlark(speculative_exec_safe)]
238    fn hash(#[starlark(require = pos)] a: &str) -> anyhow::Result<i32> {
239        // From the starlark spec:
240        // > the hash function for strings is the same as that implemented by java.lang.String.hashCode,
241        // > a simple polynomial accumulator over the UTF-16 transcoding of the string:
242        // > `s[0]*31^(n-1) + s[1]*31^(n-2) + ... + s[n-1]`
243        // As per spec the function should only support string and bytes types.
244        // We don't have support for bytes, so parameter is forced to be a string.
245
246        // Most strings are ASCII strings, try them first.
247        #[allow(clippy::never_loop)]
248        'ascii: loop {
249            let mut hash = 0i32;
250            for &b in a.as_bytes() {
251                if b > 0x7f {
252                    break 'ascii;
253                }
254                hash = hash.wrapping_mul(31i32).wrapping_add(b as i32);
255            }
256            return Ok(hash);
257        }
258
259        Ok(a.encode_utf16().fold(0i32, |hash: i32, c: u16| {
260            31i32.wrapping_mul(hash).wrapping_add(c as i32)
261        }))
262    }

266    /// ): get the length of a sequence

268    /// `len(x)` returns the number of elements in its argument.

272    /// ```
273    /// # starlark::assert::all_true(r#"

278    /// len([1,2]) == 2
279    /// len({'16': 10}) == 1
280    /// # "#);
281    /// # starlark::assert::fail(r#"
282    /// len(True)    # error: not supported
283    /// # "#, "not supported");
284    /// ```
285    #[starlark(speculative_exec_safe)]
286    fn len(#[starlark(require = pos)] a: Value) -> starlark::Result<i32> {
287        a.length()
288    }

290    /// [reversed](
291    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#reversed
292    /// ): reverse a sequence
293    ///
294    /// `reversed(x)` returns a new list containing the elements of the iterable
295    /// sequence x in reverse order.
296    ///
297    /// ```
298    /// # starlark::assert::all_true(r#"
299    /// reversed(['a', 'b', 'c'])              == ['c', 'b', 'a']
300    /// reversed(range(5))                     == [4, 3, 2, 1, 0]
301    /// reversed("stressed".elems())           == ["d", "e", "s", "s", "e", "r", "t", "s"]
302    /// reversed({"one": 1, "two": 2}.keys())  == ["two", "one"]
303    /// # "#);
304    /// ```
305    #[starlark(speculative_exec_safe)]
306    fn reversed<'v>(
307        #[starlark(require = pos)] a: ValueOfUnchecked<'v, StarlarkIter<Value<'v>>>,
308        heap: &'v Heap,
309    ) -> starlark::Result<Vec<Value<'v>>> {
310        let mut v: Vec<Value> = a.get().iterate(heap)?.collect();
311        v.reverse();
312        Ok(v)
313    }

315    /// [sorted](
316    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#sorted
317    /// ): sort a sequence
318    ///
319    /// `sorted(x)` returns a new list containing the elements of the iterable
320    /// sequence x, in sorted order.  The sort algorithm is stable.
321    ///
322    /// The optional named parameter `reverse`, if true, causes `sorted` to
323    /// return results in reverse sorted order.
324    ///
325    /// The optional named parameter `key` specifies a function of one
326    /// argument to apply to obtain the value's sort key.
327    /// The default behavior is the identity function.
328    ///
329    /// ```
330    /// # starlark::assert::all_true(r#"
331    /// sorted([3, 1, 4, 1, 5, 9])                               == [1, 1, 3, 4, 5, 9]
332    /// sorted([3, 1, 4, 1, 5, 9], reverse=True)                 == [9, 5, 4, 3, 1, 1]
333    /// sorted(["two", "three", "four"], key=len)                == ["two", "four", "three"] # shortest to longest
334    /// sorted(["two", "three", "four"], key=len, reverse=True)  == ["three", "four", "two"] # longest to shortest
335    /// # "#);
336    /// ```
337    // This function is not spec-safe, because it may call `key` function
338    // which might be not spec-safe.
339    fn sorted<'v>(
340        #[starlark(require = pos)] x: ValueOfUnchecked<'v, StarlarkIter<Value<'v>>>,
341        #[starlark(require = named)] key: Option<Value<'v>>,
342        #[starlark(require = named, default = false)] reverse: bool,
343        eval: &mut Evaluator<'v, '_, '_>,
344    ) -> starlark::Result<AllocList<impl IntoIterator<Item = Value<'v>> + use<'v>>> {
345        let it = x.get().iterate(eval.heap())?;
346        let mut it = match key {
347            None => it.map(|x| (x, x)).collect(),
348            Some(key) => {
349                let mut v = Vec::new();
350                for el in it {
351                    v.push((el, key.invoke_pos(&[el], eval)?));
352                }
353                v
354            }
355        };
356
357        let mut compare_ok = Ok(());
358
359        it.sort_by(|x: &(Value, Value), y: &(Value, Value)| {
360            let ord_or_err = if reverse {
361                x.1.compare(y.1).map(Ordering::reverse)
362            } else {
363                x.1.compare(y.1)
364            };
365            match ord_or_err {
366                Ok(r) => r,
367                Err(e) => {
368                    compare_ok = Err(e);
369                    Ordering::Equal // does not matter
370                }
371            }
372        });
373
374        compare_ok?;
375
376        Ok(AllocList(it.into_iter().map(|x| x.0)))
377    }

381    /// ): returns a string describing the type of its operand.
