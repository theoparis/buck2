38#[derive(Clone, Debug, Eq, PartialEq, Hash, Ord, PartialOrd, Allocative)]
39struct ZipType;
40
41impl TyCustomFunctionImpl for ZipType {
42    fn as_callable(&self) -> TyCallable {
43        // TODO(nga): this should be obtained from function signature from function definition.
44        TyCallable::new(ParamSpec::args(Ty::iter(Ty::any())), Ty::list(Ty::any()))
45    }
46
47    fn validate_call(
48        &self,
49        _span: Span,
50        args: &TyCallArgs,
51        oracle: TypingOracleCtx,
52    ) -> Result<Ty, TypingOrInternalError> {
53        let mut iter_item_types: Vec<Ty> = Vec::new();
54        for pos in &args.pos {
55            let item_ty = oracle.iter_item(pos.as_ref())?;
56            iter_item_types.push(item_ty);
57        }
58        if args.args.is_some() {
59            Ok(Ty::list(Ty::any()))
60        } else {
61            Ok(Ty::list(Ty::tuple(iter_item_types)))
62        }
63    }
64}

66#[starlark_module]
67pub(crate) fn register_zip(globals: &mut GlobalsBuilder) {
68    /// [zip](
69    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#zip
70    /// ): zip several iterables together
71    ///
72    /// `zip()` returns a new list of n-tuples formed from corresponding
73    /// elements of each of the n iterable sequences provided as arguments to
74    /// `zip`.  That is, the first tuple contains the first element of each of
75    /// the sequences, the second element contains the second element of each
76    /// of the sequences, and so on.  The result list is only as long as the
77    /// shortest of the input sequences.
78    ///
79    /// ```
80    /// # starlark::assert::all_true(r#"
81    /// zip()                           == []
82    /// zip(range(5))                   == [(0,), (1,), (2,), (3,), (4,)]
83    /// zip(range(5), "abc".elems())    == [(0, "a"), (1, "b"), (2, "c")]
84    /// # "#);
85    /// ```
86    #[starlark(speculative_exec_safe, ty_custom_function = ZipType)]
87    fn zip<'v>(
88        #[starlark(args)] args: UnpackTuple<ValueOfUnchecked<'v, StarlarkIter<FrozenValue>>>,
89        heap: &'v Heap,
90    ) -> starlark::Result<Vec<Value<'v>>> {
91        let mut v = Vec::new();
92        let mut first = true;
93        for arg in args.items {
94            let mut idx = 0;
95            for e in arg.get().iterate(heap)? {
96                if first {
97                    v.push(heap.alloc((e,)));
98                    idx += 1;
99                } else if idx < v.len() {
100                    v[idx] = v[idx].add(heap.alloc((e,)), heap)?;
101                    idx += 1;
102                }
103            }
104            v.truncate(idx);
105            first = false;
106        }
107        Ok(v)
108    }
109}
