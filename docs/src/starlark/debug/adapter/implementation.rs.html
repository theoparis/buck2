56pub(crate) fn prepare_dap_adapter(
57    client: Box<dyn DapAdapterClient>,
58) -> (impl DapAdapter, impl DapAdapterEvalHook) {
59    let (sender, receiver) = std::sync::mpsc::channel::<ToEvalMessage>();
60    let state = Arc::new(SharedAdapterState {
61        client,
62        breakpoints: Arc::new(Mutex::new(BreakpointConfig::new())),
63        disable_breakpoints: Arc::new(0usize.into()),
64    });
65
66    (
67        DapAdapterImpl {
68            state: state.clone(),
69            sender,
70        },
71        DapAdapterEvalHookImpl::new(state, receiver),
72    )
73}

90fn evaluate_expr<'v>(
91    state: &SharedAdapterState,
92    eval: &mut Evaluator<'v, '_, '_>,
93    expr: String,
94) -> anyhow::Result<Value<'v>> {
95    // We don't want to trigger breakpoints during an evaluate,
96    // not least because we currently don't allow reenterant evaluate
97    state.disable_breakpoints.fetch_add(1, Ordering::SeqCst);
98    // Don't use `?`, we need to reset disable_breakpoints.
99    let ast = AstModule::parse("interactive", expr, &Dialect::AllOptionsInternal);
100    // This technically loses structured access to the diagnostic information. However, it's
101    // completely unused, so there's not much point in converting all of this code to using
102    // `starlark::Error`, only for buck2 to then go and blindly turn it into a `anyhow::Error`
103    // anyway.
104    let res = ast
105        .and_then(|ast| eval.eval_statements(ast))
106        .into_anyhow_result();
107    state.disable_breakpoints.fetch_sub(1, Ordering::SeqCst);
108    res
109}

111impl<'e> BeforeStmtFuncDyn<'e> for DapAdapterEvalHookImpl {
112    fn call<'v>(
113        &mut self,
114        span_loc: FileSpanRef,
115        continued: bool,
116        eval: &mut Evaluator<'v, '_, 'e>,
117    ) -> crate::Result<()> {
118        // The debug adapter should only break on the "initial" instruction that
119        // makes up any given statement. "Continued" instructions are part of
120        // the still-executing/previous statement, and should be ignored.
121        if continued {
122            return Ok(());
123        }
124
125        let stop = if self.state.disable_breakpoints.load(Ordering::SeqCst) > 0 {
126            false
127        } else {
128            let breaks = self.state.breakpoints.lock().unwrap();
129            let breakpoint = breaks.at(span_loc);
130            match breakpoint {
131                Some(Breakpoint {
132                    condition: Some(condition),
133                    ..
134                }) => match evaluate_expr(&self.state, eval, condition.to_owned()) {
135                    Ok(v) => v.to_bool(),
136                    Err(_) => {
137                        // If failed to evaluate the condition, stop.
138                        // TODO(nga): print the error.
139                        true
140                    }
141                },
142                Some(..) => true,
143                None => false,
144            }
145        };
146
147        let step_stop = match self.step {
148            None => false,
149            Some((StepKind::Into, _)) => true,
150            // These aren't quite right because we only get called before statements and so we could
151            // return from the current function and be in an expression that then calls another function
152            // without hitting a new statement in the outer function.
153            Some((StepKind::Over, stack_size)) => eval.call_stack_count() <= stack_size,
154            Some((StepKind::Out, stack_size)) => eval.call_stack_count() < stack_size,
155        };
156
157        if stop || step_stop {
158            self.step = None;
159            self.state.client.event_stopped()?;
160            loop {
161                let msg = self.receiver.recv();
162                match msg.map(|msg| msg(span_loc, eval)) {
163                    Ok(Next::Continue) => break,
164                    Ok(Next::Step(kind)) => {
165                        self.step = Some((kind, eval.call_stack_count()));
166                        break;
167                    }
168                    Ok(Next::RemainPaused) => {}
169                    Err(..) => {
170                        // DapAdapter has been dropped so we'll continue.
171                        break;
172                    }
173                }
174            }
175        }
176        Ok(())
177    }
178}

202#[derive(Debug)]
203struct BreakpointConfig {
204    // maps a source filename to the breakpoint spans for the file
205    breakpoints: HashMap<String, HashMap<Span, Breakpoint>>,
206}
207
208impl BreakpointConfig {
209    fn new() -> Self {
210        Self {
211            breakpoints: HashMap::new(),
212        }
213    }
214
215    fn at(&self, span_loc: FileSpanRef) -> Option<&Breakpoint> {
216        self.breakpoints
217            .get(span_loc.filename())
218            .and_then(|file_breaks| file_breaks.get(&span_loc.span))
219    }
220
221    fn set_breakpoints(
222        &mut self,
223        source: &str,
224        breakpoints: &ResolvedBreakpoints,
225    ) -> anyhow::Result<()> {
226        if breakpoints.0.is_empty() {
227            self.breakpoints.remove(source);
228        } else {
229            self.breakpoints.insert(
230                source.to_owned(),
231                breakpoints
232                    .0
233                    .iter()
234                    .filter_map(|x| x.clone())
235                    .map(|x| (x.span.span, x))
236                    .collect(),
237            );
238        }
239        Ok(())
240    }
241}

243#[derive(Debug)]
244struct SharedAdapterState {
245    client: Box<dyn DapAdapterClient>,
246    // These breakpoints must all match statements as per before_stmt.
247    // Those values for which we abort the execution.
248    breakpoints: Arc<Mutex<BreakpointConfig>>,
249    // Set while we are doing evaluate calls (>= 1 means disable)
250    disable_breakpoints: Arc<AtomicUsize>,
251}

253#[derive(Debug, Clone, Copy, Dupe)]
254enum Next {
255    Continue,
256    RemainPaused,
257    Step(StepKind),
258}

260fn convert_frame(id: usize, name: String, location: Option<FileSpan>) -> StackFrame {
261    let mut s = StackFrame {
262        id: id as i64,
263        name,
264        column: 0,
265        line: 0,
266        end_column: None,
267        end_line: None,
268        module_id: None,
269        presentation_hint: None,
270        source: None,
271    };
272    if let Some(loc) = location {
273        let span = loc.resolve_span();
274        s.line = span.begin.line as i64 + 1;
275        s.column = span.begin.column as i64 + 1;
276        s.end_line = Some(span.end.line as i64 + 1);
277        s.end_column = Some(span.end.column as i64 + 1);
278        s.source = Some(Source {
279            path: Some(loc.filename().to_owned()),
280            ..Source::default()
281        })
282    }
283    s
284}

286impl DapAdapter for DapAdapterImpl {
287    fn set_breakpoints(
288        &self,
289        source: &str,
290        breakpoints: &ResolvedBreakpoints,
291    ) -> anyhow::Result<()> {
292        self.state
293            .breakpoints
294            .lock()
295            .unwrap()
296            .set_breakpoints(source, breakpoints)
297    }
298
299    fn top_frame(&self) -> anyhow::Result<Option<StackFrame>> {
300        self.with_ctx(Box::new(|span, eval| {
301            let frame = eval.call_stack_top_frame();
302            let name = frame.map_or("".to_owned(), |v| v.name);
303            Ok(Some(convert_frame(0, name, Some(span.to_file_span()))))
304        }))
305    }
306
307    fn stack_trace(&self, _: StackTraceArguments) -> anyhow::Result<StackTraceResponseBody> {
308        // Our model of a Frame and the debugger model are a bit different.
309        // We record the location of the call, but DAP wants the location we are at.
310        // We also have them in the wrong order
311        self.with_ctx(Box::new(|span, eval| {
312            let frames = eval.call_stack().into_frames();
313            let mut next = Some(span.to_file_span());
314            let mut res = Vec::with_capacity(frames.len() + 1);
315            for (i, x) in frames.iter().rev().enumerate() {
316                res.push(convert_frame(i, x.name.clone(), next));
317                next = x.location.dupe();
318            }
319            res.push(convert_frame(frames.len(), "Root".to_owned(), next));
320            Ok(StackTraceResponseBody {
321                total_frames: Some(res.len() as i64),
322                stack_frames: res,
323            })
324        }))
325    }
326
327    fn scopes(&self) -> anyhow::Result<ScopesInfo> {
328        self.with_ctx(Box::new(|_, eval| {
329            let vars = eval.local_variables();
330            Ok(ScopesInfo {
331                num_locals: vars.len(),
332            })
333        }))
334    }
335
336    fn variables(&self) -> anyhow::Result<VariablesInfo> {
337        self.with_ctx(Box::new(|_, eval| {
338            let vars = eval.local_variables();
339            Ok(VariablesInfo {
340                locals: vars
341                    .into_iter()
342                    .map(|(name, value)| Variable::from_value(PathSegment::Attr(name), value))
343                    .collect(),
344            })
345        }))
346    }
347
348    fn inspect_variable(&self, path: VariablePath) -> anyhow::Result<InspectVariableInfo> {
349        let state = self.state.dupe();
350        self.with_ctx(Box::new(move |_span, eval| {
351            let access_path = &path.access_path;
352            let mut value = match &path.scope {
353                super::Scope::Local(name) => {
354                    let mut vars = eval.local_variables();
355                    // since vars is owned within this closure scope we can just remove value from the map
356                    // obtaining owned variable as the rest of the map will be dropped anyway
357                    vars.shift_remove(name).ok_or_else(|| {
358                        anyhow::Error::msg(format!("Local variable {name} not found"))
359                    })
360                }
361                super::Scope::Expr(expr) => evaluate_expr(&state, eval, expr.to_owned()),
362            }?;
363
364            for p in access_path.iter() {
365                value = p.get(&value, eval.heap()).into_anyhow_result()?;
366            }
367            InspectVariableInfo::try_from_value(value, eval.heap()).into_anyhow_result()
368        }))
369    }
370
371    fn continue_(&self) -> anyhow::Result<()> {
372        self.inject_next(Next::Continue);
373        Ok(())
374    }
375
376    fn step(&self, kind: StepKind) -> anyhow::Result<()> {
377        self.inject_next(Next::Step(kind));
378        Ok(())
379    }
380
381    fn evaluate(&self, expr: &str) -> anyhow::Result<EvaluateExprInfo> {
382        let state = self.state.dupe();
383        let expression = expr.to_owned();
384        self.with_ctx(Box::new(move |_, eval| {
385            match evaluate_expr(&state, eval, expression.clone()) {
386                Err(e) => Err(e),
387                Ok(v) => Ok(EvaluateExprInfo::from_value(&v)),
388            }
389        }))
390    }
391}

393impl DapAdapterImpl {
394    fn inject<T: 'static + Send>(
395        &self,
396        f: Box<dyn Fn(FileSpanRef, &mut Evaluator) -> (Next, T) + Send>,
397    ) -> T {
398        let (sender, receiver) = channel();
399        self.sender
400            .send(Box::new(move |span, eval| {
401                let (next, res) = f(span, eval);
402                sender.send(res).unwrap();
403                next
404            }))
405            .unwrap();
406        receiver.recv().unwrap()
407    }
408
409    fn inject_next(&self, next: Next) {
410        self.inject(Box::new(move |_, _| (next, ())))
411    }
412
413    fn with_ctx<T: 'static + Send>(
414        &self,
415        f: Box<dyn Fn(FileSpanRef, &mut Evaluator) -> T + Send>,
416    ) -> T {
417        self.inject(Box::new(move |span, eval| {
418            (Next::RemainPaused, f(span, eval))
419        }))
420    }
421}

436pub(crate) fn resolve_breakpoints(
437    args: &SetBreakpointsArguments,
438    ast: &AstModule,
439) -> anyhow::Result<ResolvedBreakpoints> {
440    let poss: HashMap<usize, FileSpan> = ast
441        .stmt_locations()
442        .iter()
443        .map(|span| (span.resolve_span().begin.line, span.dupe()))
444        .collect();
445    Ok(ResolvedBreakpoints(args.breakpoints.as_ref().map_or(
446        Vec::new(),
447        |v| {
448            v.map(|x| {
449                poss.get(&(x.line as usize - 1)).map(|span| Breakpoint {
450                    span: span.clone(),
451                    condition: x.condition.clone(),
452                })
453            })
454        },
455    )))
456}

458pub(crate) fn resolved_breakpoints_to_dap(
459    breakpoints: &ResolvedBreakpoints,
460) -> SetBreakpointsResponseBody {
461    SetBreakpointsResponseBody {
462        breakpoints: breakpoints.0.map(|x| breakpoint(x.is_some())),
463    }
