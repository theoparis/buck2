50/// Information about a "structural variable" inspected by a debugger
51/// this currently has DAP-like semantic meaning that every complex object returned
52/// by debugger from the stack or from the heap can be broken down into "variables"
53/// this is how structured data is managed by the debugger.
54/// Something similar to LLDB's SBValue
55pub struct Variable {
56    /// Name of the variable.
57    pub name: PathSegment,
58    /// The value as a String.
59    pub value: String,
60    /// The variables type.
61    pub type_: String,
62    /// Indicates whether there are children available for a given variable.
63    pub has_children: bool,
64}
65
66/// Represents the scope of a variable.
67#[derive(Clone, Debug)]
68pub enum Scope {
69    /// A local variable's scope, identified by its name.
70    Local(String),
71    /// A scope determined by a particular expression.
72    #[allow(dead_code)]
73    Expr(String),
74}
75
76/// Represents a variable's "access path" for a local variable or watch expression.
77///
78/// # Examples
79///
80/// - For path `var1.field1[0]`, the scope is `Local("var1")` and the access path is `["field1", 0]`.
81/// - For path `someObject.method().something`, the scope is `Expr("someObject.method().something")`. The access path
82///   includes segments inside the evaluated result of `someObject.method().something` if it returns a complex object.
83#[derive(Clone, Debug)]
84pub struct VariablePath {
85    scope: Scope,
86    access_path: Vec<PathSegment>,
87}
88
89impl VariablePath {
90    /// creates new instance of VariablePath from a given expression
91    pub fn new_expression(expr: impl Into<String>) -> VariablePath {
92        VariablePath {
93            scope: Scope::Expr(expr.into()),
94            access_path: vec![],
95        }
96    }
97
98    /// creates new instance of VariablePath from a given local variable
99    pub fn new_local(scope: impl Into<String>) -> VariablePath {
100        VariablePath {
101            scope: Scope::Local(scope.into()),
102            access_path: vec![],
103        }
104    }
105    /// creates a child segment of given access path
106    pub fn make_child(&self, path: PathSegment) -> VariablePath {
107        // TODO(vmakaev): figure out if need to optimize memory usage and build persistent data structure
108        let mut access_path = self.access_path.clone();
109        access_path.push(path);
110
111        Self {
112            scope: self.scope.clone(),
113            access_path,
114        }
115    }
116}

118/// Represents a segment in an access expression.
119///
120/// For the given expression `name.field1.array\[0\]`, the segments are "field1", "array", and "0".
121#[derive(Clone, Debug)]
122pub enum PathSegment {
123    /// Represents a path segment that accesses array-like types (i.e., types indexable by numbers).
124    Index(i32),
125    /// Represents a path segment that accesses object-like types (i.e., types keyed by strings).
126    Attr(String),
127    /// Represents a path segment that accesses dict items by key.
128    Key(String),
129}
130
131impl Display for PathSegment {
132    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
133        match self {
134            PathSegment::Index(x) => write!(f, "{x}"),
135            PathSegment::Attr(x) => f.write_str(x),
136            PathSegment::Key(x) => write!(f, "\"{x}\""),
137        }
138    }
139}
140
141impl PathSegment {
142    fn get<'v>(&self, v: &Value<'v>, heap: &'v Heap) -> crate::Result<Value<'v>> {
143        match self {
144            PathSegment::Index(i) => v.at(heap.alloc(*i), heap).map_err(Into::into),
145            PathSegment::Attr(key) => v.get_attr_error(key.as_str(), heap),
146            PathSegment::Key(i) => v.at(heap.alloc(i.to_owned()), heap).map_err(Into::into),
147        }
148    }
149}

151impl Variable {
152    /// Helper to convert to the DAP Variable type.
153    pub fn to_dap(self) -> debugserver_types::Variable {
154        debugserver_types::Variable {
155            name: self.name.to_string(),
156            value: self.value,
157            type_: Some(self.type_),
158            evaluate_name: None,
159            indexed_variables: None,
160            named_variables: None,
161            presentation_hint: None,
162            variables_reference: 0,
163        }
164    }
165
166    fn tuple_value_as_str<'v>(v: Value<'v>) -> String {
167        match v.length() {
168            Ok(size) if size > 0 => format!("<tuple, size={size}>"),
169            _ => "()".to_owned(),
170        }
171    }
172
173    fn list_value_as_str<'v>(v: Value<'v>) -> String {
174        match v.length() {
175            Ok(size) if size > 0 => format!("<list, size={size}>"),
176            _ => "[]".to_owned(),
177        }
178    }
179
180    fn dict_value_as_str<'v>(v: Value<'v>) -> String {
181        match v.length() {
182            Ok(size) if size > 0 => format!("<dict, size={size}>"),
183            _ => "{}".to_owned(),
184        }
185    }
186
187    fn struct_like_value_as_str<'v>(v: Value<'v>) -> String {
188        let attrs = v.dir_attr();
189        format!("<type:{}, size={}>", v.get_type(), attrs.len())
190    }
191
192    pub(crate) fn truncate_string(mut str_value: String, mut max_len: usize) -> String {
193        if str_value.len() > max_len {
194            // Find a valid UTF-8 cut-off point that's within our max length
195            while max_len > 0 && !str_value.is_char_boundary(max_len) {
196                max_len -= 1;
197            }
198            if max_len > 0 {
199                str_value.truncate(max_len);
200                str_value.push_str("...(truncated)");
201            }
202        }
203        str_value
204    }
205
206    pub(crate) fn value_as_str<'v>(v: &Value<'v>) -> String {
207        if Self::has_children(v) {
208            match v.get_type() {
209                "list" => Self::list_value_as_str(*v),
210                "tuple" => Self::tuple_value_as_str(*v),
211                "dict" => Self::dict_value_as_str(*v),
212                _ => Self::struct_like_value_as_str(*v),
213            }
214        } else {
215            match v.get_type() {
216                "function" => "<function>".to_owned(),
217                _ => {
218                    const MAX_STR_LEN: usize = 10000;
219                    Self::truncate_string(v.to_str(), MAX_STR_LEN)
220                }
221            }
222        }
223    }
224
225    /// creates a new instance of Variable from a given starlark value
226    pub fn from_value<'v>(name: PathSegment, v: Value<'v>) -> Self {
227        Self {
228            name,
229            value: Self::value_as_str(&v),
230            type_: v.get_type().to_owned(),
231            has_children: Self::has_children(&v),
232        }
233    }
234
235    pub(crate) fn has_children<'v>(v: &Value<'v>) -> bool {
236        match v.get_type() {
237            "function" | "never" | "NoneType" | "bool" | "int" | "float" | "string" => false,
238            "list" | "tuple" | "dict" => match v.length() {
239                Ok(length) => length > 0,
240                _ => false,
241            },
242            _ => true,
243        }
244    }
245}

247/// The kind of debugger step, used for next/stepin/stepout requests.
248#[derive(Debug, Clone, Dupe, Copy)]
249pub enum StepKind {
250    /// Step "into" the statement. This is generally used on a function call to stop in the
251    /// function call. In practice, this will stop on the next statement.
252    Into,
253    /// Step "over" the statement. This will stop on the next statement in the current function
254    /// after the current one (so will step "over" a function call).
255    Over,
256    /// Step "out" of the current function. This will stop on the next statement after this
257    /// function returns.
258    Out,
259}

274/// Information about expression evaluation result
275pub struct EvaluateExprInfo {
276    /// The value as a String.
277    pub result: String,
278    /// The variables type.
279    pub type_: String,
280    /// Indicates whether there are children available for a given variable.
281    pub has_children: bool,
282}
283
284impl InspectVariableInfo {
285    fn try_from_dict<'v>(value_dict: DictRef<'v>) -> crate::Result<Self> {
286        let key_segments = value_dict
287            .iter()
288            .map(|(key, value)| (PathSegment::Key(key.to_str()), value))
289            .collect::<Vec<_>>();
290
291        Ok(Self {
292            sub_values: key_segments
293                .into_iter()
294                .map(|(path_segment, value)| Variable::from_value(path_segment, value))
295                .collect::<Vec<_>>(),
296        })
297    }
298
299    fn try_from_struct_like<'v>(v: Value<'v>, heap: &'v Heap) -> crate::Result<Self> {
300        Ok(Self {
301            sub_values: v
302                .dir_attr()
303                .into_iter()
304                .map(|child_name| {
305                    let child_value = v.get_attr_error(&child_name, heap)?;
306                    let segment = PathSegment::Attr(child_name);
307                    Ok(Variable::from_value(segment, child_value))
308                })
309                .collect::<crate::Result<Vec<_>>>()?,
310        })
311    }
312
313    fn try_from_array_like<'v>(v: Value<'v>, heap: &'v Heap) -> crate::Result<Self> {
314        let len = v.length()?;
315        Ok(Self {
316            sub_values: (0..len)
317                .map(|i| {
318                    let index = heap.alloc(i);
319                    v.at(index, heap)
320                        .map(|v| Variable::from_value(PathSegment::Index(i), v))
321                })
322                .collect::<crate::Result<Vec<_>>>()?,
323        })
324    }
325
326    /// Trying to create InspectVariableInfo from a given starlark value
327    pub fn try_from_value<'v>(v: Value<'v>, heap: &'v Heap) -> crate::Result<Self> {
328        match v.get_type() {
329            "dict" => Self::try_from_dict(
330                DictRef::from_value(v).ok_or(anyhow::Error::msg("not a dictionary"))?,
331            ),
332            "struct" => Self::try_from_struct_like(v, heap),
333            "list" | "tuple" => Self::try_from_array_like(v, heap),
334            "bool" | "int" | "float" | "string" => Ok(Default::default()),
335            "function" | "never" | "NoneType" => Ok(Default::default()),
336            // this branch will catch Ty::basic(name)
337            _ => Self::try_from_struct_like(v, heap),
338        }
339    }
340}

353/// The DapAdapter accepts DAP requests and updates the hooks in the running evaluator.
354pub trait DapAdapter: Debug + Send + 'static {
355    /// Sets multiple breakpoints for a file (and clears existing ones).
356    ///
357    /// See <https://microsoft.github.io/debug-adapter-protocol/specification#Requests_SetBreakpoints>
358    fn set_breakpoints(
359        &self,
360        source: &str,
361        breakpoints: &ResolvedBreakpoints,
362    ) -> anyhow::Result<()>;
363
364    /// Gets the top stack frame, may be None if entered from native.
365    fn top_frame(&self) -> anyhow::Result<Option<StackFrame>>;
366
367    /// Gets a stacktrace from the current execution state.
368    ///
369    /// See <https://microsoft.github.io/debug-adapter-protocol/specification#Requests_StackTrace>
370    fn stack_trace(&self, args: StackTraceArguments) -> anyhow::Result<StackTraceResponseBody>;
371
372    /// Gets the variables scope for a frame.
373    ///
374    /// See <https://microsoft.github.io/debug-adapter-protocol/specification#Requests_Scopes>
375    fn scopes(&self) -> anyhow::Result<ScopesInfo>;
376
377    /// Gets variables for the current scope
378    ///
379    /// See <https://microsoft.github.io/debug-adapter-protocol/specification#Requests_Variables>
380    fn variables(&self) -> anyhow::Result<VariablesInfo>;
381
382    /// Gets all child variables for the given access path
383    ///
384    /// See <https://microsoft.github.io/debug-adapter-protocol/specification#Requests_Variables>
385    fn inspect_variable(&self, path: VariablePath) -> anyhow::Result<InspectVariableInfo>;
386
387    /// Resumes execution.
388    ///
389    /// See <https://microsoft.github.io/debug-adapter-protocol/specification#Requests_Continue>
390    fn continue_(&self) -> anyhow::Result<()>;
391
392    /// Continues execution until some condition.
393    ///
394    /// See <https://microsoft.github.io/debug-adapter-protocol/specification#Requests_Next>
395    /// <https://microsoft.github.io/debug-adapter-protocol/specification#Requests_StepIn>
396    /// <https://microsoft.github.io/debug-adapter-protocol/specification#Requests_StepOut>
397    fn step(&self, kind: StepKind) -> anyhow::Result<()>;
398    /// Evaluates in expression in the context of the top-most frame.
399    ///
400    /// See <https://microsoft.github.io/debug-adapter-protocol/specification#Requests_Evaluate>
401    fn evaluate(&self, expr: &str) -> anyhow::Result<EvaluateExprInfo>;
402}

429/// This is sort of the evaluation side of the DapAdapter. It's expected that these are on different threads
430/// (the starlark evaluation is single-threaded, so certainly the DapAdapter itself doesn't do interesting
431/// things there).
432pub trait DapAdapterEvalHook: Debug + Send + 'static {
433    /// Hooks the evaluator for this DapAdapter.
434    fn add_dap_hooks(self: Box<Self>, eval: &mut Evaluator<'_, '_, '_>);
435}

437/// The DAP capabilities that the adapter supports.
438pub fn dap_capabilities() -> Capabilities {

441        supports_evaluate_for_hovers: Some(true),
442        supports_set_variable: Some(true),
443        supports_step_in_targets_request: Some(true),
444        supports_conditional_breakpoints: Some(true),
445        ..Capabilities::default()
446    }
447}
