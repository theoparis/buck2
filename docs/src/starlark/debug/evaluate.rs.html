18use crate::collections::SmallMap;
19use crate::debug::inspect::to_scope_names_by_local_slot_id;
20use crate::eval::Evaluator;
21use crate::eval::runtime::slots::LocalSlotIdCapturedOrNot;
22use crate::syntax::AstModule;
23use crate::values::FrozenStringValue;
24use crate::values::Value;
25
26impl<'v> Evaluator<'v, '_, '_> {
27    /// Evaluate statements in the existing context. This function is designed for debugging,
28    /// not production use.
29    ///
30    /// There are lots of health warnings on this code. Might not work with frozen modules, unassigned variables,
31    /// nested definitions etc. It would be a bad idea to rely on the results of continued execution
32    /// after evaluating stuff randomly.
33    pub fn eval_statements(&mut self, statements: AstModule) -> crate::Result<Value<'v>> {
34        // We are doing a lot of funky stuff here. It's amazing anything works, so let's not push our luck with GC.
35        self.disable_gc();
36
37        // Everything must be evaluated with the current heap (or we'll lose memory), which means
38        // the current module (eval.module_env).
39        // We also want access to the module variables (fine), the locals (need to move them over),
40        // and the frozen variables (move them over).
41        // Afterwards, we want to put everything back - locals can move back to locals, modules
42        // can stay where they are, but frozen values are discarded.
43
44        // We want all the local variables to be available to the module, so we capture
45        // everything before, shove the local variables into the module, and then revert after
46        let original_module: SmallMap<FrozenStringValue, Option<Value<'v>>> = self
47            .module_env
48            .mutable_names()
49            .all_names_and_slots()
50            .into_iter()
51            .map(|(name, slot)| (name, self.module_env.slots().get_slot(slot)))
52            .collect();
53
54        // Push all the frozen variables into the module
55        if let Some(frozen) = self.top_frame_def_frozen_module(true)? {
56            for (name, slot) in frozen.names.symbols() {
57                if let Some(value) = frozen.get_slot(slot) {
58                    self.module_env.set(&name, value.to_value())
59                }
60            }
61        }
62
63        // Push all local variables into the module
64        let locals = self
65            .call_stack
66            .to_function_values()
67            .into_iter()
68            .rev()
69            .find_map(to_scope_names_by_local_slot_id);
70        if let Some(names) = &locals {
71            for (slot, name) in names.iter().enumerate() {
72                if let Some(value) = self
73                    .current_frame
74                    .get_slot_slow(LocalSlotIdCapturedOrNot(slot as u32))
75                {
76                    self.module_env.set(name, value)
77                }
78            }
79        }
80
81        let globals = self.top_frame_def_info_for_debugger()?.globals;
82        let res = self.eval_module(statements, &globals);
83
84        // Now put the Module back how it was before we started, as best we can
85        // and move things into locals if that makes sense
86        if let Some(names) = &locals {
87            for (slot, name) in names.iter().enumerate() {
88                if let Some(value) = self.module_env.get(name) {
89                    self.current_frame
90                        .set_slot_slow(LocalSlotIdCapturedOrNot(slot as u32), value)
91                }
92            }
93            for (name, slot) in self.module_env.mutable_names().all_names_and_slots() {
94                match original_module.get(&name) {
95                    None => self.module_env.mutable_names().hide_name(&name),
96                    Some(Some(value)) => self.module_env.slots().set_slot(slot, *value),
97                    _ => {} // No way to unassign a previously assigned value yet
98                }
99            }
100        }
101
102        res
103    }
104}

106#[cfg(test)]
107mod tests {
108    use itertools::Itertools;
109    use starlark_derive::starlark_module;
110    use starlark_syntax::error::StarlarkResultExt;
111
112    use super::*;
113    use crate as starlark;
114    use crate::assert;
115    use crate::environment::GlobalsBuilder;
116    use crate::syntax::Dialect;
117    use crate::wasm::is_wasm;
118
119    #[starlark_module]
120    fn debugger(builder: &mut GlobalsBuilder) {
121        fn debug_evaluate<'v>(
122            code: String,
123            eval: &mut Evaluator<'v, '_, '_>,
124        ) -> anyhow::Result<Value<'v>> {
125            let ast = AstModule::parse("interactive", code, &Dialect::AllOptionsInternal)
126                .into_anyhow_result()?;
127            eval.eval_statements(ast).into_anyhow_result()
128        }
129    }
130
131    #[test]
132    fn test_debug_evaluate() {
133        if is_wasm() {
134            return;
135        }
136
137        let mut a = assert::Assert::new();
138        a.disable_static_typechecking();
139        a.globals_add(debugger);
140        let check = r#"
141assert_eq(debug_evaluate("1+2"), 3)
142x = 10
143assert_eq(debug_evaluate("x"), 10)
144assert_eq(debug_evaluate("x = 5"), None)
145assert_eq(x, 5)
146y = [20]
147debug_evaluate("y.append(30)")
148assert_eq(y, [20, 30])
149"#;
150        // Check evaluation works at the root
151        a.pass(check);
152        // And inside functions
153        a.pass(&format!(
154            "def local():\n{}\nlocal()",
155            check.lines().map(|x| format!("    {x}")).join("\n")
156        ));
157
158        // Check we get the right stack frames
159        a.pass(
160            r#"
161def foo(x, y, z):
162    return bar(y)
163def bar(x):
164    return debug_evaluate("x")
165assert_eq(foo(1, 2, 3), 2)
166"#,
167        );
168
169        // Check we can access module-level and globals
170        a.pass(
171            r#"
172x = 7
173def bar(y):
174    return debug_evaluate("x + y")
175assert_eq(bar(4), 4 + 7)
176"#,
177        );
178
179        // Check module-level access works in imported modules
180        a.module(
181            "test",
182            r#"
183x = 7
184z = 2
185def bar(y):
186    assert_eq(x, 7)
187    debug_evaluate("x = 20")
188    assert_eq(x, 7) # doesn't work for frozen variables
189    return debug_evaluate("x + y + z")
190"#,
191        );
192        a.pass("load('test', 'bar'); assert_eq(bar(4), 4 + 7 + 2)");
193    }
