55pub(crate) struct TypingContext<'a> {
56    pub(crate) oracle: TypingOracleCtx<'a>,
57    // We'd prefer this to be a &mut self,
58    // but that makes writing the code more fiddly, so just RefCell the errors
59    pub(crate) errors: RefCell<Vec<TypingError>>,
60    pub(crate) approximoations: RefCell<Vec<Approximation>>,
61    pub(crate) types: UnorderedMap<BindingId, Ty>,
62    pub(crate) module_var_types: &'a ModuleVarTypes,
63}
64
65impl TypingContext<'_> {
66    pub(crate) fn approximation(&self, category: &'static str, message: impl Debug) -> Ty {
67        self.approximoations
68            .borrow_mut()
69            .push(Approximation::new(category, message));
70        Ty::any()
71    }
72
73    fn result_to_ty(&self, result: Result<Ty, TypingError>) -> Ty {
74        match result {
75            Ok(x) => x,
76            Err(e) => {
77                self.errors.borrow_mut().push(e);
78                Ty::never()
79            }
80        }
81    }
82
83    fn result_to_ty_with_internal_error(
84        &self,
85        result: Result<Ty, TypingOrInternalError>,
86    ) -> Result<Ty, InternalError> {
87        match result {
88            Ok(x) => Ok(x),
89            Err(TypingOrInternalError::Internal(e)) => Err(e),
90            Err(TypingOrInternalError::Typing(e)) => {
91                self.errors.borrow_mut().push(e);
92                Ok(Ty::never())
93            }
94        }
95    }
96
97    fn validate_call(&self, fun: &Ty, args: &TyCallArgs, span: Span) -> Result<Ty, InternalError> {
98        self.result_to_ty_with_internal_error(self.oracle.validate_call(span, fun, args))
99    }
100
101    fn from_iterated(&self, ty: &Ty, span: Span) -> Ty {
102        self.result_to_ty(self.oracle.iter_item(Spanned { node: ty, span }))
103    }
104
105    pub(crate) fn validate_type(
106        &self,
107        got: Spanned<&Ty>,
108        require: &Ty,
109    ) -> Result<(), InternalError> {
110        if let Err(e) = self.oracle.validate_type(got, require) {
111            match e {
112                TypingOrInternalError::Typing(e) => {
113                    self.errors.borrow_mut().push(e);
114                }
115                TypingOrInternalError::Internal(e) => {
116                    return Err(e);
117                }
118            }
119        }
120        Ok(())
121    }
122
123    fn expr_dot(&self, ty: &Ty, attr: &str, span: Span) -> Ty {
124        self.result_to_ty(self.oracle.expr_dot(span, ty, attr))
125    }
126
127    fn expr_index(
128        &self,
129        span: Span,
130        array: &CstExpr,
131        index: &CstExpr,
132    ) -> Result<Ty, InternalError> {
133        let array_ty = self.expression_type(array)?;
134
135        // Hack for `list[str]`: list of `list` is just "function", and we don't want
136        // to make it custom type and have overly complex machinery for handling it.
137        // So we just special case it here.
138        // TODO(nga): push this code down to `OracleTypingCtx`.
139        if array_ty.is_function() {
140            if let ExprP::Identifier(v0) = &array.node {
141                if v0.ident == "list" {
142                    // TODO: make this "eval_type" or something.
143                    return Ok(Ty::any());
144                }
145            }
146        }
147
148        let index = self.expression_type_spanned(index)?;
149        self.result_to_ty_with_internal_error(self.oracle.expr_index(span, array_ty, index))
150    }
151
152    fn expression_un_op(
153        &self,
154        span: Span,
155        arg: &CstExpr,
156        un_op: TypingUnOp,
157    ) -> Result<Ty, InternalError> {
158        let ty = self.expression_type(arg)?;
159        Ok(self.result_to_ty(self.oracle.expr_un_op(span, ty, un_op)))
160    }

162    pub(crate) fn expression_bind_type(&self, x: &BindExpr) -> Result<Ty, InternalError> {
163        match x {
164            BindExpr::Expr(x) => self.expression_type(x),
165            BindExpr::GetIndex(i, x) => Ok(self.oracle.indexed(&self.expression_bind_type(x)?, *i)),
166            BindExpr::Iter(x) => Ok(self.from_iterated(&self.expression_bind_type(x)?, x.span())),
167            BindExpr::AssignModify(lhs, op, rhs) => {
168                let span = lhs.span;
169                let rhs = self.expression_type_spanned(rhs)?;
170                let lhs = self.expression_assign_spanned(lhs)?;
171                let attr = match op {
172                    AssignOp::Add => TypingBinOp::Add,
173                    AssignOp::Subtract => TypingBinOp::Sub,
174                    AssignOp::Multiply => TypingBinOp::Mul,
175                    AssignOp::Divide => TypingBinOp::Div,
176                    AssignOp::FloorDivide => TypingBinOp::FloorDiv,
177                    AssignOp::Percent => TypingBinOp::Percent,
178                    AssignOp::BitAnd => TypingBinOp::BitAnd,
179                    AssignOp::BitOr => TypingBinOp::BitOr,
180                    AssignOp::BitXor => TypingBinOp::BitXor,
181                    AssignOp::LeftShift => TypingBinOp::LeftShift,
182                    AssignOp::RightShift => TypingBinOp::RightShift,
183                };
184                self.result_to_ty_with_internal_error(
185                    self.oracle.expr_bin_op_ty(span, lhs, attr, rhs),
186                )
187            }
188            BindExpr::SetIndex(id, index, e) => {
189                let index = self.expression_type_spanned(index)?;
190                let e = self.expression_bind_type(e)?;
191                let mut res = Vec::new();
192                // We know about list and dict, everything else we just ignore
193                if self.types[id].is_list() {
194                    // If we know it MUST be a list, then the index must be an int
195                    self.validate_type(index.as_ref(), &Ty::int())?;
196                }
197                for ty in self.types[id].iter_union() {
198                    match ty {
199                        TyBasic::List(_) => {
200                            res.push(Ty::list(e.clone()));
201                        }
202                        TyBasic::Dict(..) => {
203                            res.push(Ty::dict(index.node.clone(), e.clone()));
204                        }
205                        _ => {
206                            // Either it's not something we can apply this to, in which case do nothing.
207                            // Or it's an Any, in which case we aren't going to change its type or spot errors.
208                        }
209                    }
210                }
211                Ok(Ty::unions(res))
212            }
213            BindExpr::ListAppend(id, e) => {
214                if self.oracle.probably_a_list(&self.types[id])? {
215                    Ok(Ty::list(self.expression_type(e)?))
216                } else {
217                    // It doesn't seem to be a list, so let's assume the append is non-mutating
218                    Ok(Ty::never())
219                }
220            }
221            BindExpr::ListExtend(id, e) => {
222                if self.oracle.probably_a_list(&self.types[id])? {
223                    Ok(Ty::list(
224                        self.from_iterated(&self.expression_type(e)?, e.span),
225                    ))
226                } else {
227                    // It doesn't seem to be a list, so let's assume the extend is non-mutating
228                    Ok(Ty::never())
229                }
230            }
231        }
232    }

234    /// Used to get the type of an expression when used as part of a ModifyAssign operation
235    fn expression_assign(&self, x: &CstAssignTarget) -> Result<Ty, InternalError> {
236        match &**x {
237            AssignTargetP::Tuple(_) => Ok(self.approximation("expression_assignment", x)),
238            AssignTargetP::Index(a_b) => self.expr_index(x.span, &a_b.0, &a_b.1),
239            AssignTargetP::Dot(_, _) => Ok(self.approximation("expression_assignment", x)),
240            AssignTargetP::Identifier(x) => {
241                if let Some(i) = x.payload {
242                    if let Some(ty) = self.types.get(&i) {
243                        return Ok(ty.clone());
244                    }
245                }
246                Err(InternalError::msg(
247                    "Unknown identifier",
248                    x.span,
249                    self.oracle.codemap,
250                ))
251            }
252        }
253    }
254
255    fn expression_assign_spanned(&self, x: &CstAssignTarget) -> Result<Spanned<Ty>, InternalError> {
256        Ok(Spanned {
257            span: x.span,
258            node: self.expression_assign(x)?,
259        })
260    }

262    /// We don't need the type out of the clauses (it doesn't change the overall type),
263    /// but it is important we see through to the nested expressions to raise errors
264    fn check_comprehension(
265        &self,
266        for_: &ForClauseP<CstPayload>,
267        clauses: &[ClauseP<CstPayload>],
268    ) -> Result<(), InternalError> {
269        self.expression_type(&for_.over)?;
270        for x in clauses {
271            match x {
272                ClauseP::For(x) => self.expression_type(&x.over)?,
273                ClauseP::If(x) => self.expression_type(x)?,
274            };
275        }
276        Ok(())
277    }

279    pub(crate) fn expression_type_spanned(
280        &self,
281        x: &CstExpr,
282    ) -> Result<Spanned<Ty>, InternalError> {
283        Ok(Spanned {
284            span: x.span,
285            node: self.expression_type(x)?,
286        })
287    }

289    fn expr_bin_op(
290        &self,
291        span: Span,
292        lhs: &CstExpr,
293        op: BinOp,
294        rhs: &CstExpr,
295    ) -> Result<Ty, InternalError> {
296        let lhs = self.expression_type_spanned(lhs)?;
297        let rhs = self.expression_type_spanned(rhs)?;
298        self.result_to_ty_with_internal_error(self.oracle.expr_bin_op(span, lhs, op, rhs))
299    }

301    fn expr_call(
302        &self,
303        span: Span,
304        f: &CstExpr,
305        args: &CallArgsP<CstPayload>,
306    ) -> Result<Ty, InternalError> {
307        let args = CallArgsUnpack::unpack(args, self.oracle.codemap)
308            .map_err(InternalError::from_eval_exception)?;
309
310        let CallArgsUnpack {
311            pos,
312            named,
313            star,
314            star_star,
315        } = args;
316
317        let mut pos_ty: Vec<Spanned<Ty>> = Vec::new();
318        for pos in pos {
319            pos_ty.push(Spanned {
320                span: pos.span,
321                node: self.expression_type(&pos.node.expr())?,
322            });
323        }
324
325        let mut named_ty: Vec<Spanned<(&str, Ty)>> = Vec::new();
326        for named in named {
327            let Some(name) = named.name() else {
328                return Err(InternalError::msg(
329                    "Named argument without name",
330                    named.span,
331                    self.oracle.codemap,
332                ));
333            };
334            named_ty.push(Spanned {
335                span: named.span,
336                node: (name, self.expression_type(&named.node.expr())?),
337            });
338        }
339
340        let args_ty = if let Some(star) = star {
341            let ty = self.expression_type_spanned(&star.node.expr())?;
342            self.from_iterated(&ty, star.span);
343            Some(ty)
344        } else {
345            None
346        };
347
348        let kwargs_ty = if let Some(star_star) = star_star {
349            let ty = self.expression_type_spanned(&star_star.node.expr())?;
350            self.validate_type(ty.as_ref(), &Ty::dict(Ty::string(), Ty::any()))?;
351            Some(ty)
352        } else {
353            None
354        };
355
356        let args_ty = TyCallArgs {
357            pos: pos_ty,
358            named: named_ty,
359            args: args_ty,
360            kwargs: kwargs_ty,
361        };
362
363        let f_ty = self.expression_type(f)?;
364        // If we can't resolve the types of the arguments, we can't validate the call,
365        // but we still know the type of the result since the args don't impact that
366        self.validate_call(&f_ty, &args_ty, span)
367    }

369    fn expr_slice(
370        &self,
371        span: Span,
372        x: &CstExpr,
373        start: Option<&CstExpr>,
374        stop: Option<&CstExpr>,
375        stride: Option<&CstExpr>,
376    ) -> Result<Ty, InternalError> {
377        for e in [start, stop, stride].iter().copied().flatten() {
378            self.validate_type(self.expression_type_spanned(e)?.as_ref(), &Ty::int())?;
379        }
380        Ok(self.result_to_ty(self.oracle.expr_slice(span, self.expression_type(x)?)))
381    }

383    fn expr_ident(&self, x: &CstIdent) -> Ty {
384        match &x.node.payload {
385            Some(ResolvedIdent::Slot(Slot::Module(module_slot_id), _)) => self
386                .module_var_types
387                .types
388                .get(module_slot_id)
389                .cloned()
390                .unwrap_or_else(Ty::any),
391            Some(ResolvedIdent::Slot(_, i)) => {
392                if let Some(ty) = self.types.get(i) {
393                    ty.clone()
394                } else {
395                    // All types must be resolved to this point,
396                    // this code is unreachable.
397                    Ty::any()
398                }
399            }
400            Some(ResolvedIdent::Global(g)) => Ty::of_value(g.to_value()),
401            None => {
402                // All identifiers must be resolved at this point,
403                // but we don't stop after scope resolution error,
404                // so this code is reachable.
405                Ty::any()
406            }
407        }
408    }

410    pub(crate) fn expression_type(&self, x: &CstExpr) -> Result<Ty, InternalError> {
411        let span = x.span;
412        match &**x {
413            ExprP::Tuple(xs) => Ok(Ty::tuple(xs.try_map(|x| self.expression_type(x))?)),
414            ExprP::Dot(a, b) => Ok(self.expr_dot(&self.expression_type(a)?, b, b.span)),
415            ExprP::Call(f, args) => self.expr_call(span, f, args),
416            ExprP::Index(a_b) => self.expr_index(span, &a_b.0, &a_b.1),
417            ExprP::Index2(a_i0_i1) => {
418                let (a, i0, i1) = &**a_i0_i1;
419                self.expression_type(a)?;
420                self.expression_type(i0)?;
421                self.expression_type(i1)?;
422                Ok(Ty::any())
423            }
424            ExprP::Slice(x, start, stop, stride) => self.expr_slice(
425                span,
426                x,
427                start.as_deref(),
428                stop.as_deref(),
429                stride.as_deref(),
430            ),
431            ExprP::Identifier(x) => Ok(self.expr_ident(x)),
432            ExprP::Lambda(_) => {
433                self.approximation("We don't type check lambdas", ());
434                Ok(Ty::any_callable())
435            }
436            ExprP::Literal(x) => match x {
437                AstLiteral::Int(_) => Ok(Ty::int()),
438                AstLiteral::Float(_) => Ok(Ty::float()),
439                AstLiteral::String(_) => Ok(Ty::string()),
440                AstLiteral::Ellipsis => Ok(Ty::any()),
441            },
442            ExprP::Not(x) => {
443                if self.expression_type(x)?.is_never() {
444                    Ok(Ty::never())
445                } else {
446                    Ok(Ty::bool())
447                }
448            }
449            ExprP::Minus(x) => self.expression_un_op(span, x, TypingUnOp::Minus),
450            ExprP::Plus(x) => self.expression_un_op(span, x, TypingUnOp::Plus),
451            ExprP::BitNot(x) => self.expression_un_op(span, x, TypingUnOp::BitNot),
452            ExprP::Op(lhs, op, rhs) => self.expr_bin_op(span, lhs, *op, rhs),
453            ExprP::If(c_t_f) => {
454                let c = self.expression_type(&c_t_f.0)?;
455                let t = self.expression_type(&c_t_f.1)?;
456                let f = self.expression_type(&c_t_f.2)?;
457                if c.is_never() {
458                    Ok(Ty::never())
459                } else {
460                    Ok(Ty::union2(t, f))
461                }
462            }
463            ExprP::List(xs) => {
464                let ts = xs.try_map(|x| self.expression_type(x))?;
465                Ok(Ty::list(Ty::unions(ts)))
466            }
467            ExprP::Dict(xs) => {
468                let (ks, vs) = xs
469                    .try_map(|(k, v)| Ok((self.expression_type(k)?, self.expression_type(v)?)))?
470                    .into_iter()
471                    .unzip();
472                Ok(Ty::dict(Ty::unions(ks), Ty::unions(vs)))
473            }
474            ExprP::ListComprehension(a, b, c) => {
475                self.check_comprehension(b, c)?;
476                Ok(Ty::list(self.expression_type(a)?))
477            }
478            ExprP::DictComprehension(k_v, b, c) => {
479                self.check_comprehension(b, c)?;
480                Ok(Ty::dict(
481                    self.expression_type(&k_v.0)?,
482                    self.expression_type(&k_v.1)?,
483                ))
484            }
485            ExprP::FString(_) => Ok(Ty::string()),
486        }
487    }
