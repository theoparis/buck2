18use std::hash::Hash;
19use std::ops::Deref;
20use std::slice;
21
22use allocative::Allocative;
23use dupe::Dupe;
24
25use crate::typing::small_arc_vec::SmallArcVec1;
26
27#[derive(Debug, Clone, Dupe, Allocative)]
28enum SmallArcVec1OrStaticImpl<T: 'static> {
29    Arc(SmallArcVec1<T>),
30    Static(&'static [T]),
31}
32
33#[derive(Debug, Clone, Dupe, Allocative)]
34pub(crate) struct SmallArcVec1OrStatic<T: 'static>(SmallArcVec1OrStaticImpl<T>);
35
36impl<T: 'static> Default for SmallArcVec1OrStatic<T> {
37    fn default() -> Self {
38        Self::new_static(&[])
39    }
40}
41
42impl<T: 'static> Deref for SmallArcVec1OrStatic<T> {
43    type Target = [T];
44
45    fn deref(&self) -> &Self::Target {
46        self.as_slice()
47    }
48}
49
50impl<T: PartialEq + 'static> PartialEq for SmallArcVec1OrStatic<T> {
51    fn eq(&self, other: &Self) -> bool {
52        self.as_slice() == other.as_slice()
53    }
54}
55
56impl<T: Eq + 'static> Eq for SmallArcVec1OrStatic<T> {}
57
58impl<T: Hash + 'static> Hash for SmallArcVec1OrStatic<T> {
59    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
60        self.as_slice().hash(state)
61    }
62}
63
64impl<T: PartialOrd + 'static> PartialOrd for SmallArcVec1OrStatic<T> {
65    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
66        self.as_slice().partial_cmp(other.as_slice())
67    }
68}
69
70impl<T: Ord + 'static> Ord for SmallArcVec1OrStatic<T> {
71    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
72        self.as_slice().cmp(other.as_slice())
73    }
74}
75
76impl<T: 'static> SmallArcVec1OrStatic<T> {
77    pub(crate) fn new_static(x: &'static [T]) -> Self {
78        SmallArcVec1OrStatic(SmallArcVec1OrStaticImpl::Static(x))
79    }
80
81    pub(crate) fn clone_from_slice(x: &[T]) -> Self
82    where
83        T: Clone,
84    {
85        if x.is_empty() {
86            Self::new_static(&[])
87        } else {
88            SmallArcVec1OrStatic(SmallArcVec1OrStaticImpl::Arc(
89                SmallArcVec1::clone_from_slice(x),
90            ))
91        }
92    }
93
94    pub(crate) fn as_slice(&self) -> &[T] {
95        match &self.0 {
96            SmallArcVec1OrStaticImpl::Arc(x) => x,
97            SmallArcVec1OrStaticImpl::Static(x) => x,
98        }
99    }
100}
101
102impl<'a, T> IntoIterator for &'a SmallArcVec1OrStatic<T> {
103    type Item = &'a T;
104    type IntoIter = slice::Iter<'a, T>;
105
106    fn into_iter(self) -> Self::IntoIter {
107        self.as_slice().iter()
108    }
109}
