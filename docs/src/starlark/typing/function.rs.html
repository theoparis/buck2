38/// Custom function typechecker.
39pub trait TyCustomFunctionImpl:
40    Debug + Eq + Ord + Hash + Allocative + Send + Sync + 'static
41{
42    fn is_type(&self) -> bool {
43        false
44    }
45
46    fn validate_call(
47        &self,
48        span: Span,
49        args: &TyCallArgs,
50        oracle: TypingOracleCtx,
51    ) -> Result<Ty, TypingOrInternalError>;
52
53    fn as_callable(&self) -> TyCallable;
54
55    fn as_function(&self) -> Option<&TyFunction> {
56        None
57    }
58}
59
60#[derive(
61    Allocative,
62    Eq,
63    PartialEq,
64    Hash,
65    Ord,
66    PartialOrd,
67    Debug,
68    derive_more::Display
69)]
70#[display(
71    "def({}) -> {}",
72    self.0.as_callable().params(),
73    self.0.as_callable().result(),
74)]
75pub struct TyCustomFunction<F: TyCustomFunctionImpl>(pub F);
76
77impl<F: TyCustomFunctionImpl> TyCustomImpl for TyCustomFunction<F> {
78    fn as_name(&self) -> Option<&str> {
79        Some("function")
80    }
81
82    fn validate_call(
83        &self,
84        span: Span,
85        args: &TyCallArgs,
86        oracle: TypingOracleCtx,
87    ) -> Result<Ty, TypingOrInternalError> {
88        self.0.validate_call(span, args, oracle)
89    }
90
91    fn as_callable(&self) -> Option<TyCallable> {
92        Some(self.0.as_callable())
93    }
94
95    fn as_function(&self) -> Option<&TyFunction> {
96        self.0.as_function()
97    }
98
99    fn bin_op(
100        &self,
101        bin_op: TypingBinOp,
102        _rhs: &TyBasic,
103        _ctx: &TypingOracleCtx,
104    ) -> Result<Ty, TypingNoContextOrInternalError> {
105        match bin_op {
106            // `str | list`.
107            TypingBinOp::BitOr if self.0.is_type() => Ok(Ty::basic(TyBasic::Type)),
108            _ => Err(TypingNoContextOrInternalError::Typing),
109        }
110    }
111
112    fn index(
113        &self,
114        _item: &TyBasic,
115        _ctx: &TypingOracleCtx,
116    ) -> Result<Ty, TypingNoContextOrInternalError> {
117        // TODO(nga): this is hack for `enum` (type) which pretends to be a function.
118        //   Should be a custom type.
119        Ok(Ty::any())
120    }
121
122    fn attribute(&self, _attr: &str) -> Result<Ty, TypingNoContextError> {
123        Err(TypingNoContextError)
124    }
125
126    fn matcher<T: TypeMatcherAlloc>(&self, factory: T) -> T::Result {
127        factory.callable()
128    }
129}

131/// A function.
132#[derive(Debug, Clone, PartialEq, Eq, Hash, PartialOrd, Ord, Allocative)]
133pub struct TyFunction {
134    /// The `.type` property of the function, often `""`.
135    pub(crate) type_attr: Option<Ty>,
136    pub(crate) callable: TyCallable,
137}
138
139impl TyFunction {
140    /// Constructor.
141    pub fn new_with_type_attr(params: ParamSpec, result: Ty, type_attr: Ty) -> Self {
142        // TODO(nga): validate params are in correct order.
143        TyFunction {
144            type_attr: Some(type_attr),
145            callable: TyCallable::new(params, result),
146        }
147    }
148
149    /// Constructor.
150    pub fn new(params: ParamSpec, result: Ty) -> Self {
151        TyFunction {
152            type_attr: None,
153            callable: TyCallable::new(params, result),
154        }
155    }
156
157    /// Callable signature of the function.
158    pub fn callable(&self) -> &TyCallable {
159        &self.callable
160    }
161}
162
163impl TyCustomFunctionImpl for TyFunction {
164    fn is_type(&self) -> bool {
165        self.type_attr.is_some()
166    }
167
168    fn validate_call(
169        &self,
170        span: Span,
171        args: &TyCallArgs,
172        oracle: TypingOracleCtx,
173    ) -> Result<Ty, TypingOrInternalError> {
174        oracle.validate_fn_call(span, &self.callable, args)
175    }
176
177    fn as_callable(&self) -> TyCallable {
178        self.callable.dupe()
179    }
180
181    fn as_function(&self) -> Option<&TyFunction> {
182        Some(self)
183    }
184}
