38/// Controls the formatting to use when parsing `DocString`s from raw docstrings
39#[derive(Copy, Clone, Dupe)]
40pub enum DocStringKind {
41    /// Docstrings provided by users in starlark files, following python-y documentation style.
42    ///
43    /// For functions, they are the piece in `"""` that come right after the `def foo():` line,
44    /// and they have sections for additional details. An example from a starlark file might be:
45    ///
46    /// ```python
47    /// """ Module level docs here """
48    ///
49    /// def some_function(val: "string") -> "string":
50    ///     """ This function takes a string and returns it.
51    ///
52    ///     This is where an explanation might go, but I have none
53    ///
54    ///     Args:
55    ///         val: This is the value that gets returned
56    ///
57    ///     Returns:
58    ///         The original value, because identity functions are fun.
59    /// ```
60    Starlark,
61    /// Docstrings used with `#[starlark_module]` in rust.
62    ///
63    /// These are the documentation strings prefixed by `///` (like these docs) on
64    /// `#[starlark_module]`, and the functions / attributes within it. It supports
65    /// a section `# Arguments`, and `# Returns`, and removes some lines from code
66    /// blocks that are valid for rustdoc, but not useful for people using these
67    /// functions via starlark. An example might be something like:
68    ///
69    /// ```
70    /// # use starlark::starlark_module;
71    /// # use starlark::environment::MethodsBuilder;
72    /// # use starlark::values::Value;
73    ///
74    /// /// These are where the module / object level docs go
75    /// #[starlark_module]
76    /// fn add_some_value(builder: &mut MethodsBuilder) {
77    ///     /// attr1 is an attribute that does nothing interesting.
78    ///     #[starlark(attribute)]
79    ///     fn attr1<'v>(this: Value<'v>) -> starlark::Result<String> {
80    ///         let _ = this;
81    ///         Ok("attr1".to_owned())
82    ///     }
83    ///     /// Copies a string
84    ///     ///
85    ///     /// This is where details would be, if this were
86    ///     /// a more interesting function.
87    ///     ///
88    ///     /// # Arguments
89    ///     /// * `s`: This is string that is returned.
90    ///     ///
91    ///     /// # Returns
92    ///     /// The a copy of the original string.
93    ///     fn copy_string<'v>(this: Value<'v>, s: &str) -> anyhow::Result<String> {
94    ///         let _ = this;
95    ///         Ok(s.to_owned())
96    ///     }
97    /// }
98    /// ```
99    Rust,
100}
101
102impl DocString {
103    /// Extracts the docstring from a function or module body, iff the first
104    /// statement is a string literal.
105    pub(crate) fn extract_raw_starlark_docstring<P: AstPayload>(
106        body: &AstStmtP<P>,
107    ) -> Option<String> {
108        if let StmtP::Statements(stmts) = &body.node {
109            if let Some(Spanned {
110                node:
111                    StmtP::Expression(Spanned {
112                        node: ExprP::Literal(AstLiteral::String(s)),
113                        ..
114                    }),
115                ..
116            }) = stmts.first()
117            {
118                return Some(s.node.to_owned());
119            }
120        };
121        None
122    }
123
124    fn split_summary_details(s: &str) -> Option<(&str, &str)> {
125        let mut summary_len = 0;
126        for line in s.split_inclusive('\n') {
127            if line.trim().is_empty() {
128                let details_start = summary_len + line.len();
129                return Some((s[..summary_len].trim(), &s[details_start..]));
130            } else {
131                summary_len += line.len();
132            }
133        }
134        None
135    }
136
137    // Remove any newlines (and surrounding whitespace) in the summary, and
138    // replace them with a single space.
139    fn normalize_summary(summary: &str) -> String {
140        let mut res = String::with_capacity(summary.len());
141        for line in summary.lines() {
142            if !res.is_empty() {
143                res.push(' ');
144            }
145            res.push_str(line.trim());
146        }
147        res
148    }
149
150    /// Do common work to parse a docstring (dedenting, splitting summary and details, etc)
151    pub fn from_docstring(kind: DocStringKind, user_docstring: &str) -> Option<DocString> {
152        let trimmed_docs = user_docstring.trim();
153        if trimmed_docs.is_empty() {
154            None
155        } else {
156            let split: Option<(&str, &str)> = Self::split_summary_details(trimmed_docs);
157            let (summary, details) = match split {
158                Some((summary, details)) if !summary.is_empty() && !details.is_empty() => {
159                    // Dedent the details separately so that people can have the summary on the
160                    // same line as the opening quotes, and the details indented on subsequent
161                    // lines.
162                    let details = match kind {
163                        DocStringKind::Starlark => textwrap::dedent(details).trim().to_owned(),
164                        DocStringKind::Rust => {
165                            Self::remove_rust_comments(textwrap::dedent(details).trim())
166                        }
167                    };
168                    (summary, Some(details))
169                }
170                _ => (trimmed_docs, None),
171            };
172
173            let summary = Self::normalize_summary(summary);
174
175            Some(DocString { summary, details })
176        }
177    }
178
179    /// Removes rustdoc-style commented out lines from code blocks.
180    fn remove_rust_comments(details: &str) -> String {
181        static CODEBLOCK_RE: Lazy<Regex> = Lazy::new(|| {
182            RegexBuilder::new(r"```(\w*)\n.*?```")
183                .dot_matches_new_line(true)
184                .build()
185                .expect("regex to compile")
186        });
187        static COMMENT_RE: Lazy<Regex> = Lazy::new(|| {
188            RegexBuilder::new(r"^# .*$\n")
189                .multi_line(true)
190                .build()
191                .expect("regex to compile")
192        });
193        CODEBLOCK_RE
194            .replace_all(details, |caps: &regex::Captures| {
195                match caps.get(1).expect("language group").as_str() {
196                    "" | "rust" => COMMENT_RE
197                        .replace_all(caps.get(0).expect("$0 to exist").as_str(), "")
198                        .to_string(),
199                    _ => caps.get(0).expect("$0 to exist").as_str().to_owned(),
200                }
201            })
202            .to_string()
203    }
204
205    /// Join lines up, dedent them, and trim them
206    fn join_and_dedent_lines(lines: &[String]) -> String {
207        textwrap::dedent(&lines.join("\n")).trim().to_owned()
208    }
209
210    /// Parse the sections out of a docstring's `details` text, and remove the requested sections from the text.
211    ///
212    /// "sections" are the various things in doc strings like "Arguments:", "Returns:", etc
213    ///
214    /// # Returns
215    /// - A new instance of `DocString`, with the requested sections, if found, removed.
216    /// - A mapping of section name, converted to lower case, to the cleaned up section text
217    ///     i.e. dedented, section header not present, etc for any found sections.
218    fn parse_and_remove_sections(
219        self,
220        kind: DocStringKind,
221        requested_sections: &[&str],
222    ) -> (Self, HashMap<String, String>) {
223        let mut sections = HashMap::new();
224
225        let mut finish_section =
226            |current_section: &mut Option<String>, current_section_text: &mut Vec<String>| {
227                if let Some(s) = current_section.take() {
228                    sections.insert(s, DocString::join_and_dedent_lines(current_section_text));
229                    current_section_text.clear();
230                }
231            };
232
233        static STARLARK_SECTION_RE: Lazy<Regex> =
234            Lazy::new(|| Regex::new(r"^([\w -]+):\s*$").unwrap());
235        static STARLARK_INDENTED_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(?:\s|$)").unwrap());
236        static RUST_SECTION_RE: Lazy<Regex> =
237            Lazy::new(|| Regex::new(r"^# ([\w -]+)\s*$").unwrap());
238        static RUST_INDENTED_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^.*").unwrap());
239
240        let (section_re, indented_re) = match kind {
241            DocStringKind::Starlark => (&STARLARK_SECTION_RE, &STARLARK_INDENTED_RE),
242            DocStringKind::Rust => (&RUST_SECTION_RE, &RUST_INDENTED_RE),
243        };
244
245        if let Some(details) = self.details {
246            let mut new_details = vec![];
247            let mut current_section = None;
248            let mut current_section_text = vec![];
249
250            for line in details.lines() {
251                if let Some(matches) = section_re.captures(line) {
252                    finish_section(&mut current_section, &mut current_section_text);
253
254                    let found_section = matches.get(1).unwrap().as_str().to_ascii_lowercase();
255                    if requested_sections.contains(&found_section.as_str()) {
256                        current_section = Some(found_section);
257                    } else {
258                        new_details.push(line.to_owned());
259                    }
260                } else if current_section.is_some() && indented_re.is_match(line) {
261                    current_section_text.push(line.to_owned());
262                } else {
263                    new_details.push(line.to_owned());
264                    finish_section(&mut current_section, &mut current_section_text);
265                }
266            }
267
268            finish_section(&mut current_section, &mut current_section_text);
269
270            let joined_details = new_details.join("\n").trim().to_owned();
271            let details = match joined_details.is_empty() {
272                true => None,
273                false => Some(joined_details),
274            };
275            (
276                Self {
277                    summary: self.summary,
278                    details,
279                },
280                sections,
281            )
282        } else {
283            (self, sections)
284        }
285    }
286}

288impl DocFunction {
289    /// Parses function documentation out of a docstring
290    ///
291    /// # Arguments
292    /// * `kind`: The kind of docstring. This determines the formatting that is parsed.
293    /// * `params`: The parameters of the function.
294    /// * `return_type`: The return type. This is pulled from typing info / directly from users,
295    ///                  so it cannot be inferred generically.
296    /// * `raw_docstring`: The raw docstring to be parsed and potentially modified,
297    ///                    removing the sections detailing arguments and return values.
298    ///                    The format is determined by `kind`.
299    pub fn from_docstring(
300        kind: DocStringKind,
301        mut params: DocParams,
302        return_type: Ty,
303        raw_docstring: Option<&str>,
304    ) -> Self {
305        match raw_docstring.and_then(|raw| DocString::from_docstring(kind, raw)) {
306            Some(ds) => {
307                let (function_docstring, sections) =
308                    ds.parse_and_remove_sections(kind, &["arguments", "args", "returns", "return"]);
309
310                match sections.get("arguments").or_else(|| sections.get("args")) {
311                    Some(args) => {
312                        let entries = Self::parse_params(kind, args);
313                        for x in &mut params.doc_params_mut() {
314                            let DocParam { name, docs, .. } = x;
315                            match entries.get(name) {
316                                Some(raw) => *docs = DocString::from_docstring(kind, raw),
317                                None => {}
318                            }
319                        }
320                    }
321                    _ => (),
322                }
323
324                let return_docs = sections
325                    .get("return")
326                    .or_else(|| sections.get("returns"))
327                    .and_then(|raw| DocString::from_docstring(kind, raw));
328
329                DocFunction {
330                    docs: Some(function_docstring),
331                    params,
332                    ret: DocReturn {
333                        docs: return_docs,
334                        typ: return_type,
335                    },
336                }
337            }
338            None => DocFunction {
339                docs: None,
340                params,
341                ret: DocReturn {
342                    docs: None,
343                    typ: return_type,
344                },
345            },
346        }
347    }
348
349    /// Parse out parameter docs from an "Args:" section of a docstring
350    ///
351    /// `args_section` should be dedented, and generally should just be the `args` key of
352    /// the `DocString::parse_and_remove_sections()` function call. This is done as a
353    /// separate function to reduce the number of times that sections are parsed out of
354    /// docstring (e.g. if a user wants both the `Args:` and `Returns:` sections)
355    fn parse_params(kind: DocStringKind, args_section: &str) -> HashMap<String, String> {
356        static STARLARK_ARG_RE: Lazy<Regex> =
357            Lazy::new(|| Regex::new(r"^\*{0,2}(\w+):\s*(.*)").unwrap());
358        static RUST_ARG_RE: Lazy<Regex> =
359            Lazy::new(|| Regex::new(r"^(?:\* )?`(\w+)`:?\s*(.*)").unwrap());
360
361        static INDENTED_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"^(?:\s|$)").unwrap());
362
363        let arg_re = match kind {
364            DocStringKind::Starlark => &STARLARK_ARG_RE,
365            DocStringKind::Rust => &RUST_ARG_RE,
366        };
367
368        let mut ret = HashMap::new();
369        let mut current_arg = None;
370        let mut current_text = vec![];
371
372        for line in args_section.lines() {
373            if let Some(matches) = arg_re.captures(line) {
374                if let Some(a) = current_arg.take() {
375                    ret.insert(a, DocString::join_and_dedent_lines(&current_text));
376                }
377
378                current_arg = Some(matches.get(1).unwrap().as_str().to_owned());
379
380                let doc_match = matches.get(2).unwrap();
381                current_text = vec![format!(
382                    "{}{}",
383                    " ".repeat(doc_match.start()),
384                    doc_match.as_str()
385                )];
386            } else if current_arg.is_some() && INDENTED_RE.is_match(line) {
387                current_text.push(line.to_owned());
388            }
389        }
390
391        if let Some(a) = current_arg.take() {
392            ret.insert(a, DocString::join_and_dedent_lines(&current_text));
393        }
394
395        ret
396    }
397}

405        assert_eq!(
406            DocString::from_docstring(DocStringKind::Starlark, " "),
407            None

411                DocStringKind::Starlark,

416                details: None,
417            })

419        assert_eq!(
