18//! A [Starlark interpreter in Rust](https://github.com/facebook/starlark-rust).
19//! Starlark is a deterministic version of Python, with [a specification](https://github.com/bazelbuild/starlark/blob/master/spec.md),
20//! used by (amongst others) the [Buck](https://buck.build) and [Bazel](https://bazel.build) build systems.
21//!
22//! To evaluate a simple file:
23//!
24//! ```
25//! # fn run() -> starlark::Result<()> {
26//! use starlark::environment::Globals;
27//! use starlark::environment::Module;
28//! use starlark::eval::Evaluator;
29//! use starlark::syntax::AstModule;
30//! use starlark::syntax::Dialect;
31//! use starlark::values::Value;
32//!
33//! let content = r#"
34//! def hello():
35//!    return "hello"
36//! hello() + " world!"
37//! "#;
38//!
39//! // We first parse the content, giving a filename and the Starlark
40//! // `Dialect` we'd like to use (we pick standard).
41//! let ast: AstModule =
42//!     AstModule::parse("hello_world.star", content.to_owned(), &Dialect::Standard)?;
43//!
44//! // We create a `Globals`, defining the standard library functions available.
45//! // The `standard` function uses those defined in the Starlark specification.
46//! let globals: Globals = Globals::standard();
47//!
48//! // We create a `Module`, which stores the global variables for our calculation.
49//! let module: Module = Module::new();
50//!
51//! // We create an evaluator, which controls how evaluation occurs.
52//! let mut eval: Evaluator = Evaluator::new(&module);
53//!
54//! // And finally we evaluate the code using the evaluator.
55//! let res: Value = eval.eval_module(ast, &globals)?;
56//! assert_eq!(res.unpack_str(), Some("hello world!"));
57//! # Ok(())
58//! # }
59//! # fn main(){ run().unwrap(); }
60//! ```
61//!
62//! From this example there are lots of ways to extend it, which we do so below.
63//!
64//! ## Call Rust functions from Starlark
65//!
66//! We want to define a function in Rust (that computes quadratics), and then call it from Starlark.
67//! We define the function using the [`#[starlark_module]`](macro@starlark_module) attribute, and add it to
68//! a [`Globals`](environment::Globals) object.
69//!
70//! ```
71//! #[macro_use]
72//! extern crate starlark;
73//! # fn run() -> starlark::Result<()> {
74//! use starlark::environment::{GlobalsBuilder, Module};
75//! use starlark::eval::Evaluator;
76//! use starlark::syntax::{AstModule, Dialect};
77//! use starlark::values::Value;
78//!
79//! // This defines the function that is visible to Starlark
80//! #[starlark_module]
81//! fn starlark_quadratic(builder: &mut GlobalsBuilder) {
82//!     fn quadratic(a: i32, b: i32, c: i32, x: i32) -> anyhow::Result<i32> {
83//!         Ok(a * x * x + b * x + c)
84//!     }
85//! }
86//!
87//! // We build our globals to make the function available in Starlark
88//! let globals = GlobalsBuilder::new().with(starlark_quadratic).build();
89//! let module = Module::new();
90//! let mut eval = Evaluator::new(&module);
91//!
92//! // Let's test calling the function from Starlark code
93//! let starlark_code = r#"
94//! quadratic(4, 2, 1, x = 8)
95//! "#;
96//!
97//! let ast = AstModule::parse("quadratic.star", starlark_code.to_owned(), &Dialect::Standard)?;
98//! let res = eval.eval_module(ast, &globals)?;
99//! assert_eq!(res.unpack_i32(), Some(273)); // Verify that we got an `int` return value of 4 * 8^2 + 2 * 8 + 1 = 273
100//! # Ok(())
101//! # }
102//! # fn main(){ run().unwrap(); }
103//! ```
104//!
105//! ## Collect Starlark values
106//!
107//! If we want to use Starlark as an enhanced JSON, we can define an `emit` function
108//! to "write out" a JSON value, and use the [`Evaluator.extra`](eval::Evaluator::extra) field to store it.
109//!
110//! ```
111//! #[macro_use]
112//! extern crate starlark;
113//! # fn run() -> starlark::Result<()> {
114//! use std::cell::RefCell;
115//!
116//! use starlark::any::ProvidesStaticType;
117//! use starlark::environment::GlobalsBuilder;
118//! use starlark::environment::Module;
119//! use starlark::eval::Evaluator;
120//! use starlark::syntax::AstModule;
121//! use starlark::syntax::Dialect;
122//! use starlark::values::Value;
123//! use starlark::values::ValueLike;
124//! use starlark::values::none::NoneType;
125//!
126//! let content = r#"
127//! emit(1)
128//! emit(["test"])
129//! emit({"x": "y"})
130//! "#;
131//!
132//! // Define a store in which to accumulate JSON strings
133//! #[derive(Debug, ProvidesStaticType, Default)]
134//! struct Store(RefCell<Vec<String>>);
135//!
136//! impl Store {
137//!     fn add(&self, x: String) {
138//!         self.0.borrow_mut().push(x)
139//!     }
140//! }
141//!
142//! #[starlark_module]
143//! fn starlark_emit(builder: &mut GlobalsBuilder) {
144//!     fn emit(x: Value, eval: &mut Evaluator) -> anyhow::Result<NoneType> {
145//!         // We modify extra (which we know is a Store) and add the JSON of the
146//!         // value the user gave.
147//!         eval.extra
148//!             .unwrap()
149//!             .downcast_ref::<Store>()
150//!             .unwrap()
151//!             .add(x.to_json()?);
152//!         Ok(NoneType)
153//!     }
154//! }
155//!
156//! let ast = AstModule::parse("json.star", content.to_owned(), &Dialect::Standard)?;
157//! // We build our globals adding some functions we wrote
158//! let globals = GlobalsBuilder::new().with(starlark_emit).build();
159//! let module = Module::new();
160//! let store = Store::default();
161//! {
162//!     let mut eval = Evaluator::new(&module);
163//!     // We add a reference to our store
164//!     eval.extra = Some(&store);
165//!     eval.eval_module(ast, &globals)?;
166//! }
167//! assert_eq!(&*store.0.borrow(), &["1", "[\"test\"]", "{\"x\":\"y\"}"]);
168//! # Ok(())
169//! # }
170//! # fn main(){ run().unwrap(); }
171//! ```
172//!
173//! ## Enable Starlark extensions (e.g. types)
174//!
175//! Our Starlark supports a number of extensions, including type annotations, which are
176//! controlled by the [`Dialect`](syntax::Dialect) type.
177//!
178//! ```
179//! # fn run() -> starlark::Result<()> {
180//! use starlark::environment::Globals;
181//! use starlark::environment::Module;
182//! use starlark::eval::Evaluator;
183//! use starlark::syntax::AstModule;
184//! use starlark::syntax::Dialect;
185//! use starlark::syntax::DialectTypes;
186//!
187//! let content = r#"
188//! def takes_int(x: int):
189//!     pass
190//! takes_int("test")
191//! "#;
192//!
193//! // Make the dialect enable types
194//! let dialect = Dialect {
195//!     enable_types: DialectTypes::Enable,
196//!     ..Dialect::Standard
197//! };
198//! // We could equally have done `dialect = Dialect::Extended`.
199//! let ast = AstModule::parse("json.star", content.to_owned(), &dialect)?;
200//! let globals = Globals::standard();
201//! let module = Module::new();
202//! let mut eval = Evaluator::new(&module);
203//! let res = eval.eval_module(ast, &globals);
204//! // We expect this to fail, since it is a type violation
205//! assert!(
206//!     res.unwrap_err()
207//!         .to_string()
208//!         .contains("Value `test` of type `string` does not match the type annotation `int`")
209//! );
210//! # Ok(())
211//! # }
212//! # fn main(){ run().unwrap(); }
213//! ```
214//!
215//! ## Enable the `load` statement
216//!
217//! You can have Starlark load files imported by the user.
218//! That requires that the loaded modules are first frozen with [`Module.freeze`](environment::Module::freeze).
219//! There is no requirement that the files are on disk, but that would be a common pattern.
220//!
221//! ```
222//! # fn run() -> starlark::Result<()> {
223//! use starlark::environment::FrozenModule;
224//! use starlark::environment::Globals;
225//! use starlark::environment::Module;
226//! use starlark::eval::Evaluator;
227//! use starlark::eval::ReturnFileLoader;
228//! use starlark::syntax::AstModule;
229//! use starlark::syntax::Dialect;
230//!
231//! // Get the file contents (for the demo), in reality use `AstModule::parse_file`.
232//! fn get_source(file: &str) -> &str {
233//!     match file {
234//!         "a.star" => "a = 7",
235//!         "b.star" => "b = 6",
236//!         _ => {
237//!             r#"
238//! load('a.star', 'a')
239//! load('b.star', 'b')
240//! ab = a * b
241//! "#
242//!         }
243//!     }
244//! }
245//!
246//! fn get_module(file: &str) -> starlark::Result<FrozenModule> {
247//!     let ast = AstModule::parse(file, get_source(file).to_owned(), &Dialect::Standard)?;
248//!
249//!     // We can get the loaded modules from `ast.loads`.
250//!     // And ultimately produce a `loader` capable of giving those modules to Starlark.
251//!     let mut loads = Vec::new();
252//!     for load in ast.loads() {
253//!         loads.push((load.module_id.to_owned(), get_module(load.module_id)?));
254//!     }
255//!     let modules = loads.iter().map(|(a, b)| (a.as_str(), b)).collect();
256//!     let mut loader = ReturnFileLoader { modules: &modules };
257//!
258//!     let globals = Globals::standard();
259//!     let module = Module::new();
260//!     {
261//!         let mut eval = Evaluator::new(&module);
262//!         eval.set_loader(&mut loader);
263//!         eval.eval_module(ast, &globals)?;
264//!     }
265//!     // After creating a module we freeze it, preventing further mutation.
266//!     // It can now be used as the input for other Starlark modules.
267//!     Ok(module.freeze()?)
268//! }
269//!
270//! let ab = get_module("ab.star")?;
271//! assert_eq!(ab.get("ab").unwrap().unpack_i32(), Some(42));
272//! # Ok(())
273//! # }
274//! # fn main(){ run().unwrap(); }
275//! ```
276//!
277//! ## Call a Starlark function from Rust
278//!
279//! You can extract functions from Starlark, and call them from Rust, using [`eval_function`](eval::Evaluator::eval_function).
280//!
281//! ```
282//! # fn run() -> starlark::Result<()> {
283//! use starlark::environment::Globals;
284//! use starlark::environment::Module;
285//! use starlark::eval::Evaluator;
286//! use starlark::syntax::AstModule;
287//! use starlark::syntax::Dialect;
288//! use starlark::values::Value;
289//!
290//! let content = r#"
291//! def quadratic(a, b, c, x):
292//!     return a*x*x + b*x + c
293//! quadratic
294//! "#;
295//!
296//! let ast = AstModule::parse("quadratic.star", content.to_owned(), &Dialect::Standard)?;
297//! let globals = Globals::standard();
298//! let module = Module::new();
299//! let mut eval = Evaluator::new(&module);
300//! let quad = eval.eval_module(ast, &globals)?;
301//! let heap = module.heap();
302//! let res = eval.eval_function(
303//!     quad,
304//!     &[heap.alloc(4), heap.alloc(2), heap.alloc(1)],
305//!     &[("x", heap.alloc(8))],
306//! )?;
307//! assert_eq!(res.unpack_i32(), Some(273));
308//! # Ok(())
309//! # }
310//! # fn main(){ run().unwrap(); }
311//! ```
312//!
313//! ## Defining Rust objects that are used from Starlark
314//!
315//! Finally, we can define our own types in Rust which live in the Starlark heap.
316//! Such types are relatively complex, see the details at [`StarlarkValue`](values::StarlarkValue).
317//!
318//! ```
319//! # fn run() -> starlark::Result<()> {
320//! use std::fmt::Display;
321//! use std::fmt::Write;
322//! use std::fmt::{self};
323//!
324//! use allocative::Allocative;
325//! use starlark::environment::Globals;
326//! use starlark::environment::Module;
327//! use starlark::eval::Evaluator;
328//! use starlark::starlark_simple_value;
329//! use starlark::syntax::AstModule;
330//! use starlark::syntax::Dialect;
331//! use starlark::values::Heap;
332//! use starlark::values::NoSerialize;
333//! use starlark::values::ProvidesStaticType;
334//! use starlark::values::StarlarkValue;
335//! use starlark::values::Value;
336//! use starlark::values::ValueError;
337//! use starlark::values::ValueLike;
338//! use starlark_derive::starlark_value;
339//!
340//! // Define complex numbers
341//! #[derive(Debug, PartialEq, Eq, ProvidesStaticType, NoSerialize, Allocative)]
342//! struct Complex {
343//!     real: i32,
344//!     imaginary: i32,
345//! }
346//! starlark_simple_value!(Complex);
347//!
348//! impl Display for Complex {
349//!     fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
350//!         write!(f, "{} + {}i", self.real, self.imaginary)
351//!     }
352//! }
353//!
354//! #[starlark_value(type = "complex")]
355//! impl<'v> StarlarkValue<'v> for Complex {
356//!     // How we add them
357//!     fn add(&self, rhs: Value<'v>, heap: &'v Heap) -> Option<starlark::Result<Value<'v>>> {
358//!         if let Some(rhs) = rhs.downcast_ref::<Self>() {
359//!             Some(Ok(heap.alloc(Complex {
360//!                 real: self.real + rhs.real,
361//!                 imaginary: self.imaginary + rhs.imaginary,
362//!             })))
363//!         } else {
364//!             None
365//!         }
366//!     }
367//! }
368//!
369//! let content = "str(a + b)";
370//!
371//! let ast = AstModule::parse("complex.star", content.to_owned(), &Dialect::Standard)?;
372//! let globals = Globals::standard();
373//! let module = Module::new();
374//! // We inject some complex numbers into the module before we start.
375//! let a = module.heap().alloc(Complex {
376//!     real: 1,
377//!     imaginary: 8,
378//! });
379//! module.set("a", a);
380//! let b = module.heap().alloc(Complex {
381//!     real: 4,
382//!     imaginary: 2,
383//! });
384//! module.set("b", b);
385//! let mut eval = Evaluator::new(&module);
386//! let res = eval.eval_module(ast, &globals)?;
387//! assert_eq!(res.unpack_str(), Some("5 + 10i"));
388//! # Ok(())
389//! # }
390//! # fn main(){ run().unwrap(); }
391//! ```
