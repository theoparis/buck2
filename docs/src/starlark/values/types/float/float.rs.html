53const WRITE_PRECISION: usize = 6;
54
55fn write_non_finite<W: fmt::Write>(output: &mut W, f: f64) -> fmt::Result {
56    debug_assert!(f.is_nan() || f.is_infinite());
57    if f.is_nan() {
58        write!(output, "nan")
59    } else {
60        write!(
61            output,
62            "{}inf",
63            if f.is_sign_positive() { "+" } else { "-" }
64        )
65    }
66}
67
68pub(crate) fn write_decimal<W: fmt::Write>(output: &mut W, f: f64) -> fmt::Result {
69    if !f.is_finite() {
70        write_non_finite(output, f)
71    } else {
72        write!(output, "{f:.WRITE_PRECISION$}")
73    }
74}
75
76pub(crate) fn write_scientific<W: fmt::Write>(
77    output: &mut W,
78    f: f64,
79    exponent_char: char,
80    strip_trailing_zeros: bool,
81) -> fmt::Result {
82    if !f.is_finite() {
83        write_non_finite(output, f)
84    } else {
85        let abs = f.abs();
86        let exponent = if f == 0.0 {
87            0
88        } else {
89            abs.log10().floor() as i32
90        };
91        let normal = if f == 0.0 {
92            0.0
93        } else {
94            abs / 10f64.powf(exponent as f64)
95        };
96
97        // start with "-" for a negative number
98        if f.is_sign_negative() {
99            output.write_char('-')?
100        }
101
102        // use the whole integral part of normal (a single digit)
103        output.write_fmt(format_args!("{}", normal.trunc()))?;
104
105        // calculate the fractional tail for given precision
106        let mut tail = (normal.fract() * 10f64.powf(WRITE_PRECISION as f64)).round() as u64;
107        let mut rev_tail = [0u8; WRITE_PRECISION];
108        let mut rev_tail_len = 0;
109        let mut removing_trailing_zeros = strip_trailing_zeros;
110        for _ in 0..WRITE_PRECISION {
111            let tail_digit = tail % 10;
112            if tail_digit != 0 || !removing_trailing_zeros {
113                removing_trailing_zeros = false;
114                rev_tail[rev_tail_len] = tail_digit as u8;
115                rev_tail_len += 1;
116            }
117            tail /= 10;
118        }
119
120        // write fractional part
121        if rev_tail_len != 0 {
122            output.write_char('.')?;
123        }
124        for digit in rev_tail[0..rev_tail_len].iter().rev() {
125            output.write_char((b'0' + digit) as char)?;
126        }
127
128        // add exponent part
129        output.write_char(exponent_char)?;
130        output.write_fmt(format_args!("{exponent:+03}"))
131    }
132}

134pub(crate) fn write_compact<W: fmt::Write>(
135    output: &mut W,
136    f: f64,
137    exponent_char: char,
138) -> fmt::Result {
139    if !f.is_finite() {
140        write_non_finite(output, f)
141    } else {
142        let abs = f.abs();
143        let exponent = if f == 0.0 {
144            0
145        } else {
146            abs.log10().floor() as i32
147        };
148
149        if exponent.abs() >= WRITE_PRECISION as i32 {
150            // use scientific notation if exponent is outside of our precision (but strip 0s)
151            write_scientific(output, f, exponent_char, true)
152        } else if f.fract() == 0.0 {
153            // make sure there's a fractional part even if the number doesn't have it
154            output.write_fmt(format_args!("{f:.1}"))
155        } else {
156            // rely on the built-in formatting otherwise
157            output.write_fmt(format_args!("{f}"))
158        }
159    }
160}

162/// Runtime representation of Starlark `float` type.
163#[derive(Clone, Dupe, Copy, Debug, ProvidesStaticType, Serialize, Allocative)]
164#[serde(transparent)]
165pub struct StarlarkFloat(pub f64);
166
167impl StarlarkFloat {
168    /// The result of calling `type()` on floats.
169    pub const TYPE: &'static str = "float";
170
171    pub(crate) fn compare_impl(a: f64, b: f64) -> Ordering {
172        // According to the spec (https://github.com/bazelbuild/starlark/blob/689f54426951638ef5b7c41a14d8fc48e65c5f77/spec.md#floating-point-numbers)
173        // All NaN values compare equal to each other, but greater than any non-NaN float value.
174        if let Some(ord) = a.partial_cmp(&b) {
175            ord
176        } else {
177            a.is_nan().cmp(&b.is_nan())
178        }
179    }
180
181    pub(crate) fn floor_div_impl(a: f64, b: f64) -> anyhow::Result<f64> {
182        if b == 0.0 {
183            Err(ValueError::DivisionByZero.into())
184        } else {
185            Ok((a / b).floor())
186        }
187    }
188
189    pub(crate) fn percent_impl(a: f64, b: f64) -> anyhow::Result<f64> {
190        if b == 0.0 {
191            Err(ValueError::DivisionByZero.into())
192        } else {
193            let r = a % b;
194            if r == 0.0 {
195                Ok(0.0)
196            } else {
197                Ok(if b.signum() != r.signum() { r + b } else { r })
198            }
199        }
200    }
201}

208    }

236impl<'v> UnpackValue<'v> for StarlarkFloat {
237    type Error = Infallible;

253#[starlark_value(type = StarlarkFloat::TYPE)]
254impl<'v> StarlarkValue<'v> for StarlarkFloat {
255    fn equals(&self, other: Value) -> crate::Result<bool> {
256        Ok(Some(NumRef::Float(StarlarkFloat(self.0))) == other.unpack_num())
257    }
258
259    fn collect_repr(&self, s: &mut String) {
260        write!(s, "{self}").unwrap()
261    }
262
263    fn to_bool(&self) -> bool {
264        self.0 != 0.0
265    }
266
267    fn write_hash(&self, hasher: &mut StarlarkHasher) -> crate::Result<()> {
268        hasher.write_u64(NumRef::from(self.0).get_hash_64());
269        Ok(())
270    }
271
272    fn get_hash(&self, _private: Private) -> crate::Result<StarlarkHashValue> {
273        Ok(NumRef::Float(*self).get_hash())
274    }
275
276    fn plus(&self, heap: &'v Heap) -> crate::Result<Value<'v>> {
277        Ok(heap.alloc(*self))
278    }
279
280    fn minus(&self, heap: &'v Heap) -> crate::Result<Value<'v>> {
281        Ok(heap.alloc(StarlarkFloat(-self.0)))
282    }
283
284    fn add(&self, other: Value, heap: &'v Heap) -> Option<crate::Result<Value<'v>>> {
285        Some(Ok(heap.alloc(NumRef::Float(*self) + other.unpack_num()?)))
286    }
287
288    fn sub(&self, other: Value, heap: &'v Heap) -> crate::Result<Value<'v>> {
289        match other.unpack_num() {
290            None => ValueError::unsupported_with(self, "-", other),
291            Some(other) => Ok(heap.alloc(NumRef::Float(*self) - other)),
292        }
293    }
294
295    fn mul(&self, other: Value<'v>, heap: &'v Heap) -> Option<crate::Result<Value<'v>>> {
296        Some(Ok(heap.alloc(NumRef::Float(*self) * other.unpack_num()?)))
297    }
298
299    fn div(&self, other: Value, heap: &'v Heap) -> crate::Result<Value<'v>> {
300        match other.unpack_num() {
301            None => ValueError::unsupported_with(self, "/", other),
302            Some(other) => Ok(heap.alloc(NumRef::Float(*self).div(other)?)),
303        }
304    }
305
306    fn percent(&self, other: Value, heap: &'v Heap) -> crate::Result<Value<'v>> {
307        match other.unpack_num() {
308            Some(other) => Ok(heap.alloc(NumRef::Float(*self).percent(other)?)),
309            None => ValueError::unsupported_with(self, "%", other),
310        }
311    }
312
313    fn floor_div(&self, other: Value, heap: &'v Heap) -> crate::Result<Value<'v>> {
314        match other.unpack_num() {
315            None => ValueError::unsupported_with(self, "//", other),
316            Some(other) => Ok(heap.alloc(NumRef::Float(*self).floor_div(other)?)),
317        }
318    }
319
320    fn bin_op_ty(op: TypingBinOp, rhs: &TyBasic) -> Option<Ty> {
321        typecheck_num_bin_op(NumTy::Float, op, rhs)
322    }
323
324    fn compare(&self, other: Value) -> crate::Result<Ordering> {
325        match other.unpack_num() {
326            None => ValueError::unsupported_with(self, "compare", other),
327            Some(other) => Ok(NumRef::Float(*self).cmp(&other)),
328        }
329    }
330}
