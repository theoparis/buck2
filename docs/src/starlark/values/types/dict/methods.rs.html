38#[starlark_module]
39pub(crate) fn dict_methods(registry: &mut MethodsBuilder) {
40    /// [dict.clear](
41    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#dict·clear
42    /// ): clear a dictionary
43    ///
44    /// `D.clear()` removes all the entries of dictionary D and returns `None`.
45    /// It fails if the dictionary is frozen or if there are active iterators.
46    ///
47    /// ```
48    /// # starlark::assert::is_true(r#"
49    /// x = {"one": 1, "two": 2}
50    /// x.clear()
51    /// x == {}
52    /// # "#);
53    /// ```
54    fn clear(this: Value) -> anyhow::Result<NoneType> {
55        let mut this = DictMut::from_value(this)?;
56        this.aref.clear();
57        Ok(NoneType)
58    }
59
60    /// [dict.get](
61    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#dict·get
62    /// ): return an element from the dictionary.
63    ///
64    /// `D.get(key[, default])` returns the dictionary value corresponding to
65    /// the given key. If the dictionary contains no such value, `get`
66    /// returns `None`, or the value of the optional `default` parameter if
67    /// present.
68    ///
69    /// `get` fails if `key` is unhashable.
70    ///
71    /// ```
72    /// # starlark::assert::is_true(r#"
73    /// x = {"one": 1, "two": 2}
74    /// # (
75    /// x.get("one") == 1
76    /// # and
77    /// x.get("three") == None
78    /// # and
79    /// x.get("three", 0) == 0
80    /// # )"#);
81    /// ```
82    #[starlark(speculative_exec_safe)]
83    fn get<'v>(
84        this: DictRef<'v>,
85        #[starlark(require = pos)] key: Value<'v>,
86        #[starlark(require = pos)] default: Option<Value<'v>>,
87    ) -> starlark::Result<Value<'v>> {
88        match this.get(key)? {
89            None => Ok(default.unwrap_or_else(Value::new_none)),
90            Some(x) => Ok(x),
91        }
92    }
93
94    /// [dict.items](
95    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#dict·items
96    /// ): get list of (key, value) pairs.
97    ///
98    /// `D.items()` returns a new list of key/value pairs, one per element in
99    /// dictionary D, in the same order as they would be returned by a `for`
100    /// loop.
101    ///
102    /// ```
103    /// # starlark::assert::is_true(r#"
104    /// x = {"one": 1, "two": 2}
105    /// x.items() == [("one", 1), ("two", 2)]
106    /// # "#);
107    /// ```
108    fn items<'v>(
109        this: DictRef<'v>,
110        heap: &'v Heap,
111    ) -> anyhow::Result<ValueOfUnchecked<'v, UnpackList<(Value<'v>, Value<'v>)>>> {
112        Ok(heap.alloc_typed_unchecked(AllocList(this.iter())).cast())
113    }
114
115    /// [dict.keys](
116    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#dict·keys
117    /// ): get the list of keys of the dictionary.
118    ///
119    /// `D.keys()` returns a new list containing the keys of dictionary D, in
120    /// the same order as they would be returned by a `for` loop.
121    ///
122    /// ```
123    /// # starlark::assert::is_true(r#"
124    /// x = {"one": 1, "two": 2}
125    /// x.keys() == ["one", "two"]
126    /// # "#);
127    /// ```
128    #[starlark(speculative_exec_safe)]
129    fn keys<'v>(
130        this: DictRef<'v>,
131        heap: &'v Heap,
132    ) -> anyhow::Result<ValueOfUnchecked<'v, &'v ListRef<'v>>> {
133        Ok(ValueOfUnchecked::new(heap.alloc(AllocList(this.keys()))))
134    }
135
136    /// [dict.pop](
137    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#dict·pop
138    /// ): return an element and remove it from a dictionary.
139    ///
140    /// `D.pop(key[, default])` returns the value corresponding to the specified
141    /// key, and removes it from the dictionary.  If the dictionary contains no
142    /// such value, and the optional `default` parameter is present, `pop`
143    /// returns that value; otherwise, it fails.
144    ///
145    /// `pop` fails if `key` is unhashable, or the dictionary is frozen or has
146    /// active iterators.
147    ///
148    /// ```
149    /// # starlark::assert::is_true(r#"
150    /// x = {"one": 1, "two": 2}
151    /// # (
152    /// x.pop("one") == 1
153    /// # and
154    /// x == {"two": 2}
155    /// # and
156    /// x.pop("three", 0) == 0
157    /// # and
158    /// x.pop("three", None) == None
159    /// # )"#);
160    /// ```
161    ///
162    /// Failure:
163    ///
164    /// ```
165    /// # starlark::assert::fail(r#"
166    /// {'one': 1}.pop('four')   # error: not found
167    /// # "#, "not found");
168    /// ```
169    fn pop<'v>(
170        this: Value<'v>,
171        #[starlark(require = pos)] key: Value<'v>,
172        #[starlark(require = pos)] default: Option<Value<'v>>,
173    ) -> starlark::Result<Value<'v>> {
174        let mut me = DictMut::from_value(this)?;
175        match me.aref.remove_hashed(key.get_hashed()?) {
176            Some(x) => Ok(x),
177            None => match default {
178                Some(v) => Ok(v),
179                None => {
180                    mem::drop(me);
181                    Err(anyhow::anyhow!(
182                        "Key `{}` not found in dictionary `{}`",
183                        key.to_repr(),
184                        this.to_repr()
185                    )
186                    .into())
187                }
188            },
189        }
190    }
191
192    /// [dict.popitem](
193    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#dict·popitem
194    /// ): returns and removes the first key/value pair of a dictionary.
195    ///
196    /// `D.popitem()` returns the first key/value pair, removing it from the
197    /// dictionary.
198    ///
199    /// `popitem` fails if the dictionary is empty, frozen, or has active
200    /// iterators.
201    ///
202    /// ```
203    /// # starlark::assert::is_true(r#"
204    /// x = {"one": 1, "two": 2}
205    /// # (
206    /// x.popitem() == ("one", 1)
207    /// # and
208    /// x.popitem() == ("two", 2)
209    /// # and
210    /// x == {}
211    /// # )"#);
212    /// ```
213    ///
214    /// Failure:
215    ///
216    /// ```
217    /// # starlark::assert::fail(r#"
218    /// {}.popitem()   # error: empty dict
219    /// # "#, "empty dict");
220    /// ```
221    fn popitem<'v>(this: Value<'v>) -> anyhow::Result<(Value<'v>, Value<'v>)> {
222        let mut this = DictMut::from_value(this)?;
223
224        // TODO(nga): this implementation is O(N).
225        //   https://github.com/bazelbuild/starlark/issues/286
226
227        match this.aref.content.shift_remove_index(0) {
228            Some((k, v)) => Ok((k, v)),
229            None => Err(anyhow::anyhow!("Cannot .popitem() on an empty dictionary")),
230        }
231    }
232
233    /// [dict.setdefault](
234    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#dict·setdefault
235    /// ): get a value from a dictionary, setting it to a new value if not
236    /// present.
237    ///
238    /// `D.setdefault(key[, default])` returns the dictionary value
239    /// corresponding to the given key. If the dictionary contains no such
240    /// value, `setdefault`, like `get`, returns `None` or the value of the
241    /// optional `default` parameter if present; `setdefault` additionally
242    /// inserts the new key/value entry into the dictionary.
243    ///
244    /// `setdefault` fails if the key is unhashable or if the dictionary is
245    /// frozen.
246    ///
247    /// ```
248    /// # starlark::assert::is_true(r#"
249    /// x = {"one": 1, "two": 2}
250    /// # (
251    /// x.setdefault("one") == 1
252    /// # and
253    /// x.setdefault("three", 0) == 0
254    /// # and
255    /// x == {"one": 1, "two": 2, "three": 0}
256    /// # and
257    /// x.setdefault("four") == None
258    /// # and
259    /// x == {"one": 1, "two": 2, "three": 0, "four": None}
260    /// # )"#)
261    /// ```
262    fn setdefault<'v>(
263        this: Value<'v>,
264        #[starlark(require = pos)] key: Value<'v>,
265        #[starlark(require = pos)] default: Option<Value<'v>>,
266    ) -> starlark::Result<Value<'v>> {
267        let mut this = DictMut::from_value(this)?;
268        let key = key.get_hashed()?;
269        match this.aref.content.entry_hashed(key) {
270            starlark_map::small_map::Entry::Occupied(e) => Ok(*e.get()),
271            starlark_map::small_map::Entry::Vacant(e) => {
272                let default = default.unwrap_or_else(Value::new_none);
273                e.insert(default);
274                Ok(default)
275            }
276        }
277    }

279    /// [dict.update](
280    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#dict·update
281    /// ): update values in the dictionary.
282    ///
283    /// `D.update([pairs][, name=value[, ...])` makes a sequence of key/value
284    /// insertions into dictionary D, then returns `None.`
285    ///
286    /// If the positional argument `pairs` is present, it must be
287    /// another `dict`, or some other iterable.
288    /// If it is another `dict`, then its key/value pairs are inserted into D.
289    /// If it is an iterable, it must provide a sequence of pairs (or other
290    /// iterables of length 2), each of which is treated as a key/value pair
291    /// to be inserted into D.
292    ///
293    /// For each `name=value` argument present, the name is converted to a
294    /// string and used as the key for an insertion into D, with its
295    /// corresponding value being `value`.
296    ///
297    /// `update` fails if the dictionary is frozen.
298    ///
299    /// ```
300    /// # starlark::assert::is_true(r#"
301    /// x = {}
302    /// x.update([("a", 1), ("b", 2)], c=3)
303    /// x.update({"d": 4})
304    /// x.update(e=5)
305    /// x == {"a": 1, "b": 2, "c": 3, "d": 4, "e": 5}
306    /// # "#);
307    /// ```
308    fn update<'v>(
309        this: Value<'v>,
310        #[starlark(require = pos)] pairs: Option<
311            ValueOfUnchecked<'v, Either<DictRef<'v>, StarlarkIter<(Value<'v>, Value<'v>)>>>,
312        >,
313        #[starlark(kwargs)] kwargs: DictRef<'v>,
314        heap: &'v Heap,
315    ) -> starlark::Result<NoneType> {
316        let pairs = if pairs.map(|x| x.get().ptr_eq(this)) == Some(true) {
317            // someone has done `x.update(x)` - that isn't illegal, but we will have issues
318            // with trying to iterate over x while holding x for mutation, and it doesn't do
319            // anything useful, so just change pairs back to None
320            None
321        } else {
322            pairs.map(|x| x.get())
323        };
324
325        let mut this = DictMut::from_value(this)?;
326        if let Some(pairs) = pairs {
327            match DictRef::from_value(pairs) {
328                Some(dict) => {
329                    for (k, v) in dict.iter_hashed() {
330                        this.aref.insert_hashed(k, v);
331                    }
332                }
333                _ => {
334                    for v in pairs.iterate(heap)? {
335                        let mut it = v.iterate(heap)?;
336                        // `StarlarkIterator` is fused.
337                        let (Some(k), Some(v), None) = (it.next(), it.next(), it.next()) else {
338                            return Err(anyhow::anyhow!(
339                            "dict.update expect a list of pairs or a dictionary as first argument, got a list of non-pairs.",
340                        ).into());
341                        };
342                        this.aref.insert_hashed(k.get_hashed()?, v);
343                    }
344                }
345            }
346        }
347
348        for (k, v) in kwargs.iter_hashed() {
349            this.aref.insert_hashed(k, v);
350        }
351        Ok(NoneType)
352    }
353
354    /// [dict.values](
355    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#dict·values
356    /// ): get the list of values of the dictionary.
357    ///
358    /// `D.values()` returns a new list containing the dictionary's values, in
359    /// the same order as they would be returned by a `for` loop over the
360    /// dictionary.
361    ///
362    /// ```
363    /// # starlark::assert::is_true(r#"
364    /// x = {"one": 1, "two": 2}
365    /// x.values() == [1, 2]
366    /// # "#);
367    /// ```
368    #[starlark(speculative_exec_safe)]
369    fn values<'v>(
370        this: DictRef<'v>,
371        heap: &'v Heap,
372    ) -> anyhow::Result<ValueOfUnchecked<'v, &'v ListRef<'v>>> {
373        Ok(ValueOfUnchecked::new(heap.alloc_list_iter(this.values())))
374    }
375}

377#[cfg(test)]
378mod tests {
379    use crate::assert;
380    use crate::assert::Assert;
381
382    #[test]
383    fn test_error_codes() {
384        assert::fail(r#"x = {"one": 1}; x.pop("four")"#, "not found");
385        assert::fail("x = {}; x.popitem()", "empty");
386    }
387
388    #[test]
389    fn test_dict_add() {
390        assert::fail("{1: 2} + {3: 4}", "not supported");
391    }
392
393    #[test]
394    fn test_dict_with_duplicates() {
395        // In Starlark spec this is a runtime error. In Python it's fine.
396        // We make it a runtime error, plus have a lint that checks for it statically.
397        assert::fails("{40+2: 2, 6*7: 3}", &["key repeated", "42"]);
398        // Also check we fail if the entire dictionary is static (a different code path).
399        assert::fails("{42: 2, 42: 3}", &["key repeated", "42"]);
400    }
401
402    #[test]
403    fn test_dict_update_with_self_pos() {
404        assert::eq("{3: 4, 1: 2}", "d = {3: 4, 1: 2}; d.update(d); d");
405    }
406
407    #[test]
408    fn test_dict_update_with_self_as_kwargs() {
409        assert::eq("{'a': 1, 'b': 2}", "d = {'a': 1, 'b': 2}; d.update(**d); d");
410    }
411
412    #[test]
413    fn test_frozen_dict_cannot_be_updated_with_self_pos() {
414        let mut a = Assert::new();
415        a.module("d.star", "D = {7: 8, 9: 0}");
416        a.fail(
417            r#"
418load('d.star', 'D')
419
420D.update(D)
421"#,
422            "Immutable",
423        );
424    }
425
426    #[test]
427    fn test_frozen_dict_cannot_be_updated_with_self_as_kwargs() {
428        let mut a = Assert::new();
429        a.module("d.star", "D = {'x': 17, 'y': 19}");
430        a.fail(
431            r#"
432load('d.star', 'D')
433D.update(**D)
434"#,
435            "Immutable",
436        );
437    }
