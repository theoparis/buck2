70#[derive(thiserror::Error, Debug)]
71enum EnumError {
72    #[error("enum values must all be distinct, but repeated `{0}`")]
73    DuplicateEnumValue(String),
74    #[error("Unknown enum element `{0}`, given to `{1}`")]
75    InvalidElement(String, String),
76}

78#[doc(hidden)]
79pub trait EnumCell: Freeze {
80    type TyEnumDataOpt: Debug;
81
82    fn get_or_init_ty(
83        ty: &Self::TyEnumDataOpt,
84        f: impl FnOnce() -> crate::Result<Arc<TyEnumData>>,
85    ) -> crate::Result<()>;
86    fn get_ty(ty: &Self::TyEnumDataOpt) -> Option<&Arc<TyEnumData>>;
87}
88
89impl<'v> EnumCell for Value<'v> {
90    type TyEnumDataOpt = OnceCell<Arc<TyEnumData>>;
91
92    fn get_or_init_ty(
93        ty: &Self::TyEnumDataOpt,
94        f: impl FnOnce() -> crate::Result<Arc<TyEnumData>>,
95    ) -> crate::Result<()> {
96        ty.get_or_try_init(f)?;
97        Ok(())
98    }
99
100    fn get_ty(ty: &Self::TyEnumDataOpt) -> Option<&Arc<TyEnumData>> {
101        ty.get()
102    }
103}
104
105impl EnumCell for FrozenValue {
106    type TyEnumDataOpt = Option<Arc<TyEnumData>>;
107
108    fn get_or_init_ty(
109        ty: &Self::TyEnumDataOpt,
110        f: impl FnOnce() -> crate::Result<Arc<TyEnumData>>,
111    ) -> crate::Result<()> {
112        let _ignore = (ty, f);
113        Ok(())
114    }
115
116    fn get_ty(ty: &Self::TyEnumDataOpt) -> Option<&Arc<TyEnumData>> {
117        ty.as_ref()
118    }
119}

121/// The type of an enumeration, created by `enum()`.
122#[derive(Debug, Trace, Coerce, NoSerialize, ProvidesStaticType, Allocative)]
123#[repr(C)]
124// Deliberately store fully populated values
125// for each entry, so we can produce enum values with zero allocation.
126pub struct EnumTypeGen<V: EnumCell> {
127    pub(crate) id: TypeInstanceId,
128    #[allocative(skip)] // TODO(nga): do not skip.
129    // TODO(nga): teach derive to do something like `#[trace(static)]`.
130    #[trace(unsafe_ignore)]
131    pub(crate) ty_enum_data: V::TyEnumDataOpt,
132    // The key is the value of the enumeration
133    // The value is a value of type EnumValue
134    #[allocative(skip)] // TODO(nga): do not skip.
135    elements: UnsafeCell<SmallMap<V, V>>,
136}
137
138impl<'v> Freeze for EnumTypeGen<Value<'v>> {
139    type Frozen = EnumTypeGen<FrozenValue>;
140
141    fn freeze(self, freezer: &Freezer) -> FreezeResult<Self::Frozen> {
142        let EnumTypeGen {
143            id,
144            ty_enum_data: ty_enum_type,
145            elements,
146        } = self;
147        let ty_enum_type = ty_enum_type.into_inner();
148        let elements = elements.freeze(freezer)?;
149        Ok(EnumTypeGen {
150            id,
151            ty_enum_data: ty_enum_type,
152            elements,
153        })
154    }
155}
156
157unsafe impl<V: EnumCell + Freeze> Send for EnumTypeGen<V> {}
158unsafe impl<V: EnumCell + Freeze> Sync for EnumTypeGen<V> {}
159
160impl<'v, V: EnumCell + ValueLike<'v>> Display for EnumTypeGen<V> {
161    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
162        fmt_container(f, "enum(", ")", self.elements().iter().map(|(k, _v)| k))
163    }
164}

166/// Unfrozen enum type.
167pub type EnumType<'v> = EnumTypeGen<Value<'v>>;
168/// Frozen enum type.
169pub type FrozenEnumType = EnumTypeGen<FrozenValue>;
170
171impl<'v> EnumType<'v> {
172    pub(crate) fn new(
173        elements: Vec<StringValue<'v>>,
174        heap: &'v Heap,
175    ) -> crate::Result<ValueTyped<'v, EnumType<'v>>> {
176        // We are constructing the enum and all elements in one go.
177        // They both point at each other, which adds to the complexity.
178        let id = TypeInstanceId::r#gen();
179        let typ = heap.alloc_typed(EnumType {
180            id,
181            ty_enum_data: OnceCell::new(),
182            elements: UnsafeCell::new(SmallMap::new()),
183        });
184
185        let mut res = SmallMap::with_capacity(elements.len());
186        for (i, x) in elements.iter().enumerate() {
187            let v = heap.alloc(EnumValue {
188                id,
189                typ: typ.to_value(),
190                index: i as i32,
191                value: x.to_value(),
192            });
193            if res.insert_hashed(x.to_value().get_hashed()?, v).is_some() {
194                return Err(crate::Error::new_other(EnumError::DuplicateEnumValue(
195                    x.to_string(),
196                )));
197            }
198        }
199
200        // Here we tie the cycle
201        unsafe {
202            // SAFETY: we own unique reference to `t`.
203            *typ.elements.get() = res;
204        }
205        Ok(typ)
206    }
207}
208
209impl<V: EnumCell + Freeze> EnumTypeGen<V> {
210    pub(crate) fn elements(&self) -> &SmallMap<V, V> {
211        // Safe because we never mutate the elements after construction.
212        unsafe { &*self.elements.get() }
213    }
214}
215
216impl<'v, V> EnumTypeGen<V>
217where
218    Value<'v>: Equivalent<V>,
219    V: ValueLike<'v> + EnumCell,
220{
221    pub(crate) fn ty_enum_data(&self) -> Option<&Arc<TyEnumData>> {
222        V::get_ty(&self.ty_enum_data)
223    }
224
225    pub(crate) fn construct(&self, val: Value<'v>) -> crate::Result<V> {
226        match self.elements().get_hashed_by_value(val.get_hashed()?) {
227            Some(v) => Ok(*v),
228            None => Err(crate::Error::new_other(EnumError::InvalidElement(
229                val.to_str(),
230                self.to_string(),
231            ))),
232        }
233    }
234}

236#[starlark_value(type = FUNCTION_TYPE)]
237impl<'v, V> StarlarkValue<'v> for EnumTypeGen<V>
238where
239    Self: ProvidesStaticType<'v>,
240    Value<'v>: Equivalent<V>,
241    V: ValueLike<'v> + EnumCell,
242{
243    type Canonical = FrozenEnumType;
244
245    // TODO(nga): replace `Color("RED")` with `Color.RED`.
246    //   https://www.internalfb.com/tasks/?t=183515013
247    fn invoke(
248        &self,
249        _me: Value<'v>,
250        args: &Arguments<'v, '_>,
251        eval: &mut Evaluator<'v, '_, '_>,
252    ) -> crate::Result<Value<'v>> {
253        args.no_named_args()?;
254        let val = args.positional1(eval.heap())?;
255        Ok(self.construct(val)?.to_value())
256    }
257
258    fn length(&self) -> crate::Result<i32> {
259        Ok(self.elements().len() as i32)
260    }
261
262    fn at(&self, index: Value, _heap: &'v Heap) -> crate::Result<Value<'v>> {
263        let i = convert_index(index, self.elements().len() as i32)? as usize;
264        // Must be in the valid range since convert_index checks that, so just unwrap
265        Ok(self
266            .elements()
267            .get_index(i)
268            .map(|x| *x.1)
269            .unwrap()
270            .to_value())
271    }
272
273    unsafe fn iterate(&self, me: Value<'v>, _heap: &'v Heap) -> crate::Result<Value<'v>> {
274        Ok(me)
275    }
276
277    unsafe fn iter_size_hint(&self, index: usize) -> (usize, Option<usize>) {
278        debug_assert!(index <= self.elements().len());
279        let rem = self.elements().len() - index;
280        (rem, Some(rem))
281    }
282
283    unsafe fn iter_next(&self, index: usize, _heap: &'v Heap) -> Option<Value<'v>> {
284        self.elements().values().nth(index).map(|v| v.to_value())
285    }
286
287    unsafe fn iter_stop(&self) {}
288
289    fn get_methods() -> Option<&'static Methods> {
290        static RES: MethodsStatic = MethodsStatic::new();
291        RES.methods(enum_type_methods)
292    }
293
294    fn eval_type(&self) -> Option<Ty> {
295        self.ty_enum_data().map(|t| t.ty_enum_value.dupe())
296    }
297
298    fn typechecker_ty(&self) -> Option<Ty> {
299        self.ty_enum_data().map(|t| t.ty_enum_type.dupe())
300    }
301
302    fn export_as(
303        &self,
304        variable_name: &str,
305        _eval: &mut Evaluator<'v, '_, '_>,
306    ) -> crate::Result<()> {
307        V::get_or_init_ty(&self.ty_enum_data, || {
308            let ty_enum_value = Ty::custom(TyUser::new(
309                variable_name.to_owned(),
310                TyStarlarkValue::new::<EnumValue>(),
311                self.id,
312                TyUserParams {
313                    matcher: Some(TypeMatcherFactory::new(EnumTypeMatcher { id: self.id })),
314                    ..TyUserParams::default()
315                },
316            )?);
317            let ty_enum_type = Ty::custom(TyUser::new(
318                format!("enum[{variable_name}]"),
319                TyStarlarkValue::new::<EnumType>(),
320                TypeInstanceId::r#gen(),
321                TyUserParams {
322                    index: Some(TyUserIndex {
323                        index: Ty::int(),
324                        result: ty_enum_value.dupe(),
325                    }),
326                    iter_item: Some(ty_enum_value.dupe()),
327                    callable: Some(TyCallable::new(
328                        ParamSpec::pos_only(
329                            [
330                                // TODO(nga): we can do better parameter type.
331                                Ty::any(),
332                            ],
333                            [],
334                        ),
335                        ty_enum_value.dupe(),
336                    )),
337                    ..TyUserParams::default()
338                },
339            )?);
340            Ok(Arc::new(TyEnumData {
341                name: variable_name.to_owned(),
342                id: self.id,
343                ty_enum_value,
344                ty_enum_type,
345            }))
346        })
347    }
348}

350#[starlark_module]
351fn enum_type_methods(builder: &mut MethodsBuilder) {
352    #[starlark(attribute)]
353    fn r#type<'v>(this: Value, heap: &Heap) -> starlark::Result<Value<'v>> {
354        let this = EnumType::from_value(this).unwrap();
355        let ty_enum_type = match this {
356            Either::Left(x) => x.ty_enum_data(),
357            Either::Right(x) => x.ty_enum_data(),
358        };
359        match ty_enum_type {
360            Some(ty_enum_type) => Ok(heap.alloc(ty_enum_type.name.as_str())),
361            None => Ok(heap.alloc(EnumValue::TYPE)),
362        }
363    }
364
365    fn values<'v>(this: Value<'v>) -> anyhow::Result<AllocList<impl Iterator<Item = Value<'v>>>> {
366        let this = EnumType::from_value(this).unwrap();
367        match this {
368            Either::Left(x) => Ok(AllocList(Either::Left(x.elements().keys().copied()))),
369            Either::Right(x) => Ok(AllocList(Either::Right(
370                x.elements().keys().map(|x| x.to_value()),
371            ))),
372        }
373    }
374}
