18//! Implementation of `enum` function.
19use starlark_derive::starlark_module;
20
21use crate as starlark;
22use crate::environment::GlobalsBuilder;
23use crate::values::Heap;
24use crate::values::StringValue;
25use crate::values::Value;
26use crate::values::enumeration::EnumType;
27use crate::values::tuple::UnpackTuple;
28
29#[starlark_module]
30pub fn register_enum(builder: &mut GlobalsBuilder) {
31    /// The `enum` type represents one value picked from a set of values.
32    ///
33    /// For example:
34    ///
35    /// ```python
36    /// MyEnum = enum("option1", "option2", "option3")
37    /// ```
38    ///
39    /// This statement defines an enumeration `MyEnum` that consists of the three values `"option1"`, `"option2"` and `option3`.
40    ///
41    /// Now `MyEnum` is defined, it's possible to do the following:
42    ///
43    /// * Create values of this type with `MyEnum("option2")`. It is a runtime error if the argument is not one of the predeclared values of the enumeration.
44    /// * Get the type of the enum suitable for a type annotation with `MyEnum`.
45    /// * Given a value of the enum (for example, `v = MyEnum("option2")`), get the underlying value `v.value == "option2"` or the index in the enumeration `v.index == 1`.
46    /// * Get a list of the values that make up the array with `MyEnum.values() == ["option1", "option2", "option3"]`.
47    /// * Treat `MyEnum` a bit like an array, with `len(MyEnum) == 3`, `MyEnum[1] == MyEnum("option2")` and iteration over enums `[x.value for x in MyEnum] == ["option1", "option2", "option3"]`.
48    ///
49    /// Enumeration types store each value once, which are then efficiently referenced by enumeration values.
50    fn r#enum<'v>(
51        #[starlark(args)] args: UnpackTuple<StringValue<'v>>,
52        heap: &'v Heap,
53    ) -> starlark::Result<Value<'v>> {
54        // Every Value must either be a field or a value (the type)
55        Ok(EnumType::new(args.items, heap)?.to_value())
56    }
57}

59#[cfg(test)]
60mod tests {
61    use crate::assert;
62    use crate::assert::Assert;
63
64    #[test]
65    fn test_enum() {
66        assert::pass(
67            r#"
68enum_type = enum("option1", "option2", "option3")
69x = enum_type("option1")
70assert_eq(x.value, "option1")
71assert_eq(enum_type("option3").value, "option3")
72assert_eq(enum_type.type, "enum_type")
73"#,
74        );
75        assert::fails(
76            r#"
77enum_type = enum("option1", "option2", "option3")
78enum_type(False)"#,
79            &["Unknown enum element", "`False`", "option1"],
80        );
81        assert::fails(
82            r#"
83enum_type = enum("option1", "option2", "option3")
84enum_type("option4")"#,
85            &["Unknown enum element", "`option4`"],
86        );
87        assert::fails(
88            r#"
89enum_type = enum("option1", "option1")
90enum_type("option3")"#,
91            &["distinct", "option1"],
92        );
93        assert::pass(
94            r#"
95enum_type = enum("option1","option2")
96def foo(x: enum_type) -> enum_type:
97    return x
98foo(enum_type("option1"))"#,
99        );
100        assert::pass(
101            r#"
102v = [enum("option1","option2")]
103v_0 = v[0]
104def foo(y: v_0) -> v_0:
105    # TODO(nga): fails at compile time.
106    return noop(y)
107foo(v[0]("option1"))"#,
108        );
109        assert::pass(
110            r#"
111enum_type = enum("option1","option2")
112assert_eq(enum_type.values(), ["option1","option2"])
113assert_eq([enum_type[i].value for i in range(len(enum_type))], ["option1","option2"])
114assert_eq(enum_type("option2").index, 1)
115assert_eq([x.value for x in enum_type], ["option1","option2"])"#,
116        );
117        assert::pass(
118            r#"
119enum_type = enum("option1","option2")
120x = enum_type("option1")
121assert_eq(str(enum_type), "enum(\"option1\", \"option2\")")
122assert_eq(str(x), "enum_type(\"option1\")")
123"#,
124        );
125        assert::pass(
126            r#"
127enum_type = enum("option1","option2")
128repr(enum_type) # Check it is finite
129"#,
130        );
131    }

133    #[test]
134    fn test_enum_equality() {
135        assert::pass(
136            r#"
137enum_type = enum("option1", "option2", "option3")
138assert_eq(enum_type("option1"), enum_type("option1"))
139assert_eq(enum_type("option3"), enum_type("option3"))
140assert_ne(enum_type("option1"), enum_type("option3"))
141"#,
142        );
143
144        let mut a = Assert::new();
145        a.module(
146            "m",
147            r#"
148enum_type = enum("option1", "option2", "option3")
149enum_val = enum_type("option1")
150"#,
151        );
152        a.pass(
153            r#"
154load('m', 'enum_type', 'enum_val')
155assert_eq(enum_val, enum_type("option1"))
156assert_ne(enum_val, enum_type("option3"))
157"#,
158        );
159
160        a = Assert::new();
161        a.module(
162            "m1",
163            r#"
164rt = enum("one")
165"#,
166        );
167        a.module(
168            "m2",
169            r#"
170rt = enum("one", "two")
171"#,
172        );
173        a.pass(
174            r#"
175load('m1', r1='rt')
176load('m2', r2='rt')
177rt = enum("one")
178diff = enum("one")
179assert_ne(r1("one"), rt("one"))
180assert_ne(rt("one"), r2("one"))
181assert_ne(rt("one"), diff("one"))
182"#,
183        );
184    }

193"#,
