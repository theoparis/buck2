18//! Utility to heap allocate arrays of values.
19
20use std::fmt;
21use std::fmt::Debug;
22use std::mem;
23use std::ptr;
24
25use allocative::Allocative;
26use starlark_derive::NoSerialize;
27use starlark_derive::starlark_value;
28
29use crate as starlark;
30use crate::any::ProvidesStaticType;
31use crate::values::StarlarkValue;
32
33#[derive(derive_more::Display, ProvidesStaticType, NoSerialize, Allocative)]
34#[repr(C)]
35#[display("{:?}", self)]
36#[allocative(bound = "")]
37pub(crate) struct AnyArray<T: Debug + 'static> {
38    pub(crate) len: usize,
39    #[allocative(skip)] // TODO(nga): do not skip.
40    content: [T; 0],
41}
42
43impl<T: Debug + 'static> AnyArray<T> {
44    /// This function is unsafe because it does not initialize content array,
45    /// but drops in in destructor.
46    pub(crate) unsafe fn new(len: usize) -> AnyArray<T> {
47        AnyArray { len, content: [] }
48    }
49
50    fn as_slice(&self) -> &[T] {
51        unsafe { std::slice::from_raw_parts(self.content.as_ptr(), self.len) }
52    }
53
54    pub(crate) fn offset_of_content() -> usize {
55        memoffset::offset_of!(Self, content)
56    }
57}
58
59impl<T: Debug + 'static> Debug for AnyArray<T> {
60    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
61        f.debug_tuple("AnyArray").field(&self.as_slice()).finish()
62    }
63}
64
65// Currently we always allocate in `drop` heap, which is a bit inefficient.
66// But we don't allocate much.
67impl<T: Debug + 'static> Drop for AnyArray<T> {
68    fn drop(&mut self) {
69        unsafe {
70            ptr::drop_in_place(ptr::slice_from_raw_parts_mut(
71                self.content.as_mut_ptr(),
72                self.len,
73            ));
74        }
75    }
76}
77
78// We rely on allocation of `AnyArray` in correct heap (drop or non-drop).
79// This struct has zero length array of `T`, so check it actually declares it has drop.
80const _: () = assert!(mem::needs_drop::<AnyArray<String>>());
81
82#[starlark_value(type = "AnyArray")]
83impl<'v, T: Debug + 'static> StarlarkValue<'v> for AnyArray<T> {
84    type Canonical = Self;
85}

87#[cfg(test)]
88mod tests {
89    use std::sync::Arc;
90    use std::sync::atomic::AtomicU32;
91    use std::sync::atomic::Ordering;
92
93    use dupe::Dupe;
94
95    use crate::values::FrozenHeap;
96
97    #[test]
98    fn test_drop() {
99        let counter1 = Arc::new(AtomicU32::new(0));
100        let counter2 = Arc::new(AtomicU32::new(0));
101
102        #[derive(Debug, Clone, Dupe)]
103        struct IncrementOnDrop(Arc<AtomicU32>);
104
105        impl Drop for IncrementOnDrop {
106            fn drop(&mut self) {
107                self.0.fetch_add(1, Ordering::SeqCst);
108            }
109        }
110
111        let heap = FrozenHeap::new();
112        let values = heap.alloc_any_slice(&[
113            IncrementOnDrop(counter1.dupe()),
114            IncrementOnDrop(counter1.dupe()),
115            IncrementOnDrop(counter2.dupe()),
116            IncrementOnDrop(counter1.dupe()),
117            IncrementOnDrop(counter2.dupe()),
118        ]);
119
120        assert_eq!(5, values.len());
121
122        assert!(Arc::ptr_eq(&counter1, &values[0].0));
123        assert!(Arc::ptr_eq(&counter1, &values[1].0));
124        assert!(Arc::ptr_eq(&counter2, &values[2].0));
125        assert!(Arc::ptr_eq(&counter1, &values[3].0));
126        assert!(Arc::ptr_eq(&counter2, &values[4].0));
127
128        // First drop happens when we clone values for the allocation.
129        assert_eq!(3, counter1.load(Ordering::SeqCst));
130        assert_eq!(2, counter2.load(Ordering::SeqCst));
131
132        drop(heap);
133
134        assert_eq!(6, counter1.load(Ordering::SeqCst));
135        assert_eq!(4, counter2.load(Ordering::SeqCst));
136    }
137
138    #[test]
139    fn test_allocation_size() {
140        let heap = FrozenHeap::new();
141        heap.alloc_any_slice(&[1, 2, 3]);
142        let quake = heap.alloc_str("quake");
143        // Test array allocation did not overwrite the string.
144        assert_eq!(quake.as_str(), "quake");
145    }
146}
