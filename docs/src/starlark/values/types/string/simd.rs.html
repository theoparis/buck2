18/// Fixed length byte vector API.
19#[allow(dead_code)] // FIXME: Investigate if this is really needed, fails on Mac
20pub(crate) trait Vector: Copy {
21    /// Fill the vector with given byte value.
22    unsafe fn splat(byte: u8) -> Self;
23    /// Load the vector from given memory address.
24    unsafe fn load_unaligned(ptr: *const u8) -> Self;
25    /// Store the vector to given memory address.
26    unsafe fn store_unaligned(self, ptr: *mut u8);
27    /// **signed** element-wise comparison of the vector.
28    unsafe fn cmplt(self, other: Self) -> Self;
29    /// Element-wise comparison. Result elements contain 0 for false or 0xff for true.
30    unsafe fn cmpeq(self, other: Self) -> Self;
31    /// Bitwise or.
32    unsafe fn or(self, other: Self) -> Self;
33    /// Mask of the most significant bit of each element.
34    /// For 16-bytes vector this instruction fills lower 16 bits of the result.
35    unsafe fn movemask(self) -> u32;
36}

38/// Run different code depending on whether SIMD is available or not.
39pub(crate) trait SwitchHaveSimd<R>
40where
41    Self: Sized,
42{
43    /// This function is called when SIMD is not available.
44    fn no_simd(self) -> R;
45    /// This function is called when SIMD is available.
46    #[allow(dead_code)] // FIXME: Investigate if this is really needed, fails on Mac
47    fn simd<V: Vector>(self) -> R;
48
49    /// Call either `simd` or `no_simd` function.
50    fn switch(self) -> R {
51        // Any x86_64 supports SSE2.
52        #[cfg(target_feature = "sse2")]
53        {
54            #[cfg(target_arch = "x86")]
55            use std::arch::x86::*;
56            #[cfg(target_arch = "x86_64")]
57            use std::arch::x86_64::*;
58
59            if true {
60                return self.simd::<__m128i>();
61            }
62        }
63
64        self.no_simd()
65    }
66}
67
68#[cfg(target_feature = "sse2")]
69mod sse2 {
70    #[cfg(target_arch = "x86")]
71    use std::arch::x86::*;
72    #[cfg(target_arch = "x86_64")]
73    use std::arch::x86_64::*;
74
75    use crate::values::types::string::simd::Vector;
76
77    impl Vector for __m128i {
78        #[inline(always)]
79        unsafe fn splat(byte: u8) -> Self {
80            unsafe { _mm_set1_epi8(byte as i8) }
81        }
82
83        #[inline(always)]
84        unsafe fn load_unaligned(ptr: *const u8) -> Self {
85            unsafe { _mm_loadu_si128(ptr as *const _) }
86        }
87
88        #[inline(always)]
89        unsafe fn store_unaligned(self, ptr: *mut u8) {
90            unsafe { _mm_storeu_si128(ptr as *mut _, self) }
91        }
92
93        #[inline(always)]
94        unsafe fn cmplt(self, other: Self) -> Self {
95            unsafe { _mm_cmplt_epi8(self, other) }
96        }
97
98        #[inline(always)]
99        unsafe fn cmpeq(self, other: Self) -> Self {
100            unsafe { _mm_cmpeq_epi8(self, other) }
101        }
102
103        #[inline(always)]
104        unsafe fn or(self, other: Self) -> Self {
105            unsafe { _mm_or_si128(self, other) }
106        }
107
108        #[inline(always)]
109        unsafe fn movemask(self) -> u32 {
110            unsafe { _mm_movemask_epi8(self) as u32 }
111        }
112    }
