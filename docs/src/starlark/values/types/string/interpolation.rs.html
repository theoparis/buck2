40// `i32::abs(i32::MIN)` panics as `i32::MIN` has no corresponding
41// positive value that fits inside `i32`. For this edge case,
42// let's just hardcode the results.
43const I32_MIN_OCTAL: &str = "-20000000000";
44const I32_MIN_HEX: &str = "-80000000";

46/// Operator `%` format or evaluation errors
47#[derive(Clone, Dupe, Debug, Error)]
48enum StringInterpolationError {
49    #[error("Too many arguments for format string")]
50    TooManyParameters,
51    #[error("Not enough arguments for format string")]
52    NotEnoughParameters,
53    #[error("Incomplete format")]
54    IncompleteFormat,
55    #[error("Unsupported format character: {0:?}")]
56    UnsupportedFormatCharacter(char),
57    #[error("Expecting format character (internal error)")]
58    ExpectingFormatCharacter,
59}
60
61enum PercentSFormat {
62    /// `%s`.
63    Str,
64    /// `%r`.
65    Repr,
66    /// `%d`.
67    Dec,
68    /// `%o`.
69    Oct,
70    /// `%x`.
71    Hex,
72    /// `%X`.
73    HexUpper,
74    /// `%e`.
75    Exp,
76    /// `%E`.
77    ExpUpper,
78    /// `%f` or `%F`.
79    Float,
80    /// `%g`.
81    FloatCompact,
82    /// `%G`.
83    FloatCompactUpper,
84}
85
86struct PercentFormatParser<'a> {
87    rem: &'a str,
88}
89
90struct Item<'a> {
91    literal: &'a str,
92    format: Option<PercentSFormat>,
93}
94
95impl<'a> Iterator for PercentFormatParser<'a> {
96    type Item = anyhow::Result<Item<'a>>;
97
98    #[allow(clippy::collapsible_else_if)] // Makes code more readable.
99    fn next(&mut self) -> Option<Self::Item> {
100        let index_of_percent = self.rem.bytes().position(|c| c == b'%');
101        if let Some(index_of_percent) = index_of_percent {
102            let prev_rem = self.rem;
103            let (literal, rem) = self.rem.split_at(index_of_percent);
104            match rem.as_bytes().get(1) {
105                None => return Some(Err(StringInterpolationError::IncompleteFormat.into())),
106                Some(f) => {
107                    let res = match f {
108                        b'%' => {
109                            // Include the percent in the literal.
110                            let literal = &prev_rem[..index_of_percent + 1];
111                            Item {
112                                literal,
113                                format: None,
114                            }
115                        }
116                        b's' => Item {
117                            literal,
118                            format: Some(PercentSFormat::Str),
119                        },
120                        b'r' => Item {
121                            literal,
122                            format: Some(PercentSFormat::Repr),
123                        },
124                        b'd' => Item {
125                            literal,
126                            format: Some(PercentSFormat::Dec),
127                        },
128                        b'o' => Item {
129                            literal,
130                            format: Some(PercentSFormat::Oct),
131                        },
132                        b'x' => Item {
133                            literal,
134                            format: Some(PercentSFormat::Hex),
135                        },
136                        b'X' => Item {
137                            literal,
138                            format: Some(PercentSFormat::HexUpper),
139                        },
140                        b'e' => Item {
141                            literal,
142                            format: Some(PercentSFormat::Exp),
143                        },
144                        b'E' => Item {
145                            literal,
146                            format: Some(PercentSFormat::ExpUpper),
147                        },
148                        b'f' | b'F' => Item {
149                            literal,
150                            format: Some(PercentSFormat::Float),
151                        },
152                        b'g' => Item {
153                            literal,
154                            format: Some(PercentSFormat::FloatCompact),
155                        },
156                        b'G' => Item {
157                            literal,
158                            format: Some(PercentSFormat::FloatCompactUpper),
159                        },
160                        _ => {
161                            // Note we need to find the second character, not the second byte.
162                            let Some(c) = rem.chars().nth(1) else {
163                                return Some(Err(
164                                    StringInterpolationError::ExpectingFormatCharacter.into(),
165                                ));
166                            };
167                            return Some(Err(
168                                StringInterpolationError::UnsupportedFormatCharacter(c).into(),
169                            ));
170                        }
171                    };
172                    // We reach here only if format character is ASCII,
173                    // so we can safely skip 2 bytes.
174                    self.rem = &rem[2..];
175                    Some(Ok(res))
176                }
177            }
178        } else {
179            if self.rem.is_empty() {
180                None
181            } else {
182                let literal = self.rem;
183                self.rem = "";
184                Some(Ok(Item {
185                    literal,
186                    format: None,
187                }))
188            }
189        }
190    }
191}

193pub(crate) fn percent(format: &str, value: Value) -> crate::Result<String> {
194    // NOTE(nga): use could reuse `Evaluator::string_pool` here, but
195    //   * we don't have access to `Evaluator` in `StarlarkValue::percent`
196    //   * after single %s made intrinsic, this code is not that hot now
197
198    // random guess as a baseline capacity
199    let mut res: String = String::with_capacity(format.len() + 20);
200
201    let tuple = Tuple::from_value(value);
202    let one = &[value];
203    let values = match &tuple {
204        Some(xs) => xs.content(),
205        None => one,
206    };
207    let mut values = values.iter().copied();
208    let mut next_value = || -> anyhow::Result<Value> {
209        values
210            .next()
211            .ok_or_else(|| StringInterpolationError::NotEnoughParameters.into())
212    };
213
214    // because of the way format is defined, we can deal with it as bytes
215    for item in (PercentFormatParser { rem: format }) {
216        let item = item?;
217        res.push_str(item.literal);
218        match item.format {
219            None => {}
220            Some(PercentSFormat::Str) => {
221                let arg = next_value()?;
222                match arg.unpack_str() {
223                    None => arg.collect_repr(&mut res),
224                    Some(s) => res.push_str(s),
225                }
226            }
227            Some(PercentSFormat::Repr) => next_value()?.collect_repr(&mut res),
228            Some(PercentSFormat::Dec) => {
229                let value = next_value()?;
230                match value.unpack_num() {
231                    Some(NumRef::Int(StarlarkIntRef::Small(v))) => {
232                        write!(res, "{}", v.to_i32()).unwrap()
233                    }
234                    Some(NumRef::Int(StarlarkIntRef::Big(v))) => {
235                        write!(res, "{}", v.get()).unwrap()
236                    }
237                    Some(NumRef::Float(v)) => {
238                        match NumRef::Float(StarlarkFloat(v.0.trunc())).as_int() {
239                            Some(v) => write!(res, "{v}").unwrap(),
240                            None => ValueError::unsupported_type(value, "format(%d)")?,
241                        }
242                    }
243                    None => ValueError::unsupported_type(value, "format(%d)")?,
244                }
245            }
246            Some(PercentSFormat::Oct) => {
247                let value = next_value()?;
248                match value.unpack_num() {
249                    Some(NumRef::Int(StarlarkIntRef::Small(v))) => {
250                        let v = v.to_i32();
251                        match v.checked_abs() {
252                            Some(vp) => {
253                                write!(res, "{}{:o}", if v < 0 { "-" } else { "" }, vp).unwrap()
254                            }
255                            None => write!(res, "{I32_MIN_OCTAL}").unwrap(),
256                        }
257                    }
258                    Some(NumRef::Int(StarlarkIntRef::Big(v))) => {
259                        let v = v.get();
260                        write!(
261                            res,
262                            "{}{:o}",
263                            if v.is_negative() { "-" } else { "" },
264                            v.abs()
265                        )
266                        .unwrap()
267                    }
268                    Some(NumRef::Float(_)) | None => {
269                        ValueError::unsupported_type(value, "format(%o)")?
270                    }
271                }
272            }
273            Some(PercentSFormat::Hex) => {
274                let value = next_value()?;
275                match value.unpack_num() {
276                    Some(NumRef::Int(StarlarkIntRef::Small(v))) => {
277                        let v = v.to_i32();
278                        match v.checked_abs() {
279                            Some(vp) => {
280                                write!(res, "{}{:x}", if v < 0 { "-" } else { "" }, vp).unwrap()
281                            }
282                            None => write!(res, "{I32_MIN_HEX}").unwrap(),
283                        }
284                    }
285                    Some(NumRef::Int(StarlarkIntRef::Big(v))) => {
286                        let v = v.get();
287                        write!(
288                            res,
289                            "{}{:x}",
290                            if v.is_negative() { "-" } else { "" },
291                            v.abs()
292                        )
293                        .unwrap()
294                    }
295                    Some(NumRef::Float(_)) | None => {
296                        ValueError::unsupported_type(value, "format(%x)")?
297                    }
298                }
299            }
300            Some(PercentSFormat::HexUpper) => {
301                let value = next_value()?;
302                match value.unpack_num() {
303                    Some(NumRef::Int(StarlarkIntRef::Small(v))) => {
304                        let v = v.to_i32();
305                        match v.checked_abs() {
306                            Some(vp) => {
307                                write!(res, "{}{:X}", if v < 0 { "-" } else { "" }, vp).unwrap()
308                            }
309                            None => write!(res, "{I32_MIN_HEX}").unwrap(),
310                        }
311                    }
312                    Some(NumRef::Int(StarlarkIntRef::Big(v))) => {
313                        let v = v.get();
314                        write!(
315                            res,
316                            "{}{:X}",
317                            if v.is_negative() { "-" } else { "" },
318                            v.abs()
319                        )
320                        .unwrap()
321                    }
322                    Some(NumRef::Float(_)) | None => {
323                        ValueError::unsupported_type(value, "format(%X)")?
324                    }
325                }
326            }
327            Some(PercentSFormat::Exp) => {
328                let v = NumRef::unpack_param(next_value()?)?.as_float();
329                float::write_scientific(&mut res, v, 'e', false).unwrap()
330            }
331            Some(PercentSFormat::ExpUpper) => {
332                let v = NumRef::unpack_param(next_value()?)?.as_float();
333                float::write_scientific(&mut res, v, 'E', false).unwrap()
334            }
335            Some(PercentSFormat::Float) => {
336                let v = NumRef::unpack_param(next_value()?)?.as_float();
337                float::write_decimal(&mut res, v).unwrap()
338            }
339            Some(PercentSFormat::FloatCompact) => {
340                let v = NumRef::unpack_param(next_value()?)?.as_float();
341                float::write_compact(&mut res, v, 'e').unwrap()
342            }
343            Some(PercentSFormat::FloatCompactUpper) => {
344                let v = NumRef::unpack_param(next_value()?)?.as_float();
345                float::write_compact(&mut res, v, 'E').unwrap()
346            }
347        }
348    }
349    if values.next().is_some() {
350        Err(crate::Error::new_other(
351            StringInterpolationError::TooManyParameters,
352        ))
353    } else {
354        Ok(res)
355    }
356}

358/// Try parse `"aaa%sbbb"` and return `("aaa", "bbb")`.
359pub(crate) fn parse_percent_s_one(format: &str) -> Option<(String, String)> {
360    let mut before = String::with_capacity(format.len());
361    let mut chars = format.chars();
362    loop {
363        match chars.next()? {
364            '%' => match chars.next()? {
365                '%' => before.push('%'),
366                's' => break,
367                _ => return None,
368            },
369            c => before.push(c),
370        }
371    }
372    let mut after = String::with_capacity(format.len() - before.len());
373    loop {
374        match chars.next() {
375            Some('%') => match chars.next()? {
376                '%' => after.push('%'),
377                _ => return None,
378            },
379            Some(c) => after.push(c),
380            None => break,
381        }
382    }
383    Some((before, after))
384}

386/// Evaluate `"<before>%s<after>" % arg`.
387pub(crate) fn percent_s_one<'v>(
388    before: &str,
389    arg: Value<'v>,
390    after: &str,
391    heap: &'v Heap,
392) -> crate::Result<StringValue<'v>> {
393    Ok(match StringValue::new(arg) {
394        Some(arg) => heap.alloc_str_concat3(before, &arg, after),
395        None => {
396            let one = match Tuple::from_value(arg) {
397                Some(tuple) => match tuple.content() {
398                    [] => {
399                        return Err(crate::Error::new_other(
400                            StringInterpolationError::NotEnoughParameters,
401                        ));
402                    }
403                    [value] => *value,
404                    [_, _, ..] => {
405                        return Err(crate::Error::new_other(
406                            StringInterpolationError::TooManyParameters,
407                        ));
408                    }
409                },
410                None => arg,
411            };
412            format_one(before, one, after, heap)
413        }
414    })
415}

432    #[test]
433    fn test_parse_percent_s_one() {
434        assert_eq!(
435            Some(("abc".to_owned(), "def".to_owned())),
436            parse_percent_s_one("abc%sdef")
437        );
438        assert_eq!(
439            Some(("a%b".to_owned(), "c%d%".to_owned())),
440            parse_percent_s_one("a%%b%sc%%d%%")
441        );
442        assert_eq!(None, parse_percent_s_one("a%"));
443        assert_eq!(None, parse_percent_s_one("a%s%"));
444        assert_eq!(None, parse_percent_s_one("a%s%s"));
445        assert_eq!(None, parse_percent_s_one("%d"));
446    }
