32/// Try parse `"aaa{}bbb"` and return `("aaa", "bbb")`.
33pub(crate) fn parse_format_one(s: &str) -> Option<(String, String)> {
34    let mut parser = FormatParser::new(s);
35    let mut before = String::with_capacity(s.len());
36    loop {
37        match parser.next().ok()?? {
38            FormatToken::Text(text) => before.push_str(text),
39            FormatToken::Escape(e) => before.push_str(e.as_str()),
40            FormatToken::Capture {
41                capture: "",
42                conv: FormatConv::Str,
43                pos: _,
44            } => break,
45            FormatToken::Capture { .. } => return None,
46        }
47    }
48
49    let mut after = String::with_capacity(s.len() - before.len());
50    loop {
51        match parser.next().ok()? {
52            Some(FormatToken::Text(text)) => after.push_str(text),
53            Some(FormatToken::Escape(e)) => after.push_str(e.as_str()),
54            Some(FormatToken::Capture { .. }) => return None,
55            None => break,
56        }
57    }
58
59    Some((before, after))
60}

62/// Evaluate `"<before>{}<after>".format(arg)`.
63pub(crate) fn format_one<'v>(
64    before: &str,
65    arg: Value<'v>,
66    after: &str,
67    heap: &'v Heap,
68) -> StringValue<'v> {
69    match StringValue::new(arg) {
70        Some(arg) => heap.alloc_str_concat3(before, &arg, after),
71        None => {
72            let mut result = String::with_capacity(before.len() + after.len() + 10);
73            result.push_str(before);
74            arg.collect_repr(&mut result);
75            result.push_str(after);
76            heap.alloc_str(&result)
77        }
78    }
79}

81/// The format string can either have explicit indices,
82/// or grab things sequentially, but not both.
83/// FormatArgs knows which we are doing and keeps them in mind.
84struct FormatArgs<'v, T: Iterator<Item = Value<'v>>> {
85    // Initially we have the iterator set and the args empty.
86    // If we ever ask by index, we decant the iterator into args.
87    iterator: T,
88    args: Vec<Value<'v>>,
89    by_index: bool,
90    by_order: bool,
91}
92
93impl<'v, T: Iterator<Item = Value<'v>>> FormatArgs<'v, T> {
94    fn new(iterator: T) -> Self {
95        Self {
96            iterator,
97            args: Vec::new(),
98            by_index: false,
99            by_order: false,
100        }
101    }
102
103    fn next_ordered(&mut self) -> anyhow::Result<Value<'v>> {
104        if self.by_index {
105            Err(anyhow::anyhow!(
106                "Cannot mix manual field specification and automatic field numbering in format string",
107            ))
108        } else {
109            self.by_order = true;
110            match self.iterator.next() {
111                None => Err(anyhow::anyhow!("Not enough parameters in format string")),
112                Some(x) => Ok(x),
113            }
114        }
115    }
116
117    fn by_index(&mut self, index: usize) -> anyhow::Result<Value<'v>> {
118        if self.by_order {
119            Err(anyhow::anyhow!(
120                "Cannot mix manual field specification and automatic field numbering in format string",
121            ))
122        } else {
123            if !self.by_index {
124                self.args.extend(&mut self.iterator);
125                self.by_index = true;
126            }
127            match self.args.get(index) {
128                None => Err(ValueError::IndexOutOfBound(index as i32).into()),
129                Some(v) => Ok(*v),
130            }
131        }
132    }
133}

135pub(crate) fn format<'v>(
136    this: &str,
137    args: impl Iterator<Item = Value<'v>>,
138    kwargs: Dict<'v>,
139    string_pool: &mut StringPool,
140    heap: &'v Heap,
141) -> anyhow::Result<StringValue<'v>> {
142    let mut parser = FormatParser::new(this);
143    let mut result = string_pool.alloc();
144    let mut args = FormatArgs::new(args);
145    while let Some(token) = parser.next()? {
146        match token {
147            FormatToken::Text(text) => result.push_str(text),
148            FormatToken::Escape(e) => result.push_str(e.as_str()),
149            FormatToken::Capture {
150                capture,
151                conv,
152                pos: _,
153            } => format_capture(capture, conv, &mut args, &kwargs, &mut result)?,
154        }
155    }
156    let r = heap.alloc_str(&result);
157    string_pool.release(result);
158    Ok(r)
159}

161fn format_capture<'v, T: Iterator<Item = Value<'v>>>(
162    field: &str,
163    conv: FormatConv,
164    args: &mut FormatArgs<'v, T>,
165    kwargs: &Dict,
166    result: &mut String,
167) -> anyhow::Result<()> {
168    let conv_s = |x: Value, result: &mut String| x.collect_str(result);
169    let conv_r = |x: Value, result: &mut String| x.collect_repr(result);
170    let conv: &dyn Fn(Value, &mut String) = match conv {
171        FormatConv::Str => &conv_s,
172        FormatConv::Repr => &conv_r,
173    };
174    if field.is_empty() {
175        conv(args.next_ordered()?, result);
176        Ok(())
177    } else if field.bytes().all(|c| c.is_ascii_digit()) {
178        let i = usize::from_str(field).map_err(|e| {
179            anyhow::anyhow!("Error parsing `{field}` as a format string index: {e}")
180        })?;
181        conv(args.by_index(i)?, result);
182        Ok(())
183    } else {
184        if let Some(x) = field.bytes().find(|c| match c {
185            b'.' | b',' | b'[' | b']' => true,
186            _ => false,
187        }) {
188            return Err(anyhow::anyhow!(
189                "Invalid character '{}' inside replacement field",
190                char::from(x)
191            ));
192        }
193        match kwargs.get_str(field) {
194            None => Err(ValueError::KeyNotFound(field.to_owned()).into()),
195            Some(v) => {
196                conv(v, result);
197                Ok(())
198            }
199        }
200    }
201}

217    fn format_capture_for_test<'v, T: Iterator<Item = Value<'v>>>(
218        capture: &str,
219        conv: FormatConv,
220        args: &mut FormatArgs<'v, T>,
221        kwargs: &Dict,
222    ) -> anyhow::Result<String> {
223        let mut result = String::new();
224        format_capture(capture, conv, args, kwargs, &mut result)?;
225        Ok(result)
226    }

228    #[test]
229    fn test_format_capture() {
230        let heap = Heap::new();
231        let original_args = [heap.alloc("1"), heap.alloc("2"), heap.alloc("3")];
232        let mut args = FormatArgs::new(original_args.iter().copied());
233        let mut kwargs = SmallMap::new();
234
235        kwargs.insert_hashed(heap.alloc_str("a").get_hashed(), heap.alloc("x"));
236        kwargs.insert_hashed(heap.alloc_str("b").get_hashed(), heap.alloc("y"));
237        kwargs.insert_hashed(heap.alloc_str("c").get_hashed(), heap.alloc("z"));
238        let kwargs = Dict::new(coerce(kwargs));
239        assert_eq!(
240            format_capture_for_test("", FormatConv::Str, &mut args, &kwargs).unwrap(),
241            "1"
242        );
243        assert_eq!(
244            format_capture_for_test("", FormatConv::Str, &mut args, &kwargs).unwrap(),
245            "2"
246        );
247        assert_eq!(
248            format_capture_for_test("", FormatConv::Repr, &mut args, &kwargs).unwrap(),
249            "\"3\""
250        );
251        assert_eq!(
252            format_capture_for_test("a", FormatConv::Repr, &mut args, &kwargs).unwrap(),
253            "\"x\""
254        );
255        assert_eq!(
256            format_capture_for_test("a", FormatConv::Str, &mut args, &kwargs).unwrap(),
257            "x"
258        );
259        assert!(format_capture_for_test("1", FormatConv::Str, &mut args, &kwargs).is_err());
260        let mut args = FormatArgs::new(original_args.iter().copied());
261        assert_eq!(
262            format_capture_for_test("1", FormatConv::Str, &mut args, &kwargs).unwrap(),
263            "2"
264        );
265        assert!(format_capture_for_test("", FormatConv::Str, &mut args, &kwargs).is_err());
266    }

268    #[test]
269    fn test_format() {
270        assert::eq("'a{x}b{y}c{}'.format(1, x=2, y=3)", "'a2b3c1'");
271        assert::eq("'a{x}b{{y}}c{}'.format(1, x=2)", "'a2b{y}c1'");
272    }

274    #[test]
275    fn test_parse_format_one() {
276        assert_eq!(
277            Some(("abc".to_owned(), "def".to_owned())),
278            parse_format_one("abc{}def")
279        );
280        assert_eq!(
281            Some(("abc".to_owned(), "def".to_owned())),
282            parse_format_one("abc{!s}def")
283        );
284        assert_eq!(None, parse_format_one("abc{!r}def"));
285        assert_eq!(
286            Some(("a{b".to_owned(), "c}d{".to_owned())),
287            parse_format_one("a{{b{}c}}d{{")
288        );
289        assert_eq!(None, parse_format_one("a{"));
290        assert_eq!(None, parse_format_one("a{}{}"));
291        assert_eq!(None, parse_format_one("{x}"));
292    }
