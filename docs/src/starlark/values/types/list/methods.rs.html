18//! Methods for the `list` type.
19
20use starlark_derive::starlark_module;
21use starlark_syntax::convert_indices::convert_index;
22use starlark_syntax::convert_indices::convert_indices;

24use crate as starlark;
25use crate::environment::MethodsBuilder;
26use crate::values::Heap;
27use crate::values::Value;
28use crate::values::ValueError;
29use crate::values::ValueOfUnchecked;
30use crate::values::list::ListRef;
31use crate::values::none::NoneOr;
32use crate::values::none::NoneType;
33use crate::values::types::list::value::ListData;
34use crate::values::typing::StarlarkIter;

36#[starlark_module]
37pub(crate) fn list_methods(builder: &mut MethodsBuilder) {
38    /// [list.append](
39    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#list·append
40    /// ): append an element to a list.
41    ///
42    /// `L.append(x)` appends `x` to the list L, and returns `None`.
43    ///
44    /// `append` fails if the list is frozen or has active iterators.
45    ///
46    /// ```
47    /// # starlark::assert::is_true(r#"
48    /// x = []
49    /// x.append(1)
50    /// x.append(2)
51    /// x.append(3)
52    /// x == [1, 2, 3]
53    /// # "#);
54    /// ```
55    fn append<'v>(
56        this: Value<'v>,
57        #[starlark(require = pos)] el: Value<'v>,
58        heap: &'v Heap,
59    ) -> anyhow::Result<NoneType> {
60        let this = ListData::from_value_mut(this)?;
61        this.push(el, heap);
62        Ok(NoneType)
63    }
64
65    /// [list.clear](
66    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#list·clear
67    /// ): clear a list
68    ///
69    /// `L.clear()` removes all the elements of the list L and returns `None`.
70    /// It fails if the list is frozen or if there are active iterators.
71    ///
72    /// ```
73    /// # starlark::assert::is_true(r#"
74    /// x = [1, 2, 3]
75    /// x.clear()
76    /// x == []
77    /// # "#);
78    /// ```
79    fn clear(this: Value) -> anyhow::Result<NoneType> {
80        let this = ListData::from_value_mut(this)?;
81        this.clear();
82        Ok(NoneType)
83    }
84
85    /// [list.extend](
86    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#list·extend
87    /// ): extend a list with another iterable's content.
88    ///
89    /// `L.extend(x)` appends the elements of `x`, which must be iterable, to
90    /// the list L, and returns `None`.
91    ///
92    /// `extend` fails if `x` is not iterable, or if the list L is frozen or has
93    /// active iterators.
94    ///
95    /// ```
96    /// # starlark::assert::is_true(r#"
97    /// x = []
98    /// x.extend([1, 2, 3])
99    /// x.extend(["foo"])
100    /// x == [1, 2, 3, "foo"]
101    /// # "#);
102    /// ```
103    fn extend<'v>(
104        this: Value<'v>,
105        #[starlark(require = pos)] other: ValueOfUnchecked<'v, StarlarkIter<Value<'v>>>,
106        heap: &'v Heap,
107    ) -> starlark::Result<NoneType> {
108        let res = ListData::from_value_mut(this)?;
109        if this.ptr_eq(other.get()) {
110            // If the types alias, we can't borrow the `other` for iteration.
111            // But we can do something smarter to double the elements
112            res.double(heap);
113        } else {
114            let it = other.get().iterate(heap)?;
115            res.extend(it, heap);
116        }
117        Ok(NoneType)
118    }
119
120    /// [list.index](
121    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#list·index
122    /// ): get the index of an element in the list.
123    ///
124    /// `L.index(x[, start[, end]])` finds `x` within the list L and returns its
125    /// index.
126    ///
127    /// The optional `start` and `end` parameters restrict the portion of
128    /// list L that is inspected.  If provided and not `None`, they must be list
129    /// indices of type `int`. If an index is negative, `len(L)` is effectively
130    /// added to it, then if the index is outside the range `[0:len(L)]`, the
131    /// nearest value within that range is used; see [Indexing](#indexing).
132    ///
133    /// `index` fails if `x` is not found in L, or if `start` or `end`
134    /// is not a valid index (`int` or `None`).
135    ///
136    /// ```
137    /// # starlark::assert::is_true(r#"
138    /// x = ["b", "a", "n", "a", "n", "a"]
139    /// # (
140    /// x.index("a") == 1      # bAnana
141    /// # and
142    /// x.index("a", 2) == 3   # banAna
143    /// # and
144    /// x.index("a", -2) == 5  # bananA
145    /// # )"#);
146    /// ```
147    #[starlark(speculative_exec_safe)]
148    fn index<'v>(
149        this: &ListRef<'v>,
150        #[starlark(require = pos)] needle: Value<'v>,
151        #[starlark(require = pos, default = NoneOr::None)] start: NoneOr<i32>,
152        #[starlark(require = pos, default = NoneOr::None)] end: NoneOr<i32>,
153    ) -> starlark::Result<i32> {
154        let (start, end) =
155            convert_indices(this.len() as i32, start.into_option(), end.into_option());
156        if let Some(haystack) = this.get(start..end) {
157            for (i, x) in haystack.iter().enumerate() {
158                if x.equals(needle)? {
159                    return Ok((i + start) as i32);
160                }
161            }
162        }
163        Err(anyhow::anyhow!("Element '{}' not found in '{}'", needle, this).into())
164    }
165
166    /// [list.insert](
167    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#list·insert
168    /// ): insert an element in a list.
169    ///
170    /// `L.insert(i, x)` inserts the value `x` in the list L at index `i`,
171    /// moving higher-numbered elements along by one.  It returns `None`.
172    ///
173    /// As usual, the index `i` must be an `int`. If its value is negative,
174    /// the length of the list is added, then its value is clamped to the
175    /// nearest value in the range `[0:len(L)]` to yield the effective index.
176    ///
177    /// `insert` fails if the list is frozen or has active iterators.
178    ///
179    /// ```
180    /// # starlark::assert::is_true(r#"
181    /// x = ["b", "c", "e"]
182    /// x.insert(0, "a")
183    /// x.insert(-1, "d")
184    /// x == ["a", "b", "c", "d", "e"]
185    /// # "#);
186    /// ```
187    fn insert<'v>(
188        this: Value<'v>,
189        #[starlark(require = pos)] index: i32,
190        #[starlark(require = pos)] el: Value<'v>,
191        heap: &'v Heap,
192    ) -> anyhow::Result<NoneType> {
193        let this = ListData::from_value_mut(this)?;
194        let index = convert_index(this.len() as i32, index);
195        this.insert(index, el, heap);
196        Ok(NoneType)
197    }
198
199    /// [list.pop](
200    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#list·pop
201    /// ): removes and returns the last element of a list.
202    ///
203    /// `L.pop([index])` removes and returns the last element of the list L, or,
204    /// if the optional index is provided, at that index.
205    ///
206    /// `pop` fails if the index is negative or not less than the length of
207    /// the list, of if the list is frozen or has active iterators.
208    ///
209    /// ```
210    /// # starlark::assert::is_true(r#"
211    /// x = [1, 2, 3]
212    /// # (
213    /// x.pop() == 3
214    /// # and
215    /// x.pop() == 2
216    /// # and
217    /// x == [1]
218    /// # )"#);
219    /// ```
220    fn pop<'v>(
221        this: Value<'v>,
222        #[starlark(require = pos)] index: Option<i32>,
223    ) -> anyhow::Result<Value<'v>> {
224        let this = ListData::from_value_mut(this)?;
225        let index = index.unwrap_or_else(|| (this.len() as i32) - 1);
226        if index < 0 || index >= this.len() as i32 {
227            return Err(ValueError::IndexOutOfBound(index).into());
228        }
229        Ok(this.remove(index as usize))
230    }
231
232    /// [list.remove](
233    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#list·remove
234    /// ): remove a value from a list
235    ///
236    /// `L.remove(x)` removes the first occurrence of the value `x` from the
237    /// list L, and returns `None`.
238    ///
239    /// `remove` fails if the list does not contain `x`, is frozen, or has
240    /// active iterators.
241    ///
242    /// ```
243    /// # starlark::assert::is_true(r#"
244    /// x = [1, 2, 3, 2]
245    /// x.remove(2)
246    /// # t = (
247    /// x == [1, 3, 2]
248    /// # )
249    /// x.remove(2)
250    /// # (t and (
251    /// x == [1, 3]
252    /// # ))"#);
253    /// ```
254    ///
255    /// A subsequent call to `x.remove(2)` would yield an error because the
256    /// element won't be found.
257    ///
258    /// ```
259    /// # starlark::assert::fail(r#"
260    /// x = [1, 2, 3, 2]
261    /// x.remove(2)
262    /// x.remove(2)
263    /// x.remove(2) # error: not found
264    /// # "#, "not found");
265    /// ```
266    fn remove<'v>(
267        this: Value<'v>,
268        #[starlark(require = pos)] needle: Value<'v>,
269    ) -> anyhow::Result<NoneType> {
270        // Written in two separate blocks so we ensure we give up the
271        // immutable borrow before making the mutable borrow.
272        let position = {
273            let this = ListRef::from_value(this).unwrap();
274            let position = this.iter().position(|v| v == needle);
275            match position {
276                Some(i) => i,
277                None => {
278                    return Err(anyhow::anyhow!(
279                        "Element '{}' not found in list '{}'",
280                        needle,
281                        this
282                    ));
283                }
284            }
285        };
286        {
287            // now mutate it with no further value calls
288            let this = ListData::from_value_mut(this)?;
289            this.remove(position);
290            Ok(NoneType)
291        }
292    }
293}
