44#[derive(Allocative, Hash, Eq, PartialEq, Ord, PartialOrd, Clone, Debug)]
45struct ListType;
46
47static LIST: Lazy<TyFunction> = Lazy::new(|| {
48    TyFunction::new_with_type_attr(
49        ParamSpec::pos_only([], [Ty::iter(Ty::any())]),
50        Ty::any_list(),
51        Ty::any_list(),
52    )
53});
54
55impl TyCustomFunctionImpl for ListType {
56    fn is_type(&self) -> bool {
57        true
58    }
59
60    fn as_callable(&self) -> TyCallable {
61        LIST.callable.dupe()
62    }
63
64    fn validate_call(
65        &self,
66        span: Span,
67        args: &TyCallArgs,
68        oracle: TypingOracleCtx,
69    ) -> Result<Ty, TypingOrInternalError> {
70        oracle.validate_fn_call(span, &LIST.callable, args)?;
71
72        if let Some(arg) = args.pos.first() {
73            // This is infallible after the check above.
74            let item = oracle.iter_item(Spanned {
75                span,
76                node: &arg.node,
77            })?;
78            return Ok(Ty::list(item));
79        }
80
81        Ok(Ty::any_list())
82    }
83}
84
85#[starlark_module]
86pub(crate) fn register_list(globals: &mut GlobalsBuilder) {
87    /// [list](
88    /// https://github.com/bazelbuild/starlark/blob/master/spec.md#list
89    /// ): construct a list.
90    ///
91    /// `list(x)` returns a new list containing the elements of the
92    /// iterable sequence x.
93    ///
94    /// With no argument, `list()` returns a new empty list.
95    ///
96    /// ```
97    /// # starlark::assert::all_true(r#"
98    /// list()        == []
99    /// list((1,2,3)) == [1, 2, 3]
100    /// # "#);
101    /// # starlark::assert::fail(r#"
102    /// list("strings are not iterable") # error: not supported
103    /// # "#, r#"not supported on type"#);
104    /// ```
105    #[starlark(
106    as_type = FrozenList,
107    speculative_exec_safe,
108    special_builtin_function = SpecialBuiltinFunction::List,
109    ty_custom_function = ListType,
110    )]
111    fn list<'v>(
112        #[starlark(require = pos)] a: Option<ValueOfUnchecked<'v, StarlarkIter<Value<'v>>>>,
113        heap: &'v Heap,
114    ) -> starlark::Result<ValueOfUnchecked<'v, &'v ListRef<'v>>> {
115        Ok(ValueOfUnchecked::new(if let Some(a) = a {
116            if let Some(xs) = ListRef::from_value(a.get()) {
117                heap.alloc_list(xs.content())
118            } else {
119                let it = a.get().iterate(heap)?;
120                heap.alloc(AllocList(it))
121            }
122        } else {
123            heap.alloc(AllocList::EMPTY)
124        }))
125    }
126}
