32#[starlark_module]
33pub(crate) fn register_record(builder: &mut GlobalsBuilder) {
34    /// A `record` type represents a set of named values, each with their own type.
35    ///
36    /// For example:
37    ///
38    /// ```python
39    /// MyRecord = record(host=str, port=int)
40    /// ```
41    ///
42    /// This above statement defines a record `MyRecord` with 2 fields, the first named `host` that must be of type `str`, and the second named `port` that must be of type `int`.
43    ///
44    /// Now `MyRecord` is defined, it's possible to do the following:
45    ///
46    /// * Create values of this type with `MyRecord(host="localhost", port=80)`. It is a runtime error if any arguments are missed, of the wrong type, or if any unexpected arguments are given.
47    /// * Get the type of the record suitable for a type annotation with `MyRecord.type`.
48    /// * Get the fields of the record. For example, `v = MyRecord(host="localhost", port=80)` will provide `v.host == "localhost"` and `v.port == 80`. Similarly, `dir(v) == ["host", "port"]`.
49    ///
50    /// It is also possible to specify default values for parameters using the `field` function.
51    ///
52    /// For example:
53    ///
54    /// ```python
55    /// MyRecord = record(host=str, port=field(int, 80))
56    /// ```
57    ///
58    /// Now the `port` field can be omitted, defaulting to `80` is not present (for example, `MyRecord(host="localhost").port == 80`).
59    ///
60    /// Records are stored deduplicating their field names, making them more memory efficient than dictionaries.
61    fn record<'v>(
62        #[starlark(kwargs)] kwargs: SmallMap<String, Value<'v>>,
63        eval: &mut Evaluator<'v, '_, '_>,
64    ) -> anyhow::Result<RecordType<'v>> {
65        // Every Value must either be a field or a value (the type)
66        let mut mp = SmallMap::with_capacity(kwargs.len());
67        for (k, v) in kwargs.into_iter_hashed() {
68            let field = match Field::from_value(v) {
69                None => Field::new(TypeCompiled::new(v, eval.heap())?, None),
70                Some(v) => v.dupe(),
71            };
72            mp.insert_hashed(k, field);
73        }
74        Ok(RecordType::new(mp))
75    }
76
77    /// Creates a field record. Used as an argument to the `record` function.
78    ///
79    /// ```
80    /// # starlark::assert::is_true(r#"
81    /// rec_type = record(host=field(str), port=field(int), mask=field(int, default=255))
82    /// rec = rec_type(host="localhost", port=80)
83    /// rec.port == 80
84    /// rec.mask == 255
85    /// # "#);
86    /// ```
87    fn field<'v>(
88        #[starlark(require = pos)] typ: Value<'v>,
89        default: Option<Value<'v>>,
90        eval: &mut Evaluator<'v, '_, '_>,
91    ) -> starlark::Result<Field<'v>> {
92        // We compile the type even if we don't have a default to raise the error sooner
93        let compiled = TypeCompiled::new(typ, eval.heap())?;
94        if let Some(d) = default {
95            compiled.check_type(d, Some("default"))?;
96        }
97        Ok(Field::new(compiled, default))
98    }
99}

101#[cfg(test)]
102mod tests {
103    use crate::assert;

106    #[test]
107    fn test_record_pass() {
108        assert::pass(
109            r#"
110rec_type = record(host=str, port=int)
111rec1 = rec_type(host = "test", port=80)
112rec2 = rec_type(host = "test", port=90)
113assert_eq(rec1, rec1)
114assert_eq(rec1 == rec2, False)
115assert_eq(rec1.host, "test")
116assert_eq(rec1.port, 80)
117assert_eq(dir(rec1), ["host", "port"])
118"#,
119        );
120    }

197    #[test]
198    fn test_record_equality() {
199        assert::pass(
200            r#"
201rec_type = record(host=str, port=field(int, 80))
202assert_eq(rec_type(host="s"), rec_type(host="s"))
203assert_eq(rec_type(host="s"), rec_type(host="s", port=80))
204assert_ne(rec_type(host="s"), rec_type(host="t"))
205"#,
206        );
207
208        let mut a = Assert::new();
209        a.module(
210            "m",
211            r#"
212rec_type = record(host=str, port=field(int, 80))
213rec_val = rec_type(host="s")
214"#,
215        );
216        a.pass(
217            r#"
218load('m', 'rec_type', 'rec_val')
219assert_eq(rec_val, rec_type(host="s"))
220assert_ne(rec_val, rec_type(host="t"))
221"#,
222        );
223
224        a = Assert::new();
225        a.module(
226            "m",
227            r#"
228rt = record(host=str)
229"#,
230        );
231        a.pass(
232            r#"
233load('m', r1='rt')
234rt = record(host=str)
235diff = record(host=str)
236assert_ne(r1(host="test"), rt(host="test"))
237assert_ne(r1(host="test"), diff(host="test"))
238"#,
239        );
240    }
