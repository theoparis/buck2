50/// An actual record.
51#[derive(Clone, Debug, Trace, Coerce, Freeze, ProvidesStaticType, Allocative)]
52#[repr(C)]
53pub struct RecordGen<V: ValueLifetimeless> {
54    pub(crate) typ: V, // Must be RecordType
55    pub(crate) values: Box<[V]>,
56}
57
58impl<'v, V: ValueLike<'v>> Display for RecordGen<V> {
59    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
60        let name = self.record_type_name().unwrap_or("anon");
61        fmt_keyed_container(f, &format!("record[{name}]("), ")", "=", self.iter())
62    }
63}
64
65starlark_complex_value!(pub Record);
66
67impl<'v, V: ValueLike<'v>> RecordGen<V> {
68    /// `type(x)` for records.
69    pub const TYPE: &'static str = "record";
70
71    fn get_record_type(&self) -> Either<&'v RecordType<'v>, &'v FrozenRecordType> {
72        // Safe to unwrap because we always ensure typ is RecordType
73        RecordType::from_value(self.typ.to_value()).unwrap()
74    }
75
76    fn record_type_name(&self) -> Option<&'v str> {
77        match self.get_record_type() {
78            Either::Left(x) => Some(&x.ty_record_data.get()?.name),
79            Either::Right(x) => Some(&x.ty_record_data.as_ref()?.name),
80        }
81    }
82
83    pub(crate) fn record_type_id(&self) -> TypeInstanceId {
84        match self.get_record_type() {
85            Either::Left(x) => x.id,
86            Either::Right(x) => x.id,
87        }
88    }
89
90    fn get_record_fields(&self) -> &'v SmallMap<String, FieldGen<Value<'v>>> {
91        record_fields(self.get_record_type())
92    }
93
94    /// Iterate over the elements in the record.
95    pub fn iter<'a>(&'a self) -> impl ExactSizeIterator<Item = (&'v str, V)> + 'a
96    where
97        'v: 'a,
98    {
99        self.get_record_fields()
100            .keys()
101            .map(String::as_str)
102            .zip(self.values.iter().copied())
103    }
104}
105
106#[starlark_value(type = Record::TYPE)]
107impl<'v, V: ValueLike<'v>> StarlarkValue<'v> for RecordGen<V>
108where
109    Self: ProvidesStaticType<'v>,
110{
111    fn equals(&self, other: Value<'v>) -> crate::Result<bool> {
112        match Record::from_value(other) {
113            Some(other) if self.typ.equals(other.typ)? => {
114                equals_slice(&self.values, &other.values, |x, y| x.equals(*y))
115            }
116            _ => Ok(false),
117        }
118    }
119
120    fn get_attr(&self, attribute: &str, heap: &'v Heap) -> Option<Value<'v>> {
121        self.get_attr_hashed(Hashed::new(attribute), heap)
122    }
123
124    fn get_attr_hashed(&self, attribute: Hashed<&str>, _heap: &'v Heap) -> Option<Value<'v>> {
125        let i = self.get_record_fields().get_index_of_hashed(attribute)?;
126        Some(self.values[i].to_value())
127    }
128
129    fn write_hash(&self, hasher: &mut StarlarkHasher) -> crate::Result<()> {
130        self.typ.write_hash(hasher)?;
131        for v in &*self.values {
132            v.write_hash(hasher)?;
133        }
134        Ok(())
135    }
136
137    fn dir_attr(&self) -> Vec<String> {
138        self.get_record_fields().keys().cloned().collect()
139    }
140
141    fn typechecker_ty(&self) -> Option<Ty> {
142        Some(
143            self.get_record_type()
144                .either(|r| r.instance_ty(), |r| r.instance_ty()),
145        )
146    }
147}
148
149impl<'v, V: ValueLike<'v>> Serialize for RecordGen<V> {
150    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
151    where
152        S: serde::Serializer,
153    {
154        serializer.collect_map(self.iter())
155    }
156}
