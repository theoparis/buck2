58impl<'v, V: ValueLike<'v>> StructGen<'v, V> {
59    /// The result of calling `type()` on a struct.
60    pub(crate) const TYPE: &'static str = "struct";
61
62    /// Create a new [`Struct`].
63    pub(crate) fn new(fields: SmallMap<V::String, V>) -> Self {
64        Self { fields }
65    }
66
67    /// Iterate over the elements in the struct.
68    pub(crate) fn iter<'a>(&'a self) -> impl ExactSizeIterator<Item = (StringValue<'v>, V)> + 'a
69    where
70        'v: 'a,
71    {
72        self.fields
73            .iter()
74            .map(|(name, value)| (name.to_string_value(), *value))
75    }
76
77    fn self_ty(&self) -> Ty {
78        Ty::custom(TyStruct {
79            fields: self
80                .fields
81                .iter()
82                .map(|(name, value)| (ArcStr::from(name.as_str()), Ty::of_value(value.to_value())))
83                .collect(),
84            extra: false,
85        })
86    }
87}

99/// The result of calling `struct()`.
100#[derive(Clone, Default, Debug, Trace, Freeze, ProvidesStaticType, Allocative)]
101#[repr(C)]
102pub(crate) struct StructGen<'v, V: ValueLike<'v>> {
103    /// The fields in a struct.
104    pub(crate) fields: SmallMap<V::String, V>,
105}

109impl<'v, V: ValueLike<'v>> Display for StructGen<'v, V> {
110    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
111        fmt_keyed_container(
112            f,
113            "struct(",
114            ")",
115            "=",
116            self.iter().map(|(k, v)| (k.as_str(), v)),
117        )
118    }
119}

121#[starlark_value(type = Struct::TYPE)]
122impl<'v, V: ValueLike<'v>> StarlarkValue<'v> for StructGen<'v, V>
123where
124    Self: ProvidesStaticType<'v>,
125{
126    fn collect_repr_cycle(&self, collector: &mut String) {
127        collector.push_str("struct(...)");
128    }
129
130    fn equals(&self, other: Value<'v>) -> crate::Result<bool> {
131        match Struct::from_value(other) {
132            None => Ok(false),
133            Some(other) => {
134                equals_small_map(coerce(&self.fields), &other.fields, |x, y| x.equals(*y))
135            }
136        }
137    }
138
139    fn compare(&self, other: Value<'v>) -> crate::Result<Ordering> {
140        match Struct::from_value(other) {
141            None => ValueError::unsupported_with(self, "cmp()", other),
142            Some(other) => compare_small_map(
143                coerce(&self.fields),
144                &other.fields,
145                |k| k.as_str(),
146                |x, y| x.compare(*y),
147            ),
148        }
149    }
150
151    fn get_attr(&self, attribute: &str, heap: &'v Heap) -> Option<Value<'v>> {
152        self.get_attr_hashed(Hashed::new(attribute), heap)
153    }
154
155    fn get_attr_hashed(&self, attribute: Hashed<&str>, _heap: &'v Heap) -> Option<Value<'v>> {
156        coerce(&self.fields).get_hashed(attribute).copied()
157    }
158
159    fn write_hash(&self, hasher: &mut StarlarkHasher) -> crate::Result<()> {
160        // Must use unordered hash because equality is unordered,
161        // and `a = b  =>  hash(a) = hash(b)`.
162        let mut unordered_hasher = UnorderedHasher::new();
163
164        for (k, v) in self.fields.iter_hashed() {
165            // Should hash key and value together, so two structs
166            // `a=1 b=2` and `a=2 b=1` would produce different hashes.
167            let mut entry_hasher = StarlarkHasher::new();
168            k.hash().hash(&mut entry_hasher);
169            v.write_hash(&mut entry_hasher)?;
170            unordered_hasher.write_hash(entry_hasher.finish());
171        }
172
173        hasher.write_u64(unordered_hasher.finish());
174
175        Ok(())
176    }
177
178    fn dir_attr(&self) -> Vec<String> {
179        self.fields.keys().map(|x| x.as_str().to_owned()).collect()
180    }
181
182    fn documentation(&self) -> DocItem {
183        // This treats structs as being value-like, and intentionally generates bad docs in the case
184        // of namespace-like usage. See
185        // <https://fb.workplace.com/groups/starlark/permalink/1463680027654154/> for some
186        // additional discussion
187        let typ = self.self_ty();
188        DocItem::Member(DocMember::Property(DocProperty { docs: None, typ }))
189    }
190
191    fn get_type_starlark_repr() -> Ty {
192        Ty::any_struct()
193    }
194
195    fn typechecker_ty(&self) -> Option<Ty> {
196        Some(self.self_ty())
197    }
198}

200impl<'v, V: ValueLike<'v>> Serialize for StructGen<'v, V> {
201    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
202    where
203        S: serde::Serializer,
204    {
205        serializer.collect_map(self.iter())
206    }
207}

209#[cfg(test)]

212    use crate::assert;
213
214    #[test]
215    fn test_repr() {
216        assert::eq("repr(struct(a=1, b=[]))", "'struct(a=1, b=[])'");
217        assert::eq("str(struct(a=1, b=[]))", "'struct(a=1, b=[])'");
218    }
219
220    #[test]
221    fn test_repr_cycle() {
222        assert::eq(
223            "l = []; s = struct(f=l); l.append(s); repr(s)",
224            "'struct(f=[struct(...)])'",
225        );
226        assert::eq(
227            "l = []; s = struct(f=l); l.append(s); str(s)",
228            "'struct(f=[struct(...)])'",
229        );
230    }
231
232    #[test]
233    fn test_to_json_cycle() {
234        assert::fail(
235            "l = []; s = struct(f=l); l.append(s); json.encode(s)",
236            "Cycle detected when serializing value of type `struct` to JSON",
237        );
238    }
239
240    #[test]
241    fn test_to_json() {
242        assert::all_true(
243            r#"
244json.encode(struct(key = None)) == '{"key":null}'
245json.encode(struct(key = True)) == '{"key":true}'
246json.encode(struct(key = False)) == '{"key":false}'
247json.encode(struct(key = 42)) == '{"key":42}'
248json.encode(struct(key = 'value')) == '{"key":"value"}'
249json.encode(struct(key = 'value"')) == '{"key":"value\\\""}'
250json.encode(struct(key = 'value\\')) == '{"key":"value\\\\"}'
251json.encode(struct(key = 'value/')) == '{"key":"value/"}'
252json.encode(struct(key = 'value\u0008')) == '{"key":"value\\b"}'
253json.encode(struct(key = 'value\u000C')) == '{"key":"value\\f"}'
254json.encode(struct(key = 'value\n')) == '{"key":"value\\n"}'
255json.encode(struct(key = 'value\r')) == '{"key":"value\\r"}'
256json.encode(struct(key = 'value\t')) == '{"key":"value\\t"}'
257json.encode(struct(foo = 42, bar = "some")) == '{"foo":42,"bar":"some"}'
258json.encode(struct(foo = struct(bar = "some"))) == '{"foo":{"bar":"some"}}'
259json.encode(struct(foo = ["bar/", "some"])) == '{"foo":["bar/","some"]}'
260json.encode(struct(foo = [struct(bar = "some")])) == '{"foo":[{"bar":"some"}]}'
261"#,
262        );
263    }
