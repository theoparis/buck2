39/// Allocate arbitrary value on the starlark heap without implementing full [`StarlarkValue`].
40///
41/// This is useful for data not directly visible to starlark code.
42///
43/// This type is for "complex" values (with tracing during GC). For no GC version check
44/// [`StarlarkAny`](crate::values::types::any::StarlarkAny).
45#[derive(Trace, Freeze, Allocative, ProvidesStaticType, NoSerialize)]
46pub struct StarlarkAnyComplex<T> {
47    /// The value.
48    pub value: T,
49}
50
51impl<'v, T> StarlarkAnyComplex<T>
52where
53    Self: StarlarkValue<'v>,
54{
55    /// Construct a new `StarlarkAnyComplex` value, which can be allocated on the heap.
56    pub fn new(value: T) -> StarlarkAnyComplex<T> {
57        StarlarkAnyComplex { value }
58    }
59
60    /// Obtain the value from a `Value`, if it is a `StarlarkAnyComplex<T>`.
61    pub fn get(value: Value<'v>) -> Option<&'v T> {
62        value.downcast_ref::<Self>().map(|x| &x.value)
63    }
64
65    /// Obtain the value from a `Value`, if it is a `StarlarkAnyComplex<T>`.
66    pub fn get_err(value: Value<'v>) -> crate::Result<&'v T> {
67        value.downcast_ref_err::<Self>().map(|x| &x.value)
68    }
69}
70
71// Proper `Debug` is hard to require from users because of `Freeze` and `ProvidesStaticType`.
72impl<T> Debug for StarlarkAnyComplex<T> {
73    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
74        f.debug_struct(any::type_name::<T>())
75            .finish_non_exhaustive()
76    }
77}
78
79impl<T> Display for StarlarkAnyComplex<T> {
80    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
81        Debug::fmt(self, f)
82    }
83}
84
85#[starlark_value(type = "any_complex")]
86impl<'v, T> StarlarkValue<'v> for StarlarkAnyComplex<T>
87where
88    T: Allocative + ProvidesStaticType<'v> + 'v,
89    T::StaticType: Sized,
90{
91    type Canonical = Self;
92}
93
94impl<'v, T> AllocValue<'v> for StarlarkAnyComplex<T>
95where
96    Self: StarlarkValue<'v> + Freeze,
97    T: Trace<'v>,
98    <Self as Freeze>::Frozen: StarlarkValue<'static>,
99{
100    fn alloc_value(self, heap: &'v Heap) -> Value<'v> {
101        heap.alloc_complex(self)
102    }
103}

105#[cfg(test)]
106mod tests {
107    use allocative::Allocative;
108    use starlark_derive::ProvidesStaticType;
109    use starlark_derive::Trace;
110
111    use crate as starlark;
112    use crate::const_frozen_string;
113    use crate::environment::Module;
114    use crate::values::Freeze;
115    use crate::values::FreezeResult;
116    use crate::values::Freezer;
117    use crate::values::FrozenStringValue;
118    use crate::values::FrozenValue;
119    use crate::values::StringValue;
120    use crate::values::Value;
121    use crate::values::list::AllocList;
122    use crate::values::types::any_complex::StarlarkAnyComplex;
123
124    #[test]
125    fn test_any_complex() {
126        #[derive(Trace, Allocative, ProvidesStaticType)]
127        struct UnfrozenData<'v> {
128            string: StringValue<'v>,
129            other: Value<'v>,
130        }
131
132        impl<'v> Freeze for UnfrozenData<'v> {
133            type Frozen = FrozenData;
134
135            fn freeze(self, freezer: &Freezer) -> FreezeResult<Self::Frozen> {
136                Ok(FrozenData {
137                    string: self.string.freeze(freezer)?,
138                    other: freezer.freeze(self.other)?,
139                })
140            }
141        }
142
143        #[derive(Allocative, ProvidesStaticType)]
144        struct FrozenData {
145            string: FrozenStringValue,
146            other: FrozenValue,
147        }
148
149        let module = Module::new();
150
151        let data = module.heap().alloc(StarlarkAnyComplex::new(UnfrozenData {
152            string: module.heap().alloc_str("aaa"),
153            other: module.heap().alloc(AllocList([1, 2])),
154        }));
155
156        assert_eq!(
157            const_frozen_string!("aaa"),
158            StarlarkAnyComplex::<UnfrozenData>::get_err(data)
159                .unwrap()
160                .string
161        );
162
163        module.set_extra_value(data);
164
165        let module = module.freeze().unwrap();
166
167        let data = module.extra_value().unwrap();
168        assert_eq!(
169            const_frozen_string!("aaa"),
170            StarlarkAnyComplex::<FrozenData>::get_err(data.to_value())
171                .unwrap()
172                .string
173        );
174    }
175}
