68#[derive(Debug, Error)]
69enum TypingError {
70    /// The value does not have the specified type
71    #[error("Value `{0}` of type `{1}` does not match the type annotation `{2}` for {3}")]
72    TypeAnnotationMismatch(String, String, String, String),
73    /// The given type annotation does not represent a type
74    #[error("Type `{0}` is not a valid type annotation")]
75    InvalidTypeAnnotation(String),
76    #[error("`{{A: B}}` cannot be used as type, perhaps you meant `dict[A, B]`")]
77    Dict,
78    #[error("`[X]` cannot be used as type, perhaps you meant `list[X]`")]
79    List,
80    /// The given type annotation does not exist, but the user might have forgotten quotes around
81    /// it
82    #[error(r#"Found `{0}` instead of a valid type annotation. Perhaps you meant `"{1}"`?"#)]
83    PerhapsYouMeant(String, String),
84    #[error("Value of type `{1}` does not match type `{2}`: {0}")]
85    ValueDoesNotMatchType(String, &'static str, String),
86    #[error("String literals are not allowed in type expressions: `{0}`")]
87    StringLiteralNotAllowed(String),
88}

155#[starlark_value(type = "type")]
156impl<'v, T: 'static> StarlarkValue<'v> for TypeCompiledImplAsStarlarkValue<T>
157where
158    T: TypeMatcher,
159{
160    type Canonical = TypeCompiledImplAsStarlarkValue<DummyTypeMatcher>;
161
162    fn type_matches_value(&self, value: Value<'v>, _private: Private) -> bool {
163        self.type_compiled_impl.matches(value)
164    }
165
166    fn provide(&'v self, demand: &mut Demand<'_, 'v>) {
167        demand.provide_ref_static::<dyn TypeCompiledDyn>(self);
168    }
169
170    fn write_hash(&self, hasher: &mut StarlarkHasher) -> crate::Result<()> {
171        Hash::hash(&self.ty, hasher);
172        Ok(())
173    }
174
175    fn equals(&self, other: Value<'v>) -> crate::Result<bool> {
176        let Some(other) = other.downcast_ref::<Self>() else {
177            return Ok(false);
178        };
179        Ok(self.ty == other.ty)
180    }
181
182    fn eval_type(&self) -> Option<Ty> {
183        // `TypeCompiled::new` handles this type explicitly,
184        // but implement this function to make proc-macro generate `bit_or`.
185        // Also safer to be explicit here.
186        Some(self.ty.clone())
187    }
188
189    fn get_methods() -> Option<&'static Methods>
190    where
191        Self: Sized,
192    {
193        static RES: MethodsStatic = MethodsStatic::new();
194        RES.methods(type_compiled_methods)
195    }
196}

204#[starlark_module]
205fn type_compiled_methods(methods: &mut MethodsBuilder) {
206    /// True iff the value matches this type.
207    fn matches<'v>(this: Value<'v>, value: Value<'v>) -> anyhow::Result<bool> {
208        Ok(this.get_ref().type_matches_value(value))
209    }
210
211    /// Error if the value does not match this type.
212    fn check_matches<'v>(this: Value<'v>, value: Value<'v>) -> anyhow::Result<NoneType> {
213        if !this.get_ref().type_matches_value(value) {
214            return Err(TypingError::ValueDoesNotMatchType(
215                value.to_repr(),
216                value.get_type(),
217                TypeCompiled(this).to_string(),
218            )
219            .into());
220        }
221        Ok(NoneType)
222    }
223}

225/// Wrapper for a [`Value`] that acts like a runtime type matcher.
226#[derive(
227    Debug,
228    Allocative,
229    Freeze,
230    Trace,
231    Clone,
232    Copy,
233    Dupe,
234    Coerce,
235    ProvidesStaticType
236)]
237#[repr(transparent)]
238pub struct TypeCompiled<V: ValueLifetimeless>(
239    /// `V` is `TypeCompiledImplAsStarlarkValue`.
240    V,
241);
242
243impl<'v, V: ValueLike<'v>> Display for TypeCompiled<V> {
244    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
245        match self.downcast() {
246            Ok(t) => Display::fmt(&t.as_ty_dyn(), f),
247            Err(_) => {
248                // This is unreachable, but we should not panic in `Display`.
249                Display::fmt(&self.0, f)
250            }
251        }
252    }
253}

269impl<'v, V: ValueLike<'v>> TypeCompiled<V> {
270    pub(crate) fn unchecked_new(value: V) -> Self {
271        TypeCompiled(value)
272    }
273
274    fn downcast(self) -> anyhow::Result<&'v dyn TypeCompiledDyn> {
275        self.to_value()
276            .0
277            .request_value::<&dyn TypeCompiledDyn>()
278            .ok_or_else(|| anyhow::anyhow!("Not TypeCompiledImpl (internal error)"))
279    }
280
281    /// Check if given value matches this type.
282    pub fn matches(&self, value: Value<'v>) -> bool {
283        self.0.to_value().get_ref().type_matches_value(value)
284    }
285
286    /// Get the typechecker type for this runtime type.
287    pub fn as_ty(&self) -> &'v Ty {
288        self.downcast().unwrap().as_ty_dyn()
289    }
290
291    /// True if `TypeCompiled` matches any type at runtime.
292    /// However, compile-time/lint typechecker may still check the type.
293    pub(crate) fn is_runtime_wildcard(self) -> bool {
294        self.downcast().unwrap().is_runtime_wildcard_dyn()
295    }
296
297    #[cold]
298    #[inline(never)]
299    fn check_type_error(self, value: Value<'v>, arg_name: Option<&str>) -> crate::Result<()> {
300        Err(crate::Error::new_other(
301            TypingError::TypeAnnotationMismatch(
302                value.to_str(),
303                value.get_type().to_owned(),
304                self.to_string(),
305                match arg_name {
306                    None => "return type".to_owned(),
307                    Some(x) => format!("argument `{x}`"),
308                },
309            ),
310        ))
311    }
312
313    pub(crate) fn check_type(self, value: Value<'v>, arg_name: Option<&str>) -> crate::Result<()> {
314        if self.matches(value) {
315            Ok(())
316        } else {
317            self.check_type_error(value, arg_name)
318        }
319    }
320
321    pub(crate) fn to_value(self) -> TypeCompiled<Value<'v>> {
322        TypeCompiled(self.0.to_value())
323    }
324
325    pub(crate) fn to_inner(self) -> V {
326        self.0
327    }
328
329    pub(crate) fn write_hash(self, hasher: &mut StarlarkHasher) -> crate::Result<()> {
330        self.to_value().0.write_hash(hasher)
331    }
332
333    // Dead code, but may become useful in the future.
334    pub(crate) fn _equals(self, other: Self) -> crate::Result<bool> {
335        self.to_value().0.equals(other.to_value().0)
336    }
337}

340    fn hash<H: Hasher>(&self, state: &mut H) {
341        match self.0.to_value().get_hash() {
342            Ok(h) => h.hash(state),
343            Err(_) => {

345            }
346        }

362impl<'v, V: ValueLike<'v>> TypeCompiled<V> {
363    /// Reallocate the type in a frozen heap.
364    pub fn to_frozen(self, heap: &FrozenHeap) -> TypeCompiled<FrozenValue> {
365        if let Some(v) = self.0.to_value().unpack_frozen() {
366            TypeCompiled(v)
367        } else {
368            self.to_value().downcast().unwrap().to_frozen_dyn(heap)
369        }
370    }

373// These functions are small, but are deliberately out-of-line so we get better
374// information in profiling about the origin of these closures
375impl<'v> TypeCompiled<Value<'v>> {
376    pub(crate) fn alloc(
377        type_compiled_impl: impl TypeMatcher,
378        ty: Ty,
379        heap: &'v Heap,
380    ) -> TypeCompiled<Value<'v>> {
381        TypeCompiled(heap.alloc_simple(TypeCompiledImplAsStarlarkValue {
382            type_compiled_impl,
383            ty,
384        }))
385    }
386
387    pub(crate) fn type_list_of(
388        t: TypeCompiled<Value<'v>>,
389        heap: &'v Heap,
390    ) -> TypeCompiled<Value<'v>> {
391        TypeCompiledFactory::alloc_ty(&Ty::list(t.as_ty().clone()), heap)
392    }
393
394    pub(crate) fn type_set_of(
395        t: TypeCompiled<Value<'v>>,
396        heap: &'v Heap,
397    ) -> TypeCompiled<Value<'v>> {
398        TypeCompiledFactory::alloc_ty(&Ty::set(t.as_ty().clone()), heap)
399    }
400
401    pub(crate) fn type_any_of_two(
402        t0: TypeCompiled<Value<'v>>,
403        t1: TypeCompiled<Value<'v>>,
404        heap: &'v Heap,
405    ) -> TypeCompiled<Value<'v>> {
406        let ty = Ty::union2(t0.as_ty().clone(), t1.as_ty().clone());
407        TypeCompiledFactory::alloc_ty(&ty, heap)
408    }
409
410    pub(crate) fn type_any_of(
411        ts: Vec<TypeCompiled<Value<'v>>>,
412        heap: &'v Heap,
413    ) -> TypeCompiled<Value<'v>> {
414        let ty = Ty::unions(ts.into_map(|t| t.as_ty().clone()));
415        TypeCompiledFactory::alloc_ty(&ty, heap)
416    }
417
418    pub(crate) fn type_dict_of(
419        kt: TypeCompiled<Value<'v>>,
420        vt: TypeCompiled<Value<'v>>,
421        heap: &'v Heap,
422    ) -> TypeCompiled<Value<'v>> {
423        let ty = Ty::dict(kt.as_ty().clone(), vt.as_ty().clone());
424        TypeCompiledFactory::alloc_ty(&ty, heap)
425    }
426
427    /// Parse `[t1, t2, ...]` as type.
428    fn from_list(t: &ListRef<'v>, heap: &'v Heap) -> anyhow::Result<TypeCompiled<Value<'v>>> {
429        match t.content() {
430            [] | [_] => Err(TypingError::List.into()),
431            ts @ [_, _, ..] => {
432                // A union type, can match any
433                let ts = ts.try_map(|t| TypeCompiled::new(*t, heap))?;
434                Ok(TypeCompiled::type_any_of(ts, heap))
435            }
436        }
437    }
438
439    pub(crate) fn from_ty(ty: &Ty, heap: &'v Heap) -> Self {
440        TypeCompiledFactory::alloc_ty(ty, heap)
441    }
442
443    /// Evaluate type annotation at runtime.
444    pub fn new(ty: Value<'v>, heap: &'v Heap) -> anyhow::Result<Self> {
445        if let Some(s) = StringValue::new(ty) {
446            return Err(TypingError::StringLiteralNotAllowed(s.to_string()).into());
447        } else if ty.is_none() {
448            Ok(TypeCompiledFactory::alloc_ty(&Ty::none(), heap))
449        } else if let Some(t) = Tuple::from_value(ty) {
450            let elems = t
451                .content()
452                .try_map(|t| anyhow::Ok(TypeCompiled::new(*t, heap)?.as_ty().clone()))?;
453            Ok(TypeCompiled::from_ty(&Ty::tuple(elems), heap))
454        } else if let Some(t) = ListRef::from_value(ty) {
455            TypeCompiled::from_list(t, heap)
456        } else if ty.request_value::<&dyn TypeCompiledDyn>().is_some() {
457            // This branch is optimization: `TypeCompiledAsStarlarkValue` implements `eval_type`,
458            // but this branch avoids copying the type.
459            Ok(TypeCompiled(ty))
460        } else {
461            match ty.get_ref().eval_type() {
462                Some(ty) => Ok(TypeCompiled::from_ty(&ty, heap)),
463                _ => Err(invalid_type_annotation(ty, heap).into()),
464            }
465        }
466    }
467}

469impl TypeCompiled<FrozenValue> {
470    /// Evaluate type annotation at runtime.
471    pub(crate) fn new_frozen(ty: FrozenValue, frozen_heap: &FrozenHeap) -> anyhow::Result<Self> {
472        // TODO(nga): trip to a heap is not free.
473        let heap = Heap::new();
474        let ty = TypeCompiled::new(ty.to_value(), &heap)?;
475        Ok(ty.to_frozen(frozen_heap))
476    }
477
478    /// `typing.Any`.
479    pub fn any() -> TypeCompiled<FrozenValue> {
480        static ANYTHING: AValueRepr<
481            AValueImpl<'static, AValueBasic<TypeCompiledImplAsStarlarkValue<IsAny>>>,
482        > = TypeCompiledImplAsStarlarkValue::alloc_static(IsAny, Ty::any());
483
484        TypeCompiled::unchecked_new(FrozenValue::new_repr(&ANYTHING))
485    }
486}
487
488fn invalid_type_annotation<'v>(ty: Value<'v>, heap: &'v Heap) -> TypingError {
489    if DictRef::from_value(ty).is_some() {
490        TypingError::Dict
491    } else if ListRef::from_value(ty).is_some() {
492        TypingError::List
493    } else if let Some(name) = ty
494        .get_attr("type", heap)
495        .ok()
496        .flatten()
497        .and_then(|v| v.unpack_str())
498    {
499        TypingError::PerhapsYouMeant(ty.to_str(), name.into())
500    } else {
501        TypingError::InvalidTypeAnnotation(ty.to_str())
502    }
