18//! Define a common set of errors.
19
20use thiserror::Error;
21
22use crate::values::StarlarkValue;
23use crate::values::Value;
24
25/// Common errors returned by Starlark evaluation.
26#[derive(Debug, Error)]
27#[allow(missing_docs)] // Self-explanatory.
28pub enum ValueError {
29    #[error("Operation `{op}` not supported on type `{typ}`")]
30    OperationNotSupported { op: String, typ: String },
31    #[error("Operation `{op}` not supported for types `{left}` and `{right}`")]
32    OperationNotSupportedBinary {
33        op: String,
34        left: String,
35        right: String,
36    },
37    #[error("Cannot divide by zero")]
38    DivisionByZero,
39    #[error("Integer overflow")]
40    IntegerOverflow,
41    #[error("Negative shift count")]
42    NegativeShiftCount,
43    #[error("Type of parameters mismatch")]
44    IncorrectParameterType,
45    #[error("Type of parameter `{0}` doesn't match")]
46    IncorrectParameterTypeNamed(String),
47    #[error("Missing this parameter")]
48    MissingThis,
49    #[error("Missing required parameter `{0}`")]
50    MissingRequired(String),
51    #[error("Index `{0}` is out of bound")]
52    IndexOutOfBound(i32),
53    #[error("Key `{0}` was not found")]
54    KeyNotFound(String),
55    #[error("Immutable")]
56    CannotMutateImmutableValue,
57    #[error("This operation mutates an iterable for an iterator while iterating.")]
58    MutationDuringIteration,
59    #[error("Object of type `{0}` has no attribute `{1}`")]
60    NoAttr(String, String),
61    #[error("Object of type `{0}` has no attribute `{1}`, did you mean `{2}`?")]
62    NoAttrDidYouMean(String, String, String),
63}
64
65impl From<ValueError> for crate::Error {
66    fn from(e: ValueError) -> Self {
67        crate::Error::new_kind(crate::ErrorKind::Value(anyhow::Error::new(e)))
68    }
69}
70
71#[derive(Debug, Error)]
72pub(crate) enum ControlError {
73    #[error("Value of type `{0}` is not hashable")]
74    NotHashableValue(String),
75    #[error("Too many recursion levels")]
76    TooManyRecursionLevel,
77}
78
79impl ValueError {
80    #[cold]
81    pub(crate) fn unsupported_owned<T>(
82        left: &str,
83        op: &str,
84        right: Option<&str>,
85    ) -> crate::Result<T> {
86        match right {
87            None => Err(ValueError::OperationNotSupported {
88                op: op.to_owned(),
89                typ: left.to_owned(),
90            }
91            .into()),
92            Some(right) => Err(ValueError::OperationNotSupportedBinary {
93                op: op.to_owned(),
94                left: left.to_owned(),
95                right: right.to_owned(),
96            }
97            .into()),
98        }
99    }
100
101    /// Helper to create an [`OperationNotSupported`](ValueError::OperationNotSupported) error.
102    #[cold]
103    pub fn unsupported<'v, T, V: StarlarkValue<'v>>(_left: &V, op: &str) -> crate::Result<T> {
104        Self::unsupported_owned(V::TYPE, op, None)
105    }
106
107    #[cold]
108    pub(crate) fn unsupported_type<T>(left: Value, op: &str) -> crate::Result<T> {
109        Self::unsupported_owned(left.get_type(), op, None)
110    }
111
112    /// Helper to create an [`OperationNotSupported`](ValueError::OperationNotSupportedBinary) error.
113    #[cold]
114    pub fn unsupported_with<'v, T, V: StarlarkValue<'v>>(
115        _left: &V,
116        op: &str,
117        right: Value,
118    ) -> crate::Result<T> {
119        Self::unsupported_owned(V::TYPE, op, Some(right.get_type()))
120    }
121}
