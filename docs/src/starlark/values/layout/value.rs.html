139/// A Starlark value. The lifetime argument `'v` corresponds to the [`Heap`](crate::values::Heap) it is stored on.
140///
141/// Many of the methods simply forward to the underlying [`StarlarkValue`](crate::values::StarlarkValue).
142/// The [`Display`](std::fmt::Display) trait is equivalent to the `repr()` function in Starlark.
143#[derive(Clone_, Copy_, Dupe_, ProvidesStaticType, Allocative)]
144#[allocative(skip)] // Value is owned by heap.
145// One possible change: moving to Forward during GC.

151unsafe impl<'v> CoerceKey<Value<'v>> for FrozenValue {}

165impl Display for Value<'_> {
166    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
167        match repr_stack_push(*self) {
168            Ok(_guard) => {
169                // We want to reuse Display for `repr`, so that means that
170                // strings must display "with quotes", so we get everything consistent.
171                Display::fmt(self.get_ref().as_display(), f)
172            }
173            Err(..) => {
174                let mut recursive = String::new();
175                self.get_ref().collect_repr_cycle(&mut recursive);
176                write!(f, "{recursive}")
177            }
178        }
179    }
180}

188fn debug_value(typ: &str, v: Value, f: &mut fmt::Formatter) -> fmt::Result {
189    // When value is being moved during GC or freeze,
190    // `Value` pointee is not a proper value, but a GC-related information.
191    // Regular operations like `.to_repr()` crash, but `Debug` should work.
192    if let Some(x) = v.0.unpack_ptr() {
193        if let AValueOrForwardUnpack::Forward(fwd) = x.unpack() {
194            return f.debug_tuple(typ).field(&fwd).finish();
195        }
196    }
197    f.debug_tuple(typ).field(v.get_ref().as_debug()).finish()
198}

202        debug_value("Value", *self, f)
203    }

240/// A [`Value`] that can never be changed. Can be converted back to a [`Value`] with [`to_value`](FrozenValue::to_value).
241///
242/// A [`FrozenValue`] exists on a [`FrozenHeap`](crate::values::FrozenHeap), which in turn can be kept
243/// alive by a [`FrozenHeapRef`](crate::values::FrozenHeapRef). If the frozen heap gets dropped
244/// while a [`FrozenValue`] from it still exists, the program will probably segfault, so be careful
245/// when working directly with [`FrozenValue`]s. See the type [`OwnedFrozenValue`](crate::values::OwnedFrozenValue)
246/// for a little bit more safety.
247#[derive(Clone, Copy, Dupe, ProvidesStaticType, Allocative)]
248// One possible change: moving from Blackhole during GC
249pub struct FrozenValue(
250    #[allocative(skip)] // Because it is owned by the heap.
251    pub(crate)  FrozenPointer<'static>,
252);
253
254// These can both be shared, but not obviously, because we hide a fake RefCell in Pointer to stop
255// it having variance.
256unsafe impl Send for FrozenValue {}
257unsafe impl Sync for FrozenValue {}

373    pub(crate) fn unpack_integer<I>(self) -> crate::Result<Option<I>>
374    where
375        I: TryFrom<i32>,
376        I: TryFrom<&'v BigInt>,
377    {
378        let Some(num) = StarlarkIntRef::unpack_value_opt(self) else {
379            return Ok(None);
380        };
381        let option = match num {
382            StarlarkIntRef::Small(x) => I::try_from(x.to_i32()).ok(),
383            StarlarkIntRef::Big(x) => x.unpack_integer(),
384        };
385        match option {
386            Some(i) => Ok(Some(i)),
387            None => Err(crate::Error::new_value(IntegerTooBigError {
388                integer_type: any::type_name::<I>(),
389                value: num.to_string(),
390            })),
391        }
392    }

442    /// Mostly useful for when you want to convert the string to a `dyn` trait, but can't

727    /// Check this value can be "called" with given parameter types, and provided return type.
728    ///
729    /// This check is done optimistically: when it is not known
730    /// whether the value is compatible with given arguments, return `Ok(())`.
731    ///
732    /// This operation is expensive.
733    pub fn check_callable_with<'a>(
734        self,
735        pos: impl IntoIterator<Item = &'a Ty>,
736        named: impl IntoIterator<Item = (&'a str, &'a Ty)>,
737        args: Option<&Ty>,
738        kwargs: Option<&Ty>,
739        ret: &Ty,
740    ) -> crate::Result<()> {
741        let pos = Vec::from_iter(pos);
742        let named = Vec::from_iter(named);
743        self.check_callable_with_impl(&pos, &named, args, kwargs, ret)
744    }
745
746    fn check_callable_with_impl<'a>(
747        self,
748        pos: &[&Ty],
749        named: &[(&'a str, &Ty)],
750        args: Option<&Ty>,
751        kwargs: Option<&Ty>,
752        ret: &Ty,
753    ) -> crate::Result<()> {
754        // First, provide a good error message when the value is not callable
755        // without invoking a typechecker.
756        self.check_callable()?;
757
758        let sig = TyCallable::new(
759            ParamSpec::new_parts(
760                pos.iter().map(|ty| (ParamIsRequired::Yes, (*ty).dupe())),
761                [],
762                args.duped(),
763                named
764                    .iter()
765                    .map(|(n, ty)| (ArcStr::from(*n), ParamIsRequired::Yes, (*ty).dupe())),
766                kwargs.duped(),
767            )?,
768            ret.dupe(),
769        );
770
771        let ty = Ty::of_value(self);
772        if !ty.check_call(
773            pos.iter().copied().duped(),
774            named.iter().map(|(n, ty)| (*n, (*ty).dupe())),
775            args.duped(),
776            kwargs.duped(),
777            ret.dupe(),
778        ) {
779            return Err(value_error!(
780                "Value `{}` is not compatible with the signature `{}`",
781                self.to_string_for_type_error(),
782                sig
783            ));
784        }
785
786        Ok(())
787    }

805    /// Add two [`Value`]s together. Will first try using [`add`](StarlarkValue::add),
806    /// before falling back to [`radd`](StarlarkValue::radd).
807    pub fn add(self, other: Value<'v>, heap: &'v Heap) -> crate::Result<Value<'v>> {
808        // Fast special case for ints.
809        if let Some(ls) = self.unpack_inline_int() {
810            if let Some(rs) = other.unpack_inline_int() {
811                // On overflow take the slow path below.
812                if let Some(sum) = ls.checked_add(rs) {
813                    return Ok(heap.alloc(sum));
814                }
815            }
816        }
817
818        // Addition of string is super common and pretty cheap, so have a special case for it.
819        if let Some(ls) = self.unpack_str() {
820            if let Some(rs) = other.unpack_str() {
821                if ls.is_empty() {
822                    return Ok(other);
823                } else if rs.is_empty() {
824                    return Ok(self);
825                } else {
826                    return Ok(heap.alloc_str_concat(ls, rs).to_value());
827                }
828            }
829        }
830
831        match self.get_ref().add(other, heap) {
832            Some(v) => v,
833            _ => match other.get_ref().radd(self, heap) {
834                Some(v) => v,
835                _ => ValueError::unsupported_owned(self.get_type(), "+", Some(other.get_type())),
836            },
837        }
838    }

1008    #[cold]
1009    fn display_for_type_error(self) -> impl Display + 'v {
1010        fn split_at_safe(s: &str, index: usize) -> (&str, &str) {
1011            for index in index..s.len() {
1012                if s.is_char_boundary(index) {
1013                    return s.split_at(index);
1014                }
1015            }
1016            (s, "")
1017        }
1018
1019        struct DisplayWithTypeImpl<'v>(Value<'v>);
1020
1021        impl<'v> Display for DisplayWithTypeImpl<'v> {
1022            fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
1023                let mut repr = self.0.to_repr();
1024
1025                let max_len = 60usize;
1026
1027                if repr.len() > max_len && repr.chars().count() > max_len {
1028                    let truncated = "<<...>>";
1029
1030                    // 1/3 from back, 2/3 from front, because front is usually more interesting.
1031                    let take_from_back = max_len.saturating_sub(truncated.len()) / 3;
1032                    let take_from_front = take_from_back * 2;
1033
1034                    // Resulting repr is approximately `max_len` long.
1035                    repr = format!(
1036                        "{}{}{}",
1037                        split_at_safe(&repr, take_from_front).0,
1038                        truncated,
1039                        split_at_safe(&repr, repr.len().saturating_sub(take_from_back)).1
1040                    );
1041                }
1042
1043                write!(f, "{} (repr: {})", self.0.get_type(), repr)
1044            }
1045        }
1046
1047        DisplayWithTypeImpl(self)
1048    }

1214    /// `self == b` is `ptr_eq`.
1215    pub(crate) fn eq_is_ptr_eq(self) -> bool {
1216        // Note `int` is not `ptr_eq` because `int` can be equal to `float`.
1217
1218        // If a value does not override equality, it is `ptr_eq`.
1219        !self.to_value().get_ref().vtable().starlark_value.HAS_equals
1220            // Strings of length <= 1 are statically allocated.
1221            || matches!(self.unpack_str(), Some(s) if s.len() <= 1)
1222            // Empty tuple is statically allocated.
1223            || matches!(Tuple::from_value(self.to_value()), Some(t) if t.len() == 0)
1224    }

1286/// Abstract over [`Value`] and [`FrozenValue`].
1287///
1288/// The methods on this trait are those required to implement containers,
1289/// allowing implementations of [`ComplexValue`](crate::values::ComplexValue)
1290/// to be agnostic of their contained type.
1291/// For details about each function, see the documentation for [`Value`],
1292/// which provides the same functions (and more).
1293pub trait ValueLike<'v>:
1294    ValueLifetimeless + Trace<'v> + CoerceKey<Value<'v>> + ProvidesStaticType<'v> + 'v
1295{
1296    /// `StringValue` or `FrozenStringValue`.
1297    type String: StringValueLike<'v>;
1298
1299    /// Produce a [`Value`] regardless of the type you are starting with.
1300    fn to_value(self) -> Value<'v>;
1301
1302    /// Convert from [`FrozenValue`].
1303    fn from_frozen_value(v: FrozenValue) -> Self;
1304
1305    /// Call this value as a function with given arguments.
1306    fn invoke(
1307        self,
1308        args: &Arguments<'v, '_>,
1309        eval: &mut Evaluator<'v, '_, '_>,
1310    ) -> crate::Result<Value<'v>> {
1311        self.to_value().invoke(args, eval)
1312    }
1313
1314    /// Hash the value.
1315    fn write_hash(self, hasher: &mut StarlarkHasher) -> crate::Result<()>;
1316
1317    /// Get hash value.
1318    fn get_hashed(self) -> crate::Result<Hashed<Self>> {
1319        let hash = if let Some(s) = self.to_value().unpack_starlark_str() {
1320            s.get_hash()
1321        } else {
1322            self.to_value().get_hash()?
1323        };
1324        Ok(Hashed::new_unchecked(hash, self))
1325    }
1326
1327    /// `repr(x)`.
1328    fn collect_repr(self, collector: &mut String);
1329
1330    /// `str(x)`.
1331    fn collect_str(self, collector: &mut String) {
1332        if let Some(s) = self.to_value().unpack_str() {
1333            collector.push_str(s);
1334        } else {
1335            self.collect_repr(collector);
1336        }
1337    }
1338
1339    /// `x == other`.
1340    ///
1341    /// This operation can only return error on stack overflow.
1342    fn equals(self, other: Value<'v>) -> crate::Result<bool>;
1343
1344    /// `x <=> other`.
1345    fn compare(self, other: Value<'v>) -> crate::Result<Ordering>;
1346
1347    /// Get a reference to underlying data or [`None`]
1348    /// if contained object has different type than requested.
1349    fn downcast_ref<T: StarlarkValue<'v>>(self) -> Option<&'v T>;
1350
1351    /// Get a reference to underlying data or [`Err`]
1352    /// if contained object has different type than requested.
1353    fn downcast_ref_err<T: StarlarkValue<'v>>(self) -> crate::Result<&'v T> {
1354        match self.downcast_ref() {
1355            Some(v) => Ok(v),
1356            None => Err(crate::Error::new_value(ValueValueError::WrongType(
1357                T::TYPE,
1358                self.to_value().to_string_for_type_error(),
1359            ))),
1360        }
1361    }
1362}

1385    fn downcast_ref<T: StarlarkValue<'v>>(self) -> Option<&'v T> {
1386        if T::static_type_id() == StarlarkStr::static_type_id() {
1387            if self.is_str() {
1388                // SAFETY: we just checked this is string, and requested type is string.
1389                Some(unsafe { self.downcast_ref_unchecked() })
1390            } else {
1391                None
1392            }
1393        } else if PointerI32::type_is_pointer_i32::<T>() {
1394            if self.unpack_inline_int().is_some() {
1395                // SAFETY: we just checked this is int, and requested type is int.
1396                Some(unsafe { self.downcast_ref_unchecked() })
1397            } else {
1398                None
1399            }
1400        } else {
1401            self.get_ref().downcast_ref::<T>()
1402        }
1403    }
1404
1405    fn collect_repr(self, collector: &mut String) {
1406        match repr_stack_push(self) {
1407            Ok(_guard) => {
1408                self.get_ref().collect_repr(collector);
1409            }
1410            Err(..) => {
1411                self.get_ref().collect_repr_cycle(collector);
1412            }
1413        }
1414    }
1415
1416    fn write_hash(self, hasher: &mut StarlarkHasher) -> crate::Result<()> {
1417        self.get_ref().write_hash(hasher)
1418    }
1419
1420    #[inline]
1421    fn equals(self, other: Value<'v>) -> crate::Result<bool> {
1422        self.equals(other)
1423    }
1424
1425    fn compare(self, other: Value<'v>) -> crate::Result<Ordering> {
1426        let _guard = stack_guard::stack_guard()?;
1427        self.get_ref().compare(other)
1428    }
