18// We use pointer tagging on the bottom three bits:
19// ?00 => frozen pointer
20// ?01 => mutable pointer
21// ?10 => int (32 bit)
22// third bit is a tag set by the user (get_user_tag)
23
24// We group our bytes based on the tag info, not traditional alignment.
25#![allow(clippy::unusual_byte_groupings)]

43/// Tagged pointer logically equivalent to `*mut AValueHeader`.
44#[derive(Clone, Copy, Dupe, PartialEq, Eq, Hash, Allocative)]
45pub(crate) struct RawPointer(pub(crate) NonZeroUsize);
46
47impl Debug for RawPointer {
48    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
49        f.debug_tuple("RawPointer")
50            .field(&format_args!("0x{:x}", self.ptr_value()))
51            .finish()
52    }
53}
54
55impl RawPointer {
56    #[inline]
57    pub(crate) unsafe fn new_unchecked(ptr: usize) -> RawPointer {
58        unsafe {
59            debug_assert!(ptr != 0);
60            let ptr = RawPointer(NonZeroUsize::new_unchecked(ptr));
61
62            // Run debug assertions.
63            let _ignore = PointerTags::from_pointer(ptr);
64
65            ptr
66        }
67    }
68
69    #[inline]
70    pub(crate) fn new_int(i: InlineInt) -> RawPointer {
71        let ptr = ((i.to_i32() as isize) << INT_SHIFT) as usize | TAG_INT;
72        unsafe { Self::new_unchecked(ptr) }
73    }
74
75    #[inline]
76    pub(crate) fn new_unfrozen(ptr: &AValueHeader, is_string: bool) -> RawPointer {
77        let ptr = cast::ptr_to_usize(ptr);
78        debug_assert!(ptr & TAG_MASK == 0);
79        let ptr = if is_string { ptr | TAG_STR } else { ptr };
80        let ptr = ptr | TAG_UNFROZEN;
81        unsafe { Self::new_unchecked(ptr) }
82    }
83
84    #[inline]
85    pub(crate) fn new_frozen(ptr: &AValueHeader, is_string: bool) -> RawPointer {
86        let ptr = cast::ptr_to_usize(ptr);
87        debug_assert!(ptr & TAG_MASK == 0);
88        let ptr = if is_string { ptr | TAG_STR } else { ptr };
89        unsafe { Self::new_unchecked(ptr) }
90    }
91
92    #[inline]
93    pub(crate) fn ptr_value(self) -> usize {
94        self.0.get()
95    }
96
97    #[inline]
98    pub(crate) fn tags(self) -> PointerTags {
99        PointerTags::from_pointer(self)
100    }
101
102    #[inline]
103    pub(crate) fn is_str(self) -> bool {
104        self.tags().is_str()
105    }
106
107    #[inline]
108    pub(crate) fn is_int(self) -> bool {
109        self.tags().is_int()
110    }
111
112    #[inline]
113    pub(crate) fn is_unfrozen(self) -> bool {
114        self.tags().is_unfrozen()
115    }
116
117    #[inline]
118    pub(crate) fn unpack_int(self) -> Option<InlineInt> {
119        if !self.is_int() {
120            None
121        } else {
122            unsafe { Some(self.unpack_int_unchecked()) }
123        }
124    }
125
126    #[inline]
127    pub(crate) unsafe fn unpack_pointer_i32_unchecked(self) -> &'static PointerI32 {
128        unsafe {
129            debug_assert!(self.is_int());
130            debug_assert!(self.0.get() & !INT_DATA_MASK == TAG_INT);
131
132            PointerI32::from_raw_pointer_unchecked(self)
133        }
134    }
135
136    /// Unpack integer when it is known to be not a pointer.
137    #[inline]
138    pub(crate) unsafe fn unpack_int_unchecked(self) -> InlineInt {
139        debug_assert!(self.is_int());
140        debug_assert!(self.0.get() & !INT_DATA_MASK == TAG_INT);
141
142        InlineInt::new_unchecked(((self.0.get() as isize) >> INT_SHIFT) as i32)
143    }
144
145    #[inline]
146    pub(crate) unsafe fn unpack_ptr_no_int_unchecked<'v>(self) -> &'v AValueOrForward {
147        unsafe {
148            debug_assert!(!self.is_int());
149            let ptr = self.0.get() & !(TAG_STR | TAG_UNFROZEN);
150            cast::usize_to_ptr(ptr)
151        }
152    }
153}

155// A structure that is morally a `PointerUnpack`, but gets encoded in one
156// pointer sized lump. The two types P1 and P2 are arbitrary pointers (which we
157// instantiate to FrozenValueMem and ValueMem)
158#[derive(Clone, Copy, Dupe)]
159pub(crate) struct Pointer<'p> {
160    ptr: RawPointer,
161    // Make sure we are invariant in all the types/lifetimes, and are not `Send` or `Sync`
162    // See https://stackoverflow.com/questions/62659221/why-does-a-program-compile-despite-an-apparent-lifetime-mismatch
163    _phantom: PhantomData<*mut &'p ()>,
164}
165
166// Similar to `Pointer` but allows widening lifetime, which is valid operation for frozen pointers.
167#[derive(Clone, Copy, Dupe)]
168pub(crate) struct FrozenPointer<'p> {
169    ptr: RawPointer,
170    phantom: PhantomData<&'p AValueHeader>,
171}
172
173fn _test_lifetime_covariant<'a>(p: FrozenPointer<'static>) -> FrozenPointer<'a> {
174    p
175}

177assert_eq_size!(Pointer<'static>, usize);
178assert_eq_size!(Option<Pointer<'static>>, usize);
179assert_eq_size!(FrozenPointer<'static>, usize);
180assert_eq_size!(Option<FrozenPointer<'static>>, usize);

182#[allow(dead_code)] // False positive.
183const TAG_BITS: usize = 3;
184const TAG_MASK: usize = 0b111;
185#[allow(clippy::assertions_on_constants)]
186const _: () = assert!(TAG_MASK == (1 << TAG_BITS) - 1);
187
188const TAG_INT: usize = 0b010;
189const TAG_STR: usize = 0b100;
190// Pointer to an object, which is not frozen.
191// Note, an object can be changed from unfrozen to frozen, not vice versa.
192const TAG_UNFROZEN: usize = 0b001;
193
194/// All possible tag values, three least significant bits of a pointer.
195#[repr(usize)]
196#[derive(Copy, Clone, Eq, PartialEq, Debug)]
197pub(crate) enum PointerTags {
198    Int = TAG_INT,
199    StrUnfrozen = TAG_STR | TAG_UNFROZEN,
200    StrFrozen = TAG_STR,
201    OtherUnfrozen = TAG_UNFROZEN,
202    OtherFrozen = 0,
203}
204
205impl PointerTags {
206    #[inline]
207    unsafe fn from_usize_unchecked(x: usize) -> Self {
208        debug_assert!(
209            x == PointerTags::Int as usize
210                || x == PointerTags::StrUnfrozen as usize
211                || x == PointerTags::StrFrozen as usize
212                || x == PointerTags::OtherUnfrozen as usize
213                || x == PointerTags::OtherFrozen as usize
214        );
215        unsafe { mem::transmute(x) }
216    }
217
218    #[inline]
219    fn from_pointer(ptr: RawPointer) -> Self {
220        unsafe { Self::from_usize_unchecked(ptr.0.get() & TAG_MASK) }
221    }
222
223    #[inline]
224    fn to_usize(self) -> usize {
225        self as usize
226    }
227
228    /// String value, frozen or not.
229    #[inline]
230    fn is_str(self) -> bool {
231        self.to_usize() & TAG_STR != 0
232    }
233
234    /// Inline integer.
235    #[inline]
236    fn is_int(self) -> bool {
237        self == PointerTags::Int
238    }
239
240    /// Not frozen, not an integer.
241    #[inline]
242    fn is_unfrozen(self) -> bool {
243        self.to_usize() & TAG_UNFROZEN != 0
244    }
245}

247/// All possible tag values for frozen pointers, three least significant bits of a pointer.
248#[repr(usize)]
249#[derive(Copy, Clone, Eq, PartialEq, Debug)]
250enum _FrozenPointerTags {
251    Int = TAG_INT,
252    Str = TAG_STR,
253    Other = 0,
254}
255
256/// `ThinBoxSliceFrozenValue` depends on all `FrozenPointer` values having the bottom bit unset.
257/// We keep a check here to ensure that remains true.
258#[allow(dead_code)] // False positive.
259const TAG_NICHE: usize = 0b1;
260
261const _: () = assert!(
262    _FrozenPointerTags::Int as usize & TAG_NICHE == 0
263        && _FrozenPointerTags::Str as usize & TAG_NICHE == 0
264        && _FrozenPointerTags::Other as usize & TAG_NICHE == 0
265);
266
267/// `InlineInt` is shift by this number of bits to the left to be stored in a pointer.
268const INT_SHIFT: usize = mem::size_of::<usize>() * 8 - InlineInt::BITS;
269const INT_DATA_MASK: usize = ((1usize << InlineInt::BITS) - 1) << INT_SHIFT;
270#[allow(clippy::assertions_on_constants)]
271const _: () = assert!(INT_SHIFT >= TAG_BITS);
272
273#[inline]
274unsafe fn untag_pointer<'a>(x: usize) -> &'a AValueOrForward {
275    unsafe { cast::usize_to_ptr(x & !TAG_MASK) }
276}

278impl<'p> Pointer<'p> {
279    #[inline]
280    unsafe fn new(ptr: RawPointer) -> Pointer<'p> {
281        Pointer {
282            ptr,
283            _phantom: PhantomData,
284        }
285    }
286
287    #[inline]
288    pub(crate) unsafe fn new_unfrozen_usize_with_str_tag(x: usize) -> Self {
289        unsafe {
290            debug_assert!((x & TAG_MASK & !TAG_STR) == 0);
291            Self::new(RawPointer::new_unchecked(x | TAG_UNFROZEN))
292        }
293    }
294
295    #[inline]
296    pub(crate) fn new_unfrozen(x: &'p AValueHeader, is_string: bool) -> Self {
297        unsafe { Self::new(RawPointer::new_unfrozen(x, is_string)) }
298    }
299
300    #[inline]
301    pub(crate) fn is_str(self) -> bool {
302        self.ptr.is_str()
303    }
304
305    #[inline]
306    pub(crate) fn is_unfrozen(self) -> bool {
307        self.ptr.is_unfrozen()
308    }
309
310    #[inline]
311    pub(crate) fn unpack(self) -> Either<&'p AValueOrForward, &'static PointerI32> {
312        if !self.ptr.is_int() {
313            Either::Left(unsafe { self.ptr.unpack_ptr_no_int_unchecked() })
314        } else {
315            Either::Right(unsafe { PointerI32::from_raw_pointer_unchecked(self.ptr) })
316        }
317    }
318
319    #[inline]
320    pub(crate) fn unpack_int(self) -> Option<InlineInt> {
321        self.ptr.unpack_int()
322    }
323
324    #[inline]
325    pub(crate) fn unpack_ptr(self) -> Option<&'p AValueOrForward> {
326        if !self.ptr.is_int() {
327            Some(unsafe { untag_pointer(self.ptr.0.get()) })
328        } else {
329            None
330        }
331    }
332
333    /// Unpack pointer when it is known to be not an integer.
334    #[inline]
335    pub(crate) unsafe fn unpack_ptr_no_int_unchecked(self) -> &'p AValueOrForward {
336        unsafe {
337            let p = self.ptr.0.get();
338            debug_assert!(!self.ptr.is_int());
339            untag_pointer(p)
340        }
341    }
342
343    /// Unpack integer when it is known to be not a pointer.
344    #[inline]
345    pub(crate) unsafe fn unpack_pointer_i32_unchecked(self) -> &'static PointerI32 {
346        unsafe { self.ptr.unpack_pointer_i32_unchecked() }
347    }
348
349    #[inline]
350    pub(crate) fn ptr_eq(self, other: Pointer<'_>) -> bool {
351        self.ptr == other.ptr
352    }
353
354    #[inline]
355    pub(crate) fn raw(self) -> RawPointer {
356        self.ptr
357    }
358
359    #[inline]
360    pub(crate) unsafe fn cast_lifetime<'p2>(self) -> Pointer<'p2> {
361        Pointer {
362            ptr: self.ptr,
363            _phantom: PhantomData,
364        }
365    }
366
367    #[inline]
368    pub(crate) unsafe fn to_frozen_pointer_unchecked(self) -> FrozenPointer<'p> {
369        unsafe { FrozenPointer::new(self.ptr) }
370    }
371}

373impl<'p> FrozenPointer<'p> {
374    #[inline]
375    pub(crate) unsafe fn new(ptr: RawPointer) -> FrozenPointer<'p> {
376        debug_assert!(!ptr.is_unfrozen());
377        FrozenPointer {
378            ptr,
379            phantom: PhantomData,
380        }
381    }
382
383    #[inline]
384    pub(crate) fn new_frozen_usize_with_str_tag(x: usize) -> Self {
385        debug_assert!((x & TAG_MASK & !TAG_STR) == 0);
386        unsafe { Self::new(RawPointer::new_unchecked(x)) }
387    }
388
389    #[inline]
390    pub(crate) fn new_frozen(x: &'p AValueHeader, is_str: bool) -> Self {
391        unsafe { Self::new(RawPointer::new_frozen(x, is_str)) }
392    }
393
394    #[inline]
395    pub(crate) fn new_int(x: InlineInt) -> Self {
396        unsafe { Self::new(RawPointer::new_int(x)) }
397    }
398
399    /// It is safe to bitcast `FrozenPointer` to `Pointer`
400    /// but not vice versa.
401    #[inline]
402    pub(crate) fn to_pointer(self) -> Pointer<'p> {
403        Pointer {
404            ptr: self.ptr,
405            _phantom: PhantomData,
406        }
407    }
408
409    #[inline]
410    pub(crate) fn raw(self) -> RawPointer {
411        self.ptr
412    }
413
414    /// Unpack pointer when it is known to be not an integer.
415    #[inline]
416    pub(crate) unsafe fn unpack_ptr_no_int_unchecked(self) -> &'p AValueOrForward {
417        unsafe {
418            debug_assert!(!self.ptr.is_int());
419            self.ptr.unpack_ptr_no_int_unchecked()
420        }
421    }
422
423    /// Unpack integer when it is known to be not a pointer.
424    #[inline]
425    pub(crate) unsafe fn unpack_pointer_i32_unchecked(self) -> &'static PointerI32 {
426        unsafe { self.ptr.unpack_pointer_i32_unchecked() }
427    }
428
429    /// Unpack pointer when it is known to be frozen, not an integer, not a string.
430    #[inline]
431    pub(crate) unsafe fn unpack_ptr_no_int_no_str_unchecked(self) -> &'p AValueOrForward {
432        unsafe {
433            debug_assert!(self.ptr.tags() == PointerTags::OtherFrozen);
434            cast::usize_to_ptr(self.ptr.0.get())
435        }
436    }

442    fn check(x: InlineInt) {
443        assert_eq!(x, RawPointer::new_int(x).unpack_int().unwrap());
444    }
