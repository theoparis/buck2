19use std::cmp;
20use std::mem;

22use crate::values::layout::aligned_size::AlignedSize;
23use crate::values::layout::heap::allocator::alloc::chain::ChunkChain;
24use crate::values::layout::heap::allocator::alloc::chunk::Chunk;
25use crate::values::layout::heap::allocator::alloc::chunk_part::ChunkPart;

28/// Minimum usable cached allocation.
29const MIN_USABLE_ALLOC: AlignedSize = AlignedSize::new_bytes(
30    (
31        // * All chunks are used to store chains, so we need to include chain header.
32        // * We need to be able to store at least one object after the header.
33        ChunkChain::HEADER_SIZE.bytes() + MIN_ALLOC.bytes()
34    ) as usize,
35);
36
37#[derive(Debug, Default)]
38struct PerThreadChunkCache {
39    /// Keep a few last chunks.
40    ///
41    /// Frozen heap has two arenas: drop and non-drop. So we should keep at least two chunks.
42    last_chunks: [ChunkPart; 4],
43}
44
45impl PerThreadChunkCache {
46    /// Save a chunk to the thread-local cache if it is large enough.
47    fn store(&mut self, mut chunk: ChunkPart) {
48        for next in &mut self.last_chunks {
49            // Keep the largest chunks in the pool.
50            if chunk.len() > next.len() {
51                mem::swap(next, &mut chunk);
52            }
53        }
54    }
55
56    /// Fetch a chunk from the thread-local cache if the cache has a chunk large enough.
57    fn fetch(&mut self, len: AlignedSize) -> Option<ChunkPart> {
58        for next in &mut self.last_chunks {
59            // Pick any chunk which is large enough.
60            if next.len() >= len {
61                let result = mem::take(next);
62                return Some(result);
63            }
64        }
65        None
66    }
67}
68
69thread_local! {
70    static PER_THREAD_ALLOCATOR: RefCell<PerThreadChunkCache> = RefCell::new(PerThreadChunkCache::default());
71}
72
73fn next_chunk_size(chunk_count_in_bump: usize) -> AlignedSize {
74    // Replicate `bumpalo` behavior: 512 in the first chunk, double each next,
75    // but not greater than 2G.
76    // TODO(nga): we should stop doubling after 1M or so.
77    let size = AlignedSize::new_bytes(
78        512u32
79            .checked_shl(chunk_count_in_bump.try_into().unwrap())
80            .unwrap() as usize,
81    );
82    if size.bytes() == 0 {
83        AlignedSize::new_bytes(1 << 31)
84    } else {
85        size
86    }
87}
88
89/// Allocate chunk which is large enough for given number of words.
90pub(crate) fn thread_local_alloc_at_least(
91    len: AlignedSize,
92    chunk_count_in_bump: usize,
93) -> ChunkPart {
94    let chunk = match PER_THREAD_ALLOCATOR.with_borrow_mut(|allocator| allocator.fetch(len)) {
95        Some(chunk) => chunk,
96        _ => {
97            let next_chunk_size = next_chunk_size(chunk_count_in_bump) - Chunk::HEADER_SIZE;
98            let len = cmp::max(len, next_chunk_size);
99            ChunkPart::alloc_at_least(len)
100        }
101    };
102    debug_assert!(chunk.len() >= len);
103    chunk
104}
105
106/// Release chunk part to thread-local pool.
107#[allow(clippy::if_same_then_else)]
108#[inline]
109pub(crate) fn thread_local_release(chunk: ChunkPart) {
110    if chunk.is_full() {
111        // Chunk part is the full chunk. Better return it to malloc.
112        drop(chunk)
113    } else if chunk.len() < MIN_USABLE_ALLOC {
114        // It is not reusable.
115        drop(chunk)
116    } else if chunk.chunk_ref_count() == 1 {
117        // We could reuse the chunk, but since it is not shared,
118        // better return it to malloc.
119        drop(chunk);
120    } else {
121        PER_THREAD_ALLOCATOR.with_borrow_mut(|allocator| allocator.store(chunk));
122    }
123}

125#[cfg(test)]
126mod tests {
127    use crate::values::layout::aligned_size::AlignedSize;
128    use crate::values::layout::heap::allocator::alloc::chunk_part::ChunkPart;
129    use crate::values::layout::heap::allocator::alloc::per_thread::PerThreadChunkCache;
130    use crate::values::layout::heap::repr::AValueHeader;
131
132    #[test]
133    fn test_release_partial() {
134        let mut allocator = PerThreadChunkCache::default();
135        let chunk = ChunkPart::alloc_at_least(AlignedSize::new_bytes(10 * AValueHeader::ALIGN));
136        let (a, b) = chunk.split_at_offset(AlignedSize::new_bytes(5 * AValueHeader::ALIGN));
137        let old_a_ptr = a.begin().as_ptr();
138        let old_b_ptr = b.begin().as_ptr();
139        allocator.store(a);
140        allocator.store(b);
141        let a = allocator
142            .fetch(AlignedSize::new_bytes(3 * AValueHeader::ALIGN))
143            .unwrap();
144        let b = allocator
145            .fetch(AlignedSize::new_bytes(3 * AValueHeader::ALIGN))
146            .unwrap();
147        assert!(
148            std::ptr::eq(old_a_ptr, a.begin().as_ptr())
149                || std::ptr::eq(old_a_ptr, b.begin().as_ptr())
150        );
151        assert!(
152            std::ptr::eq(old_b_ptr, a.begin().as_ptr())
153                || std::ptr::eq(old_b_ptr, b.begin().as_ptr())
154        );
155    }
