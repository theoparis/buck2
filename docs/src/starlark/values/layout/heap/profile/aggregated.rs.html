51/// A mapping from function Value to FunctionId, which must be continuous
52#[derive(Default)]
53struct FunctionIds {
54    values: HashMap<RawPointer, StringId>,
55    strings: StringIndex,
56}
57
58impl FunctionIds {
59    fn get_value(&mut self, x: Value) -> StringId {
60        match self.values.entry(x.ptr_value()) {
61            hash_map::Entry::Occupied(v) => *v.get(),
62            hash_map::Entry::Vacant(outer) => {
63                let function_id = self.strings.index(&x.to_str());
64                outer.insert(function_id);
65                function_id
66            }
67        }
68    }
69}

71/// A stack frame, its caller and the functions it called, and the allocations it made itself.
72struct StackFrameData {
73    callees: SmallMap<StringId, StackFrameBuilder>,
74    allocs: HeapSummary,
75    /// Time spent in this frame excluding callees.
76    /// Double, because enter/exit are recorded twice, in drop and non-drop heaps.
77    time_x2: SmallDuration,
78    /// How many times this function was called (with this stack).
79    /// Double.
80    calls_x2: u32,
81}
82
83#[derive(Clone, Dupe)]
84struct StackFrameBuilder(Rc<RefCell<StackFrameData>>);
85
86impl StackFrameBuilder {
87    fn new() -> Self {
88        Self(Rc::new(RefCell::new(StackFrameData {
89            callees: Default::default(),
90            allocs: Default::default(),
91            time_x2: SmallDuration::default(),
92            calls_x2: 0,
93        })))
94    }
95
96    /// Enter a new stack frame.
97    fn push(&self, function: StringId) -> Self {
98        let mut this = self.0.borrow_mut();
99
100        let callee = this
101            .callees
102            .entry(function)
103            .or_insert_with(StackFrameBuilder::new);
104
105        callee.dupe()
106    }
107
108    fn build(&self) -> StackFrame {
109        StackFrame {
110            callees: self
111                .0
112                .borrow()
113                .callees
114                .iter()
115                .map(|(f, s)| (*f, s.build()))
116                .collect(),
117            allocs: self.0.borrow().allocs.clone(),
118            time_x2: self.0.borrow().time_x2,
119            calls_x2: self.0.borrow().calls_x2,
120        }
121    }
122}

124/// An accumulator for stack frames that lets us visit the heap.
125pub(crate) struct StackCollector {
126    /// Timestamp of last call enter or exit.
127    last_time: Option<ProfilerInstant>,
128    ids: FunctionIds,
129    current: Vec<StackFrameBuilder>,
130    /// What we are collecting.
131    /// When unset, we are collecting allocated memory (not retained).
132    /// When set, must be set to correct heap type (unfrozen or frozen), we are traversing.
133    retained: Option<HeapKind>,
134}
135
136impl StackCollector {
137    pub(crate) fn new(retained: Option<HeapKind>) -> Self {
138        Self {
139            ids: FunctionIds::default(),
140            current: vec![StackFrameBuilder::new()],
141            last_time: None,
142            retained,
143        }
144    }
145}
146
147impl<'v> ArenaVisitor<'v> for StackCollector {
148    fn enter_bump(&mut self) {
149        self.last_time = None;
150    }
151
152    fn regular_value(&mut self, value: &'v AValueOrForward) {
153        let value = match (value.unpack(), self.retained) {
154            (AValueOrForwardUnpack::Header(header), None) => unsafe {
155                header.unpack_value(HeapKind::Unfrozen)
156            },
157            (AValueOrForwardUnpack::Forward(forward), Some(retained)) => unsafe {
158                forward.forward_ptr().unpack_value(retained)
159            },
160            _ => return,
161        };
162
163        let frame = match self.current.last() {
164            Some(frame) => frame,
165            None => return,
166        };
167
168        // Value allocated in this frame, record it!
169        let typ = value.vtable().type_name;
170        let mut frame = frame.0.borrow_mut();
171        frame.allocs.add(
172            typ,
173            AllocCounts {
174                count: 1,
175                bytes: value.get_ref().total_memory(),
176            },
177        );
178    }
179
180    fn call_enter(&mut self, function: Value<'v>, time: ProfilerInstant) {
181        if let Some(last_time) = self.last_time {
182            self.current.last_mut().unwrap().0.borrow_mut().time_x2 +=
183                time.duration_since(last_time);
184            self.current.last_mut().unwrap().0.borrow_mut().calls_x2 += 1;
185        }
186
187        let frame = match self.current.last() {
188            Some(frame) => frame,
189            None => return,
190        };
191
192        // New frame, enter it.
193        let id = self.ids.get_value(function);
194        let new_frame = frame.push(id);
195        self.current.push(new_frame);
196
197        self.last_time = Some(time)
198    }
199
200    fn call_exit(&mut self, time: ProfilerInstant) {
201        if let Some(last_time) = self.last_time {
202            self.current.last_mut().unwrap().0.borrow_mut().time_x2 +=
203                time.duration_since(last_time);
204        }
205        self.current.pop().unwrap();
206        self.last_time = Some(time);
207    }
208}

210/// Aggregated stack frame data.
211#[derive(Clone, Default, Allocative)]
212pub(crate) struct StackFrame {
213    /// Aggregated callees.
214    pub(crate) callees: SmallMap<StringId, StackFrame>,
215    /// Aggregated allocations in this frame, without callees.
216    pub(crate) allocs: HeapSummary,
217    /// Time spend in this frame excluding callees.
218    /// `x2` because enter/exit are recorded twice, in drop and non-drop heaps.
219    pub(crate) time_x2: SmallDuration,
220    /// How many times this frame was called with the same callers.
221    /// `x2` because enter/exit are recorded twice, in drop and non-drop heaps.
222    pub(crate) calls_x2: u32,
223}
224
225impl StackFrame {
226    fn merge_callees<'a>(
227        frames: &'a [StackFrameWithContext<'a>],
228        strings: &mut StringIndex,
229    ) -> SmallMap<StringId, StackFrame> {
230        let mut group_by_callee: SmallMap<&str, Vec<StackFrameWithContext>> = SmallMap::new();
231        for frame in frames {
232            for (name, callee) in frame.callees() {
233                group_by_callee.entry(name).or_default().push(callee);
234            }
235        }
236        group_by_callee
237            .into_iter()
238            .map(|(name, frames)| {
239                let name = strings.index(name);
240                (name, StackFrame::merge(frames, strings))
241            })
242            .collect()
243    }
244
245    fn merge<'a>(
246        frames: impl IntoIterator<Item = StackFrameWithContext<'a>>,
247        strings: &mut StringIndex,
248    ) -> StackFrame {
249        let frames = Vec::from_iter(frames);
250        let callees = StackFrame::merge_callees(&frames, strings);
251        let allocs = HeapSummary::merge(frames.iter().map(|f| &f.frame.allocs));
252        let time_x2 = frames.iter().map(|f| f.frame.time_x2).sum();
253        let calls_x2 = frames.iter().map(|f| f.frame.calls_x2).sum();
254        StackFrame {
255            callees,
256            allocs,
257            time_x2,
258            calls_x2,
259        }
260    }
261
262    #[cfg(test)]
263    pub(crate) fn normalize_for_golden_tests(&mut self) {
264        for (_, v) in &mut self.callees {
265            v.normalize_for_golden_tests();
266        }
267        self.allocs.normalize_for_golden_tests();
268    }
269}
270
271struct StackFrameWithContext<'c> {
272    frame: &'c StackFrame,
273    strings: &'c StringIndex,
274}
275
276impl<'c> StackFrameWithContext<'c> {
277    fn callees(&self) -> impl Iterator<Item = (&'c ArcStr, StackFrameWithContext<'c>)> + '_ {
278        self.frame.callees.iter().map(move |(id, callee)| {
279            (
280                self.strings.get(*id),
281                StackFrameWithContext {
282                    frame: callee,
283                    strings: self.strings,
284                },
285            )
286        })
287    }
288
289    /// Accumulate this stack frame's data into the given FlameGraphNode
290    fn gen_flame_graph_data(&self, node: &mut FlameGraphNode) {
291        for (k, v) in &self.frame.allocs.summary {
292            node.child((*k).into()).add(v.bytes as u64);
293        }
294
295        for (id, frame) in self.callees() {
296            let child_node = node.child(id.dupe());
297            frame.gen_flame_graph_data(child_node);
298        }
299    }
300}

302/// Aggregated heap profiling data when heap profiling is enabled.
303///
304/// Can be:
305/// * written as CSV or flamegraph
306/// * merged with another data
307#[derive(Clone, Allocative)]
308pub(crate) struct AggregateHeapProfileInfo {
309    pub(crate) strings: StringIndex,
310    pub(crate) root: StackFrame,
311}
312
313impl Debug for AggregateHeapProfileInfo {
314    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
315        f.debug_struct("AggregateHeapProfileInfo")
316            .finish_non_exhaustive()
317    }
318}
319
320impl Default for AggregateHeapProfileInfo {
321    fn default() -> AggregateHeapProfileInfo {
322        let strings = StringIndex::default();
323        AggregateHeapProfileInfo {
324            root: StackFrame::default(),
325            strings,
326        }
327    }
328}
329
330impl AggregateHeapProfileInfo {
331    pub(crate) fn collect(heap: &Heap, retained: Option<HeapKind>) -> AggregateHeapProfileInfo {
332        let mut collector = StackCollector::new(retained);
333        unsafe {
334            heap.visit_arena(HeapKind::Unfrozen, &mut collector);
335        }
336        assert_eq!(1, collector.current.len());
337        AggregateHeapProfileInfo {
338            strings: collector.ids.strings,
339            root: collector.current.pop().unwrap().build(),
340        }
341    }
342
343    fn root(&self) -> StackFrameWithContext {
344        StackFrameWithContext {
345            frame: &self.root,
346            strings: &self.strings,
347        }
348    }
349
350    /// Merge aggregated heap profile from multiple sources (e.g. from several runs).
351    pub fn merge<'a>(
352        profiles: impl IntoIterator<Item = &'a AggregateHeapProfileInfo>,
353    ) -> AggregateHeapProfileInfo {
354        let profiles: Vec<_> = Vec::from_iter(profiles);
355
356        let mut strings = StringIndex::default();
357        let roots = profiles.into_iter().map(|p| p.root());
358        let root = StackFrame::merge(roots, &mut strings);
359        AggregateHeapProfileInfo { strings, root }
360    }
361
362    /// Generate the flame graph data and return it as a string.
363    pub fn gen_flame_graph_data(&self) -> String {
364        let mut data = FlameGraphData::default();
365        self.root().gen_flame_graph_data(data.root());
366        data.write()
367    }
368
369    /// Generate per-function summary in CSV format.
370    pub fn gen_summary_csv(&self) -> String {
371        HeapSummaryByFunction::init(self).gen_csv()
372    }
373
374    #[cfg(test)]
375    pub(crate) fn normalize_for_golden_tests(&mut self) {
376        self.root.normalize_for_golden_tests();
377    }
378}

380#[derive(Debug, Allocative)]
381pub(crate) struct RetainedHeapProfile {
382    pub(crate) info: AggregateHeapProfileInfo,
383    pub(crate) mode: RetainedHeapProfileMode,
384}
385
386impl RetainedHeapProfile {
387    pub(crate) fn to_profile(&self) -> ProfileData {
388        ProfileData {
389            profile: match self.mode {
390                RetainedHeapProfileMode::FlameAndSummary => {
391                    ProfileDataImpl::HeapRetained(Box::new(self.info.clone()))
392                }
393                RetainedHeapProfileMode::Flame => {
394                    ProfileDataImpl::HeapFlameRetained(Box::new(self.info.clone()))
395                }
396                RetainedHeapProfileMode::Summary => {
397                    ProfileDataImpl::HeapSummaryRetained(Box::new(self.info.clone()))
398                }
399            },
400        }
401    }

417    fn total_alloc_count(frame: &StackFrame) -> usize {
418        frame.allocs.total().count
419            + frame

421                .values()
422                .map(|c| total_alloc_count(c.dupe()))
423                .sum::<usize>()
424    }

441    fn test_stacks_collect_retained() {
442        let heap = Heap::new();
443        heap.record_call_enter(const_frozen_string!("enter").to_value());
444        let s0 = heap.alloc_str("xxyy");
445        let s1 = heap.alloc_str("zzww");
446        heap.alloc_str("rrtt");
447        heap.record_call_exit();
448
449        let frozen_heap = FrozenHeap::new();
450        let freezer = Freezer::new(&frozen_heap);
451        freezer.freeze(s0.to_value()).unwrap();
452        freezer.freeze(s1.to_value()).unwrap();
453
454        let stacks = AggregateHeapProfileInfo::collect(&heap, Some(HeapKind::Frozen));
455        assert!(stacks.root.allocs.summary.is_empty());
456        assert_eq!(1, stacks.root.callees.len());
457        // 3 allocated, 2 retained.
458        assert_eq!(
459            2,
460            stacks
461                .root
462                .callees
463                .values()
464                .next()
465                .unwrap()
466                .allocs
467                .summary
468                .get("string")
469                .unwrap()
470                .count
471        );
472        assert_eq!(2, total_alloc_count(&stacks.root));
473    }
