30/// Information relating to a function.
31#[derive(Default, Debug, Clone)]
32pub(crate) struct FuncInfo {
33    /// Number of times this function was called
34    pub calls: usize,
35    /// Who called this function (and how many times each)
36    pub callers: SmallMap<ArcStr, usize>,
37    /// Time spent directly in this function
38    pub time: SmallDuration,
39    /// Time spent directly in this function and recursive functions.
40    pub time_rec: SmallDuration,
41    /// Allocations made by this function
42    pub alloc: SmallMap<&'static str, AllocCounts>,
43}
44
45impl FuncInfo {
46    pub(crate) fn merge<'a>(xs: impl Iterator<Item = &'a Self>) -> Self {
47        let mut result = Self::default();
48        for x in xs {
49            result.calls += x.calls;
50            result.time += x.time;
51            for (k, v) in x.alloc.iter() {
52                *result.alloc.entry(k).or_default() += *v;
53            }
54        }
55        // Recursive time doesn't accumulate nicely, the time is the right value
56        result.time_rec = result.time;
57        result
58    }
59
60    /// Total number of allocations made by this function.
61    fn alloc_count(&self) -> usize {
62        self.alloc.values().map(|x| x.count).sum()
63    }
64
65    /// Total number of bytes allocated by this function.
66    fn alloc_bytes(&self) -> usize {
67        self.alloc.values().map(|x| x.bytes).sum()
68    }
69}

71/// We morally have two pieces of information:
72/// 1. Information about each function.
73/// 2. The call stack.
74///
75/// However, we are always updating the top of the call stack,
76/// so pull out top_stack/top_info as a cache.
77pub(crate) struct HeapSummaryByFunction {
78    /// Information about all functions.
79    info: SmallMap<ArcStr, FuncInfo>,
80}
81
82impl HeapSummaryByFunction {
83    pub(crate) fn init(stacks: &AggregateHeapProfileInfo) -> HeapSummaryByFunction {
84        let mut info = HeapSummaryByFunction {
85            info: SmallMap::new(),
86        };
87        info.init_children(&stacks.root, &ArcStr::new_static("(root)"), &stacks.strings);
88        info
89    }
90
91    fn init_children(
92        &mut self,
93        frame: &StackFrame,
94        name: &ArcStr,
95        strings: &StringIndex,
96    ) -> SmallDuration {
97        let mut time_rec = SmallDuration::default();
98        for (func, child) in &frame.callees {
99            time_rec += self.init_child(*func, child, name.dupe(), strings);
100        }
101        time_rec
102    }
103
104    fn init_child(
105        &mut self,
106        func: StringId,
107        frame: &StackFrame,
108        caller: ArcStr,
109        strings: &StringIndex,
110    ) -> SmallDuration {
111        let func_str = strings.get(func);
112        self.info.entry(func_str.dupe()).or_default().time += frame.time_x2;
113        self.info.entry(func_str.dupe()).or_default().calls += frame.calls_x2 as usize;
114        *self
115            .info
116            .entry(func_str.dupe())
117            .or_default()
118            .callers
119            .entry(caller)
120            .or_insert(0) += 1;
121        for (t, allocs) in &frame.allocs.summary {
122            *self
123                .info
124                .entry(func_str.dupe())
125                .or_default()
126                .alloc
127                .entry(t)
128                .or_default() += *allocs;
129        }
130
131        let time_rec = frame.time_x2 + self.init_children(frame, func_str, strings);
132        self.info.entry(func_str.dupe()).or_default().time_rec += time_rec;
133        time_rec
134    }
135
136    fn totals(&self) -> FuncInfo {
137        FuncInfo::merge(self.info.values())
138    }
139
140    pub(crate) fn info(&self) -> Vec<(&ArcStr, &FuncInfo)> {
141        self.info.iter().collect::<Vec<_>>()
142    }
143
144    pub(crate) fn gen_csv(&self) -> String {
145        // Add a totals column
146        let totals = self.totals();
147        let mut columns: Vec<(&'static str, AllocCounts)> =
148            totals.alloc.iter().map(|(k, v)| (*k, *v)).collect();
149
150        columns.sort_by_key(|x| -(x.1.count as isize));
151
152        let mut info = self.info();
153        info.sort_by_key(|x| -(x.1.time.nanos as i128));
154
155        enum RowKind {
156            Total,
157            Func,
158        }
159
160        let totals_str = ArcStr::new_static("TOTALS");
161        let info = [(&totals_str, &totals, RowKind::Total)]
162            .into_iter()
163            .chain(info.into_iter().map(|(k, v)| (k, v, RowKind::Func)));
164
165        let mut csv = CsvWriter::new(
166            [
167                "Function",
168                "Time(s)",
169                "TimeRec(s)",
170                "Calls",
171                "Callers",
172                "TopCaller",
173                "TopCallerCount",
174                "Allocs",
175                "AllocBytes",
176            ]
177            .iter()
178            .copied()
179            .chain(columns.iter().map(|c| c.0)),
180        );
181        for (rowname, info, _row_kind) in info {
182            let blank = ArcStr::new_static("");
183            let callers = info
184                .callers
185                .iter()
186                .max_by_key(|x| x.1)
187                .unwrap_or((&blank, &0));
188            assert!(
189                info.calls % 2 == 0,
190                "we enter calls twice, for drop and non_drop"
191            );
192            // We divide calls and time by two
193            // because we could calls twice: for drop and non-drop bumps.
194            csv.write_value(&**rowname);
195            csv.write_value(info.time / 2);
196            csv.write_value(info.time_rec / 2);
197            csv.write_value(info.calls / 2);
198            csv.write_value(info.callers.len());
199            csv.write_value(callers.0.as_str());
200            csv.write_value(callers.1);
201            csv.write_value(info.alloc_count());
202            csv.write_value(info.alloc_bytes());
203            for c in &columns {
204                csv.write_value(info.alloc.get(c.0).unwrap_or(&AllocCounts::default()).count);
205            }
206            csv.finish_row();
207        }
208        csv.finish()
209    }
210}

212#[cfg(test)]
213mod tests {
214    use crate::environment::Globals;
215    use crate::environment::Module;
216    use crate::eval::Evaluator;
217    use crate::eval::runtime::profile::mode::ProfileMode;
218    use crate::syntax::AstModule;
219    use crate::syntax::Dialect;
220    use crate::values::layout::heap::profile::aggregated::AggregateHeapProfileInfo;
221    use crate::values::layout::heap::profile::summary_by_function::FuncInfo;
222    use crate::values::layout::heap::profile::summary_by_function::HeapSummaryByFunction;
223
224    // Test data is collected from both drop and non-drop heaps.
225    #[test]
226    fn drop_non_drop() {
227        let ast = AstModule::parse(
228            "x.star",
229            "\
230_ignore = {1: 2}       # allocate a dict in drop
231_ignore = str([1])     # allocate a string in non_drop
232        "
233            .to_owned(),
234            &Dialect::AllOptionsInternal,
235        )
236        .unwrap();
237
238        let globals = Globals::standard();
239        let module = Module::new();
240        let mut eval = Evaluator::new(&module);
241        eval.enable_profile(&ProfileMode::HeapSummaryAllocated)
242            .unwrap();
243
244        eval.eval_module(ast, &globals).unwrap();
245
246        let stacks = AggregateHeapProfileInfo::collect(eval.heap(), None);
247
248        let info = HeapSummaryByFunction::init(&stacks);
249
250        // Run the assertions.
251        info.gen_csv();
252
253        let total = FuncInfo::merge(info.info.values());
254        // from non-drop heap
255        assert_eq!(total.alloc.get("string").unwrap().count, 1);
256        // from drop heap
257        assert_eq!(total.alloc.get("dict").unwrap().count, 1);
258    }
