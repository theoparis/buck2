26/// Collect `N` arguments.
27///
28/// This function is called by generated code.
29#[inline]
30pub fn parse_signature<'v, const N: usize>(
31    parser: &ParametersSpec<FrozenValue>,
32    args: &Arguments<'v, '_>,
33    heap: &'v Heap,
34) -> crate::Result<[Option<Value<'v>>; N]> {
35    parser.collect_into(args, heap)
36}
37
38/// Parse positional-only arguments, required and optional.
39#[inline(always)]
40pub fn parse_positional<'v, const R: usize, const O: usize>(
41    args: &Arguments<'v, '_>,
42    heap: &'v Heap,
43) -> crate::Result<([Value<'v>; R], [Option<Value<'v>>; O])> {
44    args.no_named_args()?;
45    args.optional(heap)
46}
47
48#[inline(always)]
49pub fn parse_positional_kwargs_alloc<'v, 'a, const R: usize, const O: usize>(
50    args: &'a Arguments<'v, 'a>,
51    heap: &'v Heap,
52) -> crate::Result<([Value<'v>; R], [Option<Value<'v>>; O], Value<'v>)> {
53    let (required, optional) = args.optional(heap)?;
54    let kwargs = args.names_map()?;
55    let kwargs = heap.alloc(kwargs);
56    Ok((required, optional, kwargs))
57}

59/// Utility for checking a `this` parameter matches what you expect.
60#[inline]
61pub fn check_this<'v, T: UnpackValue<'v>>(this: Value<'v>) -> crate::Result<T> {
62    T::unpack_named_param(this, "this")
63}
64
65/// Utility for checking a required parameter matches what you expect.
66#[inline]
67pub fn check_required<'v, T: UnpackValue<'v>>(
68    name: &str,
69    x: Option<Value<'v>>,
70) -> crate::Result<T> {
71    let x = x.ok_or_else(|| ValueError::MissingRequired(name.to_owned()))?;
72    T::unpack_named_param(x, name)
73}
74
75/// Utility for checking an optional parameter matches what you expect.
76#[inline]
77pub fn check_optional<'v, T: UnpackValue<'v>>(
78    name: &str,
79    x: Option<Value<'v>>,
80) -> crate::Result<Option<T>> {
81    match x {
82        None => Ok(None),
83        Some(x) => Ok(Some(T::unpack_named_param(x, name)?)),
84    }
85}
86
87#[inline]
88pub fn check_defaulted<'v, T: UnpackValue<'v>>(
89    name: &str,
90    x: Option<Value<'v>>,
91    default: impl FnOnce() -> T,
92) -> crate::Result<T> {
93    Ok(check_optional(name, x)?.unwrap_or_else(default))
94}
95
96/// We already know the parameter is set, so we just unpack it.
97#[inline]
98pub fn check_unpack<'v, T: UnpackValue<'v>>(name: &str, x: Value<'v>) -> crate::Result<T> {
99    T::unpack_named_param(x, name)
100}
