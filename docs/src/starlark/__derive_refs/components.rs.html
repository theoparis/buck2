20use crate::__derive_refs::param_spec::NativeCallableParam;
21use crate::__derive_refs::param_spec::NativeCallableParamDefaultValue;
22use crate::__derive_refs::param_spec::NativeCallableParamSpec;

24use crate::docs::DocItem;

27use crate::docs::DocParams;

29use crate::docs::DocType;
30use crate::eval::runtime::params::display::PARAM_FMT_OPTIONAL;
31use crate::typing::Ty;
32
33/// A wrapper for the parameters to `GlobalsBuilder::set_function` and `MethodBuilder::set_method`
34pub struct NativeCallableComponents {
35    pub speculative_exec_safe: bool,
36    pub rust_docstring: Option<&'static str>,
37    pub param_spec: NativeCallableParamSpec,
38    pub return_type: Ty,
39}
40
41impl NativeCallableComponents {
42    fn doc_params(&self) -> DocParams {
43        fn doc_param(p: &NativeCallableParam) -> DocParam {
44            let NativeCallableParam { name, ty, required } = p;
45            DocParam {
46                name: (*name).to_owned(),
47                docs: None,
48                typ: ty.dupe(),
49                default_value: match required {
50                    None => None,
51                    Some(NativeCallableParamDefaultValue::Optional) => {
52                        Some(PARAM_FMT_OPTIONAL.to_owned())
53                    }
54                    Some(NativeCallableParamDefaultValue::Value(v)) => Some(v.to_value().to_repr()),
55                },
56            }
57        }
58
59        DocParams {
60            pos_only: self.param_spec.pos_only.iter().map(doc_param).collect(),
61            pos_or_named: self.param_spec.pos_or_named.iter().map(doc_param).collect(),
62            args: self.param_spec.args.as_ref().map(doc_param),
63            named_only: self.param_spec.named_only.iter().map(doc_param).collect(),
64            kwargs: self.param_spec.kwargs.as_ref().map(doc_param),
65        }
66    }
67
68    pub(crate) fn into_docs(self, as_type: Option<(Ty, DocType)>) -> DocItem {
69        let func_docs = DocFunction::from_docstring(
70            DocStringKind::Rust,
71            self.doc_params(),
72            self.return_type.clone(),
73            self.rust_docstring,
74        );
75        match as_type {
76            Some((_, ty_docs)) => DocItem::Type(DocType {
77                constructor: Some(func_docs),
78                ..ty_docs
79            }),
80            None => DocItem::Member(DocMember::Function(func_docs)),
81        }
82    }
83}
