30#[derive(Error, Debug)]
31pub(crate) enum Performance {
32    #[error("Dict copy `{0}` is more efficient as `{1}`")]
33    DictWithoutStarStar(String, String),
34
35    #[error(
36        "`{0}` eagerly evaluates all items in the iterable, and allocates an array for the results. Prefer using a for-loop."
37    )]
38    EagerAndInefficientBoolCheck(String),
39
40    #[error("`{0}` allocates a new {1} for the results. Prefer using a for-loop.")]
41    InefficientBoolCheck(String, String),
42}
43
44impl LintWarning for Performance {
45    fn severity(&self) -> EvalSeverity {
46        EvalSeverity::Warning
47    }
48
49    fn short_name(&self) -> &'static str {
50        match self {
51            Performance::DictWithoutStarStar(..) => "dict-without-star-star",
52            Performance::EagerAndInefficientBoolCheck(..) => "eager-and-inefficient-bool-check",
53            Performance::InefficientBoolCheck(..) => "inefficient-bool-check",
54        }
55    }
56}
57
58fn match_dict_copy(codemap: &CodeMap, x: &AstExpr, res: &mut Vec<LintT<Performance>>) {
59    // If we see `dict(**x)` suggest `dict(x)`
60    match &**x {
61        Expr::Call(fun, args) if args.args.len() == 1 => match (&***fun, &*args.args[0]) {
62            (Expr::Identifier(f), Argument::KwArgs(arg)) if f.node.ident == "dict" => {
63                res.push(LintT::new(
64                    codemap,
65                    x.span,
66                    Performance::DictWithoutStarStar(x.to_string(), format!("dict({})", arg.node)),
67                ))
68            }
69            _ => {}
70        },
71        _ => {}
72    }
73}
74
75fn match_inefficient_bool_check(codemap: &CodeMap, x: &AstExpr, res: &mut Vec<LintT<Performance>>) {
76    match &**x {
77        Expr::Call(fun, args) if args.args.len() == 1 => match (&***fun, &*args.args[0]) {
78            (Expr::Identifier(f), Argument::Positional(arg))
79                if f.node.ident == "any" || f.node.ident == "all" =>
80            {
81                match &**arg {
82                    // any([blah for blah in blahs])
83                    Expr::ListComprehension(_, _, _) | Expr::DictComprehension(_, _, _) => res
84                        .push(LintT::new(
85                            codemap,
86                            x.span,
87                            Performance::EagerAndInefficientBoolCheck(f.node.ident.clone()),
88                        )),
89                    // any(list(_get_some_dict()))
90                    Expr::Call(any_call, _) => match &***any_call {
91                        Expr::Identifier(any_id)
92                            if any_id.node.ident == "dict" || any_id.node.ident == "list" =>
93                        {
94                            res.push(LintT::new(
95                                codemap,
96                                x.span,
97                                Performance::InefficientBoolCheck(
98                                    x.to_string(),
99                                    any_id.node.ident.clone(),
100                                ),
101                            ))
102                        }
103                        _ => {}
104                    },
105                    _ => {}
106                }
107            }
108            _ => {}
109        },
110        _ => {}
111    }
112}
113
114fn check_call_expr(module: &AstModule, res: &mut Vec<LintT<Performance>>) {
115    fn check(codemap: &CodeMap, x: &AstExpr, res: &mut Vec<LintT<Performance>>) {
116        match_dict_copy(codemap, x, res);
117        match_inefficient_bool_check(codemap, x, res);
118        x.visit_expr(|x| check(codemap, x, res));
119    }
120    module
121        .statement()
122        .visit_expr(|x| check(module.codemap(), x, res));
123}
124
125pub(crate) fn lint(module: &AstModule) -> Vec<LintT<Performance>> {
126    let mut res = Vec::new();
127    check_call_expr(module, &mut res);
128    res
129}

131#[cfg(test)]
132mod tests {
133    use starlark_syntax::slice_vec_ext::SliceExt;
134
135    use super::*;
136    use crate::syntax::Dialect;
137
138    fn module(x: &str) -> AstModule {
139        AstModule::parse("bad.bzl", x.to_owned(), &Dialect::AllOptionsInternal).unwrap()
140    }
141
142    #[test]
143    fn test_lint_matches_dict_issue() {
144        let mut res = Vec::new();
145        check_call_expr(
146            &module(
147                r#"
148def foo(extra, **kwargs):
149    x = dict(**kwargs)
150    y = dict(extra)
151    return (x,y)
152"#,
153            ),
154            &mut res,
155        );
156        assert_eq!(
157            res.map(|x| x.to_string()),
158            &["bad.bzl:3:9-23: Dict copy `dict(**kwargs)` is more efficient as `dict(kwargs)`"]
159        );
160    }
161
162    #[test]
163    fn test_lint_matches_any_function() {
164        let mut res = Vec::new();
165        check_call_expr(
166            &module(
167                r#"
168def foo(items):
169    a = all(items)
170    b = all([item for item in items])
171    c = any([item for item in items])
172    d = all({"a": a for a in []})
173    e = any(list({}))
174    f = all(dict([]))
175    return (a,b,c,d,e,f)
176"#,
177            ),
178            &mut res,
179        );
180        assert_eq!(
181            res.map(|x| x.to_string()),
182            &[
183                "bad.bzl:4:9-38: `all` eagerly evaluates all items in the iterable, and allocates an array for the results. Prefer using a for-loop.",
184                "bad.bzl:5:9-38: `any` eagerly evaluates all items in the iterable, and allocates an array for the results. Prefer using a for-loop.",
185                "bad.bzl:6:9-34: `all` eagerly evaluates all items in the iterable, and allocates an array for the results. Prefer using a for-loop.",
186                "bad.bzl:7:9-22: `any(list({}))` allocates a new list for the results. Prefer using a for-loop.",
187                "bad.bzl:8:9-22: `all(dict([]))` allocates a new dict for the results. Prefer using a for-loop."
188            ]
189        );
190    }
191}
