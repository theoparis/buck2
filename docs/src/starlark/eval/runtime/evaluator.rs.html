120/// Holds everything about an ongoing evaluation (local variables, globals, module resolution etc).
121pub struct Evaluator<'v, 'a, 'e> {
122    // The module that is being used for this evaluation
123    pub(crate) module_env: &'v Module,
124    /// Current function (`def` or `lambda`) frame: locals and bytecode stack.
125    pub(crate) current_frame: BcFramePtr<'v>,
126    // How we deal with a `load` function.
127    pub(crate) loader: Option<&'a dyn FileLoader>,
128    // `DefInfo` of currently executed module.
129    // `DefInfo` of currently execution function can be obtained from call stack.
130    pub(crate) module_def_info: FrozenRef<'static, DefInfo>,
131    // Should we enable heap profiling or not
132    pub(crate) heap_profile: HeapProfile,
133    // Should we enable flame profiling or not
134    pub(crate) time_flame_profile: TimeFlameProfile<'v>,
135    // Is GC disabled for some reason
136    pub(crate) disable_gc: bool,
137    // If true, the interpreter prints to stderr on GC.
138    // This is used for debugging.
139    pub(crate) verbose_gc: bool,
140    // Size of the heap when we should next perform a GC.
141    pub(crate) next_gc_level: usize,
142    /// Run static typechecking of the module being evaluated.
143    pub(crate) static_typechecking: bool,
144    // Profiling or instrumentation enabled.
145    pub(crate) profile_or_instrumentation_mode: ProfileOrInstrumentationMode,
146    // Used for line profiling
147    stmt_profile: StmtProfile,
148    // Holds things that require hooking into evaluation.
149    eval_instrumentation: EvaluationInstrumentation<'a, 'e>,
150    // Total time spent in runtime typechecking.
151    // Filled only if runtime typechecking profiling is enabled.
152    pub(crate) typecheck_profile: TypecheckProfile,
153    // Used for stack-like allocation
154    alloca: Alloca,
155    // Another stack-like allocation
156    pub(crate) string_pool: StringPool,
157    /// Field that can be used for any purpose you want (can store types you define).
158    /// Typically accessed via native functions you also define.
159    pub extra: Option<&'a dyn AnyLifetime<'e>>,
160    /// Called to perform console IO each time `breakpoint` function is called.
161    pub(crate) breakpoint_handler:
162        Option<Box<dyn Fn() -> anyhow::Result<Box<dyn BreakpointConsole>>>>,
163    /// Use in implementation of `print` function.
164    pub(crate) print_handler: &'a (dyn PrintHandler + 'a),
165    /// Deprecation handler.
166    pub(crate) soft_error_handler: &'a (dyn SoftErrorHandler + 'a),
167    /// Max size of starlark stack
168    pub(crate) max_callstack_size: Option<usize>,
169    // The Starlark-level call-stack of functions.
170    // Must go last because it's quite a big structure
171    pub(crate) call_stack: CheapCallStack<'v>,
172    /// Function to check if evaluation should be cancelled early
173    pub(crate) is_cancelled: Box<dyn Fn() -> bool + 'a>,
174    /// A counter to track when to perform "infrequent" checks like cancellation, timeouts, etc
175    pub(crate) infrequent_instr_check_counter: u32,
176}

199/// Just holds things that require using EvaluationCallbacksEnabled so that we can cache whether that needs to be enabled or not.
200struct EvaluationInstrumentation<'a, 'e: 'a> {
201    // Bytecode profile.
202    bc_profile: BcProfile,
203    // Extra functions to run on each statement, usually empty
204    before_stmt: BeforeStmt<'a, 'e>,
205    heap_or_flame_profile: bool,
206    // Whether we need to instrument evaluation or not, should be set if before_stmt or bc_profile are enabled.
207    enabled: bool,
208}
209
210impl<'a, 'e: 'a> EvaluationInstrumentation<'a, 'e> {
211    fn new() -> EvaluationInstrumentation<'a, 'e> {
212        Self {
213            bc_profile: BcProfile::new(),
214            before_stmt: BeforeStmt::default(),
215            heap_or_flame_profile: false,
216            enabled: false,
217        }
218    }
219
220    fn enable_heap_or_flame_profile(&mut self) {
221        self.heap_or_flame_profile = true;
222    }
223
224    fn change<F: FnOnce(&mut EvaluationInstrumentation<'a, 'e>) -> R, R>(&mut self, f: F) -> R {
225        let r = f(self);
226        self.enabled =
227            self.bc_profile.enabled() || self.before_stmt.enabled() || self.heap_or_flame_profile;
228        r
229    }
230}

302    /// Enable profiling, allowing [`Evaluator::gen_profile`] to be used.
303    /// Profilers add overhead, and while some profilers can be used together,
304    /// it's better to run at most one profiler at a time.
305    pub fn enable_profile(&mut self, mode: &ProfileMode) -> anyhow::Result<()> {
306        if self.profile_or_instrumentation_mode != ProfileOrInstrumentationMode::None {
307            return Err(EvaluatorError::ProfileOrInstrumentationAlreadyEnabled.into());
308        }
309
310        self.profile_or_instrumentation_mode = ProfileOrInstrumentationMode::Profile(mode.dupe());
311
312        match mode {
313            ProfileMode::HeapAllocated
314            | ProfileMode::HeapRetained
315            | ProfileMode::HeapSummaryAllocated
316            | ProfileMode::HeapFlameAllocated
317            | ProfileMode::HeapSummaryRetained
318            | ProfileMode::HeapFlameRetained => {
319                self.heap_profile.enable();
320
321                match mode {
322                    ProfileMode::HeapFlameRetained => self
323                        .module_env
324                        .enable_retained_heap_profile(RetainedHeapProfileMode::Flame),
325                    ProfileMode::HeapSummaryRetained => self
326                        .module_env
327                        .enable_retained_heap_profile(RetainedHeapProfileMode::Summary),
328                    ProfileMode::HeapRetained => {
329                        self.module_env
330                            .enable_retained_heap_profile(RetainedHeapProfileMode::FlameAndSummary);
331                    }
332                    _ => {}
333                }
334
335                self.eval_instrumentation
336                    .change(|v| v.enable_heap_or_flame_profile());
337
338                // Disable GC because otherwise why lose the profile records, as we use the heap
339                // to store a complete list of what happened in linear order.
340                self.disable_gc = true;
341            }
342            ProfileMode::Statement | ProfileMode::Coverage => {
343                self.stmt_profile.enable();
344                self.before_stmt_fn(&|span, _continued, eval| eval.stmt_profile.before_stmt(span));
345            }
346            ProfileMode::TimeFlame => {
347                self.time_flame_profile.enable();
348                self.eval_instrumentation
349                    .change(|v| v.enable_heap_or_flame_profile());
350            }
351            ProfileMode::Bytecode => {
352                self.eval_instrumentation
353                    .change(|v| v.bc_profile.enable_1());
354            }
355            ProfileMode::BytecodePairs => {
356                self.eval_instrumentation
357                    .change(|v| v.bc_profile.enable_2());
358            }
359            ProfileMode::Typecheck => {
360                self.typecheck_profile.enabled = true;
361            }
362            ProfileMode::None => {}
363        }
364        Ok(())
365    }

367    /// Generate profile for a given mode.
368    /// Only valid if corresponding profiler was enabled.
369    pub fn gen_profile(&mut self) -> crate::Result<ProfileData> {
370        let mode = match &self.profile_or_instrumentation_mode {
371            ProfileOrInstrumentationMode::None => {
372                return Err(crate::Error::new_other(EvaluatorError::ProfilingNotEnabled));
373            }
374            ProfileOrInstrumentationMode::Collected => {
375                return Err(crate::Error::new_other(
376                    EvaluatorError::ProfileDataAlreadyCollected,
377                ));
378            }
379            ProfileOrInstrumentationMode::Profile(mode) => mode.dupe(),
380        };
381        self.profile_or_instrumentation_mode = ProfileOrInstrumentationMode::Collected;
382        match mode {
383            ProfileMode::HeapAllocated => self
384                .heap_profile
385                .r#gen(self.heap(), HeapProfileFormat::FlameGraphAndSummary),
386            ProfileMode::HeapSummaryAllocated => self
387                .heap_profile
388                .r#gen(self.heap(), HeapProfileFormat::Summary),
389            ProfileMode::HeapFlameAllocated => self
390                .heap_profile
391                .r#gen(self.heap(), HeapProfileFormat::FlameGraph),
392            ProfileMode::HeapSummaryRetained
393            | ProfileMode::HeapFlameRetained
394            | ProfileMode::HeapRetained => Err(crate::Error::new_other(
395                EvaluatorError::RetainedMemoryProfilingCannotBeObtainedFromEvaluator,
396            )),
397            ProfileMode::Statement => self.stmt_profile.r#gen(),
398            ProfileMode::Coverage => self.stmt_profile.gen_coverage(),
399            ProfileMode::Bytecode => self.gen_bc_profile(),
400            ProfileMode::BytecodePairs => self.gen_bc_pairs_profile(),
401            ProfileMode::TimeFlame => self.time_flame_profile.r#gen(),
402            ProfileMode::Typecheck => self.typecheck_profile.r#gen(),
403            ProfileMode::None => Ok(ProfileData {
404                profile: ProfileDataImpl::None,
405            }),
406        }
407    }

489    /// Called to add an entry to the call stack, by the function being invoked.
490    /// Called for all types of function, including those written in Rust.
491    #[inline(always)]
492    pub(crate) fn with_call_stack<R>(
493        &mut self,
494        function: Value<'v>,
495        span: Option<FrozenRef<'static, FrameSpan>>,
496        within: impl FnOnce(&mut Self) -> crate::Result<R>,
497    ) -> crate::Result<R> {
498        #[cold]
499        #[inline(never)]
500        fn add_diagnostics(mut e: crate::Error, me: &Evaluator) -> crate::Error {
501            // Make sure we capture the call_stack before popping things off it
502            e.set_call_stack(|| me.call_stack.to_diagnostic_frames(InlinedFrames::default()));
503            e
504        }
505
506        self.call_stack.push(function, span)?;
507        // Must always call .pop regardless
508        let res = within(self).map_err(|e| add_diagnostics(e, self));
509        self.call_stack.pop();
510        res
511    }

532    pub(crate) fn get_slot_module(&self, slot: ModuleSlotId) -> crate::Result<Value<'v>> {
533        // Make sure the error-path doesn't get inlined into the normal-path execution
534        #[cold]
535        #[inline(never)]
536        fn error<'v>(eval: &Evaluator<'v, '_, '_>, slot: ModuleSlotId) -> crate::Error {
537            let name = match eval.top_frame_def_frozen_module(false) {
538                Err(e) => Some(format!("<internal error: {e}>")),
539                Ok(None) => eval
540                    .module_env
541                    .mutable_names()
542                    .get_slot(slot)
543                    .map(|s| s.as_str().to_owned()),
544                Ok(Some(e)) => e.get_slot_name(slot).map(|s| s.as_str().to_owned()),
545            }
546            .unwrap_or_else(|| "<unknown>".to_owned());
547            crate::Error::new_other(EvaluatorError::LocalVariableReferencedBeforeAssignment(
548                name,
549            ))
550        }
551
552        match self.top_frame_def_frozen_module(false)? {
553            None => self.module_env.slots().get_slot(slot),
554            Some(e) => e.get_slot(slot).map(Value::new_frozen),
555        }
556        .ok_or_else(|| error(self, slot))
557    }

559    // Make sure the error-path doesn't get inlined into the normal-path execution
560    #[cold]
561    #[inline(never)]
562    pub(crate) fn local_var_referenced_before_assignment(&self, slot: LocalSlotId) -> crate::Error {
563        let def_info = match self.top_frame_def_info() {
564            Ok(def_info) => def_info,
565            Err(e) => return e,
566        };
567        let names = &def_info.used;
568        let name = names[slot.0 as usize].as_str().to_owned();
569        crate::Error::new_other(EvaluatorError::LocalVariableReferencedBeforeAssignment(
570            name,
571        ))
572    }

579    ) -> crate::Result<Value<'v>> {
580        // We access locals from explicitly passed frame because it is faster.
581        debug_assert!(self.current_frame == frame);
582
583        frame
584            .get_slot(slot.to_captured_or_not())
585            .ok_or_else(|| self.local_var_referenced_before_assignment(slot))
586    }

593        let value_captured = value_captured_get(value_captured);
594        value_captured

598    pub(crate) fn clone_slot_capture(
599        &self,
600        copy: &CopySlotFromParent,
601        target_def_info: &DefInfo,
602    ) -> Value<'v> {
603        match self.current_frame.get_slot(copy.parent) {
604            Some(value_captured) => {
605                debug_assert!(
606                    value_captured.downcast_ref::<ValueCaptured>().is_some()
607                        || value_captured
608                            .downcast_ref::<FrozenValueCaptured>()
609                            .is_some(),
610                    "slot {} ({}) is expected to be ValueCaptured, it is {:?} ({}); \
611                        def location: {}",
612                    copy.parent.0,
613                    target_def_info
614                        .used
615                        .get(copy.child.0 as usize)
616                        .copied()
617                        .unwrap_or_default()
618                        .as_str(),
619                    value_captured,
620                    value_captured.get_type(),
621                    target_def_info.signature_span,
622                );
623                value_captured
624            }
625            None => {
626                let value_captured = self.heap().alloc_complex(ValueCaptured::new(None));
627                self.current_frame.set_slot(copy.parent, value_captured);
628                value_captured
629            }
630        }
631    }

685        let func = self.call_stack.top_nth_function(0)?;
686        if let Some(func) = func.downcast_ref::<Def>() {
687            func.check_return_type(ret, self)
688        } else if let Some(func) = func.downcast_ref::<FrozenDef>() {

692        }

695    fn func_to_def_info(&self, func: Value<'_>) -> crate::Result<FrozenRef<DefInfo>> {
696        if let Some(func) = func.downcast_ref::<Def>() {
697            Ok(func.def_info)
698        } else if let Some(func) = func.downcast_ref::<FrozenDef>() {
699            Ok(func.def_info)
700        } else if func.is_none() {
701            // For module, it is `None`.
702            Ok(self.module_def_info)
703        } else {
704            Err(crate::Error::new_other(EvaluatorError::TopFrameNotDef))
705        }

715        for_debugger: bool,

757    /// Perform a garbage collection.
758    /// After this operation all [`Value`]s not reachable from the evaluator will be invalid,
759    /// and using them will lead to a segfault.
760    /// Do not call during Starlark evaluation.
761    pub unsafe fn garbage_collect(&mut self) {
762        unsafe {
763            if self.verbose_gc {
764                eprintln!(
765                    "Starlark: allocated bytes: {}, starting GC...",
766                    self.heap().allocated_bytes()
767                );
768            }
769
770            self.stmt_profile
771                .before_stmt(rust_loc!().span.file_span_ref());
772
773            self.time_flame_profile
774                .record_call_enter(const_frozen_string!("GC").to_value());
775
776            self.heap().garbage_collect(|tracer| self.trace(tracer));
777
778            self.time_flame_profile.record_call_exit();
779
780            if self.verbose_gc {
781                eprintln!(
782                    "Starlark: GC complete. Allocated bytes: {}.",
783                    self.heap().allocated_bytes()
784                );
785            }
786        }
787    }

789    /// Note that the `Drop` for the `T` will not be called. That's safe if there is no `Drop`,
790    /// or you call it yourself.
791    #[inline(always)]
792    pub(crate) fn alloca_uninit<T, R, F>(&mut self, len: usize, k: F) -> R
793    where
794        F: FnOnce(&mut [MaybeUninit<T>], &mut Self) -> R,
795    {
796        // We want to be able to access the evaluator underneath the alloca.
797        // We know that the alloca will be used in a stacked way, so that's fine.
798        let alloca = unsafe { cast::ptr_lifetime(&self.alloca) };
799        alloca.alloca_uninit(len, |xs| k(xs, self))
800    }

830    #[cold]
831    #[inline(never)]
832    fn eval_bc_with_callbacks(
833        &mut self,
834        def: Value<'v>,
835        bc: &Bc,
836    ) -> Result<Value<'v>, EvalException> {
837        debug_assert!(self.eval_instrumentation.enabled);
838        if self.eval_instrumentation.heap_or_flame_profile {
839            self.heap_profile.record_call_enter(def, self.heap());
840            self.time_flame_profile.record_call_enter(def);
841            let res = bc.run(self, &mut EvalCallbacksDisabled);
842            self.heap_profile.record_call_exit(self.heap());
843            self.time_flame_profile.record_call_exit();
844            res
845        } else {
846            bc.run(
847                self,
848                &mut EvalCallbacksEnabled {
849                    mode: match (
850                        self.eval_instrumentation.before_stmt.enabled(),
851                        self.eval_instrumentation.bc_profile.enabled(),
852                    ) {
853                        (true, false) => EvalCallbacksMode::BeforeStmt,
854                        (false, true) => EvalCallbacksMode::BcProfile,
855                        (true, true) => {
856                            return Err(EvalException::new_unknown_span(internal_error!(
857                                "both before_stmt and bc_profile are enabled"
858                            )));
859                        }
860                        (false, false) => {
861                            return Err(EvalException::new_unknown_span(internal_error!(
862                                "neither before_stmt nor bc_profile are enabled"
863                            )));
864                        }
865                    },
866                    stmt_locs: &bc.instrs.stmt_locs,
867                    bc_start_ptr: bc.instrs.start_ptr(),
868                },
869            )
870        }
871    }

873    #[inline(always)]
874    pub(crate) fn eval_bc(&mut self, def: Value<'v>, bc: &Bc) -> Result<Value<'v>, EvalException> {
875        if self.eval_instrumentation.enabled {
876            self.eval_bc_with_callbacks(def, bc)
877        } else {
878            bc.run(self, &mut EvalCallbacksDisabled)
879        }

933pub(crate) enum EvalCallbacksMode {
934    BcProfile,
935    BeforeStmt,
936}
937
938pub(crate) struct EvalCallbacksEnabled<'a> {
939    pub(crate) mode: EvalCallbacksMode,
940    pub(crate) stmt_locs: &'a BcStatementLocations,
941    pub(crate) bc_start_ptr: BcPtrAddr<'a>,
942}
943
944impl<'a> EvalCallbacksEnabled<'a> {
945    fn before_stmt(&mut self, eval: &mut Evaluator, ip: BcPtrAddr) -> crate::Result<()> {
946        let offset = ip.offset_from(self.bc_start_ptr);
947        if let Some((loc, continued)) = self.stmt_locs.stmt_at(offset) {
948            before_stmt(loc.span, continued, eval)?;
949        }
950        Ok(())
951    }
952}
953
954impl<'a> EvaluationCallbacks for EvalCallbacksEnabled<'a> {
955    #[inline(always)]
956    fn before_instr(
957        &mut self,
958        eval: &mut Evaluator,
959        ip: BcPtrAddr,
960        opcode: BcOpcode,
961    ) -> crate::Result<()> {
962        match self.mode {
963            EvalCallbacksMode::BcProfile => {
964                eval.eval_instrumentation.bc_profile.before_instr(opcode);
965                Ok(())
966            }
967            EvalCallbacksMode::BeforeStmt => self.before_stmt(eval, ip),
968        }
969    }

972// This function should be called before every meaningful statement (continued==false), and after a call that returns into a previously entered statement (continued==true).
973// The purposes are GC, profiling and debugging.
974//
975// This function is called only if `before_stmt` is set before compilation start.
976pub(crate) fn before_stmt(
977    span: FrameSpan,
978    continued: bool,
979    eval: &mut Evaluator,
980) -> crate::Result<()> {
981    assert!(
982        eval.eval_instrumentation.before_stmt.enabled(),
983        "this code should only be called if `before_stmt` is set"
984    );
985    let mut fs = eval.eval_instrumentation.change(|eval_instrumentation| {
986        mem::take(&mut eval_instrumentation.before_stmt.before_stmt)
987    });
988    let mut result = Ok(());
989    for f in &mut fs {
990        if result.is_ok() {
991            result = f.call(span.span.file_span_ref(), continued, eval);
992        }
993    }
994    let added = eval.eval_instrumentation.change(|eval_instrumentation| {
995        mem::replace(&mut eval_instrumentation.before_stmt.before_stmt, fs)
996    });
997    assert!(
998        added.is_empty(),
999        "`before_stmt` cannot be modified during evaluation"
1000    );
1001    result
