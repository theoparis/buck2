37// A value akin to Frame, but can be created cheaply, since it doesn't resolve
38// anything in advance.
39// The downside is it has a lifetime on 'v and keeps alive the whole CodeMap.
40#[derive(Clone, Copy, Dupe)]
41struct CheapFrame<'v> {
42    function: Value<'v>,
43    span: Option<FrozenRef<'static, FrameSpan>>,
44}
45
46impl CheapFrame<'_> {
47    fn location(&self) -> Option<FileSpan> {
48        self.span.map(|span| span.span.to_file_span())
49    }
50
51    fn extend_frames(&self, frames: &mut Vec<Frame>) {
52        if let Some(span) = self.span {
53            span.inlined_frames.extend_frames(frames);
54        }
55        frames.push(self.to_frame());
56    }
57
58    fn to_frame(&self) -> Frame {
59        Frame {
60            name: self.function.name_for_call_stack(),
61            location: self.location(),
62        }
63    }
64}
65
66impl Debug for CheapFrame<'_> {
67    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
68        let mut x = f.debug_struct("Frame");
69        x.field("function", &self.function);
70        x.field("span", &self.span);
71        x.finish()
72    }
73}

75#[derive(Debug, thiserror::Error)]
76enum CallStackError {
77    #[error("Requested {0}-th top frame, but stack size is {1} (internal error)")]
78    StackIsTooShallowForNthTopFrame(usize, usize),
79    #[error("Starlark call stack overflow")]
80    Overflow,
81    #[error("Starlark call stack is already allocated")]
82    AlreadyAllocated,
83}
84
85/// Starlark call stack.
86#[derive(Debug)]
87pub(crate) struct CheapCallStack<'v> {
88    count: usize,
89    stack: Box<[CheapFrame<'v>]>,
90}
91
92impl<'v> Default for CheapCallStack<'v> {
93    fn default() -> Self {
94        Self {
95            count: 0,
96            stack: Box::new(
97                [CheapFrame {
98                    function: Value::new_none(),
99                    span: None,
100                }; 0],
101            ),
102        }
103    }
104}
105
106unsafe impl<'v> Trace<'v> for CheapCallStack<'v> {
107    fn trace(&mut self, tracer: &Tracer<'v>) {
108        let (used, unused) = self.stack.split_at_mut(self.count);
109        for x in used {
110            x.function.trace(tracer);
111        }
112        // Not required, but since we are choosing not to walk those above
113        // the current stack depth, it's good practice to blank those values out
114        for x in unused {
115            x.function = Value::new_none();
116            x.span = None;
117        }
118    }
119}
120
121impl<'v> CheapCallStack<'v> {
122    // Currently, each frame typically allocates about 1K of native stack size (see `test_frame_size`),
123    // but it is a bit more complicated:
124    // * each for loop in a frame allocates more native stack
125    // * inlined functions do not allocate native stack
126    // Practically max call stack depends on native stack size,
127    // and depending on environment, it may be configured differently, for example:
128    // * macOS default stack size is 512KB
129    // * Linux default stack size is 8MB
130    // * [tokio default stack size is 2MB][1]
131    // [1] https://docs.rs/tokio/0.2.1/tokio/runtime/struct.Builder.html#method.thread_stack_size
132    pub(crate) fn alloc_if_needed(&mut self, max_size: usize) -> anyhow::Result<()> {
133        if !self.stack.is_empty() {
134            return if self.stack.len() == max_size {
135                Ok(())
136            } else {
137                Err(CallStackError::AlreadyAllocated.into())
138            };
139        }
140
141        self.stack = vec![
142            CheapFrame {
143                function: Value::new_none(),
144                span: None,
145            };
146            max_size
147        ]
148        .into_boxed_slice();
149        Ok(())
150    }
151
152    /// Push an element to the stack. It is important the each `push` is paired
153    /// with a `pop`.
154    pub(crate) fn push(
155        &mut self,
156        function: Value<'v>,
157        span: Option<FrozenRef<'static, FrameSpan>>,
158    ) -> crate::Result<()> {
159        if unlikely(self.count >= self.stack.len()) {
160            return Err(crate::Error::new_kind(ErrorKind::StackOverflow(
161                CallStackError::Overflow.into(),
162            )));
163        }
164        self.stack[self.count] = CheapFrame { function, span };
165        self.count += 1;
166        Ok(())
167    }
168
169    /// Remove the top element from the stack. Called after `push`.
170    pub(crate) fn pop(&mut self) {
171        debug_assert!(self.count >= 1);
172        // We could clear the elements, but don't need to bother
173        self.count -= 1;
174    }
175
176    /// Current size (in frames) of the stack.
177    pub(crate) fn count(&self) -> usize {
178        self.count
179    }
180
181    /// The frame at the top of the stack. May be `None` if
182    /// either there the stack is empty, or the top of the stack lacks location
183    /// information (e.g. called from Rust).
184    pub(crate) fn top_frame(&self) -> Option<Frame> {
185        Some(self.stack.last().as_ref()?.to_frame())
186    }
187
188    /// The location at the top of the stack. May be `None` if
189    /// either there the stack is empty, or the top of the stack lacks location
190    /// information (e.g. called from Rust).
191    pub(crate) fn top_location(&self) -> Option<FileSpan> {
192        if self.count == 0 {
193            None
194        } else {
195            self.stack[self.count - 1].location()
196        }
197    }
198
199    /// `n`-th element from the top of the stack.
200    pub(crate) fn top_nth_function(&self, n: usize) -> anyhow::Result<Value<'v>> {
201        self.top_nth_function_opt(n)
202            .ok_or_else(|| CallStackError::StackIsTooShallowForNthTopFrame(n, self.count).into())
203    }
204
205    pub(crate) fn top_nth_function_opt(&self, n: usize) -> Option<Value<'v>> {
206        let index = self.count.checked_sub(1).and_then(|x| x.checked_sub(n))?;
207        Some(self.stack[index].function)
208    }
209
210    pub(crate) fn to_diagnostic_frames(&self, inlined_frames: InlinedFrames) -> CallStack {
211        // The first entry is just the entire module, so skip it
212        let mut frames = Vec::new();
213        for frame in &self.stack[1..self.count] {
214            frame.extend_frames(&mut frames);
215        }
216        inlined_frames.extend_frames(&mut frames);
217        CallStack { frames }
218    }
219
220    /// List the entries on the stack as values
221    pub(crate) fn to_function_values(&self) -> Vec<Value<'v>> {
222        self.stack[1..self.count].map(|x| x.function)
223    }
224}
