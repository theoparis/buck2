70#[derive(Hash, PartialEq, Eq, Clone, Copy, Dupe)]
71struct MutableValueId(usize);
72#[derive(Hash, PartialEq, Eq, Clone, Copy, Dupe)]
73struct FrozenValueId(usize);
74
75/// Index into FlameData.values
76#[derive(Hash, PartialEq, Eq, Clone, Copy, Dupe)]
77enum ValueId {
78    // This struct is two words, can be made one.
79    Mutable(MutableValueId),
80    Frozen(FrozenValueId),
81}
82
83impl ValueId {
84    fn lookup<'a, T>(self, mutable: &'a [T], frozen: &'a [T]) -> &'a T {
85        match self {
86            ValueId::Mutable(x) => &mutable[x.0],
87            ValueId::Frozen(x) => &frozen[x.0],
88        }
89    }
90}
91
92/// Bimap between `Value` and `ValueId`.
93/// In order to optimise GC (which otherwise quickly becomes O(n^2)) we have to
94/// dedupe the values, so store them in `values`, with a fast map to get them in `map`.
95/// Whenever we GC, regenerate map.
96#[derive(Default)]
97struct ValueIndex<'v> {
98    /// Map from `MutableValueId` to `Value`.
99    mutable_values: Vec<Value<'v>>,
100    /// Map from `FrozenValueId` to `Value`.
101    frozen_values: Vec<FrozenValue>,
102    /// Map from `Value` to `MutableValueId`.
103    mutable_map: HashMap<RawPointer, MutableValueId, StarlarkHasherBuilder>,
104    /// Map from `Value` to `FrozenValueId`.
105    frozen_map: HashMap<RawPointer, FrozenValueId, StarlarkHasherBuilder>,
106}
107
108unsafe impl<'v> Trace<'v> for ValueIndex<'v> {
109    fn trace(&mut self, tracer: &Tracer<'v>) {
110        // We only need to trace mutable values.
111        self.mutable_values.trace(tracer);
112        // Have to rebuild the map, as its keyed by ValuePtr which changes on GC
113        self.mutable_map.clear();
114        for (i, x) in self.mutable_values.iter().enumerate() {
115            self.mutable_map.insert(x.ptr_value(), MutableValueId(i));
116        }
117    }
118}
119
120impl<'v> ValueIndex<'v> {
121    /// Map `Value` to `ValueId`.
122    fn index(&mut self, value: Value<'v>) -> ValueId {
123        match value.unpack_frozen() {
124            Some(frozen) => match self.frozen_map.entry(frozen.ptr_value()) {
125                Entry::Occupied(e) => ValueId::Frozen(*e.get()),
126                Entry::Vacant(e) => {
127                    let res = FrozenValueId(self.frozen_values.len());
128                    self.frozen_values.push(frozen);
129                    e.insert(res);
130                    ValueId::Frozen(res)
131                }
132            },
133            None => match self.mutable_map.entry(value.ptr_value()) {
134                Entry::Occupied(e) => ValueId::Mutable(*e.get()),
135                Entry::Vacant(e) => {
136                    let res = MutableValueId(self.mutable_values.len());
137                    self.mutable_values.push(value);
138                    e.insert(res);
139                    ValueId::Mutable(res)
140                }
141            },
142        }
143    }
144}

146enum Frame {
147    /// Entry recorded when we enter a function.
148    Push(ValueId),
149    /// Entry recorded when we exit a function.
150    Pop,
151}

153#[derive(Trace)]
154pub(crate) struct TimeFlameProfile<'v>(
155    /// `Some` means enabled.
156    Option<Box<FlameData<'v>>>,
157);

159#[derive(Default, Trace)]
160struct FlameData<'v> {
161    /// All events in the profile, i.e. function entry or exit with timestamp.
162    frames: Vec<(Frame, ProfilerInstant)>,
163    index: ValueIndex<'v>,
164}
165
166struct Stacks<'a> {
167    name: &'a str,
168    time: SmallDuration,
169    children: HashMap<ValueId, Stacks<'a>, StarlarkHasherBuilder>,
170}
171
172impl<'a> Stacks<'a> {
173    fn blank(name: &'a str) -> Self {
174        Stacks {
175            name,
176            time: SmallDuration::default(),
177            children: HashMap::with_hasher(StarlarkHasherBuilder),
178        }
179    }
180
181    fn new(
182        mutable_names: &'a [String],
183        frozen_names: &'a [String],
184        frames: &[(Frame, ProfilerInstant)],
185    ) -> Self {
186        let mut res = Stacks::blank("root");
187        let Some(mut last_time) = frames.first().map(|x| x.1) else {
188            return res;
189        };
190        res.add(
191            mutable_names,
192            frozen_names,
193            &mut frames.iter(),
194            &mut last_time,
195        );
196        res
197    }
198
199    fn add(
200        &mut self,
201        mutable_names: &'a [String],
202        frozen_names: &'a [String],
203        frames: &mut slice::Iter<(Frame, ProfilerInstant)>,
204        last_time: &mut ProfilerInstant,
205    ) {
206        while let Some((frame, time)) = frames.next() {
207            self.time += time.duration_since(*last_time);
208            *last_time = *time;
209            match frame {
210                Frame::Pop => return,
211                Frame::Push(i) => match self.children.entry(*i) {
212                    Entry::Occupied(mut e) => {
213                        e.get_mut()
214                            .add(mutable_names, frozen_names, frames, last_time)
215                    }
216                    Entry::Vacant(e) => e
217                        .insert(Stacks::blank(
218                            i.lookup(mutable_names, frozen_names).as_str(),
219                        ))
220                        .add(mutable_names, frozen_names, frames, last_time),
221                },
222            }
223        }
224    }
225
226    fn render_with_buffer(&self, node: &mut FlameGraphNode) {
227        let node = node.child(ArcStr::from(self.name));
228        let count = self.time.to_duration().as_millis();
229        if count > 0 {
230            node.add(count as u64);
231        }
232        for x in self.children.values() {
233            x.render_with_buffer(node);
234        }
235    }
236
237    fn render(&self) -> FlameGraphData {
238        let mut data = FlameGraphData::default();
239        self.render_with_buffer(data.root());
240        data
241    }
242}

244impl<'v> TimeFlameProfile<'v> {
245    pub(crate) fn new() -> Self {
246        Self(None)
247    }
248
249    pub(crate) fn enable(&mut self) {
250        self.0 = Some(Box::default());
251    }
252
253    pub(crate) fn record_call_enter(&mut self, function: Value<'v>) {
254        if let Some(x) = &mut self.0 {
255            let ind = x.index.index(function);
256            x.frames.push((Frame::Push(ind), ProfilerInstant::now()))
257        }
258    }
259
260    pub(crate) fn record_call_exit(&mut self) {
261        if let Some(x) = &mut self.0 {
262            x.frames.push((Frame::Pop, ProfilerInstant::now()))
263        }
264    }
265
266    // We could expose profile on the Heap, but it's an implementation detail that it works here.
267    pub(crate) fn r#gen(&self) -> crate::Result<ProfileData> {
268        match &self.0 {
269            None => Err(crate::Error::new_other(FlameProfileError::NotEnabled)),
270            Some(x) => Ok(Self::gen_profile(x)),
271        }
272    }
273
274    fn gen_profile(x: &FlameData) -> ProfileData {
275        // Need to write out lines which look like:
276        // root;calls1;calls2 1
277        // All the numbers at the end must be whole numbers (we use milliseconds)
278        let mutable_names = x.index.mutable_values.map(|x| x.to_repr());
279        let frozen_names = x.index.frozen_values.map(|x| x.to_value().to_repr());
280        ProfileData {
281            profile: ProfileDataImpl::TimeFlameProfile(
282                Stacks::new(&mutable_names, &frozen_names, &x.frames).render(),
283            ),
284        }
285    }
286}

289mod tests {
290    use std::collections::HashMap;

297    use crate as starlark;
298    use crate::assert::Assert;
299    use crate::environment::Globals;
300    use crate::environment::GlobalsBuilder;
301    use crate::environment::Module;
302    use crate::eval::Evaluator;
303    use crate::eval::runtime::file_loader::ReturnOwnedFileLoader;
304    use crate::eval::runtime::profile::mode::ProfileMode;
305    use crate::syntax::AstModule;
306    use crate::syntax::Dialect;
307    use crate::values::none::NoneType;
308
309    #[test]
310    fn test_time_flame_works_inside_frozen_module() {
311        #[starlark_module]
312        fn register_sleep(globals: &mut GlobalsBuilder) {
313            fn sleep() -> anyhow::Result<NoneType> {
314                thread::sleep(Duration::from_millis(2));
315                Ok(NoneType)
316            }
317        }
318
319        let mut a = Assert::new();
320        a.globals_add(register_sleep);
321        let a_bzl = a.pass_module(
322            r#"
323def foo():
324    for i in range(5):
325        # Must sleep otherwise time flame will round the duration to zero and erase it.
326        sleep()
327    "#,
328        );
329
330        let modules = HashMap::from_iter([("a.bzl".to_owned(), a_bzl)]);
331        let loader = ReturnOwnedFileLoader { modules };
332
333        let module = Module::new();
334        let mut eval = Evaluator::new(&module);
335        eval.enable_profile(&ProfileMode::TimeFlame).unwrap();
336        eval.set_loader(&loader);
337        eval.eval_module(
338            AstModule::parse(
339                "x.star",
340                r#"
341load("a.bzl", "foo")
342
343def bar():
344    for i in range(10):
345        foo()
346
347bar()
348"#
349                .to_owned(),
350                &Dialect::Standard,
351            )
352            .unwrap(),
353            &Globals::standard(),
354        )
355        .unwrap();
356
357        let profile = eval.gen_profile().unwrap().gen_flame_data().unwrap();
358        let the_line = profile
359            .lines()
360            .find(|l| l.contains("foo"))
361            .with_context(|| format!("There must be a line with `foo` in the profile: {profile:?}"))
362            .unwrap();
363        assert!(
364            the_line.contains("bar"),
365            "Profile must contain a line `bar.*foo`: {profile:?}"
366        );
367    }
