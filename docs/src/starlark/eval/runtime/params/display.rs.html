18use std::fmt;
19use std::fmt::Display;
20use std::iter;

22/// Parameter or `*` or `/` separator, but only if needed for formatting.
23pub enum FmtParam<T> {
24    /// Positional-only, positional-or-named, or named-only parameter.
25    Regular(T),
26    /// `*args` parameter.
27    Args(T),
28    /// `**kwargs` parameter.
29    Kwargs(T),
30    /// `/` separator.
31    Slash,
32    /// `*` separator.
33    Star,
34}
35
36/// Flatten parameters and insert `/` and `*` separators if needed.
37pub(crate) fn iter_fmt_param_spec<T>(
38    pos_only: impl IntoIterator<Item = T>,
39    pos_named: impl IntoIterator<Item = T>,
40    args: Option<T>,
41    named_only: impl IntoIterator<Item = T>,
42    kwargs: Option<T>,
43) -> impl Iterator<Item = FmtParam<T>> {
44    let mut pos_only = pos_only.into_iter().peekable();
45    let slash = match pos_only.peek().is_some() {
46        true => Some(FmtParam::Slash),
47        false => None,
48    };
49
50    let mut named_only = named_only.into_iter().peekable();
51    // `*args`, otherwise `*` if needed.
52    let args_or_star = match (named_only.peek().is_some(), args) {
53        (_, Some(args)) => Some(FmtParam::Args(args)),
54        (true, None) => Some(FmtParam::Star),
55        (false, None) => None,
56    };
57
58    iter::empty()
59        .chain(pos_only.map(FmtParam::Regular))
60        .chain(slash)
61        .chain(pos_named.into_iter().map(FmtParam::Regular))
62        .chain(args_or_star)
63        .chain(named_only.map(FmtParam::Regular))
64        .chain(kwargs.map(FmtParam::Kwargs))
65}

68pub(crate) const PARAM_FMT_OPTIONAL: &str = "...";

70pub(crate) struct ParamFmt<'a, T: Display, D: Display> {
71    /// Parameter name.
72    pub(crate) name: &'a str,
73    /// Parameter type. If `None`, it will be omitted.
74    pub(crate) ty: Option<T>,
75    pub(crate) default: Option<D>,
76}
77
78/// Utility to format function signature.
79pub(crate) fn fmt_param_spec<'n, T: Display, D: Display>(
80    f: &mut dyn fmt::Write,
81    pos_only: impl IntoIterator<Item = ParamFmt<'n, T, D>>,
82    pos_named: impl IntoIterator<Item = ParamFmt<'n, T, D>>,
83    args: Option<ParamFmt<'n, T, D>>,
84    named_only: impl IntoIterator<Item = ParamFmt<'n, T, D>>,
85    kwargs: Option<ParamFmt<'n, T, D>>,
86) -> fmt::Result {
87    fmt_param_spec_maybe_multiline(
88        f, None, pos_only, pos_named, args, named_only, kwargs, false,
89    )
90}

92#[allow(clippy::write_with_newline)]
93pub(crate) fn fmt_param_spec_maybe_multiline<'n, T: Display, D: Display>(
94    f: &mut dyn fmt::Write,
95    // Single-line if `None`.
96    indent: Option<&str>,
97    pos_only: impl IntoIterator<Item = ParamFmt<'n, T, D>>,
98    pos_named: impl IntoIterator<Item = ParamFmt<'n, T, D>>,
99    args: Option<ParamFmt<'n, T, D>>,
100    named_only: impl IntoIterator<Item = ParamFmt<'n, T, D>>,
101    kwargs: Option<ParamFmt<'n, T, D>>,
102    // If `true`, escape `*` to `\\*` to avoid rendering in Markdown as bold or italic.
103    escape_stars: bool,
104) -> fmt::Result {
105    struct Printer<'w> {
106        f: &'w mut dyn fmt::Write,
107        escape_stars: bool,
108    }
109
110    impl<'w> Printer<'w> {
111        fn write_param(
112            &mut self,
113            name: impl Display,
114            ty: Option<impl Display>,
115            default: Option<impl Display>,
116        ) -> fmt::Result {
117            write!(self.f, "{name}")?;
118            if let Some(ty) = ty {
119                write!(self.f, ": {ty}")?;
120            }
121            if let Some(default) = default {
122                write!(self.f, " = {default}")?;
123            }
124            Ok(())
125        }
126    }
127
128    let mut printer = Printer { f, escape_stars };
129
130    let mut iter = iter_fmt_param_spec(pos_only, pos_named, args, named_only, kwargs).peekable();
131
132    let not_empty = iter.peek().is_some();
133
134    for (i, param) in iter.enumerate() {
135        if i == 0 {
136            if let Some(indent) = indent {
137                write!(printer.f, "{indent}")?;
138            }
139        } else {
140            if let Some(indent) = indent {
141                write!(printer.f, ",\n{indent}")?;
142            } else {
143                write!(printer.f, ", ")?;
144            }
145        }
146        let star = if printer.escape_stars { "\\*" } else { "*" };
147        match param {
148            FmtParam::Regular(p) => {
149                printer.write_param(p.name, p.ty, p.default)?;
150            }
151            FmtParam::Args(p) => {
152                printer.write_param(format_args!("{}{}", star, p.name), p.ty, p.default)?;
153            }
154            FmtParam::Kwargs(p) => {
155                printer.write_param(format_args!("{}{}{}", star, star, p.name), p.ty, p.default)?;
156            }
157            FmtParam::Slash => {
158                write!(printer.f, "/")?;
159            }
160            FmtParam::Star => {
161                write!(printer.f, "{star}")?;
162            }
163        }
164    }
165
166    if not_empty && indent.is_some() {
167        write!(printer.f, ",\n")?;
168    }
169
170    Ok(())
171}
