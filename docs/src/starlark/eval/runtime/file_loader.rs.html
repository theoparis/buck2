18//! Define variants of the evaluation function with different support
19//! for the `load(...)` statement.
20
21use std::collections::HashMap;
22
23use dupe::Dupe;
24
25use crate::environment::FrozenModule;
26
27/// A trait for turning a `path` given by a `load()` statement into a [`FrozenModule`].
28pub trait FileLoader {
29    /// Open the file given by the load statement `path`.
30    fn load(&self, path: &str) -> crate::Result<FrozenModule>;
31}
32
33/// [`FileLoader`] that looks up modules by name from a [`HashMap`].
34///
35/// A list of all load statements can be obtained through
36/// [`AstModule::loads`](crate::syntax::AstModule::loads).
37/// This struct will raise an error if any requested files are not available.
38pub struct ReturnFileLoader<'a> {
39    /// Map from module name (first argument to `load` statement) to the actual module.
40    pub modules: &'a HashMap<&'a str, &'a FrozenModule>,
41}
42
43impl<'a> FileLoader for ReturnFileLoader<'a> {
44    fn load(&self, path: &str) -> crate::Result<FrozenModule> {
45        match self.modules.get(path) {
46            Some(v) => Ok((*v).dupe()),
47            None => Err(crate::Error::new_other(anyhow::anyhow!(
48                "ReturnFileLoader does not know the module `{}`",
49                path
50            ))),
51        }
52    }
53}
54
55/// Same as [`ReturnFileLoader`], but does not require fighting the borrow checker.
56#[cfg(test)]
57pub(crate) struct ReturnOwnedFileLoader {
58    pub(crate) modules: HashMap<String, FrozenModule>,
59}
60
61#[cfg(test)]
62impl FileLoader for ReturnOwnedFileLoader {
63    fn load(&self, path: &str) -> crate::Result<FrozenModule> {
64        match self.modules.get(path) {
65            Some(v) => Ok(v.dupe()),
66            None => Err(crate::Error::new_other(anyhow::anyhow!(
67                "ReturnOwnedFileLoader does not know the module `{}`",
68                path
69            ))),
70        }
71    }
72}
