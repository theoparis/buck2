40/// Ready to execute bytecode.
41#[derive(Default)]
42pub(crate) struct Bc {
43    pub(crate) instrs: BcInstrs,
44    /// Number of local variable slots.
45    pub(crate) local_count: u32,
46    /// Max stack size in values (`Value`).
47    pub(crate) max_stack_size: u32,
48    /// Max depth of loops.
49    pub(crate) max_loop_depth: LoopDepth,
50}
51
52impl Bc {
53    /// Find span for instruction.
54    #[cold]
55    #[inline(never)]
56    pub(crate) fn slow_arg_at_ptr(addr_ptr: BcPtrAddr) -> &BcInstrSlowArg {
57        let mut ptr = addr_ptr;
58        loop {
59            let opcode = ptr.get_opcode();
60            if opcode == BcOpcode::End {
61                let end_of_bc = ptr.get_instr::<InstrEnd>();
62                let BcInstrEndArg {
63                    slow_args,
64                    end_addr,
65                    ..
66                } = &end_of_bc.arg;
67                let code_start_ptr = ptr.sub(*end_addr);
68                let addr = addr_ptr.offset_from(code_start_ptr);
69                for (next_addr, next_span) in slow_args {
70                    if *next_addr == addr {
71                        return next_span;
72                    }
73                }
74                panic!("span not found for addr: {addr}");
75            }
76            ptr = ptr.add(opcode.size_of_repr());
77        }
78    }
79
80    #[cold]
81    #[inline(never)]
82    pub(crate) fn wrap_error_for_instr_ptr(
83        ptr: BcPtrAddr,
84        e: crate::Error,
85        eval: &Evaluator,
86    ) -> EvalException {
87        let span = Self::slow_arg_at_ptr(ptr).span;
88        add_span_to_expr_error(e, span, eval)
89    }
90
91    /// Run the bytecode in the current frame allocated in the evaluator.
92    ///
93    /// Frame must be allocated properly, otherwise it will likely result in memory corruption.
94    #[inline(always)]
95    pub(crate) fn run<'v, EC: EvaluationCallbacks>(
96        &self,
97        eval: &mut Evaluator<'v, '_, '_>,
98        ec: &mut EC,
99    ) -> Result<Value<'v>, EvalException> {
100        debug_assert!(eval.current_frame.is_inititalized());
101        debug_assert_eq!(self.max_stack_size, eval.current_frame.max_stack_size());
102        run_block(eval, ec, self.instrs.start_ptr())
103    }
104
105    pub(crate) fn dump_debug(&self) -> String {
106        let mut w = String::new();
107        writeln!(w, "Max stack size: {}", self.max_stack_size).unwrap();
108        writeln!(w, "Instructions:").unwrap();
109        self.instrs
110            .dump_debug()
111            .lines()
112            .for_each(|line| writeln!(w, "  {line}").unwrap());
113        w
114    }
115}

117/// Execute one instruction.
118#[cfg_attr(not(debug_assertions), inline(always))]
119fn step<'v, 'b, EC: EvaluationCallbacks>(
120    eval: &mut Evaluator<'v, '_, '_>,
121    ec: &mut EC,
122    frame: BcFramePtr<'v>,
123    ip: BcPtrAddr<'b>,
124) -> InstrControl<'v, 'b> {
125    let opcode = ip.get_opcode();
126    // println!("{}: {:?}", self.current_ip, opcode);
127
128    struct HandlerImpl<'v, 'a, 'e, 'y, 'b> {
129        eval: &'y mut Evaluator<'v, 'a, 'e>,
130        frame: BcFramePtr<'v>,
131        ip: BcPtrAddr<'b>,
132    }
133
134    impl<'v, 'a, 'e, 'y, 'b> BcOpcodeHandler<InstrControl<'v, 'b>> for HandlerImpl<'v, 'a, 'e, 'y, 'b> {
135        #[cfg_attr(not(debug_assertions), inline(always))]
136        fn handle<I: BcInstr>(self) -> InstrControl<'v, 'b> {
137            let HandlerImpl { eval, frame, ip } = self;
138            let repr = ip.get_instr::<I>();
139            I::run(eval, frame, ip, &repr.arg)
140        }
141    }
142
143    match ec.before_instr(eval, ip, opcode) {
144        Ok(()) => {}
145        Err(e) => return InstrControl::Err(e),
146    }
147    opcode.dispatch(HandlerImpl { eval, frame, ip })
148}
149
150/// Execute the code block, either a module or a function body.
151pub(crate) fn run_block<'v, EC: EvaluationCallbacks>(
152    eval: &mut Evaluator<'v, '_, '_>,
153    ec: &mut EC,
154    mut ip: BcPtrAddr,
155) -> Result<Value<'v>, EvalException> {
156    // Copy frame pointer to local variable to generate more efficient code.
157    let frame = eval.current_frame;
158
159    loop {
160        // Note most functions called from here must be carefully annotated
161        // as `#[inline(always)]` otherwise LLVM considers them too large to inline.
162        //
163        // We do inline always only in release mode because otherwise
164        // generated stack frame is too large which leads to C stack overflow in debug more.
165        ip = match step(eval, ec, frame, ip) {
166            InstrControl::Next(ip) => ip,
167            InstrControl::Return(v) => return Ok(v),
168            InstrControl::Err(e) => return Err(Bc::wrap_error_for_instr_ptr(ip, e, eval)),
169        }
170    }
171}
