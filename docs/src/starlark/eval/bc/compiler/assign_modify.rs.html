50trait AssignOnWriteBc {
51    fn write_bc(
52        self,
53        v0: BcSlotIn,
54        v1: BcSlotIn,
55        target: BcSlotOut,
56        span: FrameSpan,
57        bc: &mut BcWriter,
58    );
59}
60
61impl AssignOnWriteBc for AssignOp {
62    fn write_bc(
63        self,
64        v0: BcSlotIn,
65        v1: BcSlotIn,
66        target: BcSlotOut,
67        span: FrameSpan,
68        bc: &mut BcWriter,
69    ) {
70        let arg = (v0, v1, target);
71        match self {
72            AssignOp::Add => bc.write_instr::<InstrAddAssign>(span, arg),
73            AssignOp::Subtract => bc.write_instr::<InstrSub>(span, arg),
74            AssignOp::Multiply => bc.write_instr::<InstrMultiply>(span, arg),
75            AssignOp::Divide => bc.write_instr::<InstrDivide>(span, arg),
76            AssignOp::FloorDivide => bc.write_instr::<InstrFloorDivide>(span, arg),
77            AssignOp::Percent => bc.write_instr::<InstrPercent>(span, arg),
78            AssignOp::BitAnd => bc.write_instr::<InstrBitAnd>(span, arg),
79            AssignOp::BitOr => bc.write_instr::<InstrBitOrAssign>(span, arg),
80            AssignOp::BitXor => bc.write_instr::<InstrBitXor>(span, arg),
81            AssignOp::LeftShift => bc.write_instr::<InstrLeftShift>(span, arg),
82            AssignOp::RightShift => bc.write_instr::<InstrRightShift>(span, arg),
83        }
84    }
85}
86
87impl AssignModifyLhs {
88    /// After evaluation of `x[y] += ...`, variables `x` and `y` are definitely assigned.
89    pub(crate) fn mark_definitely_assigned_after(&self, bc: &mut BcWriter) {
90        match self {
91            AssignModifyLhs::Dot(object, _field) => object.mark_definitely_assigned_after(bc),
92            AssignModifyLhs::Array(array, index) => {
93                array.mark_definitely_assigned_after(bc);
94                index.mark_definitely_assigned_after(bc);
95            }
96            AssignModifyLhs::LocalCaptured(_) => {}
97            AssignModifyLhs::Local(local) => bc.mark_definitely_assigned(local.node),
98            AssignModifyLhs::Module(_) => {}
99        }
100    }
101
102    pub(crate) fn write_bc(
103        &self,
104        span: FrameSpan,
105        op: AssignOp,
106        rhs: &IrSpanned<ExprCompiled>,
107        bc: &mut BcWriter,
108    ) {
109        match *self {
110            AssignModifyLhs::Dot(ref object, ref field) => {
111                object.write_bc_cb(bc, |object, bc| {
112                    bc.alloc_slots_c(|lhs_rhs: BcSlotsN<2>, bc| {
113                        let field = Symbol::new(field.as_str());
114                        bc.write_instr::<InstrObjectField>(
115                            span,
116                            (object, field.clone(), lhs_rhs.get::<0>().to_out()),
117                        );
118                        rhs.write_bc(lhs_rhs.get::<1>().to_out(), bc);
119                        op.write_bc(
120                            lhs_rhs.get::<0>().to_in(),
121                            lhs_rhs.get::<1>().to_in(),
122                            lhs_rhs.get::<1>().to_out(),
123                            span,
124                            bc,
125                        );
126                        bc.write_instr::<InstrSetObjectField>(
127                            span,
128                            (lhs_rhs.get::<1>().to_in(), object, field),
129                        );
130                    })
131                });
132            }
133            AssignModifyLhs::Array(ref array, ref index) => {
134                write_n_exprs([array, index], bc, |[array, index], bc| {
135                    bc.alloc_slots_c(|slots: BcSlotsN<2>, bc| {
136                        let temp_slot = slots.get::<0>();
137                        let rhs_slot = slots.get::<1>();
138
139                        bc.write_instr::<InstrArrayIndex>(span, (array, index, temp_slot.to_out()));
140                        rhs.write_bc(rhs_slot.to_out(), bc);
141                        op.write_bc(
142                            temp_slot.to_in(),
143                            rhs_slot.to_in(),
144                            temp_slot.to_out(),
145                            span,
146                            bc,
147                        );
148                        bc.write_instr::<InstrArrayIndexSet>(
149                            span,
150                            (array, index, temp_slot.to_in()),
151                        );
152                    })
153                });
154            }
155            AssignModifyLhs::Local(s) => bc.alloc_slots_c(|lhs_rhs: BcSlotsN<2>, bc| {
156                let slot = s.node;
157                bc.write_load_local(span, slot, lhs_rhs.get::<0>().to_out());
158                rhs.write_bc(lhs_rhs.get::<1>().to_out(), bc);
159
160                op.write_bc(
161                    lhs_rhs.get::<0>().to_in(),
162                    lhs_rhs.get::<1>().to_in(),
163                    lhs_rhs.get::<1>().to_out(),
164                    span,
165                    bc,
166                );
167                bc.write_mov(span, lhs_rhs.get::<1>().to_in(), slot.to_bc_slot().to_out());
168            }),
169            AssignModifyLhs::LocalCaptured(s) => bc.alloc_slots_c(|lhs_rhs: BcSlotsN<2>, bc| {
170                let slot = s.node;
171                bc.write_load_local_captured(span, slot, lhs_rhs.get::<0>().to_out());
172                rhs.write_bc(lhs_rhs.get::<1>().to_out(), bc);
173
174                op.write_bc(
175                    lhs_rhs.get::<0>().to_in(),
176                    lhs_rhs.get::<1>().to_in(),
177                    lhs_rhs.get::<1>().to_out(),
178                    span,
179                    bc,
180                );
181                bc.write_store_local_captured(span, lhs_rhs.get::<1>().to_in(), slot);
182            }),
183            AssignModifyLhs::Module(m) => bc.alloc_slots_c(|lhs_rhs: BcSlotsN<2>, bc| {
184                let slot = m.node;
185                bc.write_instr::<InstrLoadModule>(span, (slot, lhs_rhs.get::<0>().to_out()));
186                rhs.write_bc(lhs_rhs.get::<1>().to_out(), bc);
187                op.write_bc(
188                    lhs_rhs.get::<0>().to_in(),
189                    lhs_rhs.get::<1>().to_in(),
190                    lhs_rhs.get::<1>().to_out(),
191                    span,
192                    bc,
193                );
194                bc.write_instr::<InstrStoreModule>(span, (lhs_rhs.get::<1>().to_in(), slot));
195            }),
196        }
197    }
