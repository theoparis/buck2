57impl ArgsCompiledValue {
58    /// After evaluation of function arguments like `foo(a, b=c[d], **e)`,
59    /// variables `a`, `b`, `c`, `d`, and `e` are definitely assigned.
60    fn mark_definitely_assigned_after(&self, bc: &mut BcWriter) {
61        let ArgsCompiledValue {
62            pos_named,
63            names,
64            args,
65            kwargs,
66        } = self;
67        for n in pos_named {
68            n.mark_definitely_assigned_after(bc);
69        }
70        let _ = names;
71        if let Some(args) = args {
72            args.mark_definitely_assigned_after(bc);
73        }
74        if let Some(kwargs) = kwargs {
75            kwargs.mark_definitely_assigned_after(bc);
76        }
77    }
78
79    fn write_bc(&self, bc: &mut BcWriter, k: impl FnOnce(BcCallArgsFull<Symbol>, &mut BcWriter)) {
80        write_exprs(&self.pos_named, bc, |pos_named, bc| {
81            write_expr_opt(&self.args, bc, |args, bc| {
82                write_expr_opt(&self.kwargs, bc, |kwargs, bc| {
83                    let args_full = BcCallArgsFull {
84                        pos_named,
85                        names: self.names.clone().into_boxed_slice(),
86                        args,
87                        kwargs,
88                    };
89                    k(args_full, bc)
90                })
91            })
92        });
93    }
94}

96impl CallCompiled {
97    /// After evaluation of call like `a[b](c.d)`,
98    /// variables `a`, `b`, and `c` are definitely assigned.
99    pub(crate) fn mark_definitely_assigned_after(&self, bc: &mut BcWriter) {
100        let CallCompiled { fun, args } = self;
101        fun.mark_definitely_assigned_after(bc);
102        args.mark_definitely_assigned_after(bc);
103    }
104}

106impl IrSpanned<CallCompiled> {
107    fn write_args(
108        args: &ArgsCompiledValue,
109        bc: &mut BcWriter,
110        k: impl FnOnce(Either<BcCallArgsPos, BcCallArgsFull<Symbol>>, &mut BcWriter),
111    ) {
112        if let Some(pos) = args.pos_only() {
113            write_exprs(pos, bc, |pos, bc| {
114                let args = Either::Left(BcCallArgsPos { pos });
115                k(args, bc)
116            })
117        } else {
118            args.write_bc(bc, |args_full, bc| {
119                let args = Either::Right(args_full);
120                k(args, bc)
121            })
122        }
123    }
124
125    fn write_call_frozen(
126        span: FrameSpan,
127        fun: FrozenValue,
128        args: &ArgsCompiledValue,
129        target: BcSlotOut,
130        bc: &mut BcWriter,
131    ) {
132        let file_span = bc.alloc_file_span(span);
133        if let Some(fun) = FrozenValueTyped::<FrozenDef>::new(fun) {
134            Self::write_args(args, bc, |args, bc| match args {
135                Either::Left(npops) => {
136                    bc.write_instr::<InstrCallFrozenDefPos>(span, (fun, npops, file_span, target))
137                }
138                Either::Right(args) => bc.write_instr::<InstrCallFrozenDef>(
139                    span,
140                    (fun, args.resolve(fun.as_ref()), file_span, target),
141                ),
142            })
143        } else if let Some(fun) = FrozenValueTyped::<NativeFunction>::new(fun) {
144            let fun = BcNativeFunction::new(fun);
145            Self::write_args(args, bc, |args, bc| match args {
146                Either::Left(npops) => {
147                    bc.write_instr::<InstrCallFrozenNativePos>(
148                        span,
149                        (fun, npops, file_span, target),
150                    );
151                }
152                Either::Right(args) => {
153                    bc.write_instr::<InstrCallFrozenNative>(span, (fun, args, file_span, target));
154                }
155            })
156        } else {
157            Self::write_args(args, bc, |args, bc| match args {
158                Either::Left(npops) => {
159                    bc.write_instr::<InstrCallFrozenPos>(span, (fun, npops, file_span, target));
160                }
161                Either::Right(args) => {
162                    bc.write_instr::<InstrCallFrozen>(span, (fun, args, file_span, target));
163                }
164            })
165        }
166    }
167
168    fn write_call_method(
169        target: BcSlotOut,
170        span: FrameSpan,
171        this: &IrSpanned<ExprCompiled>,
172        symbol: &Symbol,
173        args: &ArgsCompiledValue,
174        bc: &mut BcWriter,
175    ) {
176        this.write_bc_cb(bc, |this, bc| {
177            let file_span = bc.alloc_file_span(span);
178            let known_method = get_known_method(symbol.as_str());
179            if let Some(pos) = args.pos_only() {
180                write_exprs(pos, bc, |pos, bc| {
181                    if let Some(known_method) = known_method {
182                        bc.write_instr::<InstrCallMaybeKnownMethodPos>(
183                            span,
184                            (
185                                this,
186                                symbol.clone(),
187                                known_method,
188                                BcCallArgsPos { pos },
189                                file_span,
190                                target,
191                            ),
192                        );
193                    } else {
194                        bc.write_instr::<InstrCallMethodPos>(
195                            span,
196                            (
197                                this,
198                                symbol.clone(),
199                                BcCallArgsPos { pos },
200                                file_span,
201                                target,
202                            ),
203                        );
204                    }
205                });
206            } else {
207                args.write_bc(bc, |args, bc| {
208                    if let Some(known_method) = known_method {
209                        bc.write_instr::<InstrCallMaybeKnownMethod>(
210                            span,
211                            (this, symbol.clone(), known_method, args, file_span, target),
212                        );
213                    } else {
214                        bc.write_instr::<InstrCallMethod>(
215                            span,
216                            (this, symbol.clone(), args, file_span, target),
217                        );
218                    }
219                })
220            }
221        })
222    }
223
224    pub(crate) fn write_bc(&self, target: BcSlotOut, bc: &mut BcWriter) {
225        if let Some(arg) = self.as_len() {
226            return arg.write_bc_cb(bc, |arg, bc| {
227                bc.write_instr::<InstrLen>(self.span, (arg, target));
228            });
229        } else if let Some(arg) = self.as_type() {
230            return arg.write_bc_cb(bc, |arg, bc| {
231                bc.write_instr::<InstrType>(self.span, (arg, target));
232            });
233        } else if let Some((x, t)) = self.as_isinstance() {
234            if let Ok(t) = TypeCompiled::new_frozen(t, bc.heap) {
235                return x.write_bc_cb(bc, |x, bc| {
236                    bc.write_instr::<InstrIsInstance>(self.span, (x, t, target));
237                });
238            }
239        }
240
241        let span = self.span;
242        let file_span = bc.alloc_file_span(span);
243        match self.method() {
244            None => match self.fun.as_value() {
245                Some(f) => Self::write_call_frozen(span, f, &self.args, target, bc),
246                None => {
247                    self.fun.write_bc_cb(bc, |fun, bc| {
248                        Self::write_args(&self.args, bc, |args, bc| match args {
249                            Either::Left(npops) => bc
250                                .write_instr::<InstrCallPos>(span, (fun, npops, file_span, target)),
251                            Either::Right(args) => {
252                                bc.write_instr::<InstrCall>(span, (fun, args, file_span, target));
253                            }
254                        })
255                    });
256                }
257            },
258            Some((this, symbol, args)) => {
259                Self::write_call_method(target, span, this, symbol, args, bc)
260            }
261        }
262    }
