36/// Address relative to bytecode start.
37#[derive(
38    Eq, PartialEq, Copy, Clone, Dupe, Debug, PartialOrd, Ord, Display, Hash, Default
39)]
40#[display("@{}", _0)]
41pub(crate) struct BcAddr(pub(crate) u32);
42
43impl BcAddr {
44    #[inline]
45    pub(crate) fn offset_from(self, start: BcAddr) -> BcAddrOffset {
46        debug_assert!(self >= start);
47        BcAddrOffset(self.0 - start.0)
48    }
49
50    #[inline]
51    pub(crate) fn offset(self, offset: BcAddrOffset) -> BcAddr {
52        BcAddr(self.0 + offset.0)
53    }
54
55    #[inline]
56    pub(crate) fn offset_neg(self, offset: BcAddrOffsetNeg) -> BcAddr {
57        BcAddr(self.0 - offset.0)
58    }
59}
60
61impl Sub<u32> for BcAddr {
62    type Output = BcAddr;
63
64    fn sub(self, rhs: u32) -> BcAddr {
65        BcAddr(self.0.checked_sub(rhs).unwrap())
66    }
67}
68
69impl Add<u32> for BcAddr {
70    type Output = BcAddr;
71
72    fn add(self, rhs: u32) -> BcAddr {
73        BcAddr(self.0 + rhs)
74    }
75}
76
77impl AddAssign<u32> for BcAddr {
78    fn add_assign(&mut self, rhs: u32) {
79        self.0 += rhs;
80    }
81}

83/// Valid pointer range of bytecode.
84/// Used for debugging assertions. This objects is not created in release mode.
85#[derive(Copy, Clone, Dupe, Debug, PartialEq)]
86pub(crate) struct BcPtrRange {
87    start: *const u8,
88    /// Length in bytes.
89    len: usize,
90}
91
92impl BcPtrRange {
93    pub(crate) fn for_slice(slice: &[u64]) -> BcPtrRange {
94        BcPtrRange {
95            start: slice.as_ptr() as *const u8,
96            len: mem::size_of_val::<[u64]>(slice),
97        }
98    }
99
100    pub(crate) fn assert_in_range(&self, ptr: *const u8) {
101        let offset = unsafe { ptr.offset_from(self.start) };
102        assert!(offset >= 0);
103        assert!(offset as usize <= self.len);
104    }
105
106    fn end(&self) -> *const u8 {
107        unsafe { self.start.add(self.len) }
108    }
109}

111/// Pointer to an instruction in memory.
112#[derive(Copy, Clone, Dupe, PartialOrd, PartialEq, Debug)]
113// Lifetime here is mostly for documentation, it does not guarantee anything.
114pub(crate) struct BcPtrAddr<'b> {
115    ptr: *const u8,
116    /// When assertions enabled, we validate the pointer is in this range.
117    range: IfDebug<BcPtrRange>,
118    _marker: marker::PhantomData<&'b u8>,
119}
120
121impl<'b> BcPtrAddr<'b> {
122    /// Constructor.
123    unsafe fn new(ptr: *const u8, range: IfDebug<BcPtrRange>) -> BcPtrAddr<'b> {
124        debug_assert!(ptr as usize % BC_INSTR_ALIGN == 0);
125        range.if_debug(|range| range.assert_in_range(ptr));
126        BcPtrAddr {
127            ptr,
128            range,
129            _marker: marker::PhantomData,
130        }
131    }
132
133    /// Create a pointer for the beginning of the slice.
134    pub(crate) fn for_slice_start(slice: &'b [u64]) -> BcPtrAddr<'b> {
135        unsafe {
136            BcPtrAddr::new(
137                slice.as_ptr() as *const u8,
138                IfDebug::new(BcPtrRange::for_slice(slice)),
139            )
140        }
141    }
142
143    /// Create a pointer for the beginning of the slice.
144    pub(crate) fn for_slice_end(slice: &'b [u64]) -> BcPtrAddr<'b> {
145        unsafe {
146            BcPtrAddr::new(
147                slice.as_ptr().add(slice.len()) as *const u8,
148                IfDebug::new(BcPtrRange::for_slice(slice)),
149            )
150        }
151    }
152
153    /// Distance from current ptr to the end of instructions.
154    fn remaining_if_debug(self) -> usize {
155        unsafe { self.range.get_ref_if_debug().end().offset_from(self.ptr) as usize }
156    }
157
158    pub(crate) fn get_instr<I: BcInstr>(self) -> &'b BcInstrRepr<I> {
159        debug_assert!(self.remaining_if_debug() >= mem::size_of::<BcInstrRepr<I>>());
160        let ptr = self.ptr as *const BcInstrRepr<I>;
161        let repr = unsafe { &*ptr };
162        debug_assert_eq!(repr.header.opcode, BcOpcode::for_instr::<I>());
163        repr
164    }
165
166    pub(crate) fn get_instr_checked<I: BcInstr>(self) -> Option<&'b BcInstrRepr<I>> {
167        if self.get_opcode() == BcOpcode::for_instr::<I>() {
168            Some(self.get_instr())
169        } else {
170            None
171        }
172    }
173
174    pub(crate) fn get_instr_mut<I: BcInstr>(self) -> *mut BcInstrRepr<I> {
175        debug_assert!(
176            self.remaining_if_debug() >= mem::size_of::<BcInstrRepr<I>>(),
177            "remaining: {}, instr size: {}",
178            self.remaining_if_debug(),
179            mem::size_of::<BcInstrRepr<I>>()
180        );
181        self.ptr as *mut BcInstrRepr<I>
182    }
183
184    pub(crate) fn get_opcode(self) -> BcOpcode {
185        debug_assert!(self.remaining_if_debug() >= mem::size_of::<BcInstrHeader>());
186        let ptr = self.ptr as *const BcInstrHeader;
187        unsafe { (*ptr).opcode }
188    }
189
190    pub(crate) fn offset_from(self, start: BcPtrAddr) -> BcAddr {
191        debug_assert!(self.range.get_ref_if_debug() == start.range.get_ref_if_debug());
192        unsafe {
193            let offset = self.ptr.offset_from(start.ptr);
194            debug_assert!(offset >= 0);
195            debug_assert!(offset <= i32::MAX as isize);
196            BcAddr(offset as u32)
197        }
198    }
199
200    #[inline(always)]
201    fn sub_usize(self, offset: usize) -> BcPtrAddr<'b> {
202        unsafe { BcPtrAddr::new(self.ptr.sub(offset), self.range) }
203    }
204
205    #[inline(always)]
206    pub(crate) fn sub(self, start: BcAddr) -> BcPtrAddr<'b> {
207        self.sub_usize(start.0 as usize)
208    }
209
210    pub(crate) fn offset(self, addr: BcAddr) -> BcPtrAddr<'b> {
211        self.add(addr.0 as usize)
212    }
213
214    #[inline(always)]
215    pub(crate) fn add_rel(self, rel: BcAddrOffset) -> BcPtrAddr<'b> {
216        self.add(rel.0 as usize)
217    }
218
219    #[inline(always)]
220    pub(crate) fn add_rel_neg(self, rel: BcAddrOffsetNeg) -> BcPtrAddr<'b> {
221        self.sub_usize(rel.0 as usize)
222    }
223
224    pub(crate) fn add(self, offset: usize) -> BcPtrAddr<'b> {
225        unsafe { BcPtrAddr::new(self.ptr.add(offset), self.range) }
226    }
227
228    pub(crate) fn add_instr<I: BcInstr>(self) -> BcPtrAddr<'b> {
229        self.add_rel(BcAddrOffset::for_instr::<I>())
230    }
231}

233/// Difference between addresses.
234#[derive(Eq, PartialEq, Copy, Clone, Dupe, Debug, PartialOrd, Ord, Display)]
235pub(crate) struct BcAddrOffset(pub(crate) u32);
236
237impl BcAddrOffset {
238    /// Pointer to not yet known address.
239    pub(crate) const FORWARD: BcAddrOffset = BcAddrOffset(0xdeadbeef);
240
241    /// Size of an instruction.
242    fn for_instr<I: BcInstr>() -> BcAddrOffset {
243        <BcInstrRepr<I>>::assert_align();
244        BcAddrOffset(mem::size_of::<BcInstrRepr<I>>() as u32)
245    }
246
247    #[inline(always)]
248    pub(crate) fn neg(self) -> BcAddrOffsetNeg {
249        BcAddrOffsetNeg(self.0)
250    }
251}
