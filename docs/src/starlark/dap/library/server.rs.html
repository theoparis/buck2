18use debugserver_types as ds;
19
20use crate::dap::library::events::*;
21use crate::dap::library::requests::*;
22use crate::dap::library::stream::*;
23
24pub(crate) struct DapService {
25    _private: (),
26}
27
28impl DapService {
29    pub(crate) fn run<T: DebugServer>(f: impl FnOnce(Client) -> T) {
30        server(f(Client::new()));
31    }
32}
33
34fn server(dap: impl DebugServer) {
35    log_begin();
36
37    // Because of the eval we're running in, we probably can't see panics.
38    // So mirror them to the log file.
39    let orig_hook = std::panic::take_hook();
40    std::panic::set_hook(Box::new(move |panic_info| {
41        if let Some(s) = panic_info.payload().downcast_ref::<&str>() {
42            log(&format!("Panic occurred: {s:?}"));
43        } else {
44            log("Panic occurred: Unknown message");
45        }
46        orig_hook(panic_info);
47    }));
48
49    log("DEBUG ADAPTER STARTING");
50    loop {
51        let recv = read();
52        let r: ds::Request = serde_json::from_value(recv).unwrap();
53        assert_eq!(r.type_, "request");
54        let resp = dispatch(&dap, &r);
55        send(serde_json::to_value(resp).unwrap());
56
57        if r.command == "disconnect" {
58            break;
59        }
60    }
61    log("DEBUG ADAPTER STOPPING");
62}
