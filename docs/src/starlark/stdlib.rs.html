44/// Return the default global environment, it is not yet frozen so that a caller
45/// can refine it.
46///
47/// For example `stdlib::standard_environment().freeze().child("test")` create a
48/// child environment of this global environment that have been frozen.
49pub(crate) fn standard_environment() -> GlobalsBuilder {
50    GlobalsBuilder::new().with(register_globals)
51}

53/// The extra library definitions available in this Starlark implementation, but not in the standard.
54#[derive(PartialEq, Eq, Copy, Clone, Dupe)]
55pub enum LibraryExtension {
56    /// Definitions to support the `struct` type, the `struct()` constructor.
57    StructType,
58    /// Definitions to support the `record` type, the `record()` constructor and `field()` function.
59    RecordType,
60    /// Definitions to support the `enum` type, the `enum()` constructor.
61    EnumType,
62    /// Add a function `namespace()` which acts much like `struct()` but is clear about it's
63    /// intended use and stricter
64    NamespaceType,
65    /// A function `map(f, xs)` which applies `f` to each element of `xs` and returns the result.
66    // TODO(nga): add set: https://www.internalfb.com/tasks/?t=184017710
67    Map,
68    /// A function `filter(f, xs)` which applies `f` to each element of `xs` and returns those for which `f` returns `True`.
69    /// As a special case, `filter(None, xs)` removes all `None` values.
70    Filter,
71    /// Partially apply a function, `partial(f, *args, **kwargs)` will create a function where those `args` `kwargs`
72    /// are already applied to `f`.
73    Partial,
74    /// Add a function `debug(x)` which shows the Rust [`Debug`](std::fmt::Debug) representation of a value.
75    /// Useful when debugging, but the output should not be considered stable.
76    Debug,
77    /// Add a function `print(x)` which prints to stderr.
78    Print,
79    /// Add a function `pprint(x)` which pretty-prints to stderr.
80    Pprint,
81    /// Add a function `pstr` which is a pretty-printed version of `str`.
82    Pstr,
83    /// Add a function `prepr` which is a pretty-printed version of `repr`.
84    Prepr,
85    /// Add a function `breakpoint()` which will drop into a console-module evaluation prompt.
86    Breakpoint,
87    /// Add a function `json()` which will generate JSON for a module.
88    Json,
89    /// Provides `typing.All`, `typing.Callable` etc.
90    /// Usually used in conjunction with
91    /// [`Dialect::enable_types`](crate::syntax::Dialect::enable_types).
92    Typing,
93    /// Utilities exposing starlark-rust internals.
94    /// These are not for production use.
95    Internal,
96    /// Add a function `call_stack()` which returns a string representation of
97    /// the current call stack.
98    CallStack,
99    /// Definitions to support the `set` type, the `set()` constructor.
100    SetType,
101    // Make sure if you add anything new, you add it to `all` below.
102}
103
104impl LibraryExtension {
105    /// A list of all extensions that will be updated as new methods are added.
106    pub(crate) fn all() -> &'static [Self] {
107        use LibraryExtension::*;
108        &[
109            StructType,
110            RecordType,
111            EnumType,
112            NamespaceType,
113            Map,
114            Filter,
115            Partial,
116            Debug,
117            Print,
118            Pprint,
119            Pstr,
120            Prepr,
121            Breakpoint,
122            Json,
123            Typing,
124            Internal,
125            CallStack,
126            SetType,
127        ]
128    }
129
130    /// Add a specific extension to a [`GlobalsBuilder`].
131    pub fn add(self, builder: &mut GlobalsBuilder) {
132        use LibraryExtension::*;
133        match self {
134            StructType => register_struct(builder),
135            NamespaceType => register_namespace(builder),
136            RecordType => register_record(builder),
137            EnumType => register_enum(builder),
138            SetType => register_set(builder),
139            Map => extra::map(builder),
140            Filter => extra::filter(builder),
141            Partial => partial::partial(builder),
142            Debug => extra::debug(builder),
143            Print => extra::print(builder),
144            Pprint => extra::pprint(builder),
145            Pstr => extra::pstr(builder),
146            Prepr => extra::prepr(builder),
147            Breakpoint => breakpoint::global(builder),
148            Json => json::json(builder),
149            Typing => typing::globals::register_typing(builder),
150            Internal => register_internal(builder),
151            CallStack => call_stack::global(builder),
152        }
153    }
154}

194    #[test]
195    fn test_value_attributes() {
196        #[derive(
197            Copy,
198            Clone,
199            Debug,
200            Dupe,
201            PartialEq,
202            Display,
203            ProvidesStaticType,
204            NoSerialize,
205            Allocative
206        )]
207        struct Bool2(bool);
208        starlark_simple_value!(Bool2);
209
210        #[starlark_value(type = "bool2")]
211        impl<'v> StarlarkValue<'v> for Bool2 {
212            fn get_methods() -> Option<&'static Methods> {
213                static RES: MethodsStatic = MethodsStatic::new();
214                RES.methods(methods)
215            }
216
217            fn equals(&self, other: Value<'v>) -> crate::Result<bool> {
218                match other.downcast_ref::<Bool2>() {
219                    None => Ok(false),
220                    Some(v) => Ok(*v == *self),
221                }
222            }
223        }
224
225        impl<'v> UnpackValue<'v> for Bool2 {
226            type Error = Infallible;
227
228            fn unpack_value_impl(value: Value<'v>) -> Result<Option<Self>, Self::Error> {
229                Ok(Some(*value.downcast_ref::<Bool2>().unwrap()))
230            }
231        }
232
233        #[starlark_module]
234        fn globals(builder: &mut GlobalsBuilder) {
235            const True2: Bool2 = Bool2(true);
236            const False2: Bool2 = Bool2(false);
237        }
238
239        #[starlark_module]
240        fn methods(builder: &mut MethodsBuilder) {
241            #[starlark(attribute)]
242            fn invert1(this: Bool2) -> starlark::Result<Bool2> {
243                Ok(Bool2(!this.0))
244            }
245
246            fn invert2(this: Bool2) -> anyhow::Result<Bool2> {
247                Ok(Bool2(!this.0))
248            }
249        }
250
251        let mut a = Assert::new();
252        a.globals_add(globals);
253        a.all_true(
254            r#"
255True2 == True2
256True2 != False2
257True2.invert1 == False2
258False2.invert1 == True2
259False2.invert2() == True2
260hasattr(True2, "invert1") == True
261hasattr(True2, "invert2") == True
262hasattr(True2, "invert3") == False
263dir(False2) == ["invert1","invert2"]
264getattr(False2, "invert1") == True2
265getattr(True2, "invert1") == False2
266getattr(True2, "invert2")() == False2
267"#,
268        );
269    }
