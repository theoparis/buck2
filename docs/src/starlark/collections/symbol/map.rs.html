18//! An optimised string HashMap which goes even faster when the keys can
19//! be pre-hashed or otherwise precomputed.
20//!
21//! The two bottlenecks in our use of these hash tables are computing the hashes and comparing
22//! the resulting keys for equality. We precompute the hashes. We also use `[usize]` to do faster
23//! comparison when possible. We use the Starlark SmallHash hashes, promoted by IdHasher,
24//! so we can reuse a SmallMap hash.
25//!
26//! Benchmarks on which the `[usize]` choice was made (mac/linux, all in ns):
27//!                           8 bytes       32 bytes      64 bytes
28//! slice equality (memcmp)   3.5/3.8       3.5/ 3.0      4.5/ 4.7
29//! usize equality loop       1.0/1.4       2.7/ 3.5      3.5/ 6.0
30//! u8 equality loop          3.4/5.7      13.7/19.7     22.6/44.8
31//!
32//! Measuring some sample strings, the P50 = 21 bytes, P75 = 27, P95 = 35,
33//! so we can reasonably expect to hit the smaller cases most often.
34
35use std::fmt;
36use std::fmt::Debug;
37use std::mem;
38
39use allocative::Allocative;
40use hashbrown::HashTable;
41use starlark_derive::Trace;
42use starlark_map::Hashed;
43
44use crate as starlark;
45use crate::collections::symbol::symbol::Symbol;
46use crate::values::StringValue;
47
48// We use a RawTable (the thing that underlies HashMap) so we can look up efficiently
49// and easily by Symbol and str, without being limited by `Borrow` traits.
50#[derive(Clone, Trace, Allocative)]
51pub(crate) struct SymbolMap<T>(HashTable<(Symbol, T)>);
52
53impl<T: Debug> Debug for SymbolMap<T> {
54    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
55        f.debug_map()
56            .entries(self.iter().map(|x| (&x.0, &x.1)))
57            .finish()
58    }
59}
60
61impl<T> SymbolMap<T> {
62    pub(crate) fn new() -> Self {
63        SymbolMap::with_capacity(0)
64    }
65
66    pub(crate) fn with_capacity(capacity: usize) -> Self {
67        SymbolMap(HashTable::with_capacity(capacity))
68    }
69
70    pub(crate) fn insert(&mut self, key: &str, value: T) -> Option<T> {
71        let s = Symbol::new(key);
72        if let Some((_, item)) = self.0.find_mut(s.hash(), |x| s == x.0) {
73            Some(mem::replace(item, value))
74        } else {
75            // This insert doesn't remove old values, so do that manually first
76            self.0.insert_unique(s.hash(), (s, value), |x| x.0.hash());
77            None
78        }
79    }
80
81    #[inline]
82    pub(crate) fn get(&self, key: &Symbol) -> Option<&T> {
83        self.0.find(key.hash(), |x| key == &x.0).map(|x| &x.1)
84    }
85
86    pub(crate) fn get_str(&self, key: &str) -> Option<&T> {
87        self.get_hashed_str(Hashed::new(key))
88    }
89
90    pub(crate) fn get_hashed_str(&self, key: Hashed<&str>) -> Option<&T> {
91        self.0
92            .find(key.hash().promote(), |x| x.0.as_str() == *key.key())
93            .map(|x| &x.1)
94    }
95
96    pub(crate) fn get_hashed_string_value(&self, key: Hashed<StringValue>) -> Option<&T> {
97        self.0
98            .find(key.hash().promote(), |x| {
99                x.0.as_aligned_padded_str() == key.key().as_aligned_padded_str()
100            })
101            .map(|x| &x.1)
102    }
103
104    pub(crate) fn len(&self) -> usize {
105        self.0.len()
106    }
107
108    pub(crate) fn iter<'a>(&'a self) -> impl ExactSizeIterator<Item = &'a (Symbol, T)> + 'a {
109        self.0.iter()
110    }
111
112    pub(crate) fn keys<'a>(&'a self) -> impl ExactSizeIterator<Item = &'a Symbol> + 'a {
113        self.iter().map(|x| &x.0)
114    }
115
116    pub(crate) fn values<'a>(&'a self) -> impl ExactSizeIterator<Item = &'a T> + 'a {
117        self.iter().map(|x| &x.1)
118    }
119}
