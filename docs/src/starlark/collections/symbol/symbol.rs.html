19use std::fmt::Debug;
20use std::intrinsics::copy_nonoverlapping;
21use std::mem;
22use std::slice;
23use std::str;

34/// A pre-hashed string used for efficient dictionary lookup.
35#[derive(Clone, Trace, Allocative)]
36pub(crate) struct Symbol {
37    hash: u64,
38    len: u32,
39    payload: Box<[usize]>,
40    small_hash: StarlarkHashValue,
41}
42
43unsafe impl Coerce<Symbol> for Symbol {}
44
45impl Debug for Symbol {
46    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
47        self.as_str().fmt(f)
48    }
49}
50
51impl PartialEq for Symbol {
52    fn eq(&self, other: &Self) -> bool {
53        if self.len != other.len {
54            return false;
55        }
56
57        let p1 = &*self.payload;
58        let p2 = &*other.payload;
59        // Important to use the payload len, which is in u64 units, rather than len which is in u8
60        for i in 0..self.payload.len() {
61            // Safe because we checked the lengths at the start
62            if unsafe { p1.get_unchecked(i) != p2.get_unchecked(i) } {
63                return false;
64            }
65        }
66        true
67    }
68}
69
70impl Eq for Symbol {}
71
72impl Symbol {
73    pub(crate) fn new(x: &str) -> Self {
74        Self::new_hashed(Hashed::new(x))
75    }
76
77    pub(crate) fn new_hashed(x: Hashed<&str>) -> Self {
78        let small_hash = x.hash();
79        let hash = small_hash.promote();
80        let len = x.key().len();
81        let len_words = len.div_ceil(mem::size_of::<usize>());
82        let mut payload = vec![0; len_words]; // 0 pad it at the end
83        unsafe {
84            copy_nonoverlapping(x.key().as_ptr(), payload.as_mut_ptr() as *mut u8, len);
85        }
86        Self {
87            hash,
88            len: len.try_into().unwrap(),
89            payload: payload.into_boxed_slice(),
90            small_hash,
91        }
92    }
93
94    #[inline]
95    pub(crate) fn hash(&self) -> u64 {
96        self.hash
97    }
98
99    pub(crate) fn as_str(&self) -> &str {
100        // All safe because we promise we started out with a str
101        unsafe {
102            let s = slice::from_raw_parts(self.payload.as_ptr() as *const u8, self.len as usize);
103            str::from_utf8_unchecked(s)
104        }
105    }
106
107    #[inline]
108    pub(crate) fn as_aligned_padded_str(&self) -> AlignedPaddedStr {
109        unsafe { AlignedPaddedStr::new(self.len as usize, self.payload.as_ptr()) }
110    }
111
112    pub(crate) fn as_str_hashed(&self) -> Hashed<&str> {
113        Hashed::new_unchecked(self.small_hash, self.as_str())
114    }
115
116    pub(crate) fn small_hash(&self) -> StarlarkHashValue {
117        self.small_hash
118    }
119}
