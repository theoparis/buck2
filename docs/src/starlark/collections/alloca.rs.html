33/// Holds the allocation.
34struct Buffer {
35    ptr: NonNull<u8>,
36    layout: Layout,
37}
38
39impl Buffer {
40    fn alloc(layout: Layout) -> Buffer {
41        let ptr = unsafe { alloc(layout) };
42        let ptr = NonNull::new(ptr).unwrap();
43        Buffer { ptr, layout }
44    }
45
46    fn ptr(&self) -> *mut u8 {
47        self.ptr.as_ptr()
48    }
49
50    fn end(&self) -> *mut u8 {
51        unsafe { self.ptr.as_ptr().add(self.layout.size()) }
52    }
53
54    fn size_words(&self) -> usize {
55        self.layout.size() / ALIGN
56    }
57}
58
59impl Drop for Buffer {
60    fn drop(&mut self) {
61        unsafe { dealloc(self.ptr.as_ptr().cast(), self.layout) }
62    }
63}
64
65/// We'd love to use the real `alloca`, but don't want to blow through the stack space,
66/// so define our own wrapper.
67/// We use a single continuous buffer. When it needs upgrading, we double it and keep the old one around.
68pub(crate) struct Alloca {
69    // An alternative design would be to bake the <T> into the type, so all allocations are of the same type.
70    // Benchmarking that, even if the most optimistic scenario (just reallocating a single element)
71    // the performance difference is only 2%, so keep the flexibility of not needing to predeclare the type.
72    alloc: Cell<*mut Align>,
73    end: Cell<*mut Align>,
74    buffers: RefCell<Vec<Buffer>>,
75}
76
77const INITIAL_SIZE: usize = 1000000; // ~ 1Mb
78
79type Align = u64;
80const ALIGN: usize = mem::size_of::<Align>();
81
82impl Alloca {
83    pub fn new() -> Self {
84        Self::with_capacity(INITIAL_SIZE)
85    }
86
87    pub fn with_capacity(size_bytes: usize) -> Self {
88        let size_words = size_bytes.div_ceil(ALIGN);
89        let layout = Layout::array::<Align>(size_words).unwrap();
90        let buffer = Buffer::alloc(layout);
91        Self {
92            alloc: Cell::new(buffer.ptr().cast()),
93            end: Cell::new(buffer.end().cast()),
94            buffers: RefCell::new(vec![buffer]),
95        }
96    }
97
98    fn assert_state(&self) {
99        unsafe {
100            debug_assert!(self.end.get().offset_from(self.alloc.get()) >= 0);
101            debug_assert!(
102                self.end.get().offset_from(self.alloc.get()) as usize
103                    <= self.buffers.borrow().last().unwrap().size_words()
104            );
105        }
106    }
107
108    #[cold]
109    #[inline(never)]
110    fn allocate_more(&self, len: usize, one: Layout) {
111        let want =
112            Layout::from_size_align(one.size().checked_mul(len).unwrap(), one.align()).unwrap();
113        assert!(want.align() <= mem::size_of::<Align>());
114        let size_words =
115            self.buffers.borrow().last().unwrap().size_words() * 2 + want.size() / ALIGN;
116        let layout = Layout::array::<Align>(size_words).unwrap();
117        let buffer = Buffer::alloc(layout);
118        let pointer = buffer.ptr().cast();
119        let end = buffer.end().cast();
120        self.buffers.borrow_mut().push(buffer);
121        self.alloc.set(pointer);
122        self.end.set(end);
123    }
124
125    /// Convert remaining capacity in words to remaining capacity in `T`.
126    #[inline(always)]
127    fn rem_in_words_to_rem_in_t<T>(rem_in_words: usize) -> usize {
128        assert!(mem::align_of::<T>() <= ALIGN);
129        rem_in_words * ALIGN / mem::size_of::<T>()
130    }
131
132    #[inline(always)]
133    fn len_in_to_to_len_in_words<T>(len: usize) -> usize {
134        if mem::size_of::<T>() % ALIGN == 0 {
135            // Special case to make common case fast:
136            // https://rust.godbolt.org/z/adh3nzdzs
137            len * (mem::size_of::<T>() / ALIGN)
138        } else {
139            // Multiplication does not overflow because we know that `[T; len]`
140            // is within the size of the current buffer, which is smaller than `isize::MAX`.
141            (len * mem::size_of::<T>()).div_ceil(ALIGN)
142        }
143    }
144
145    /// Note that the `Drop` for the `T` will not be called. That's safe if there is no `Drop`,
146    /// or you call it yourself.
147    #[inline(always)]
148    pub fn alloca_uninit<T, R>(&self, len: usize, k: impl FnOnce(&mut [MaybeUninit<T>]) -> R) -> R {
149        self.assert_state();
150
151        assert!(mem::align_of::<T>() <= ALIGN);
152
153        let mut start = self.alloc.get();
154
155        let rem_words = unsafe { self.end.get().offset_from(start) as usize };
156        let rem_in_t = Self::rem_in_words_to_rem_in_t::<T>(rem_words);
157        if unlikely(len > rem_in_t) {
158            self.allocate_more(len, Layout::new::<T>());
159            start = self.alloc.get();
160        }
161
162        let size_words = Self::len_in_to_to_len_in_words::<T>(len);
163
164        let stop = start.wrapping_add(size_words);
165        let old = start;
166        self.alloc.set(stop);
167        let data = start as *mut MaybeUninit<T>;
168        let slice = unsafe { slice::from_raw_parts_mut(data, len) };
169        let res = k(slice);
170
171        // If the pointer changed, it means a callback called alloca again,
172        // which allocated a new buffer. So we are abandoning the current allocation here,
173        // and new allocations will use the new buffer even if the current buffer has space.
174        if likely(std::ptr::eq(self.alloc.get(), stop)) {
175            self.alloc.set(old);
176        }
177
178        self.assert_state();
179
180        res
181    }
182
183    #[inline(always)]
184    pub fn alloca_init<T, R>(
185        &self,
186        len: usize,
187        mut init: impl FnMut() -> T,
188        k: impl FnOnce(&mut [T]) -> R,
189    ) -> R {
190        self.alloca_uninit(len, |data| {
191            for x in data.iter_mut() {
192                x.write(init());
193            }
194            let data = unsafe { &mut *(data as *mut [MaybeUninit<T>] as *mut [T]) };
195            k(data)
196        })
197    }
198
199    #[allow(dead_code)] // Dead, but morally a sensible API to provide, and useful for testing
200    #[inline(always)]
201    pub fn alloca_fill<T: Copy, R>(&self, len: usize, fill: T, k: impl FnOnce(&mut [T]) -> R) -> R {
202        self.alloca_init(len, || fill, k)
203    }
204
205    fn alloca_concat_slow<T: Clone, R, F>(&self, x: &[T], y: &[T], k: F) -> R
206    where
207        F: FnOnce(&[T]) -> R,
208    {
209        // We clone the input into a slice allocated in `Alloca`,
210        // slice does not invoke `Drop`, so we do drop explicitly.
211        struct DropSliceGuard<A>(*mut [A]);
212        impl<A> Drop for DropSliceGuard<A> {
213            fn drop(&mut self) {
214                unsafe {
215                    ptr::drop_in_place(self.0);
216                }
217            }
218        }
219
220        self.alloca_uninit(x.len() + y.len(), |xy| {
221            let (x_uninit, y_uninit) = xy.split_at_mut(x.len());
222            let x = maybe_uninit_write_slice_cloned(x_uninit, x);
223            let _x_drop_guard = DropSliceGuard(x);
224            let y = maybe_uninit_write_slice_cloned(y_uninit, y);
225            let _y_drop_guard = DropSliceGuard(y);
226            let xy = unsafe { &mut *(xy as *mut [MaybeUninit<T>] as *mut [T]) };
227            k(xy)
228        })
229    }
230
231    /// Concat two slices and invoke the callback with the result.
232    /// Use either slice as is if the other is empty,
233    /// otherwise clone the elements into a temporary slice.
234    #[inline(always)]
235    pub(crate) fn alloca_concat<T: Clone, R, F>(&self, x: &[T], y: &[T], k: F) -> R
236    where
237        F: FnOnce(&[T]) -> R,
238    {
239        if x.is_empty() {
240            k(y)
241        } else if y.is_empty() {
242            k(x)
243        } else {
244            self.alloca_concat_slow(x, y, k)
245        }
246    }
247}

253    #[allow(clippy::identity_op)]
254    #[test]
255    fn test_rem_in_words_to_rem_in_t() {
256        assert_eq!(10, Alloca::rem_in_words_to_rem_in_t::<Align>(10));
257        assert_eq!(
258            1 * ALIGN / mem::size_of::<u32>(),
259            Alloca::rem_in_words_to_rem_in_t::<u32>(1)
260        );
261        assert_eq!(
262            2 * ALIGN / mem::size_of::<u32>(),
263            Alloca::rem_in_words_to_rem_in_t::<u32>(2)
264        );
265        assert_eq!(1, Alloca::rem_in_words_to_rem_in_t::<[u8; ALIGN - 1]>(1));
266        assert_eq!(1, Alloca::rem_in_words_to_rem_in_t::<[u8; ALIGN]>(1));
267        assert_eq!(0, Alloca::rem_in_words_to_rem_in_t::<[u8; ALIGN + 1]>(1));
268        assert_eq!(2, Alloca::rem_in_words_to_rem_in_t::<[u8; ALIGN - 1]>(2));
269        assert_eq!(2, Alloca::rem_in_words_to_rem_in_t::<[u8; ALIGN]>(2));
270        assert_eq!(1, Alloca::rem_in_words_to_rem_in_t::<[u8; ALIGN + 1]>(2));
271    }

273    #[test]
274    fn test_len_in_t_to_len_in_words() {
275        assert_eq!(10, Alloca::len_in_to_to_len_in_words::<Align>(10));
276        assert_eq!(1, Alloca::len_in_to_to_len_in_words::<u8>(1));
277        assert_eq!(1, Alloca::len_in_to_to_len_in_words::<u8>(ALIGN - 1));
278        assert_eq!(1, Alloca::len_in_to_to_len_in_words::<u8>(ALIGN));
279        assert_eq!(2, Alloca::len_in_to_to_len_in_words::<u8>(ALIGN + 1));
280        assert_eq!(2, Alloca::len_in_to_to_len_in_words::<u8>(2 * ALIGN - 1));
281        assert_eq!(2, Alloca::len_in_to_to_len_in_words::<u8>(2 * ALIGN));
282        assert_eq!(3, Alloca::len_in_to_to_len_in_words::<u8>(2 * ALIGN + 1));
283        assert_eq!(1, Alloca::len_in_to_to_len_in_words::<[u8; ALIGN - 1]>(1));
284        assert_eq!(1, Alloca::len_in_to_to_len_in_words::<[u8; ALIGN]>(1));
285        assert_eq!(2, Alloca::len_in_to_to_len_in_words::<[u8; ALIGN + 1]>(1));
286        assert_eq!(2, Alloca::len_in_to_to_len_in_words::<[u8; ALIGN - 1]>(2));
287        assert_eq!(2, Alloca::len_in_to_to_len_in_words::<[u8; ALIGN]>(2));
288        assert_eq!(3, Alloca::len_in_to_to_len_in_words::<[u8; ALIGN + 1]>(2));
289    }

291    #[test]
292    fn test_alloca() {
293        // Use a small capacity to encourage overflow behaviour
294        let a = Alloca::with_capacity(100);
295        a.alloca_fill(3, 8usize, |xs| {
296            xs[0] = 5;
297            xs[2] = xs[0] + xs[1] + 15;
298            a.alloca_fill(200, 18usize, |ys| {
299                assert_eq!(xs[2], 8 + 5 + 15);
300                assert_eq!(ys[0], 18);
301                assert_eq!(ys[200 - 1], 18);
302            });
303            a.alloca_fill(3, 1u64, |_| {});
304            assert_eq!(xs[2], 8 + 5 + 15);
305        })
306    }

308    #[test]
309    fn trigger_bug() {
310        let a = Alloca::with_capacity(100);
311        for _ in 0..100 {
312            a.alloca_fill(10, 17usize, |_| {
313                a.alloca_fill(1000, 19usize, |_| {});
314            });
315        }

334    fn test_alloca_concat() {
335        let a = Alloca::new();
336        let x = vec!["ab".to_owned()];
337        let y = vec!["cd".to_owned()];
338        a.alloca_concat(&x, &[], |xy| {
339            // No copy here.
340            assert_eq!(xy.as_ptr(), x.as_ptr());
341        });
342        a.alloca_concat(&[], &x, |xy| {
343            // No copy here.
344            assert_eq!(xy.as_ptr(), x.as_ptr());
345        });
346        a.alloca_concat(&x, &y, |xy| {
347            assert_eq!(&["ab".to_owned(), "cd".to_owned()], xy);
348        });
