18use crate::module::typ::StarArg;
19use crate::module::typ::StarArgPassStyle;
20
21/// Function parameters validated and split into modes.
22pub(crate) struct ParamSpec<'a> {
23    pub(crate) pos_only: Vec<&'a StarArg>,
24    pub(crate) pos_or_named: Vec<&'a StarArg>,
25    pub(crate) args: Option<&'a StarArg>,
26    pub(crate) named_only: Vec<&'a StarArg>,
27    pub(crate) kwargs: Option<&'a StarArg>,
28}
29
30impl<'a> ParamSpec<'a> {
31    pub(crate) fn split(star_args: &'a [StarArg]) -> syn::Result<ParamSpec<'a>> {
32        #[derive(PartialEq, Eq, PartialOrd, Ord)]
33        enum CurrentParamStyle {
34            PosOnly,
35            PosOrNamed,
36            NamedOnly,
37            NoMore,
38        }
39
40        let mut seen_optional = false;
41
42        let mut pos_only = Vec::new();
43        let mut pos_or_named = Vec::new();
44        let mut args = None;
45        let mut named_only = Vec::new();
46        let mut kwargs = None;
47
48        let mut last_param_style = CurrentParamStyle::PosOnly;
49        for arg in star_args {
50            match arg.pass_style {
51                StarArgPassStyle::Args => {
52                    if last_param_style >= CurrentParamStyle::NamedOnly {
53                        return Err(syn::Error::new(
54                            arg.span,
55                            "`args` cannot follow named-only parameters",
56                        ));
57                    }
58                    if args.is_some() {
59                        return Err(syn::Error::new(
60                            arg.span,
61                            "Cannot have more than one `args` parameter (internal error)",
62                        ));
63                    }
64                    args = Some(arg);
65                    last_param_style = CurrentParamStyle::NamedOnly;
66                }
67                StarArgPassStyle::Kwargs => {
68                    if last_param_style == CurrentParamStyle::NoMore {
69                        return Err(syn::Error::new(
70                            arg.span,
71                            "Cannot have more than one `kwargs` parameter",
72                        ));
73                    }
74                    if kwargs.is_some() {
75                        return Err(syn::Error::new(
76                            arg.span,
77                            "Cannot have more than one `kwargs` parameter (internal error)",
78                        ));
79                    }
80                    kwargs = Some(arg);
81                    last_param_style = CurrentParamStyle::NoMore;
82                }
83                StarArgPassStyle::PosOnly => {
84                    if last_param_style > CurrentParamStyle::PosOnly {
85                        return Err(syn::Error::new(
86                            arg.span,
87                            "Positional-only parameter after non-positional-only",
88                        ));
89                    }
90                    last_param_style = CurrentParamStyle::PosOnly;
91                    pos_only.push(arg);
92                }
93                StarArgPassStyle::PosOrNamed => {
94                    if last_param_style > CurrentParamStyle::PosOrNamed {
95                        return Err(syn::Error::new(
96                            arg.span,
97                            "Positional-or-named parameter after named-only",
98                        ));
99                    }
100                    last_param_style = CurrentParamStyle::PosOrNamed;
101                    pos_or_named.push(arg);
102                }
103                StarArgPassStyle::NamedOnly => {
104                    if last_param_style > CurrentParamStyle::NamedOnly {
105                        return Err(syn::Error::new(
106                            arg.span,
107                            "Named-only parameter cannot follow kwargs",
108                        ));
109                    }
110                    named_only.push(arg);
111                    last_param_style = CurrentParamStyle::NamedOnly;
112                }
113            }
114
115            let optional = arg.default.is_some() || arg.is_option();
116
117            if last_param_style <= CurrentParamStyle::PosOrNamed {
118                if seen_optional && !optional {
119                    return Err(syn::Error::new(
120                        arg.span,
121                        "Positional parameter without default after optional parameter",
122                    ));
123                }
124            }
125
126            seen_optional |= optional;
127        }
128        Ok(ParamSpec {
129            pos_only,
130            pos_or_named,
131            args,
132            named_only,
133            kwargs,
134        })
135    }
136}
