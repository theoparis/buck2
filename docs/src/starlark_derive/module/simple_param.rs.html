18use proc_macro2::TokenStream;
19use quote::ToTokens;
20use quote::TokenStreamExt;
21use syn::parse::Parse;
22use syn::parse::ParseStream;
23use syn::spanned::Spanned;
24
25/// Simple function parameter, subset of what Rust language supports.
26#[derive(Debug, Clone)]
27pub(crate) struct SimpleParam {
28    pub(crate) attrs: Vec<syn::Attribute>,
29    pub(crate) mutability: Option<syn::Token![mut]>,
30    pub(crate) ident: syn::Ident,
31    pub(crate) ty: syn::Type,
32}
33
34impl SimpleParam {
35    fn fn_arg(&self) -> syn::FnArg {
36        let attrs = &self.attrs;
37        let mutability = &self.mutability;
38        let ident = &self.ident;
39        let ty = &self.ty;
40        syn::parse_quote! {
41            #(#attrs)*
42            #mutability #ident: #ty
43        }
44    }
45
46    pub(crate) fn from_fn_arg(arg: syn::FnArg) -> syn::Result<Self> {
47        match arg {
48            syn::FnArg::Receiver(_) => Err(syn::Error::new(
49                arg.span(),
50                "`self` is not used in Starlark native functions",
51            )),
52            syn::FnArg::Typed(syn::PatType {
53                attrs,
54                pat,
55                colon_token: _,
56                ty,
57            }) => {
58                let syn::Pat::Ident(ident) = *pat else {
59                    return Err(syn::Error::new_spanned(
60                        pat,
61                        "Only simple identifiers are allowed in Starlark function parameters",
62                    ));
63                };
64                let syn::PatIdent {
65                    attrs: pat_ident_attrs,
66                    by_ref,
67                    mutability,
68                    ident,
69                    subpat,
70                } = ident;
71                if by_ref.is_some() {
72                    return Err(syn::Error::new_spanned(
73                        by_ref,
74                        "Starlark function parameters cannot use `ref`",
75                    ));
76                }
77                if let Some(subpat) = subpat {
78                    return Err(syn::Error::new_spanned(
79                        subpat.0,
80                        "Starlark function parameters cannot use subpatterns",
81                    ));
82                }
83                if let Some(attr) = pat_ident_attrs.into_iter().next() {
84                    return Err(syn::Error::new_spanned(
85                        attr,
86                        "Starlark function parameters cannot have attributes",
87                    ));
88                }
89                Ok(SimpleParam {
90                    attrs,
91                    mutability,
92                    ident,
93                    ty: *ty,
94                })
95            }
96        }
97    }
98}
99
100impl Parse for SimpleParam {
101    fn parse(input: ParseStream) -> syn::Result<Self> {
102        let arg = syn::FnArg::parse(input)?;
103        SimpleParam::from_fn_arg(arg)
104    }
105}
106
107impl ToTokens for SimpleParam {
108    fn to_tokens(&self, tokens: &mut TokenStream) {
109        tokens.append_all(self.fn_arg().into_token_stream())
110    }
111}
