35pub(crate) fn render(x: StarModule) -> syn::Result<TokenStream> {
36    Ok(render_impl(x)?.to_token_stream())
37}
38
39fn render_impl(x: StarModule) -> syn::Result<syn::ItemFn> {
40    let StarModule {
41        mut input,
42        docstring,
43        stmts,
44        module_kind,
45        generics,
46    } = x;
47    let statics = format_ident!("{}", module_kind.statics_type_name());
48    let stmts: Vec<_> = stmts
49        .into_iter()
50        .map(|s| render_stmt(s, &generics))
51        .collect::<syn::Result<_>>()?;
52    let set_docstring = docstring.map(|ds| quote!(globals_builder.set_docstring(#ds);));
53
54    let inner_fn = syn::ItemFn {
55        attrs: Default::default(),
56        vis: syn::Visibility::Inherited,
57        sig: syn::Signature {
58            ident: syn::Ident::new("build", input.sig.ident.span()),
59            ..input.sig.clone()
60        },
61        block: syn::parse_quote! {
62            {
63                #set_docstring
64                #( #stmts )*
65                // Mute warning if stmts is empty.
66                let _ = globals_builder;
67            }
68        },
69    };
70    let turbofish = generics.turbofish();
71    input.block = syn::parse_quote! {
72        {
73            #inner_fn
74            static RES: starlark::environment::#statics = starlark::environment::#statics::new();
75            RES.populate(build #turbofish, globals_builder);
76        }
77    };
78    Ok(input)
79}

81fn render_stmt(x: StarStmt, generics: &StarGenerics) -> syn::Result<syn::Stmt> {
82    match x {
83        StarStmt::Const(x) => Ok(render_const(x)),
84        StarStmt::Attr(x) => Ok(render_attr(x, generics)),
85        StarStmt::Fun(x) => render_fun(x, generics),
86    }
87}
88
89fn render_const(x: StarConst) -> syn::Stmt {
90    let StarConst { name, ty, value } = x;
91    let name = ident_string(&name);
92    syn::parse_quote! {
93        globals_builder.set::<#ty>(#name, #value);
94    }
95}

97fn render_attr(x: StarAttr, generics: &StarGenerics) -> syn::Stmt {
98    let StarAttr {
99        name,
100        this,
101        heap,
102        attrs,
103        return_type,
104        speculative_exec_safe,
105        body,
106        docstring,
107    } = x;
108    let name_str = ident_string(&name);
109    let name_inner = syn::Ident::new(&format!("{name_str}__inner"), name.span());
110    let docstring: syn::Expr = match docstring {
111        Some(d) => render_some(syn::parse_quote! { #d.to_owned() }),
112        None => render_none(),
113    };
114
115    let generic_decls = generics.decls();
116    let where_clause = generics.where_clause();
117    let turbofish = generics.turbofish();
118
119    let let_heap = if let Some(SpecialParam {
120        param: SimpleParam { ident, ty, .. },
121    }) = heap
122    {
123        Some(quote! { let #ident: #ty = __heap; })
124    } else {
125        None
126    };
127
128    let this_value: syn::Ident = syn::parse_quote! { s_this_value };
129    let this_return_type: &syn::Type = &this.param.ty;
130
131    let unpack = this.render_prepare(&this.param.ident, &this_value);
132
133    let inner: syn::ItemFn = syn::parse_quote! {
134        #( #attrs )*
135        #[allow(non_snake_case)] // Starlark doesn't have this convention
136        #[allow(unused_variables)]
137        fn #name_inner #generic_decls(
138            this: #this_return_type,
139            __heap: &'v starlark::values::Heap,
140        ) -> #return_type
141        #where_clause
142        {
143            #let_heap
144            #body
145        }
146    };
147
148    let outer: syn::ItemFn = syn::parse_quote! {
149        #[allow(non_snake_case)]
150        fn #name #generic_decls(
151            _ignored: std::option::Option<starlark::values::FrozenValue>,
152            #this_value: starlark::values::Value<'v>,
153            heap: &'v starlark::values::Heap,
154        ) -> starlark::Result<starlark::values::Value<'v>>
155        #where_clause
156        {
157            #unpack
158            Ok(heap.alloc(#name_inner #turbofish(this, heap)?))
159        }
160    };
161
162    syn::parse_quote! {
163        {
164            #inner
165            #outer
166
167            globals_builder.set_attribute_fn(
168                #name_str,
169                #speculative_exec_safe,
170                #docstring,
171                starlark::values::type_repr::type_repr_from_attr_impl(#name_inner #turbofish),
172                #name #turbofish
173            );
174        }
175    }
176}

178pub(crate) fn render_none() -> syn::Expr {
179    syn::parse_quote! { std::option::Option::None }
180}
181
182pub(crate) fn render_some(expr: syn::Expr) -> syn::Expr {
183    syn::parse_quote! { std::option::Option::Some(#expr) }
184}
