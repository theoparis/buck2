39// This macro does two related derivations depending on whether there are any generic parameters.
40//
41// struct A<T>(...) => coerce A<T1> to A<T2> if coerce T1 to T2 and all the fields support it
42pub fn derive_coerce(input: proc_macro::TokenStream) -> proc_macro::TokenStream {
43    let input = parse_macro_input!(input as DeriveInput);
44    match derive_coerce_impl(input) {
45        Ok(x) => x.into(),
46        Err(e) => e.to_compile_error().into(),
47    }
48}
49
50fn derive_coerce_impl(input: DeriveInput) -> syn::Result<proc_macro2::TokenStream> {
51    check_repr(&input)?;
52
53    if input.generics.type_params().count() == 0 {
54        Err(syn::Error::new_spanned(
55            &input,
56            "`Coerce` can only be derived for types with type parameters",
57        ))
58    } else {
59        derive_coerce_params(input)
60    }
61}
62
63#[derive(Copy, Clone)]
64enum ParamNameMapping {
65    From,
66    To,
67}
68
69impl ParamNameMapping {
70    fn format_ident(self, ident: &Ident) -> Ident {
71        match self {
72            ParamNameMapping::From => format_ident!("From{}", ident),
73            ParamNameMapping::To => format_ident!("To{}", ident),
74        }
75    }
76}
77
78fn derive_coerce_params(input: DeriveInput) -> syn::Result<proc_macro2::TokenStream> {
79    let lifetimes = input.generics.lifetimes().collect::<Vec<_>>();
80
81    // unsafe impl <To__T, From__T> Coerce<X<To__T>> X<From__T> where ...
82    let name = &input.ident;
83    let mut ty_args = HashSet::new();
84    let mut ty_args_to = Vec::new();
85    let mut ty_args_from = Vec::new();
86    let mut ty_args_to1 = Vec::new();
87    let mut ty_args_from1 = Vec::new();
88    for t in input.generics.type_params() {
89        ty_args.insert(t.ident.clone());
90        let to_ident = ParamNameMapping::To.format_ident(&t.ident);
91        let from_ident = ParamNameMapping::From.format_ident(&t.ident);
92        ty_args_to.push(to_ident.clone());
93        ty_args_from.push(from_ident.clone());
94        if t.bounds.is_empty() {
95            ty_args_to1.push(quote! { #to_ident });
96            ty_args_from1.push(quote! { #from_ident });
97        } else {
98            let bounds = &t.bounds;
99            ty_args_to1.push(quote! { #to_ident: #bounds });
100            ty_args_from1.push(quote! { #from_ident: #bounds });
101        }
102    }
103
104    let mut constraints = Vec::new();
105    let fields = collect_fields(&input)?;
106    for x in fields {
107        let mut to_ty = x.ty.clone();
108        let mut from_ty = x.ty.clone();
109        replace_type(&mut to_ty, &ty_args, ParamNameMapping::To)?;
110        replace_type(&mut from_ty, &ty_args, ParamNameMapping::From)?;
111        if to_ty != from_ty {
112            constraints.push(quote! { #from_ty : starlark::coerce::Coerce< #to_ty >});
113        }
114    }
115
116    Ok(quote! {
117        unsafe impl < #(#lifetimes,)* #(#ty_args_from1,)* #(#ty_args_to1,)* >
118            starlark::coerce::Coerce<#name < #(#lifetimes,)* #(#ty_args_to,)* >>
119            for #name < #(#lifetimes,)* #(#ty_args_from,)* >
120            where #(#constraints,)* {}
121    })
122}
123
124fn collect_fields(input: &DeriveInput) -> syn::Result<Vec<&Field>> {
125    match &input.data {
126        Data::Struct(x) => Ok(x.fields.iter().collect()),
127        Data::Enum(x) => Ok(x.variants.iter().flat_map(|x| &x.fields).collect()),
128        Data::Union(..) => Err(syn::Error::new_spanned(
129            input,
130            "Type-parameter cannot be a union",
131        )),
132    }
133}

135fn replace_type(
136    ty: &mut Type,
137    idents: &HashSet<Ident>,
138    mapping: ParamNameMapping,
139) -> syn::Result<()> {
140    match ty {
141        Type::Path(x)
142            if x.qself.is_none()
143                && x.path.segments.len() == 1
144                && x.path.segments[0].arguments.is_empty() =>
145        {
146            let i = &mut x.path.segments[0].ident;
147            if idents.contains(i) {
148                *i = mapping.format_ident(i);
149            }
150            Ok(())
151        }
152        _ => descend_type(ty, |ty| replace_type(ty, idents, mapping)),
153    }
154}

156/// Descend into all the nested type values within a type,
157/// or return an error if you don't know how.
158fn descend_type(ty: &mut Type, op: impl Fn(&mut Type) -> syn::Result<()>) -> syn::Result<()> {
159    match ty {
160        Type::Array(x) => op(&mut x.elem),
161        Type::Group(x) => op(&mut x.elem),
162        Type::Never(_) => Ok(()),
163        Type::Paren(x) => op(&mut x.elem),
164        Type::Path(x) => {
165            if let Some(qself) = &mut x.qself {
166                op(&mut qself.ty)?;
167            }
168            for p in x.path.segments.iter_mut() {
169                match &mut p.arguments {
170                    PathArguments::None => {}
171                    PathArguments::AngleBracketed(x) => {
172                        x.args.iter_mut().try_for_each(|x| match x {
173                            GenericArgument::Type(x) => op(x),
174                            _ => Ok(()),
175                        })?
176                    }
177                    PathArguments::Parenthesized(x) => {
178                        x.inputs.iter_mut().try_for_each(&op)?;
179                        if let ReturnType::Type(_, ty) = &mut x.output {
180                            op(ty)?;
181                        }
182                    }
183                }
184            }
185            Ok(())
186        }
187        Type::Ptr(x) => op(&mut x.elem),
188        Type::Reference(x) => op(&mut x.elem),
189        Type::Slice(x) => op(&mut x.elem),
190        Type::Tuple(xs) => xs.elems.iter_mut().try_for_each(op),
191        _ => Err(syn::Error::new_spanned(ty, "Unsupported type")),
192    }
193}

195// Taken from ref_cast: https://github.com/dtolnay/ref-cast/blob/eadd839cc0db116e4fc9bfee5dd29dde82089eed/derive/src/lib.rs#L91
196fn check_repr(input: &DeriveInput) -> syn::Result<()> {
197    let mut has_repr = false;
198    let mut errors = None;
199    let mut push_error = |error| match &mut errors {
200        Some(errors) => syn::Error::combine(errors, error),
201        None => errors = Some(error),
202    };
203
204    for attr in &input.attrs {
205        if attr.path().is_ident("repr") {
206            if let Err(error) = attr.parse_args_with(|input: ParseStream| {
207                while !input.is_empty() {
208                    let path = input.call(Path::parse_mod_style)?;
209                    if path.is_ident("C") || path.is_ident("transparent") {
210                        has_repr = true;
211                    } else if path.is_ident("packed") {
212                        // ignore
213                    } else {
214                        let meta_item_span = if input.peek(token::Paren) {
215                            let group: TokenTree = input.parse()?;
216                            quote!(#path #group)
217                        } else if input.peek(Token![=]) {
218                            let eq_token: Token![=] = input.parse()?;
219                            let value: Expr = input.parse()?;
220                            quote!(#path #eq_token #value)
221                        } else {
222                            quote!(#path)
223                        };
224                        let msg = if path.is_ident("align") {
225                            "aligned repr on struct that implements Coerce is not supported"
226                        } else {
227                            "unrecognized repr on struct that implements Coerce"
228                        };
229                        push_error(syn::Error::new_spanned(meta_item_span, msg));
230                    }
231                    if !input.is_empty() {
232                        input.parse::<Token![,]>()?;
233                    }
234                }
235                Ok(())
236            }) {
237                push_error(error);
238            }
239        }
240    }
241
242    if !has_repr {
243        let mut requires_repr = syn::Error::new(
244            Span::call_site(),
245            "Coerce trait requires #[repr(C)] or #[repr(transparent)]",
246        );
247        if let Some(errors) = errors {
248            requires_repr.combine(errors);
249        }
250        errors = Some(requires_repr);
251    }
252
253    match errors {
254        None => Ok(()),
255        Some(errors) => Err(errors),
256    }
257}
