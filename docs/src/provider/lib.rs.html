13//! `std::any::Provider` replacement (which is removed Rust stdlib).
14
15use std::any::TypeId;
16use std::marker::PhantomData;
17
18trait DemandImpl {
19    fn tag() -> TypeId;
20}
21
22struct DemandValue<T: 'static> {
23    value: Option<T>,
24}
25
26struct DemandRef<'a, T: ?Sized + 'static> {
27    value: Option<&'a T>,
28}
29
30impl<T: 'static> DemandImpl for DemandValue<T> {
31    fn tag() -> TypeId {
32        TypeId::of::<DemandValue<T>>()
33    }
34}
35
36impl<T: ?Sized + 'static> DemandImpl for DemandRef<'_, T> {
37    fn tag() -> TypeId {
38        TypeId::of::<DemandRef<'static, T>>()
39    }
40}
41
42/// Supplied value can be passed using this object.
43pub struct Demand<'a> {
44    /// What we want?
45    demand_impl_tag: TypeId,
46    // SAFETY: Must be a `&'a mut` to a `DemandRef<T>` or `DemandValue<T>`, matching the `TypeId` above
47    demand_impl: *mut (),
48    _phantom: PhantomData<&'a mut ()>,
49}
50
51impl<'a> Demand<'a> {
52    fn new<I>(demand_impl: &'a mut I) -> Demand<'a>
53    where
54        I: DemandImpl,
55    {
56        Demand {
57            demand_impl_tag: I::tag(),
58            demand_impl: demand_impl as *mut _ as *mut (),
59            _phantom: PhantomData,
60        }
61    }
62
63    /// Provide a value. Discard the value if the type is not matched.
64    pub fn provide_value<T: 'static>(&mut self, value: T) {
65        self.provide_value_with(|| value);
66    }
67
68    /// Provide a value. Discard the value if the type is not matched.
69    pub fn provide_value_with<T: 'static>(&mut self, f: impl FnOnce() -> T) {
70        if self.demand_impl_tag == DemandValue::<T>::tag() {
71            // SAFETY: We've checked that this is a pointer to a `DemandValue<T>`
72            let demand_impl = unsafe { &mut *(self.demand_impl as *mut DemandValue<T>) };
73            demand_impl.value = Some(f());
74        }
75    }
76
77    /// Provide a reference. Discard the value if the type is not matched.
78    pub fn provide_ref<T: ?Sized + 'static>(&mut self, value: &'a T) {
79        self.provide_ref_with(|| value);
80    }
81
82    /// Provide a reference. Discard the value if the type is not matched.
83    pub fn provide_ref_with<T: ?Sized + 'static>(&mut self, f: impl FnOnce() -> &'a T) {
84        if self.demand_impl_tag == DemandRef::<T>::tag() {
85            // SAFETY: We've checked that this is a pointer to a `DemandRef<T>`
86            let demand_impl = unsafe { &mut *(self.demand_impl as *mut DemandRef<T>) };
87            demand_impl.value = Some(f());
88        }
89    }
90}
91
92/// Something that can provide a value of a given type.
93pub trait Provider {
94    /// Provide a value of a given type which is specified by the `demand` object.
95    fn provide<'a>(&'a self, demand: &mut Demand<'a>);
96}
97
98/// Request a value from the provider.
99pub fn request_value<T: 'static>(provider: &(impl Provider + ?Sized)) -> Option<T> {
100    let mut demand_impl = DemandValue { value: None };
101    provider.provide(&mut Demand::new(&mut demand_impl));
102    demand_impl.value
103}
104
105/// Request a reference from the provider.
106pub fn request_ref<T: ?Sized + 'static>(provider: &(impl Provider + ?Sized)) -> Option<&T> {
107    let mut demand_impl = DemandRef { value: None };
108    provider.provide(&mut Demand::new(&mut demand_impl));
109    demand_impl.value
110}

112#[cfg(test)]
113mod tests {
114    use crate::Demand;
115    use crate::Provider;
116    use crate::request_ref;
117    use crate::request_value;
118
119    struct My {
120        s0: String,
121        s1: Box<str>,
122        x0: Vec<u32>,
123        x1: Vec<u64>,
124    }
125
126    impl Provider for My {
127        fn provide<'a>(&'a self, demand: &mut Demand<'a>) {
128            demand.provide_value::<String>(self.s0.clone());
129            demand.provide_value_with::<Box<str>>(|| self.s1.clone());
130            demand.provide_ref::<[u32]>(&self.x0);
131            demand.provide_ref_with::<[u64]>(|| &self.x1);
132        }
133    }
134
135    #[test]
136    fn test() {
137        let my = My {
138            s0: "s0".to_owned(),
139            s1: "s1".into(),
140            x0: vec![0, 1, 2],
141            x1: vec![3, 4, 5],
142        };
143
144        let provider: &dyn Provider = &my;
145
146        assert_eq!(Some("s0".to_owned()), request_value(provider));
147        assert_eq!(Some(Box::<str>::from("s1")), request_value(provider));
148        assert_eq!(Some(&[0, 1, 2][..]), request_ref::<[u32]>(provider));
149        assert_eq!(Some(&[3, 4, 5][..]), request_ref::<[u64]>(provider));
150        assert_eq!(None, request_value::<bool>(provider));
151        assert_eq!(None, request_ref::<String>(provider));
152    }
153}
