11use crate::CriticalPathVertexData;
12use crate::OptionalVertexId;
13use crate::VertexData;
14use crate::VertexId;
15use crate::graph::PathCost;
16
17/// A container that can be used to access critical path from arbitrary vertices.
18///
19/// When constructed, this must maintain the invariants that predecessors and
20/// cost_from_source are from the same graph (thus ensuring that if the path
21/// tells us a node has n predecessors, then we can actually find n-non-null
22/// entries when following the path in the predecessor graph).
23pub struct CriticalPathAccessor {
24    pub(crate) predecessors: VertexData<OptionalVertexId>,
25    pub(crate) cost_from_source: VertexData<PathCost>,
26}
27
28impl CriticalPathAccessor {
29    /// Get the critical path for a given vertex. Passing an invalid vertex ID will return None.
30    pub fn critical_path_for_vertex(
31        &self,
32        sink: VertexId,
33    ) -> Option<(PathCost, CriticalPathVertexData<VertexId>)> {
34        let cost = *self.cost_from_source.get(sink)?;
35        let critical_path = self.critical_path_for_vertex_and_cost(sink, cost);
36        Some((cost, critical_path))
37    }
38
39    pub(crate) fn critical_path_for_vertex_and_cost(
40        &self,
41        sink: VertexId,
42        cost: PathCost,
43    ) -> CriticalPathVertexData<VertexId> {
44        let critical_path_len = cost.len as usize;
45        let mut critical_path = vec![VertexId::new(0); critical_path_len];
46        let mut idx: VertexId = sink;
47        for i in 0..critical_path_len {
48            critical_path[critical_path_len - 1 - i] = idx;
49            if i != critical_path_len - 1 {
50                // Unwrap safety: we know that the predecessor cannot be null
51                // because the critical path length tells us how many
52                // predecessors we expect.
53                idx = self.predecessors[idx].into_option().unwrap();
54            }
55        }
56        CriticalPathVertexData::new(critical_path)
57    }
58}
