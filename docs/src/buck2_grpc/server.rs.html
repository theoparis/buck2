27pub struct ServerHandle {
28    channel: DropNotifier,
29    handle: JoinHandle<anyhow::Result<()>>,
30}
31
32impl ServerHandle {
33    /// Tell the server to shutdown and wait for it to exit.
34    pub async fn shutdown(self) -> anyhow::Result<()> {
35        self.channel.notify_now();
36        self.handle.await.context("Failed to join task")?
37    }
38
39    /// Obtain the JoinHandle to the task driving the server, without asking the server to
40    /// shutdown.
41    pub fn into_join_handle(self) -> JoinHandle<anyhow::Result<()>> {
42        self.channel.cancel();
43        self.handle
44    }
45}
46
47pub fn spawn_oneshot<T, L>(io: T, router: Router<L>) -> ServerHandle
48where
49    T: AsyncRead + AsyncWrite + Send + Unpin + 'static + tonic::transport::server::Connected,
50    L: Layer<Routes> + Send + 'static,
51    L::Service: Service<
52            hyper::Request<tonic::body::BoxBody>,
53            Response = hyper::Response<tonic::body::BoxBody>,
54        > + Clone
55        + Send
56        + 'static,
57    <<L as Layer<Routes>>::Service as Service<hyper::Request<tonic::body::BoxBody>>>::Future:
58        Send + 'static,
59    <<L as Layer<Routes>>::Service as Service<hyper::Request<tonic::body::BoxBody>>>::Error:
60        Into<Box<dyn std::error::Error + Send + Sync>> + Send,
61{
62    // We reserve 2 slots here: one for the connection and one for the ServerHandle's
63    // shutdown.
64    let (send, mut recv) = mpsc::channel(2);
65
66    let io = ConnectionWithExtra {
67        inner: io,
68        notifier: DropNotifier::new(send.clone()),
69    };
70
71    let fut = async move {
72        // NOTE: We immediately yield the connection we have on hand, then just report no further
73        // connections going forward. This is important to ensure that the server doesn't just
74        // exist when it sees no connections (it's a little weird that the server exists even with
75        // open connections if they're not in the middle of a request, but thats's what it is).
76        let incoming = stream::once(future::ready(io))
77            .chain(stream::once(future::pending()))
78            .map(std::io::Result::Ok);
79
80        router
81            .serve_with_incoming_shutdown(incoming, async move {
82                let _ignored = recv.recv().await;
83            })
84            .await
85            .context("Server exited with an error")?;
86
87        Ok(())
88    };
89
90    let handle = tokio::task::spawn(fut);
91
92    ServerHandle {
93        channel: DropNotifier::new(send),
94        handle,
95    }
96}

98mod drop_notifier {
99    use tokio::sync::mpsc::Sender;
100
101    /// Upon being dropped, this notifies the sender is contains.
102    pub struct DropNotifier {
103        sender: Option<Sender<()>>,
104    }
105
106    impl DropNotifier {
107        pub fn new(sender: Sender<()>) -> Self {
108            Self {
109                sender: Some(sender),
110            }
111        }
112
113        /// Notify the sender this contains (this is equivalent to dropping the DropNotifier).
114        pub fn notify_now(self) {
115            drop(self)
116        }
117
118        /// Drop this DropNotifier without notifying. This is useful in cases where "defusing" the
119        /// DropNotifier is desired.
120        pub fn cancel(mut self) {
121            self.sender.take();
122        }
123    }
124
125    impl Drop for DropNotifier {
126        fn drop(&mut self) {
127            if let Some(sender) = self.sender.as_ref() {
128                let _ = sender.try_send(());
129            }
130        }
131    }
132}

134mod connection_with_extra {
135    use std::io;
136    use std::pin::Pin;
137    use std::task::Context;
138    use std::task::Poll;
139
140    use pin_project::pin_project;
141    use tokio::io::AsyncRead;
142    use tokio::io::AsyncWrite;
143    use tokio::io::ReadBuf;
144    use tonic::transport::server::Connected;
145
146    use super::DropNotifier;
147
148    /// A connection (AsyncWrite + AsyncRead + Connected), with a field for a DropNotifier. This
149    /// acts a connection, but lets us observe when it's dropped.
150    #[pin_project]
151    pub struct ConnectionWithExtra<I> {
152        #[pin]
153        pub inner: I,
154        pub notifier: DropNotifier,
155    }
156
157    impl<I> AsyncWrite for ConnectionWithExtra<I>
158    where
159        I: AsyncWrite,
160    {
161        fn poll_write(
162            self: Pin<&mut Self>,
163            cx: &mut Context<'_>,
164            buf: &[u8],
165        ) -> Poll<Result<usize, io::Error>> {
166            self.project().inner.poll_write(cx, buf)
167        }
168
169        fn poll_flush(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), io::Error>> {
170            self.project().inner.poll_flush(cx)
171        }
172
173        fn poll_shutdown(
174            self: Pin<&mut Self>,
175            cx: &mut Context<'_>,
176        ) -> Poll<Result<(), io::Error>> {
177            self.project().inner.poll_shutdown(cx)
178        }
179    }
180
181    impl<I> AsyncRead for ConnectionWithExtra<I>
182    where
183        I: AsyncRead,
184    {
185        fn poll_read(
186            self: Pin<&mut Self>,
187            cx: &mut Context<'_>,
188            buf: &mut ReadBuf<'_>,
189        ) -> Poll<io::Result<()>> {
190            self.project().inner.poll_read(cx, buf)
191        }
192    }
193
194    impl<I> Connected for ConnectionWithExtra<I>
195    where
196        I: Connected,
197    {
198        type ConnectInfo = I::ConnectInfo;
199
200        fn connect_info(&self) -> Self::ConnectInfo {
201            self.inner.connect_info()
202        }
203    }
204}
