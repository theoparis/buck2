11//! Used by golden tests written in rust, particularly useful for starlark tests
12//!
13//! This is mostly a copy from starlark_syntax/src/golden_test_template.rs.
14//! Copied so we wouldn't need to depend on starlark_syntax for tests.
15
16use std::env;
17use std::fmt::Write;
18use std::fs;
19
20use buck2_error::BuckErrorContext;
21
22const REGENERATE_VAR_NAME: &str = "BUCK2_RUST_REGENERATE_GOLDEN_TESTS";
23
24#[allow(clippy::write_literal)] // We mark generated files as generated, but not this file.
25fn make_golden(output: &str) -> String {
26    let mut golden = String::new();
27    writeln!(golden, "# {at}generated", at = "@").unwrap();
28    writeln!(
29        golden,
30        "# To regenerate, append -- --env {REGENERATE_VAR_NAME}=1 and re-run the test"
31    )
32    .unwrap();
33    writeln!(golden).unwrap();
34    writeln!(golden, "{}", output.trim_end()).unwrap();
35    golden
36}
37
38/// Common code for golden tests.
39pub fn golden_test_template(golden_rel_path: &str, output: &str) {
40    assert!(golden_rel_path.contains(".golden"));
41
42    let manifest_dir =
43        env::var("CARGO_MANIFEST_DIR").expect("`CARGO_MANIFEST_DIR` variable must be set");
44    let golden_file_path = format!("{manifest_dir}/{golden_rel_path}");
45    let output_with_prefix = make_golden(output);
46
47    if env::var(REGENERATE_VAR_NAME).is_ok() {
48        fs::write(&golden_file_path, &output_with_prefix)
49            .with_buck_error_context(|| format!("Writing `{golden_file_path}`"))
50            .unwrap();
51    } else {
52        let expected = fs::read_to_string(&golden_file_path)
53            .with_buck_error_context(|| format!("Reading `{golden_file_path}`"))
54            .unwrap();
55
56        let expected = if cfg!(windows) {
57            // Git may check out files on Windows with \r\n as line separator.
58            // We could configure git, but it's more reliable to handle it in the test.
59            expected.replace("\r\n", "\n")
60        } else {
61            expected
62        };
63        assert_eq!(expected, output_with_prefix);
64    }
65}

67/// Duplicate of `starlark::tests::util::trim_rust_backtrace` to avoid exposing test internals.
68/// There's no anyhow API to print error without rust backtrace
69/// ([issue](https://github.com/dtolnay/anyhow/issues/300)).
70pub fn trim_rust_backtrace(error: &str) -> &str {
71    match error.find("\nStack backtrace:") {
72        Some(pos) => error[..pos].trim_end(),
73        None => error.trim_end(),
74    }
