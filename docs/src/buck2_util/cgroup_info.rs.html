11use std::fs;
12use std::path::MAIN_SEPARATOR;
13
14use buck2_error::BuckErrorContext;
15
16const SLICE_EXT: &str = ".slice";
17
18#[derive(Debug)]
19pub struct CGroupInfo {
20    pub path: String,
21}
22
23impl CGroupInfo {
24    pub async fn read_async() -> buck2_error::Result<CGroupInfo> {
25        Self::read_from_str(
26            &tokio::fs::read_to_string("/proc/self/cgroup")
27                .await
28                .with_buck_error_context(|| "Failed to read /proc/self/cgroup")?,
29        )
30    }
31
32    pub fn read() -> buck2_error::Result<CGroupInfo> {
33        Self::read_from_str(
34            &fs::read_to_string("/proc/self/cgroup")
35                .with_buck_error_context(|| "Failed to read /proc/self/cgroup")?,
36        )
37    }
38
39    fn read_from_str(s: &str) -> buck2_error::Result<CGroupInfo> {
40        s.lines()
41            .nth(0)
42            .buck_error_context("Failed to read the first line of /proc/self/cgroup")
43            .map(|s| CGroupInfo::parse(&s))?
44    }
45
46    pub fn parse(cgroup: &str) -> buck2_error::Result<CGroupInfo> {
47        let cgroup = cgroup
48            .splitn(3, ':')
49            .nth(2)
50            .buck_error_context("Failed to parse cgroup path")?;
51        Ok(CGroupInfo {
52            path: format!("/sys/fs/cgroup{cgroup}"),
53        })
54    }
55
56    pub fn join_hierarchically(&self, parts: &[&str]) -> buck2_error::Result<String> {
57        let mut name = self
58            .get_slice_name()
59            .buck_error_context("Can't find slice in cgroup path")?
60            .to_owned();
61        let mut path = self
62            .get_slice()
63            .buck_error_context("Can't get slice name in cgroup path")?
64            .to_owned();
65        let name_separator = '-';
66        for part in parts {
67            name.push(name_separator);
68            name.push_str(part);
69            path.push(MAIN_SEPARATOR);
70            path.push_str(&name);
71            path.push_str(SLICE_EXT);
72        }
73        Ok(path)
74    }
75
76    // Returns the path to the closest slice in a cgroup path,
77    // or None if there is no slice in the path
78    pub fn get_slice(&self) -> Option<&str> {
79        self.path
80            .rfind(SLICE_EXT)
81            .map(|i| &self.path[..i + SLICE_EXT.len()])
82    }
83
84    // Returns the path to the closest slice in a cgroup path,
85    // or None if there is no slice in the path
86    fn get_slice_name(&self) -> Option<&str> {
87        let slice = self.get_slice()?;
88        let name_start = slice.rfind(MAIN_SEPARATOR)?;
89        Some(&slice[name_start + 1..slice.len() - SLICE_EXT.len()])
90    }
91}

93#[cfg(test)]
94mod tests {
95    use std::path::Path;
96
97    use super::*;
98
99    #[test]
100    fn test_cgroup_info_parse() {
101        let cgroup = "0::/user.slice/user-532497.slice/user@532497.service/buck2.slice/buck2-daemon.fbsource.v2.slice/buck2-daemon.fbsource.v2-forkserver.slice/buck2-daemon.fbsource.v2-forkserver-f80cd8522e809ed63d6bffd0ff21637a81c760928a3ecf01cc3ef5d9046dd6d5:145.slice";
102        assert_eq!(
103            "/sys/fs/cgroup/user.slice/user-532497.slice/user@532497.service/buck2.slice/buck2-daemon.fbsource.v2.slice/buck2-daemon.fbsource.v2-forkserver.slice/buck2-daemon.fbsource.v2-forkserver-f80cd8522e809ed63d6bffd0ff21637a81c760928a3ecf01cc3ef5d9046dd6d5:145.slice",
104            CGroupInfo::parse(cgroup).unwrap().path
105        );
106    }
107
108    #[test]
109    fn test_cgroup_get_slice() {
110        let cgroup = "0::/user.slice/user-532497.slice/user@532497.service/buck2.slice/buck2-daemon.fbsource.v2.slice/buck2-daemon-fbsource-v2.scope";
111        let info = CGroupInfo::parse(cgroup).unwrap();
112        assert_eq!(
113            "/sys/fs/cgroup/user.slice/user-532497.slice/user@532497.service/buck2.slice/buck2-daemon.fbsource.v2.slice/buck2-daemon-fbsource-v2.scope",
114            info.path
115        );
116        assert_eq!(
117            Some(
118                "/sys/fs/cgroup/user.slice/user-532497.slice/user@532497.service/buck2.slice/buck2-daemon.fbsource.v2.slice"
119            ),
120            info.get_slice()
121        );
122        assert_eq!(Some("buck2-daemon.fbsource.v2"), info.get_slice_name());
123    }
124
125    #[test]
126    fn test_cgroup_join() {
127        let cgroup = "0::/user.slice/user-532497.slice/user@532497.service/buck2.slice/buck2-daemon.fbsource.v2.slice/buck2-daemon-fbsource-v2.scope";
128        let info = CGroupInfo::parse(cgroup).unwrap();
129        let path = info
130            .join_hierarchically(&[
131                "forkserver",

133            ])
134            .unwrap();
135        assert_eq!(

137            path
138        );
139    }

141    #[tokio::test]
142    async fn test_cgroup_info_read_async() {
143        // check if cgroups are supported
144        if Path::new("/sys/fs/cgroup").exists() {
145            let info = CGroupInfo::read_async().await.unwrap();
146            assert!(Path::new(&info.path).exists());
147        }
148    }
