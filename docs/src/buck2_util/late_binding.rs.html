11use std::sync::OnceLock;
12
13use buck2_error::BuckErrorContext;
14
15/// Value (typically a function pointer or a trait pointer) that is initialized at program start.
16///
17/// # Motivation
18///
19/// Late binding pattern is used to call code defined in the downstream crates.
20///
21/// buck2 compilation speed is very important for us, so we try to make smaller crates
22/// with as few dependencies between them as possible. So when we change one line of code,
23/// we recompile as little as possible.
24///
25/// # Example
26///
27/// We have a crate `buck2_node` that defines target graph data types
28/// and a function to compute a target node.
29///
30/// To compute a target node, we need dependency on heavy `starlark` crate.
31/// So we use late binding here. Actual implementation lives in downstream crate
32/// `buck2_interpreter_for_build`, but code that needs to compute a target node,
33/// calls the function defined in `buck2_node` crate without compile time dependency
34/// on `buck2_interpreter_for_build` or `starlark` crates.
35///
36/// The best example of this pattern is probably `buck2_bxl` crate:
37/// the only target depending on `buck2_bxl` is the final binary with `buck2.rs` file.
38/// So when `buck2_bxl` changes, we recompile only `buck2_bxl` and `buck2.rs`.
39/// At the moment of writing this comment, `buck2_bxl` had only one public function,
40/// that is `init_late_bindings()`, that initializes all late bindings.
41///
42/// # Usage
43///
44/// `LateBinding` is stored in a static variable, and must be initialized explicitly.
45/// We tried to use `#[ctor]` crate for initialization, but unfortunately it does not work
46/// reliably on macOS. Some details are in [this post](https://fburl.com/ctor).
47///
48/// Suppose you have a function `fn foo()` initialized with `LateBinding`.
49/// The convention is this:
50/// * in the interface crate define a static variable
51///   `static FOO: LateBinding<fn()> = LateBinding::new("FOO");`
52///   ([example](https://fburl.com/code/rvxqbf4f)).
53/// * in the implementation crate define an implementation like `fn foo() { ... }`,
54///   and next to the implementation, define a function like `fn init_foo() { FOO.init(foo); }`
55///   ([example](https://fburl.com/code/0wd4xoql)).
56/// * in the root of the implementation crate, define a function `fn init_late_bindings() { ... }`
57///   that calls `init_foo()` and other `init_*` functions
58///   ([example](https://fburl.com/code/wbj4tt25)).
59/// * in the file `app/buck2/bin/buck2.rs` call `init_late_bindings()` of the corresponding crate
60///   ([example](https://fburl.com/code/maorfzdy)).
61/// * In the test crates that need to call this function, also call `init_late_bindings()`
62///   of the corresponding crates
63///   ([example](https://fburl.com/code/ynd8ylo1)).
64///   Note, to use `#[ctor]` in test crate, it should be placed in `#[test]`
65///   to avoid rust compiler and linker erase the initialization code.
66///
67/// # Safety
68///
69/// `LateBinding` is safe. All late bindings are initialized in the main function
70/// before any threads start. If late binding is not initialized,
71/// it will panic with the name of the late binding.
72///
73/// # Performance
74///
75/// `LateBinding` is relatively fast: it is extra indirection. However,
76/// combined with boxed futures, it results in extra allocations of the boxed futures:
77/// we cannot put `fn() -> impl Future` in the `LateBinding` directly, so we have to box it.
78///
79/// # Alternatives
80///
81/// There are two alternatives to late binding, that can be used in similar situations.
82/// These are similar to trait downcasting, which is not supported natively in the Rust language.
83///
84/// * `Provider` from `shed/provider` crate
85/// * Starlark's `StarlarkValue`/`Value` support similar API to `provider::Provider`
86///
87/// # Drawbacks
88///
89/// Late binding (or `std::any::Provider`) is not ergonomic. Code navigation is harder
90/// because of extra steps: instead of jumping to the definition of the function,
91/// one has to jump to the definition of the `LateBinding` and then "find usages"
92/// of that late binding to find where the implementation lives.
93pub struct LateBinding<T> {
94    /// Name for diagnostic.
95    name: &'static str,
96    symbol: OnceLock<T>,
97}
98
99impl<T> LateBinding<T> {
100    pub const fn new(name: &'static str) -> LateBinding<T> {
101        LateBinding {
102            name,
103            symbol: OnceLock::new(),
104        }
105    }
106
107    pub fn init(&self, symbol: T) {
108        if self.symbol.set(symbol).is_err() {
109            panic!("{} already set", self.name);
110        }
111    }
112
113    #[inline]
114    pub fn get(&self) -> buck2_error::Result<&T> {
115        self.symbol
116            .get()
117            .with_internal_error(|| format!("{} not set", self.name))
118    }
119}
