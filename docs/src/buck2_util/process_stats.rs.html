11use std::default::Default;
12
13#[derive(Default)]
14pub struct ProcessStats {
15    pub rss_bytes: Option<u64>,
16    pub max_rss_bytes: Option<u64>,
17    pub user_cpu_us: Option<u64>,
18    pub system_cpu_us: Option<u64>,
19}
20
21#[cfg(unix)]
22pub fn process_stats() -> ProcessStats {
23    use crate::process_stats::proc_self_stat::ProcSelfStat;
24
25    let usage = unsafe {
26        let mut usage: libc::rusage = std::mem::zeroed();
27        match libc::getrusage(libc::RUSAGE_SELF, &mut usage as *mut _) {
28            0 => usage,
29            _ => return ProcessStats::default(),
30        }
31    };
32    // POSIX didn't specify unit of ru_maxrss. Linux uses KB while BSD and
33    // OSX use bytes (despite their manpages might say differently).
34    let rss_scale = if cfg!(target_os = "linux") {
35        1024
36    } else {
37        // Assume BSD-ish
38        1
39    };
40
41    fn tv_to_micros(tv: &libc::timeval) -> u64 {
42        (1_000_000 * tv.tv_sec as u64) + (tv.tv_usec as u64)
43    }
44
45    let rss_bytes = if cfg!(target_os = "linux") {
46        // Buck2 snapshot is made once per second, so this shouldn't be too expensive.
47        ProcSelfStat::read().map(|stat| {
48            // `getconf PAGESIZE`, but practically it's always 4096.
49            stat.rss * 4096
50        })
51    } else {
52        None
53    };
54
55    ProcessStats {
56        rss_bytes,
57        max_rss_bytes: Some((usage.ru_maxrss as u64) * rss_scale),
58        user_cpu_us: Some(tv_to_micros(&usage.ru_utime)),
59        system_cpu_us: Some(tv_to_micros(&usage.ru_stime)),
60    }
61}

63#[cfg(windows)]
64pub fn process_stats() -> ProcessStats {
65    use winapi::shared::minwindef::DWORD;
66    use winapi::um::processthreadsapi::GetCurrentProcess;
67    use winapi::um::psapi::K32GetProcessMemoryInfo;
68    use winapi::um::psapi::PROCESS_MEMORY_COUNTERS;
69
70    let mut pmc: PROCESS_MEMORY_COUNTERS = unsafe { std::mem::zeroed() };
71    pmc.cb = std::mem::size_of_val(&pmc) as DWORD;
72    // Code is referenced from eden/scm/lib/procinfo/src/lib.rs
73    // API reference: https://learn.microsoft.com/en-us/windows/win32/api/psapi/ns-psapi-process_memory_counters
74    let (wss_bytes, max_wss_bytes) =
75        match unsafe { K32GetProcessMemoryInfo(GetCurrentProcess(), &mut pmc, pmc.cb) } {
76            0 => (None, None),
77            _ => (
78                Some(pmc.WorkingSetSize as u64),
79                Some(pmc.PeakWorkingSetSize as u64),
80            ),
81        };
82
83    // Technically, `GetProcessMemoryInfo` returns working set size, not resident set size. However,
84    // we log it as rss for two reasons:
85    // (1) Reading https://learn.microsoft.com/en-us/windows/win32/memory/working-set, the definition of
86    // working set size is "set of pages in the virtual address space of the process that are currently resident
87    // in physical memory", which sound just like RSS in unix.
88    // (2) We can reuse our existing logging for rss without having duplicate column and logic for wss.
89    ProcessStats {
90        rss_bytes: wss_bytes,
91        max_rss_bytes: max_wss_bytes,
92        // TODO T219499802: Add CPU stats for Windows
93        user_cpu_us: None,
94        system_cpu_us: None,
95    }
96}

98#[cfg(not(any(unix, windows)))]
99pub fn process_stats() -> ProcessStats {
100    ProcessStats::default()
101}

103pub fn process_cpu_time_us() -> Option<u64> {
104    let stats = process_stats();
105    if let (Some(user_cpu_us), Some(system_cpu_us)) = (stats.user_cpu_us, stats.system_cpu_us) {
106        Some(user_cpu_us + system_cpu_us)
107    } else {
108        None
109    }
110}

112#[cfg_attr(not(unix), allow(dead_code))]
113mod proc_self_stat {
114    use std::fs;
115
116    /// Parsed `/proc/self/stat` file.
117    pub struct ProcSelfStat {
118        /// Resident Set Size: number of pages the process has in real memory.
119        /// Raw value.
120        pub rss: u64,
121    }
122
123    impl ProcSelfStat {
124        pub fn parse(stat: &str) -> Option<ProcSelfStat> {
125            let rss = stat.split(' ').nth(23)?.parse().ok()?;
126            Some(ProcSelfStat { rss })
127        }
128
129        pub fn read() -> Option<ProcSelfStat> {
130            fs::read_to_string("/proc/self/stat")
131                .ok()
132                .and_then(|s| ProcSelfStat::parse(&s))
133        }
134    }
135}

137#[cfg(test)]
138mod tests {
139    use crate::process_stats::proc_self_stat::ProcSelfStat;
140    use crate::process_stats::process_stats;

142    #[test]
143    fn test_process_stats() {
144        let process_stats = process_stats();
145        if cfg!(unix) {
146            // Sometimes tests start too quickly and CPU counters are zero.
147            if false {
148                assert!(process_stats.user_cpu_us.unwrap() > 0);
149                assert!(process_stats.system_cpu_us.unwrap() > 0);
150            }
151        }
152        assert!(process_stats.max_rss_bytes.unwrap() > 0);
153        if cfg!(target_os = "linux") || cfg!(target_os = "windows") {
154            let rss_bytes = process_stats.rss_bytes.unwrap();
155            assert!(rss_bytes > 0);
156        }
157    }
