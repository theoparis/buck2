23/// Get the available parallelism
24///
25/// This value is cached for the lifetime of the process. The reason is that there are various
26/// components that cannot be updated to reflect the new value if it changes during the lifetime of
27/// the daemon. Caching this sacrifices some accuracy of this value in exchange for putting the
28/// daemon into a more predictable state.
29///
30/// Use `available_parallelism_fresh` if the caching is not desired
31pub fn available_parallelism() -> usize {
32    static PARALLELISM: OnceLock<usize> = OnceLock::new();
33
34    *PARALLELISM.get_or_init(available_parallelism_fresh)
35}
36
37/// Get the available parallelism
38///
39/// Unlike `available_parallelism`, this is not cached - callers using this should ensure that this
40/// value is logged somewhere
41pub fn available_parallelism_fresh() -> usize {
42    // NB: num_cpus and tokio both also use 1 as the default in case of an error
43    std::thread::available_parallelism().map_or(1, |v| v.get())
44}

46/// Default stack size for buck2.
47///
48/// We want to be independent of possible future changes to the default stack size in Rust.
49pub(crate) const THREAD_DEFAULT_STACK_SIZE: usize = {
50    if cfg!(buck2_asan) {
51        // ASAN requires much larger stack size.
52        8 << 20
53    } else if cfg!(debug_assertions) {
54        // Need 4MB for windows-debug according to D60449433.
55        4 << 20
56    } else {
57        2 << 20
58    }
59};
60
61fn thread_builder(name: &str) -> thread::Builder {
62    thread::Builder::new()
63        .stack_size(THREAD_DEFAULT_STACK_SIZE)
64        .name(name.to_owned())
65}

67pub fn thread_spawn<T, F>(name: &str, code: F) -> std::io::Result<thread::JoinHandle<T>>
68where
69    T: Send + 'static,
70    F: FnOnce() -> T + Send + 'static,
71{
72    thread_builder(name).spawn(move || {
73        on_thread_start();
74        let r = code();
75        on_thread_stop();
76        r
77    })
78}

80pub fn thread_spawn_scoped<'scope, 'env: 'scope, T, F>(
81    name: &str,
82    scope: &'scope thread::Scope<'scope, 'env>,
83    code: F,
84) -> std::io::Result<thread::ScopedJoinHandle<'scope, T>>
85where
86    T: Send + 'static,
87    F: FnOnce() -> T + Send + 'scope,
88{
89    thread_builder(name).spawn_scoped(scope, move || {

93        r
94    })

97pub(crate) fn stack_pointer() -> *const () {
98    let mut x: u32 = 0;
99    hint::black_box(&mut x as *const u32 as *const ())
100}
101
102#[derive(Copy, Clone)]
103struct ValidStackRange {
104    start: *const (),
105    end: *const (),
106}
107
108impl ValidStackRange {
109    fn full_range() -> ValidStackRange {
110        let start = usize::MAX as *const ();
111        let end = usize::MIN as *const ();
112        ValidStackRange { start, end }
113    }
114}
115
116thread_local! {
117    static STACK_RANGE: Cell<Option<ValidStackRange >> = const { Cell::new(None) };
118}
119
120pub(crate) fn on_thread_start() {
121    assert!(
122        STACK_RANGE.get().is_none(),
123        "stack range must not be set in a new thread"
124    );
125    let stack_pointer = stack_pointer();
126    // Stack grows downwards. So we add to the start and subtract from the end.
127    // Add a little bit to the start because we don't really know where the stack starts.
128    let start = (stack_pointer as usize).checked_add(0x1000).unwrap() as *const ();
129    // Subtract 3/4 to catch stack overflow before program crashes.
130    let end = (stack_pointer as usize)
131        .checked_sub(THREAD_DEFAULT_STACK_SIZE / 4 * 3)
132        .unwrap() as *const ();
133    let stack_range = ValidStackRange { start, end };
134    STACK_RANGE.set(Some(stack_range));
135}
136
137pub(crate) fn on_thread_stop() {
138    let range = STACK_RANGE.replace(None);
139    assert!(range.is_some(), "stack range must be set in a thread");
140}
141
142pub fn check_stack_overflow() -> buck2_error::Result<()> {
143    let stack_range = STACK_RANGE.get().internal_error("stack range not set")?;
144    let stack_pointer = stack_pointer();
145    if stack_pointer > stack_range.start {
146        return Err(internal_error!("stack underflow, should not happen"));
147    }
148    if stack_pointer < stack_range.end {
149        return Err(internal_error!("stack overflow"));
150    }
151    Ok(())
152}

154#[must_use]
155pub struct IgnoreStackOverflowChecksForCurrentThread {
156    prev: Option<ValidStackRange>,
157}
158
159impl Drop for IgnoreStackOverflowChecksForCurrentThread {
160    fn drop(&mut self) {
161        STACK_RANGE.set(self.prev.take());
162    }
163}
164
165/// For tests.
166pub fn ignore_stack_overflow_checks_for_current_thread() -> IgnoreStackOverflowChecksForCurrentThread
167{
168    let prev = STACK_RANGE.replace(Some(ValidStackRange::full_range()));
169    IgnoreStackOverflowChecksForCurrentThread { prev }
170}

172/// For tests.
173pub async fn ignore_stack_overflow_checks_for_future<F: Future>(f: F) -> F::Output {
174    let f = pin!(f);
175
176    struct IgnoreStackOverflowChecksForFuture<'a, F> {
177        f: Pin<&'a mut F>,
178    }
179
180    impl<F: Future> Future for IgnoreStackOverflowChecksForFuture<'_, F> {
181        type Output = F::Output;
182
183        fn poll(mut self: Pin<&mut Self>, cx: &mut std::task::Context<'_>) -> Poll<Self::Output> {
184            let _ignore = ignore_stack_overflow_checks_for_current_thread();
185            self.f.as_mut().poll(cx)
186        }
187    }
188
189    IgnoreStackOverflowChecksForFuture { f }.await
190}

192#[cfg(test)]
193pub(crate) mod tests {
194    use std::hint;
195
196    use crate::threads::check_stack_overflow;
197    use crate::threads::thread_spawn;
198
199    pub(crate) fn recursive_function(frames: u32) -> buck2_error::Result<()> {
200        let Some(frames) = frames.checked_sub(1) else {
201            return Ok(());
202        };
203
204        check_stack_overflow()?;
205
206        // Allocate a string on the stack so the compiler won't optimize the recursion away.
207        let mut x = String::new();
208        hint::black_box(&mut x);
209        recursive_function(frames)?;
210        hint::black_box(&mut x);
211        Ok(())
212    }
213
214    #[test]
215    fn test_catch_stack_overflow() {
216        let error = thread_spawn("test", || recursive_function(u32::MAX))
217            .unwrap()
218            .join()
219            .unwrap()
220            .unwrap_err();
221        assert!(error.to_string().contains("stack overflow"), "{error:?}");
222    }
223
224    #[test]
225    fn test_no_stack_overflow() {
226        let () = thread_spawn("test", || recursive_function(1000))
227            .unwrap()
228            .join()
229            .unwrap()
230            .unwrap();
231    }
