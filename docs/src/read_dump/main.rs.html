11use std::fs::File;
12use std::path::PathBuf;
13
14use clap::CommandFactory;
15use clap::FromArgMatches;
16use dice::introspection::graph::SerializedGraphNodesForKey;
17
18#[derive(Debug, clap::Parser)]
19#[clap(name = "read_dump", about = "dice dump reader")]
20pub(crate) struct Opt {
21    #[clap(name = "DICE_DUMP", help = "The dice dump")]
22    file: PathBuf,
23    #[clap(long = "out", help = "Copy the output to this path")]
24    out: Option<PathBuf>,
25}
26
27fn main() -> anyhow::Result<()> {
28    let clap = <Opt as CommandFactory>::command();
29    let matches = clap.get_matches_from(std::env::args().collect::<Vec<String>>());
30    let opt = Opt::from_arg_matches(&matches)?;
31
32    let file = File::open(opt.file)?;
33
34    let out: Vec<SerializedGraphNodesForKey> = bincode::deserialize_from(&file)?;
35
36    match opt.out {
37        Some(path) => {
38            serde_json::to_writer_pretty(File::create(path)?, &out)?;
39        }
40        None => {
41            serde_json::to_writer_pretty(std::io::stdout(), &out)?;
42        }
43    };
44
45    Ok(())
46}
