27/// Compute the number of buckets needed for a given capacity.
28/// The number of bucket is `log2(capacity) - 3`.
29#[inline]
30pub const fn buckets_for_max_capacity(max_capacity: usize) -> usize {
31    assert!(max_capacity.is_power_of_two());
32    assert!(max_capacity >= 16);
33    max_capacity.trailing_zeros() as usize - (FIRST_BUCKET_CAP_EXP as usize - 1)
34}
35
36/// `Box<[MaybeUninit<T>]>`.
37struct BucketAlloc<T> {
38    data: *mut T,
39    cap: usize,
40}
41
42impl<T> BucketAlloc<T> {
43    #[inline]
44    fn new(cap: usize) -> BucketAlloc<T> {
45        debug_assert!(cap.is_power_of_two());
46        let mut data = Vec::with_capacity(cap);
47        data.resize_with(cap, MaybeUninit::uninit);
48        let data = data.into_boxed_slice();
49        let data: *mut [MaybeUninit<T>] = Box::into_raw(data);
50        let data: *mut T = data as *mut T;
51        BucketAlloc { data, cap }
52    }
53
54    #[inline]
55    unsafe fn drop_data(&self, len: usize) {
56        assert!(len <= self.cap);
57        let data: *mut [T] = unsafe { slice::from_raw_parts_mut(self.data, len) };
58        unsafe { ptr::drop_in_place(data) };
59    }
60
61    #[inline]
62    unsafe fn from_raw_parts(data: *mut T, cap: usize) -> BucketAlloc<T> {
63        debug_assert!(cap.is_power_of_two());
64        BucketAlloc { data, cap }
65    }
66}
67
68/// Deallocate.
69impl<T> Drop for BucketAlloc<T> {
70    fn drop(&mut self) {
71        unsafe {
72            let data: *mut MaybeUninit<T> = self.data as *mut MaybeUninit<T>;
73            let data: *mut [MaybeUninit<T>] = slice::from_raw_parts_mut(data, self.cap);
74            let data: Box<[MaybeUninit<T>]> = Box::from_raw(data);
75            drop(data);
76        }
77    }
78}

80const LOCKED_FLAG: usize = 1 << (mem::size_of::<usize>() * 8 - 1);
81/// `log2` of first bucket capacity.
82const FIRST_BUCKET_CAP_EXP: u32 = 4;
83/// 16.
84const FIRST_BUCKET_CAP: usize = 1 << FIRST_BUCKET_CAP_EXP;
85
86/// Maximum number of buckets.
87pub const MAX_BUCKETS: usize = mem::size_of::<usize>() * 8 - FIRST_BUCKET_CAP_EXP as usize;
88
89/// Data structure with push and get operations.
90///
91/// * `get` is wait-free and no locks.
92/// * `push` is single-threaded.
93///
94/// Both operations are `O(1)`, hence the `Vec` part in name.
95/// But data is not stored in contiguous memory.
96// This can be parameterized by capacity when `feature(generic_const_exprs)` is stable.
97// https://github.com/rust-lang/rust/issues/76560
98pub struct LockFreeVec<T, const BUCKETS: usize = MAX_BUCKETS> {
99    /// Buckets containing the actual data.
100    /// ```ignore
101    /// [0..15] // first is special
102    /// [16..31]
103    /// [32..63]
104    /// [64..127]
105    /// [128..255]
106    /// [256..511]
107    /// ...
108    /// ```
109    /// The number of initialized fields in this array is given by the size. If, for
110    /// example, the size is 100 then the first four fields of this array are
111    /// initialized. Those fields can then also be read without need for any
112    /// synchronization.
113    buckets: [UnsafeCell<*mut T>; BUCKETS],
114    /// Size, and high bit is locked flag.
115    /// Operations are synchronized on this field.
116    size: AtomicUsize,
117}
118
119impl<T, const BUCKETS: usize> Default for LockFreeVec<T, BUCKETS> {
120    #[inline]
121    fn default() -> LockFreeVec<T, BUCKETS> {
122        LockFreeVec::new()
123    }
124}
125
126unsafe impl<T: Sync + Send, const BUCKETS: usize> Send for LockFreeVec<T, BUCKETS> {}
127unsafe impl<T: Sync + Send, const BUCKETS: usize> Sync for LockFreeVec<T, BUCKETS> {}

129impl<T, const BUCKETS: usize> LockFreeVec<T, BUCKETS> {
130    const _ASSERTS: () = {
131        assert!(BUCKETS > 0);
132        assert!(BUCKETS <= MAX_BUCKETS);
133        assert!(Self::max_capacity() < LOCKED_FLAG);
134    };
135
136    /// Empty.
137    #[inline]
138    pub const fn new() -> LockFreeVec<T, BUCKETS> {
139        LockFreeVec {
140            // SAFETY: we want zeros.
141            buckets: unsafe { MaybeUninit::zeroed().assume_init() },
142            size: AtomicUsize::new(0),
143        }
144    }
145
146    /// Maximum capacity.
147    #[inline]
148    pub const fn max_capacity() -> usize {
149        // Does not overflow because static assertions.
150        1 << (BUCKETS + FIRST_BUCKET_CAP_EXP as usize - 1)
151    }
152
153    /// Capacity of a bucket given its index.
154    #[inline]
155    fn bucket_capacity(bucket: usize) -> usize {
156        debug_assert!(bucket < BUCKETS);
157        cmp::max(
158            FIRST_BUCKET_CAP,
159            1 << (bucket + FIRST_BUCKET_CAP_EXP as usize - 1),
160        )
161    }
162
163    /// Unpack index into bucket index and offset in bucket.
164    #[inline]
165    fn unpack_index(index: usize) -> (usize, usize) {
166        debug_assert!(index < Self::max_capacity());
167        let bucket =
168            mem::size_of::<usize>() * 8 - (index >> FIRST_BUCKET_CAP_EXP).leading_zeros() as usize;
169        let index_in_bucket = index & (Self::bucket_capacity(bucket) - 1);
170        (bucket, index_in_bucket)
171    }
172
173    /// Number of elements.
174    ///
175    /// This operation has acquire semantics.
176    #[inline]
177    pub fn len(&self) -> usize {
178        // We want acquire because the returned value is to be used in `push_at`.
179        self.size.load(Ordering::Acquire) & !LOCKED_FLAG
180    }
181
182    /// Is empty?
183    #[inline]
184    pub fn is_empty(&self) -> bool {
185        self.len() == 0
186    }
187
188    /// Get element.
189    ///
190    /// Wait-free, no locks.
191    #[inline]
192    pub fn get(&self, index: usize) -> Option<&T> {
193        let size = self.len();
194        if index >= size {
195            return None;
196        }
197        let (bucket, index_in_bucket) = Self::unpack_index(index);
198        let bucket: *mut T = unsafe { *self.buckets[bucket].get() };
199        Some(unsafe { &*bucket.add(index_in_bucket) })
200    }
201
202    #[inline]
203    unsafe fn init_bucket(&self, bucket: usize) -> *mut T {
204        let bucket_ptr_ptr: *mut *mut T = self.buckets[bucket].get();
205        let bucket_ptr = unsafe { *bucket_ptr_ptr };
206        if !bucket_ptr.is_null() {
207            return bucket_ptr;
208        }
209        unsafe { self.init_bucket_slow(bucket) }
210    }
211
212    #[cold]
213    unsafe fn init_bucket_slow(&self, bucket: usize) -> *mut T {
214        let bucket_ptr_ptr: *mut *mut T = self.buckets[bucket].get();
215        assert!(unsafe { *bucket_ptr_ptr }.is_null());
216        let bucket_alloc = BucketAlloc::new(Self::bucket_capacity(bucket));
217        let bucket_ptr = bucket_alloc.data;
218        unsafe { *bucket_ptr_ptr = bucket_ptr };
219        mem::forget(bucket_alloc);
220        bucket_ptr
221    }
222
223    /// Push an element to the end.
224    ///
225    /// This operation returns error if index is not equal to the current size.
226    ///
227    /// Wait-free.
228    #[inline]
229    pub fn push_at(&self, index: usize, value: T) -> Result<(), T> {
230        if index == Self::max_capacity() {
231            panic!("capacity overflow: {}", Self::max_capacity());
232        }
233
234        if self
235            .size
236            .compare_exchange(
237                index,
238                index | LOCKED_FLAG,
239                Ordering::Acquire,
240                Ordering::Relaxed,
241            )
242            .is_err()
243        {
244            // We could retry, but we promised not to lock.
245            return Err(value);
246        }
247
248        let (bucket, index_in_bucket) = Self::unpack_index(index);
249
250        unsafe {
251            let bucket_ptr = self.init_bucket(bucket);
252            ptr::write(bucket_ptr.add(index_in_bucket), value);
253        }
254
255        self.size.store(index + 1, Ordering::Release);
256        Ok(())
257    }
258
259    /// Iterate over all elements.
260    ///
261    /// Iterates over snapshot of the vector, newly pushed elements may not be visible.
262    #[inline]
263    pub fn iter(&self) -> Iter<T, BUCKETS> {
264        let len = self.len();
265        let slice = if len == 0 {
266            &[]
267        } else {
268            unsafe {
269                slice::from_raw_parts(*self.buckets[0].get(), cmp::min(len, FIRST_BUCKET_CAP))
270            }
271        };
272        Iter {
273            vec: self,
274            bucket: 0,
275            slice: slice.iter(),
276            rem: len - slice.len(),
277        }
278    }
279}

281impl<T, const BUCKETS: usize> Drop for LockFreeVec<T, BUCKETS> {
282    fn drop(&mut self) {
283        unsafe {
284            let mut rem_len = *self.size.get_mut();
285            for (bucket, bucket_ptr_ptr) in self.buckets.iter().enumerate() {
286                let bucket_ptr: *mut T = *bucket_ptr_ptr.get();
287                if bucket_ptr.is_null() {
288                    assert_eq!(0, rem_len);
289                    break;
290                }
291
292                let bucket_len = cmp::min(rem_len, Self::bucket_capacity(bucket));
293                rem_len -= bucket_len;
294
295                let bucket_alloc: BucketAlloc<T> =
296                    BucketAlloc::from_raw_parts(bucket_ptr, Self::bucket_capacity(bucket));
297
298                bucket_alloc.drop_data(bucket_len);
299
300                drop(bucket_alloc);
301            }
302
303            assert_eq!(0, rem_len);
304        }
305    }
306}

308impl<T: Allocative, const SHARDS: usize> Allocative for LockFreeVec<T, SHARDS> {
309    fn visit<'a, 'b: 'a>(&self, visitor: &'a mut Visitor<'b>) {
310        let mut visitor = visitor.enter_self_sized::<Self>();
311        let mut rem_len = self.len();
312        for (bucket, bucket_ptr_ptr) in self.buckets.iter().enumerate() {
313            if rem_len == 0 {
314                break;
315            }
316
317            let bucket_ptr: *mut T = unsafe { *bucket_ptr_ptr.get() };
318            if bucket_ptr.is_null() {
319                break;
320            }
321
322            let bucket_len = cmp::min(rem_len, Self::bucket_capacity(bucket));
323            rem_len -= bucket_len;
324
325            let mut visitor = visitor.enter_unique(
326                allocative::Key::new("bucket"),
327                mem::size_of_val(&bucket_ptr),
328            );
329            {
330                let mut visitor = visitor.enter(
331                    allocative::Key::new("capacity"),
332                    mem::size_of::<T>() * Self::bucket_capacity(bucket),
333                );
334                for i in 0..bucket_len {
335                    let value: &T = unsafe { &*bucket_ptr.add(i) };
336                    visitor.visit_field(allocative::Key::new("item"), value);
337                }
338                visitor.exit();
339            }
340
341            visitor.exit();
342        }
343        visitor.exit();
344    }
345}

347/// Iterator over all elements.
348pub struct Iter<'a, T, const SHARDS: usize> {
349    vec: &'a LockFreeVec<T, SHARDS>,
350    /// Current bucket index.
351    bucket: usize,
352    /// Remaining elements in the current bucket.
353    slice: slice::Iter<'a, T>,
354    /// Remaining elements after the current bucket iterator.
355    rem: usize,
356}
357
358impl<'a, T, const SHARDS: usize> Iterator for Iter<'a, T, SHARDS> {
359    type Item = &'a T;
360
361    #[inline]
362    fn next(&mut self) -> Option<Self::Item> {
363        match self.slice.next() {
364            Some(value) => Some(value),
365            None if self.rem == 0 => None,
366            None => {
367                self.bucket += 1;
368                let bucket_len = cmp::min(
369                    self.rem,
370                    LockFreeVec::<T, SHARDS>::bucket_capacity(self.bucket),
371                );
372                let slice = unsafe {
373                    slice::from_raw_parts(*self.vec.buckets[self.bucket].get(), bucket_len)
374                };
375                self.rem -= slice.len();
376                self.slice = slice.iter();
377                self.slice.next()
378            }
379        }
380    }
381
382    fn size_hint(&self) -> (usize, Option<usize>) {
383        let rem = self.rem + self.slice.len();
384        (rem, Some(rem))
385    }
386}
387
388impl<T, const SHARDS: usize> ExactSizeIterator for Iter<'_, T, SHARDS> {}
389
390impl<'a, T, const SHARDS: usize> IntoIterator for &'a LockFreeVec<T, SHARDS> {
391    type Item = &'a T;
392    type IntoIter = Iter<'a, T, SHARDS>;
393
394    #[inline]
395    fn into_iter(self) -> Self::IntoIter {
396        self.iter()
397    }
398}

499    fn test_drop() {
500        struct RecordDrops(u32, Arc<Mutex<Vec<u32>>>);
501        impl Drop for RecordDrops {
502            fn drop(&mut self) {
503                self.1.lock().unwrap().push(self.0);
504            }
505        }
506
507        let drops = Arc::new(Mutex::new(Vec::new()));
508        let vec = LockFreeVec::<RecordDrops, 20>::new();
509
510        for i in 0..1000 {
511            vec.push_at(i as usize, RecordDrops(i, drops.clone()))
512                .ok()
513                .unwrap();
514        }
515
516        drop(vec);
517
518        let drops = Arc::try_unwrap(drops).unwrap().into_inner().unwrap();
519        assert_eq!((0..1000).collect::<Vec<_>>(), drops);
520    }
