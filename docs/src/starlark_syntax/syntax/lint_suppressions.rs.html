18use std::collections::HashMap;
19use std::collections::HashSet;
20
21use crate::codemap::CodeMap;
22use crate::codemap::Pos;
23use crate::codemap::Span;
24
25static LINT_SUPPRESISON_PREFIX: &str = "starlark-lint-disable ";
26
27#[derive(Debug, Clone)]
28struct SuppressionInfo {
29    /// The original span of the comment token containing the suppression
30    token_span: Span,
31    /// The span that this suppression effects
32    effective_span: Span,
33    /// Does the suppression cover the next line?
34    suppress_next_line: bool,
35}
36#[derive(Debug, Clone)]
37pub(crate) struct LintSuppressions {
38    /// A map from lint short names to spans where they are suppressed
39    suppressions: HashMap<String, Vec<SuppressionInfo>>,
40}
41
42impl LintSuppressions {
43    /// Check if a given lint short_name and span is suppressed
44    pub(crate) fn is_suppressed(&self, issue_short_name: &str, issue_span: Span) -> bool {
45        self.suppressions
46            .get(issue_short_name)
47            .map(|suppression_spans| {
48                suppression_spans.iter().any(
49                    |SuppressionInfo {
50                         token_span,
51                         effective_span,
52                         suppress_next_line,
53                     }| {
54                        // is this suppression that last thing in a suppress_next_line issue span? ...
55                        if *suppress_next_line &&
56                            // (issue_span includes line terminator)
57                            (issue_span.end() - 1) == token_span.end()
58                        {
59                            // ... then issue is not suppressed
60                            false
61                        } else {
62                            issue_span.intersects(*effective_span)
63                        }
64                    },
65                )
66            })
67            == Some(true)
68    }
69}

71/// State needed for parsing a block of comments
72#[derive(Default)]
73struct ParseState {
74    token_spans: Vec<Span>,
75    effective_spans: Vec<Span>,
76    short_names: HashSet<String>,
77    last_line: usize,
78}
79
80impl ParseState {
81    fn is_empty(&self) -> bool {
82        self.token_spans.is_empty()
83            && self.effective_spans.is_empty()
84            && self.short_names.is_empty()
85    }
86}
87
88/// Parse lint suppressions for a module and build a LintSuppressions struct
89pub(crate) struct LintSuppressionsBuilder {
90    state: ParseState,
91    suppressions: LintSuppressions,
92}
93
94impl LintSuppressionsBuilder {
95    pub(crate) fn new() -> Self {
96        Self {
97            state: ParseState::default(),
98            suppressions: LintSuppressions {
99                suppressions: HashMap::new(),
100            },
101        }
102    }
103
104    /// Call for each comment in a block of comments
105    pub(crate) fn parse_comment(
106        &mut self,
107        codemap: &CodeMap,
108        comment: &str,
109        start: usize,
110        end: usize,
111    ) {
112        let parsed_short_names = parse_lint_suppressions(comment);
113        if !parsed_short_names.is_empty() || !self.state.short_names.is_empty() {
114            if let (Ok(start_pos), Ok(end_pos)) = (start.try_into(), end.try_into()) {
115                let token_span = Span::new(Pos::new(start_pos), Pos::new(end_pos));
116                let line = codemap.find_line(Pos::new(start_pos));
117                let effective_span = codemap.line_span_trim_newline(line);
118                self.state.short_names.extend(parsed_short_names);
119                self.state.token_spans.push(token_span);
120                self.state.effective_spans.push(effective_span);
121                self.state.last_line = line;
122            }
123        }
124    }
125
126    /// Call after the last comment in a block of comments
127    pub(crate) fn end_of_comment_block(&mut self, codemap: &CodeMap) {
128        if !self.state.short_names.is_empty() {
129            self.update_lint_suppressions(codemap);
130        }
131    }
132
133    pub(crate) fn build(self) -> LintSuppressions {
134        assert!(self.state.is_empty());
135        self.suppressions
136    }
137
138    /// Update the line_suppressions hashmap with parsed lint suppressions for a block of comment
139    /// Consumes and clears the ParseState
140    fn update_lint_suppressions(&mut self, codemap: &CodeMap) {
141        let state = std::mem::take(&mut self.state);
142        let number_of_tokens = state.token_spans.len();
143        let token_span = Span::merge_all(state.token_spans.into_iter());
144        let mut effective_span = Span::merge_all(state.effective_spans.into_iter());
145        // In case the suppression comment has preceding whitespace
146        let source_before_token =
147            codemap.source_span(Span::new(effective_span.begin(), token_span.begin()));
148        let suppress_next_line = number_of_tokens > 1
149            || effective_span == token_span
150            || (effective_span.end() == token_span.end() && source_before_token.trim().is_empty());
151        if suppress_next_line {
152            // Expand the span to include the next line,
153            // in case suppression was put on the line before the issue
154            if let Some(next_line_span) = codemap.line_span_opt(state.last_line + 1) {
155                effective_span = effective_span.merge(next_line_span);
156            }
157        }
158
159        for name in state.short_names {
160            self.suppressions
161                .suppressions
162                .entry(name)
163                .or_default()
164                .push(SuppressionInfo {
165                    token_span,
166                    effective_span,
167                    suppress_next_line,
168                });
169        }
170    }
171}

173/// Parse a single comment line and extract any lint suppressions.
174fn parse_lint_suppressions(comment_line: &str) -> Vec<String> {
175    let mut res = Vec::new();
176    if let Some(short_names) = comment_line
177        .trim_start()
178        .strip_prefix(LINT_SUPPRESISON_PREFIX)
179    {
180        for name in short_names.split([' ', ',']) {
181            let trimmed = name.trim();
182            if !trimmed.is_empty() {
183                res.push(trimmed.to_owned());
184            }
185        }
186    }
187
188    res
189}
