37impl Expr {
38    /// We want to check a function call is well-formed.
39    /// Our eventual plan is to follow the Python invariants, but for now, we are closer
40    /// to the Starlark invariants.
41    ///
42    /// Python invariants are no positional arguments after named arguments,
43    /// no *args after **kwargs, no repeated argument names.
44    ///
45    /// Starlark invariants are the above, plus at most one *args and the *args must appear
46    /// after all positional and named arguments. The spec is silent on whether you are allowed
47    /// multiple **kwargs.
48    ///
49    /// We allow at most one **kwargs.
50    pub(crate) fn check_call(
51        f: AstExpr,
52        args: Vec<AstArgument>,
53        parser_state: &mut ParserState<'_>,
54    ) -> Expr {
55        let args = CallArgsP { args };
56
57        if let Err(e) = CallArgsUnpack::unpack(&args, parser_state.codemap) {
58            parser_state.errors.push(e);
59        }
60
61        Expr::Call(Box::new(f), args)
62    }
63}

65/// Validate all statements only occur where they are allowed to.
66pub(crate) fn validate_module(stmt: &AstStmt, parser_state: &mut ParserState) {
67    fn validate_params(params: &[AstParameter], parser_state: &mut ParserState) {
68        if !parser_state.dialect.enable_keyword_only_arguments {
69            for param in params {
70                if let ParameterP::NoArgs = &param.node {
71                    parser_state.error(
72                        param.span,
73                        "* keyword-only-arguments is not allowed in this dialect",
74                    );
75                }
76            }
77        }
78        if !parser_state.dialect.enable_positional_only_arguments {
79            for param in params {
80                if let ParameterP::Slash = &param.node {
81                    parser_state.error(
82                        param.span,
83                        "/ positional-only-arguments is not allowed in this dialect",
84                    );
85                }
86            }
87        }
88        if let Err(e) = DefParams::unpack(params, parser_state.codemap) {
89            parser_state.errors.push(e);
90        }
91    }
92
93    // Inside a for, we allow continue/break, unless we go beneath a def.
94    // Inside a def, we allow return.
95    // All load's must occur at the top-level.
96    // At the top-level we only allow for/if when the dialect permits it.
97    fn f(
98        stmt: &AstStmt,
99        parser_state: &mut ParserState,
100        top_level: bool,
101        inside_for: bool,
102        inside_def: bool,
103    ) {
104        let span = stmt.span;
105
106        match &stmt.node {
107            Stmt::Def(DefP { params, body, .. }) => {
108                if !parser_state.dialect.enable_def {
109                    parser_state.error(span, "`def` is not allowed in this dialect");
110                }
111                validate_params(params, parser_state);
112                f(body, parser_state, false, false, true)
113            }
114            Stmt::For(ForP { body, .. }) => {
115                if top_level && !parser_state.dialect.enable_top_level_stmt {
116                    parser_state.error(span, "`for` cannot be used outside `def` in this dialect")
117                } else {
118                    f(body, parser_state, false, true, inside_def)
119                }
120            }
121            Stmt::If(..) | Stmt::IfElse(..) => {
122                if top_level && !parser_state.dialect.enable_top_level_stmt {
123                    parser_state.error(span, "`if` cannot be used outside `def` in this dialect")
124                } else {
125                    stmt.node
126                        .visit_stmt(|x| f(x, parser_state, false, inside_for, inside_def))
127                }
128            }
129            Stmt::Break if !inside_for => {
130                parser_state.error(span, "`break` cannot be used outside of a `for` loop")
131            }
132            Stmt::Continue if !inside_for => {
133                parser_state.error(span, "`continue` cannot be used outside of a `for` loop")
134            }
135            Stmt::Return(_) if !inside_def => {
136                parser_state.error(span, "`return` cannot be used outside of a `def` function")
137            }
138            Stmt::Load(..) => {
139                if !top_level {
140                    parser_state.error(span, "`load` must only occur at the top of a module");
141                }
142                if !parser_state.dialect.enable_load {
143                    parser_state.error(span, "`load` is not allowed in this dialect");
144                }
145            }
146            _ => stmt
147                .node
148                .visit_stmt(|x| f(x, parser_state, top_level, inside_for, inside_def)),
149        }
150    }
151
152    fn expr(x: &AstExpr, parser_state: &mut ParserState) {
153        match &x.node {
154            Expr::Literal(AstLiteral::Ellipsis) => {
155                if parser_state.dialect.enable_types == DialectTypes::Disable {
156                    parser_state.error(x.span, "`...` is not allowed in this dialect");
157                }
158            }
159            Expr::Lambda(LambdaP { params, .. }) => {
160                if !parser_state.dialect.enable_lambda {
161                    parser_state.error(x.span, "`lambda` is not allowed in this dialect");
162                }
163                validate_params(params, parser_state);
164            }
165            _ => {}
166        }
167        x.node.visit_expr(|x| expr(x, parser_state));
168    }
169
170    f(stmt, parser_state, true, false, false);
171
172    stmt.visit_expr(|x| expr(x, parser_state));
173}
