11//! Cancellable results of DICE
12
13use allocative::Allocative;
14use derive_more::Display;
15use dupe::Dupe;
16use thiserror::Error;
17
18pub type CancellableResult<T> = Result<T, CancellationReason>;
19
20#[derive(Clone, Dupe, Copy, Display, Debug, Error, Allocative, PartialEq)]
21#[display("{:?}", self)]
22pub enum CancellationReason {
23    OutdatedEpoch,
24    Rejected,
25    DepsMatch,
26    WorkerFinished,
27    Cached,
28    AllDependentsDropped,
29    AllObserversDropped,
30    TransactionCancelled,
31    TransactionDropped,
32    /// Used by test code that manually cancels things.
33    ByTest,
34    /// Indicates the DiceTaskHandle was dropped without producing any result or (other) cancellation.
35    HandleDropped,
36}
