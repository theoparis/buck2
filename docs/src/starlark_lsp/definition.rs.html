51/// The location of a definition for a given identifier. See [`AstModule::find_definition_at_location`].
52#[derive(Debug, Clone, Eq, PartialEq)]
53pub(crate) enum IdentifierDefinition {
54    /// The definition was found at this location in the current file.
55    Location {
56        source: ResolvedSpan,
57        destination: ResolvedSpan,
58        name: String,
59    },
60    /// The symbol was loaded from another file. "destination" is the position within the
61    /// "load()" statement, but additionally, the path in that load statement, and the
62    /// name of the symbol within that file are provided so that additional lookups can
63    /// happen to find the original definition.
64    LoadedLocation {
65        source: ResolvedSpan,
66        destination: ResolvedSpan,
67        path: String,
68        name: String,
69    },
70    /// The symbol is the path component of a `load` statement. This is the raw string
71    /// that is in the AST, and needs to be properly resolved to a path to be useful.
72    LoadPath { source: ResolvedSpan, path: String },
73    /// A literal string in the source code. This can be a standalone string, or part of e.g. a
74    /// list, a dictionary, a return value, etc.
75    StringLiteral {
76        source: ResolvedSpan,
77        literal: String,
78    },
79    /// A named symbol was found, but it was not found in any of the current scopes. It
80    /// should be considered a global symbol, and attempted to be resolved externally.
81    Unresolved { source: ResolvedSpan, name: String },
82    /// Either the provided location was not an access of a variable, or no definition
83    /// could be found.
84    NotFound,
85}
86
87impl IdentifierDefinition {
88    fn source(&self) -> Option<ResolvedSpan> {
89        match self {
90            IdentifierDefinition::Location { source, .. }
91            | IdentifierDefinition::LoadedLocation { source, .. }
92            | IdentifierDefinition::LoadPath { source, .. }
93            | IdentifierDefinition::StringLiteral { source, .. }
94            | IdentifierDefinition::Unresolved { source, .. } => Some(*source),
95            IdentifierDefinition::NotFound => None,
96        }
97    }
98}

100/// A definition as in [`IdentifierDefinition`], but the source is within a dot expression.
101#[derive(Debug, Clone, Eq, PartialEq)]
102pub(crate) struct DottedDefinition {
103    /// The location of the member access that was requested.
104    pub(crate) source: ResolvedSpan,
105    /// The [`IdentifierDefinition`] for the left most component of the expression.
106    ///
107    /// For example, this would be the location of `x` if `source` enclosed `y` in `x.y.z`.
108    pub(crate) root_definition_location: IdentifierDefinition,
109    /// All of the identifiers up to the one that `source` includes.
110    ///
111    /// For example. if the `y` in `x.y.z` is the source, this would contain `x` and `y`.
112    pub(crate) segments: Vec<String>,
113}

115/// An inner structure that only holds references and unresolved spans. Used while
116/// finding locations for identifiers. Resolved to a fuller, owned structure
117/// ([`IdentifierDefinition`]) after the entire AST is visited.
118///
119/// This references temporary variables, so cannot be returned to external users directly.
120#[derive(Debug, Clone, Eq, PartialEq)]
121enum TempIdentifierDefinition<'a> {
122    /// The location of the definition of the symbol at the current line/column
123    Location {
124        source: Span,
125        destination: Span,
126        name: &'a str,
127    },
128    LoadedLocation {
129        source: Span,
130        destination: Span,
131        path: &'a str,
132        name: &'a str,
133    },
134    /// The definition was not found in the current scope but the name of an identifier
135    /// was found at the given position. This should be checked in outer scopes
136    /// to attempt to find the definition.
137    Name { source: Span, name: &'a str },
138    /// None of the accesses matched the position that was provided.
139    NotFound,
140}
141
142/// Reference version of [`DottedDefinitionLocation`]. Analogous to [`IdentifierDefinition`].
143#[derive(Debug, Clone, Eq, PartialEq)]
144struct TempDottedDefinition<'a> {
145    /// The location of the member access that was requested.
146    source: Span,
147    /// The [`IdentifierDefinition`] for the left most component of the expression.
148    ///
149    /// For example, this would be the location of `x` if `source` enclosed `y` in `x.y.z`.
150    root_definition_location: TempIdentifierDefinition<'a>,
151    /// The variable you are pointing at, e.g. `x.y.z` would be `x`.
152    variable: &'a AstIdent,
153    /// All of the attributes up to the one that `source` includes.
154    ///
155    /// For example. if the `y` in `x.y.z` is the source, this would contain `y`.
156    attributes: &'a [AstString],
157}
158
159#[derive(Debug, Clone, Eq, PartialEq)]
160enum TempDefinition<'a> {
161    Identifier(TempIdentifierDefinition<'a>),
162    Dotted(TempDottedDefinition<'a>),
163}
164
165impl<'a> From<TempIdentifierDefinition<'a>> for TempDefinition<'a> {
166    fn from(def: TempIdentifierDefinition<'a>) -> Self {
167        Self::Identifier(def)
168    }
169}
170
171impl<'a> From<TempDottedDefinition<'a>> for TempDefinition<'a> {
172    fn from(def: TempDottedDefinition<'a>) -> Self {
173        Self::Dotted(def)
174    }
175}

177/// The container for both definition locations of a standalone identifier, and
178/// for ones that access members (via '.' syntax)
179#[derive(Debug, Clone, Eq, PartialEq)]
180pub(crate) enum Definition {
181    Identifier(IdentifierDefinition),
182    Dotted(DottedDefinition),
183}
184
185impl From<IdentifierDefinition> for Definition {
186    fn from(def: IdentifierDefinition) -> Self {
187        Self::Identifier(def)
188    }
189}
190
191impl From<DottedDefinition> for Definition {
192    fn from(def: DottedDefinition) -> Self {
193        Self::Dotted(def)
194    }
195}

197impl Definition {
198    /// Get the "destination" of this location, but only within the current module.
199    ///
200    /// Some definition location types do not have a local definition.
201    fn local_destination(&self) -> Option<ResolvedSpan> {
202        match self {
203            Definition::Identifier(i)
204            | Definition::Dotted(DottedDefinition {
205                root_definition_location: i,
206                ..
207            }) => match i {
208                IdentifierDefinition::Location { destination, .. }
209                | IdentifierDefinition::LoadedLocation { destination, .. } => Some(*destination),
210                _ => None,
211            },
212        }
213    }
214
215    pub(crate) fn source(&self) -> Option<ResolvedSpan> {
216        match self {
217            Definition::Identifier(i) => i.source(),
218            Definition::Dotted(DottedDefinition { source, .. }) => Some(*source),
219        }
220    }
221}

234    /// Attempts to find the location where a symbol is defined in the module.
235    ///
236    /// `line` and `col` are zero based indexes of a location of the symbol to attempt to lookup.
237    ///
238    /// This method also handles scoping properly (i.e. an access of "foo" in a function
239    /// will return location of the parameter "foo", even if there is a global called "foo").
240    pub(crate) fn find_definition_at_location(&self, line: u32, col: u32) -> Definition {
241        // TODO(nmj): This should probably just store references to all of the AST nodes
242        //            when the LSPModule object is created, and then we can do a much faster
243        //            lookup, especially in cases where a file has not been changed, so the
244        //            LSPModule doesn't need to reparse anything.
245
246        let scope = scope(&self.ast);
247        let line_span = match self.ast.codemap().line_span_opt(line as usize) {
248            None => {
249                // The document got edited to add new lines, just bail out
250                return Definition::Identifier(IdentifierDefinition::NotFound);
251            }
252            Some(line_span) => line_span,
253        };
254        let current_pos = std::cmp::min(line_span.begin() + col, line_span.end());
255
256        // Finalize the results after recursing down from and back up to the the top level scope.
257        match Self::find_definition_in_scope(&scope, current_pos) {
258            TempDefinition::Identifier(def) => self
259                .get_definition_location(def, &scope, current_pos)
260                .into(),
261            TempDefinition::Dotted(def) => DottedDefinition {
262                source: self.ast.codemap().resolve_span(def.source),
263                root_definition_location: self.get_definition_location(
264                    def.root_definition_location,
265                    &scope,
266                    current_pos,
267                ),
268                segments: iter::once(def.variable.ident.clone())
269                    .chain(def.attributes.map(|s| s.node.to_owned()))
270                    .collect(),
271            }
272            .into(),
273        }
274    }

276    /// Look at the given scope and child scopes to try to find where the identifier
277    /// accessed at Pos is defined.
278    fn find_definition_in_scope(scope: &Scope, pos: Pos) -> TempDefinition<'_> {
279        /// Look for a name in the given scope, with a given source, and return the right
280        /// type of [`TempIdentifierDefinition`] based on whether / how the variable is bound.
281        fn resolve_get_in_scope<'a>(
282            scope: &'a Scope,
283            name: &'a str,
284            source: Span,
285        ) -> TempIdentifierDefinition<'a> {
286            match scope.bound.get(name) {
287                Some((Assigner::Load { path, name }, span)) => {
288                    TempIdentifierDefinition::LoadedLocation {
289                        source,
290                        destination: *span,
291                        path,
292                        name: name.as_str(),
293                    }
294                }
295                Some((_, span)) => TempIdentifierDefinition::Location {
296                    source,
297                    destination: *span,
298                    name,
299                },
300                // We know the symbol name, but it might only be available in
301                // an outer scope.
302                None => TempIdentifierDefinition::Name { source, name },
303            }
304        }
305
306        for bind in &scope.inner {
307            match bind {
308                Bind::Get(g) if g.span.contains(pos) => {
309                    return resolve_get_in_scope(scope, g.node.ident.as_str(), g.span).into();
310                }
311                Bind::Scope(inner_scope) => {
312                    match Self::find_definition_in_scope(inner_scope, pos) {
313                        TempDefinition::Identifier(TempIdentifierDefinition::Name {
314                            source,
315                            name,
316                        }) => {
317                            return resolve_get_in_scope(scope, name, source).into();
318                        }
319                        TempDefinition::Identifier(TempIdentifierDefinition::NotFound) => {}
320                        TempDefinition::Dotted(TempDottedDefinition {
321                            source,
322                            root_definition_location:
323                                TempIdentifierDefinition::Name {
324                                    source: root_source,
325                                    name: root_name,
326                                },
327                            variable,
328                            attributes,
329                        }) => {
330                            return TempDefinition::Dotted(TempDottedDefinition {
331                                source,
332                                root_definition_location: resolve_get_in_scope(
333                                    scope,
334                                    root_name,
335                                    root_source,
336                                ),
337                                variable,
338                                attributes,
339                            });
340                        }
341                        x @ TempDefinition::Identifier(TempIdentifierDefinition::Location {
342                            ..
343                        })
344                        | x @ TempDefinition::Identifier(
345                            TempIdentifierDefinition::LoadedLocation { .. },
346                        )
347                        | x @ TempDefinition::Dotted(_) => {
348                            return x;
349                        }
350                    }
351                }
352                Bind::GetDotted(dotted) => {
353                    if let Some((idx, source)) = dotted.contains(pos) {
354                        let root_identifier = &dotted.variable;
355                        let root_definition_location = resolve_get_in_scope(
356                            scope,
357                            root_identifier.node.ident.as_str(),
358                            root_identifier.span,
359                        );
360                        // If someone clicks on the "root" identifier, just treat it as a "get"
361                        match idx {
362                            None => return root_definition_location.into(),
363                            Some(idx) => {
364                                return TempDottedDefinition {
365                                    source,
366                                    root_definition_location,
367                                    variable: &dotted.variable,
368                                    attributes: &dotted.attributes[0..idx + 1],
369                                }
370                                .into();
371                            }
372                        }
373                    }
374                }
375                // For everything else, just ignore it. Note that the `Get` is ignored
376                // because we already checked the pos above.
377                Bind::Set(_, _) | Bind::Flow | Bind::Get(_) => {}
378            }
379        }
380        TempIdentifierDefinition::NotFound.into()
381    }

383    /// Converts a `TempIdentifierDefinition` to an `IdentifierDefinition`, resolving spans
384    /// against the current AST, owning strings, etc.
385    fn get_definition_location(
386        &self,
387        definition: TempIdentifierDefinition,
388        scope: &Scope,
389        current_pos: Pos,
390    ) -> IdentifierDefinition {
391        match definition {
392            TempIdentifierDefinition::Location {
393                source,
394                destination,
395                name,
396            } => IdentifierDefinition::Location {
397                source: self.ast.codemap().resolve_span(source),
398                destination: self.ast.codemap().resolve_span(destination),
399                name: name.to_owned(),
400            },
401            TempIdentifierDefinition::Name { source, name } => match scope.bound.get(name) {
402                None => IdentifierDefinition::Unresolved {
403                    source: self.ast.codemap().resolve_span(source),
404                    name: name.to_owned(),
405                },
406                Some((Assigner::Load { path, name }, span)) => {
407                    IdentifierDefinition::LoadedLocation {
408                        source: self.ast.codemap().resolve_span(source),
409                        destination: self.ast.codemap().resolve_span(*span),
410                        path: path.node.clone(),
411                        name: name.node.clone(),
412                    }
413                }
414                Some((_, span)) => IdentifierDefinition::Location {
415                    source: self.ast.codemap().resolve_span(source),
416                    destination: self.ast.codemap().resolve_span(*span),
417                    name: name.to_owned(),
418                },
419            },
420            // If we could not find the symbol, see if the current position is within
421            // a load statement (these are not exposed as Get/Set in bind).
422            TempIdentifierDefinition::NotFound => self.find_definition_from_ast(current_pos),
423            TempIdentifierDefinition::LoadedLocation {
424                source,
425                destination,
426                path,
427                name,
428            } => IdentifierDefinition::LoadedLocation {
429                source: self.ast.codemap().resolve_span(source),
430                destination: self.ast.codemap().resolve_span(destination),
431                path: path.to_owned(),
432                name: name.to_owned(),
433            },
434        }
435    }

461    /// Attempt to find the location in this module where a member of a struct (named `name`)
462    /// is defined.
463    ///
464    /// This helps with the common idiom of
465    /// ```python
466    /// ...
467    /// Foo = struct(
468    ///     member1 = _member1,
469    /// )
470    /// ```
471    ///
472    /// which is imported by other files and used as `Foo.member1`. Rather than jumping to `Foo`,
473    /// this would jump to `_member1` if it exists.
474    pub(crate) fn find_exported_symbol_and_member(
475        &self,
476        name: &str,
477        member: &str,
478    ) -> Option<ResolvedSpan> {
479        // If we can't fully resolve the symbol, try to get as "close" as possible.
480        // Whether this is the left hand side of an assign statement, or maybe the
481        // left side of a named argument in the struct that matches the member name.
482        let mut arg_span = None;
483        let mut identifier_span = None;
484        let mut symbol_to_lookup = None;
485
486        'outer: for v in top_level_stmts(self.ast.statement()) {
487            if let StmtP::Assign(assign) = &v.node {
488                let main_assign_span = match &assign.lhs.node {
489                    AssignTargetP::Identifier(main_assign_id) if main_assign_id.ident == name => {
490                        main_assign_id.span
491                    }
492                    _ => {
493                        continue 'outer;
494                    }
495                };
496                // If nothing else, go to the left hand of the assignment expression.
497                if identifier_span.is_none() {
498                    identifier_span = Some(main_assign_span);
499                }
500
501                // Look for a function call to `struct`.
502                if let ExprP::Call(function_name, args) = &assign.rhs.node {
503                    match &function_name.node {
504                        ExprP::Identifier(function_name)
505                            if function_name.node.ident == "struct" => {}
506                        _ => {
507                            continue 'outer;
508                        }
509                    }
510
511                    for arg in &args.args {
512                        if let ArgumentP::Named(arg_name, arg_expr) = &arg.node {
513                            if arg_name.node != member {
514                                continue;
515                            }
516                            if arg_span.is_none() {
517                                arg_span = Some(arg_name.span);
518                            }
519                            if let ExprP::Identifier(arg_value_id) = &arg_expr.node {
520                                symbol_to_lookup = Some(arg_value_id.span);
521                                break 'outer;
522                            }
523                            break;
524                        }
525                    }
526                }
527            }
528        }
529
530        // Try to find the symbol that is assigned, but if not, try to get to that "closest" span.
531        symbol_to_lookup
532            .and_then(|span| {
533                let resolved = self.ast.codemap().resolve_span(span);
534                self.find_definition_at_location(
535                    resolved.begin.line as u32,
536                    resolved.begin.column as u32,
537                )
538                .local_destination()
539            })
540            .or_else(|| match (arg_span, identifier_span) {
541                (Some(span), _) => Some(self.ast.codemap().resolve_span(span)),
542                (None, Some(span)) => Some(self.ast.codemap().resolve_span(span)),
543                (None, None) => None,
544            })
545    }

547    fn find_definition_from_ast(&self, pos: Pos) -> IdentifierDefinition {
548        fn visit_node(
549            codemap: &CodeMap,
550            pos: Pos,
551            ret: &mut Option<IdentifierDefinition>,
552            node: Visit<AstNoPayload>,
553        ) {
554            if ret.is_some() {
555                return;
556            }
557            match node {
558                Visit::Expr(Spanned {
559                    node: Expr::Literal(AstLiteral::String(s)),
560                    ..
561                }) if s.span.contains(pos) => {
562                    *ret = Some(IdentifierDefinition::StringLiteral {
563                        source: codemap.resolve_span(s.span),
564                        literal: s.node.to_owned(),
565                    });
566                }
567                Visit::Stmt(Spanned {
568                    node: Stmt::Load(load),
569                    ..
570                }) => {
571                    *ret = if load.module.span.contains(pos) {
572                        Some(IdentifierDefinition::LoadPath {
573                            source: codemap.resolve_span(load.module.span),
574                            path: load.module.node.to_owned(),
575                        })
576                    } else {
577                        load.args.iter().find_map(|LoadArgP { local, their, .. }| {
578                            if local.span.contains(pos) || their.span.contains(pos) {
579                                Some(IdentifierDefinition::LoadedLocation {
580                                    source: codemap.resolve_span(their.span),
581                                    destination: codemap.resolve_span(their.span),
582                                    path: load.module.node.to_owned(),
583                                    name: their.node.to_owned(),
584                                })
585                            } else {
586                                None
587                            }
588                        })
589                    }
590                }
591                v => v.visit_children(|node| visit_node(codemap, pos, ret, node)),
592            }
593        }
594
595        let mut ret = None;
596        visit_node(
597            self.ast.codemap(),
598            pos,
599            &mut ret,
600            Visit::Stmt(self.ast.statement()),
601        );
602        ret.unwrap_or(IdentifierDefinition::NotFound)
603    }

617    /// Result of parsing a starlark fixture that has range markers in it. See `FixtureWithRanges::from_fixture`
618    #[derive(Debug, Clone, PartialEq, Eq)]
619    pub(crate) struct FixtureWithRanges {
620        filename: String,
621        /// The starlark program with markers removed.
622        program: String,
623        /// The location of all of the symbols that were indicated by the test fixture.
624        ranges: HashMap<String, ResolvedSpan>,
625    }
626
627    impl FixtureWithRanges {
628        /// Parse a program that has markers in it indicating a "range" by name.
629        ///
630        /// Markers are simple `<\w+>` and `</\w+>` tags. The returned program is the
631        /// program content with these markers removed.
632        ///
633        /// For example:
634        /// ```#starlark
635        /// load("foo.star", <a>"bar"</a>);
636        ///
637        /// x = 1
638        /// <bar_highlight><bar_click>b</bar_click>ar</bar_highlight>(<x>x</x>)
639        /// ```
640        ///
641        /// Would return a program body of:
642        /// ```#starlark
643        /// load("foo.star", "bar");
644        ///
645        /// x = 1
646        /// bar(x)
647        /// ```
648        ///
649        /// And ranges of:
650        ///  `a`: 0:17-0:22
651        ///  `bar_highlight`: 3:0-3:2
652        ///  `bar_click`: 3:0-3:1
653        ///  `x`: 3:4-3:5
654        pub(crate) fn from_fixture(filename: &str, fixture: &str) -> anyhow::Result<Self> {
655            let re = regex::Regex::new(r#"<(/)?(\w+)>"#).unwrap();
656            let mut program = String::new();
657            let mut ranges: HashMap<String, (Option<usize>, Option<usize>)> = HashMap::new();
658
659            let mut fixture_idx = 0;
660            for matches in re.captures_iter(fixture) {
661                let full_tag = matches.get(0).unwrap();
662                let is_end_tag = matches.get(1).is_some();
663                let identifier = matches.get(2).unwrap().as_str().to_owned();
664
665                program.push_str(&fixture[fixture_idx..full_tag.start()]);
666                fixture_idx = full_tag.end();
667
668                match (is_end_tag, ranges.entry(identifier.clone())) {
669                    (false, Entry::Occupied(_)) => {
670                        return Err(anyhow::anyhow!("duplicate open tag `{}` found", identifier));
671                    }
672                    (false, Entry::Vacant(e)) => {
673                        e.insert((Some(program.len()), None));
674                    }
675                    (true, Entry::Occupied(mut e)) => {
676                        e.insert((e.get().0, Some(program.len())));
677                    }
678                    (true, Entry::Vacant(_)) => {
679                        return Err(anyhow::anyhow!(
680                            "found closing tag for `{}`, but no open tag",
681                            identifier
682                        ));
683                    }
684                }
685            }
686            program.push_str(&fixture[fixture_idx..fixture.len()]);
687
688            let code_map = CodeMap::new(filename.to_owned(), program.clone());
689            let spans: HashMap<String, ResolvedSpan> = ranges
690                .into_iter()
691                .map(|(id, (start, end))| {
692                    let span = Span::new(
693                        Pos::new(start.unwrap() as u32),
694                        Pos::new(end.unwrap() as u32),
695                    );
696                    (id, code_map.resolve_span(span))
697                })
698                .collect();
699
700            Ok(Self {
701                filename: filename.to_owned(),
702                program,
703                ranges: spans,
704            })
705        }
706
707        pub(crate) fn begin_line(&self, identifier: &str) -> u32 {
708            self.ranges
709                .get(identifier)
710                .expect("identifier to be present")
711                .begin
712                .line as u32
713        }
714
715        pub(crate) fn begin_column(&self, identifier: &str) -> u32 {
716            self.ranges
717                .get(identifier)
718                .expect("identifier to be present")
719                .begin
720                .column as u32
721        }
722
723        pub(crate) fn resolved_span(&self, identifier: &str) -> ResolvedSpan {
724            *self
725                .ranges
726                .get(identifier)
727                .expect("identifier to be present")
728        }
729
730        pub(crate) fn module(&self) -> starlark::Result<LspModule> {
731            Ok(LspModule::new(AstModule::parse(
732                &self.filename,
733                self.program.clone(),
734                // TODO(nga): use dialect of current module.
735                &Dialect::AllOptionsInternal,
736            )?))
737        }
738
739        pub(crate) fn program(&self) -> String {
740            self.program.clone()
741        }
