38/// The type of an exported symbol.
39/// If unknown, will use `Any`.
40#[derive(Debug, PartialEq, Eq, Clone, Hash)]
41pub(crate) enum SymbolKind {
42    /// Any kind of symbol.
43    Any,
44    /// The symbol represents something that can be called, for example
45    /// a `def` or a variable assigned to a `lambda`.
46    Function { argument_names: Vec<String> },
47}
48
49impl SymbolKind {
50    pub(crate) fn from_expr(x: &Expr) -> Self {
51        match x {
52            Expr::Lambda(lambda) => Self::Function {
53                argument_names: lambda
54                    .params
55                    .iter()
56                    .filter_map(|param| param.split().0.map(|name| name.to_string()))
57                    .collect(),
58            },
59            _ => Self::Any,
60        }
61    }
62}
63
64impl From<SymbolKind> for CompletionItemKind {
65    fn from(value: SymbolKind) -> Self {
66        match value {
67            SymbolKind::Any => CompletionItemKind::CONSTANT,
68            SymbolKind::Function { .. } => CompletionItemKind::FUNCTION,
69        }
70    }
71}

73/// A symbol. Returned from [`AstModule::exported_symbols`].
74#[derive(Debug, PartialEq, Clone)]
75pub(crate) struct Symbol {
76    /// The name of the symbol.
77    pub(crate) name: String,
78    /// The location of its definition.
79    pub(crate) span: FileSpan,
80    /// The type of symbol it represents.
81    pub(crate) kind: SymbolKind,
82    /// The documentation for this symbol.
83    pub(crate) docs: Option<DocItem>,
84}
85
86impl From<Symbol> for CompletionItem {
87    fn from(value: Symbol) -> Self {
88        let documentation = value.docs.map(|docs| {
89            Documentation::MarkupContent(MarkupContent {
90                kind: MarkupKind::Markdown,
91                value: render_doc_item_no_link(&value.name, &docs),
92            })
93        });
94        Self {
95            label: value.name,
96            kind: Some(value.kind.into()),
97            documentation,
98            ..Default::default()
99        }
100    }
101}

103pub(crate) trait AstModuleExportedSymbols {
104    /// Which symbols are exported by this module. These are the top-level assignments,
105    /// including function definitions. Any symbols that start with `_` are not exported.
106    fn exported_symbols(&self) -> Vec<Symbol>;
107}
108
109impl AstModuleExportedSymbols for AstModule {
110    fn exported_symbols(&self) -> Vec<Symbol> {
111        // Map since we only want to store the first of each export
112        // IndexMap since we want the order to match the order they were defined in
113        let mut result: SmallMap<&str, _> = SmallMap::new();
114
115        fn add<'a>(
116            me: &AstModule,
117            result: &mut SmallMap<&'a str, Symbol>,
118            name: &'a AstAssignIdent,
119            kind: SymbolKind,
120            resolve_docs: impl FnOnce() -> Option<DocItem>,
121        ) {
122            if !name.ident.starts_with('_') {
123                result.entry(&name.ident).or_insert(Symbol {
124                    name: name.ident.clone(),
125                    span: me.file_span(name.span),
126                    kind,
127                    docs: resolve_docs(),
128                });
129            }
130        }
131
132        let mut last_node = None;
133        for x in top_level_stmts(self.statement()) {
134            match &**x {
135                Stmt::Assign(assign) => {
136                    assign.lhs.visit_lvalue(|name| {
137                        let kind = SymbolKind::from_expr(&assign.rhs);
138                        add(self, &mut result, name, kind, || {
139                            last_node
140                                .and_then(|last| get_doc_item_for_assign(last, &assign.lhs))
141                                .map(|x| DocItem::Member(DocMember::Property(x)))
142                        });
143                    });
144                }
145                Stmt::AssignModify(dest, _, _) => {
146                    dest.visit_lvalue(|name| {
147                        add(self, &mut result, name, SymbolKind::Any, || {
148                            last_node
149                                .and_then(|last| get_doc_item_for_assign(last, dest))
150                                .map(|x| DocItem::Member(DocMember::Property(x)))
151                        });
152                    });
153                }
154                Stmt::Def(def) => {
155                    add(
156                        self,
157                        &mut result,
158                        &def.name,
159                        SymbolKind::Function {
160                            argument_names: def
161                                .params
162                                .iter()
163                                .filter_map(|param| param.split().0.map(|name| name.to_string()))
164                                .collect(),
165                        },
166                        || {
167                            get_doc_item_for_def(def, self.codemap())
168                                .map(|x| DocItem::Member(DocMember::Function(x)))
169                        },
170                    );
171                }
172                _ => {}
173            }
174            last_node = Some(x);
175        }
176        result.into_values().collect()
177    }
178}

180#[cfg(test)]
181mod tests {
182    use starlark::syntax::Dialect;
183    use starlark_syntax::slice_vec_ext::SliceExt;
184
185    use super::*;
186
187    fn module(x: &str) -> AstModule {
188        AstModule::parse("X", x.to_owned(), &Dialect::AllOptionsInternal).unwrap()
189    }
190
191    #[test]
192    fn test_lint_exported() {
193        let modu = module(
194            r#"
195load("test", "a")
196def b(): pass
197d = 1
198def _e(): pass
199d = 2
200"#,
201        );
202        let res = modu.exported_symbols();
203        assert_eq!(
204            res.map(|symbol| format!("{} {}", symbol.span, symbol.name)),
205            &["X:3:5-6 b", "X:4:1-2 d"]
206        );
207    }
