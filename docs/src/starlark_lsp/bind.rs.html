42#[derive(Debug, Clone, Eq, PartialEq)]
43pub(crate) enum Assigner {
44    /// Obtained from `load`. `name` is the symbol in that file, not necessarily the local name
45    Load {
46        path: AstString,
47        name: AstString,
48    },
49    Argument, // From a function argument
50    Assign,   // From an assignment
51}
52
53#[derive(Debug)]
54pub(crate) enum Bind {
55    Set(Assigner, AstAssignIdent), // Variable assigned to directly
56    Get(AstIdent),                 // Variable that is referenced
57    GetDotted(GetDotted),          // Variable is referenced, but is part of a dotted access
58    Flow,         // Flow control occurs here (if, for etc) - can arrive or leave at this point
59    Scope(Scope), // Entering a new scope (lambda/def/comprehension)
60}
61
62/// A 'get' bind that was part of a dotted member access pattern.
63///
64/// e.g. `x.y.z` would be considered a variable of `x` with attributes `y` and `z`
65#[derive(Debug, Clone)]
66pub(crate) struct GetDotted {
67    pub(crate) variable: AstIdent,
68    pub(crate) attributes: Vec<AstString>,
69}
70
71impl GetDotted {
72    fn new(variable: AstIdent, attributes: Vec<AstString>) -> Self {
73        Self {
74            variable,
75            attributes,
76        }
77    }
78
79    /// Determines if a position is contained in this dot access.
80    ///
81    /// The returned value is the index of the segment where `pos` was found, and
82    /// the span that contained it. If the Pos is inside the variable the index is None,
83    /// otherwise the index inside the attributes.
84    pub(crate) fn contains(&self, pos: Pos) -> Option<(Option<usize>, Span)> {
85        if self.variable.span.contains(pos) {
86            return Some((None, self.variable.span));
87        }
88        for (i, s) in self.attributes.iter().enumerate() {
89            if s.span.contains(pos) {
90                return Some((Some(i), s.span));
91            }
92        }
93        None
94    }
95}

97#[derive(Debug)]
98pub(crate) struct Scope {
99    pub inner: Vec<Bind>,
100    pub(crate) free: HashMap<String, Span>, // Things referred to in this scope, or inner scopes, that we don't define
101    pub(crate) bound: HashMap<String, (Assigner, Span)>, // Things bound in this scope, doesn't include inner scope bindings
102}
103
104impl Scope {
105    fn new(inner: Vec<Bind>) -> Self {
106        let mut bound: HashMap<String, _> = HashMap::new();
107        let mut free: HashMap<String, _> = HashMap::new();
108        for x in &inner {
109            match x {
110                Bind::Set(assigner, x) => {
111                    bound
112                        .entry(x.ident.clone())
113                        .or_insert((assigner.clone(), x.span));
114                }
115                Bind::Get(x) => {
116                    free.entry(x.node.ident.clone()).or_insert(x.span);
117                }
118                Bind::GetDotted(x) => {
119                    free.entry(x.variable.node.ident.clone())
120                        .or_insert(x.variable.span);
121                }
122                Bind::Scope(scope) => scope.free.iter().for_each(|(k, v)| {
123                    free.entry(k.clone()).or_insert(*v);
124                }),
125                Bind::Flow => {}
126            }
127        }
128        for x in bound.keys() {
129            free.remove(x);
130        }
131
132        Self { inner, free, bound }
133    }
134}

136fn opt_expr(x: Option<&AstExpr>, res: &mut Vec<Bind>) {
137    if let Some(x) = x {
138        expr(x, res)
139    }
140}
141
142fn opt_type_expr(x: Option<&AstTypeExpr>, res: &mut Vec<Bind>) {
143    if let Some(x) = x {
144        expr(&x.expr, res)
145    }
146}

148fn comprehension(
149    for_: &ForClause,
150    clauses: &[Clause],
151    res: &mut Vec<Bind>,
152    end: impl Fn(&mut Vec<Bind>),
153) {
154    expr(&for_.over, res);
155    let mut inner = Vec::new();
156    expr_lvalue(&for_.var, &mut inner);
157    for clause in clauses {
158        match clause {
159            Clause::For(ForClause { var, over }) => {
160                expr(over, &mut inner);
161                expr_lvalue(var, &mut inner);
162            }
163            Clause::If(x) => expr(x, &mut inner),
164        }
165    }
166    end(&mut inner);
167    res.push(Bind::Scope(Scope::new(inner)))
168}

170/// Can we interpret this as a Dotted expression? If not, just treat it normally.
171/// Note that we treat `foo.bar().baz` like the dotted expression `foo.bar.baz`.
172fn dot_access<'a>(lhs: &'a AstExpr, attribute: &'a AstString, res: &mut Vec<Bind>) {
173    // The attributes are stored in reverse.
174    // Keep recursing while we have a chain of dots, if we find something else bail out
175    fn f<'a>(lhs: &'a AstExpr, mut attributes: Vec<&'a AstString>, res: &mut Vec<Bind>) {
176        match &**lhs {
177            Expr::Identifier(id) => {
178                let attrs = attributes.into_iter().rev().cloned().collect();
179                res.push(Bind::GetDotted(GetDotted::new(id.clone(), attrs)));
180            }
181            Expr::Dot(lhs, attribute) => {
182                attributes.push(attribute);
183                f(lhs, attributes, res);
184            }
185            Expr::Call(name, args) => {
186                f(name, attributes, res);
187                // make sure that if someone does a(b).c, 'b' is bound and considered used.
188                for parameter in &args.args {
189                    expr(parameter.expr(), res);
190                }
191            }
192            _ => {
193                // We saw through a series of dots, but it isn't a dotted-identifier
194                // Bail out and just deal with the LHS normally
195                expr(lhs, res);
196            }
197        }
198    }
199    f(lhs, vec![attribute], res);
200}
201
202fn expr(x: &AstExpr, res: &mut Vec<Bind>) {
203    match &**x {
204        Expr::Identifier(x) => res.push(Bind::Get(x.clone())),
205        Expr::Lambda(LambdaP {
206            params,
207            body,
208            payload: _,
209        }) => {
210            let mut inner = Vec::new();
211            parameters(params, res, &mut inner);
212            expr(body, &mut inner);
213            res.push(Bind::Scope(Scope::new(inner)));
214        }
215        Expr::Dot(lhs, attribute) => dot_access(lhs, attribute, res),
216        Expr::ListComprehension(x, for_, clauses) => {
217            comprehension(for_, clauses, res, |res| expr(x, res))
218        }
219        Expr::DictComprehension(x, for_, clauses) => comprehension(for_, clauses, res, |res| {
220            expr(&x.0, res);
221            expr(&x.1, res)
222        }),
223
224        // Uninteresting - just recurse
225        _ => x.visit_expr(|x| expr(x, res)),
226    }
227}
228
229fn expr_lvalue(x: &AstAssignTarget, res: &mut Vec<Bind>) {
230    x.visit_expr(|x| expr(x, res));
231    x.visit_lvalue(|x| res.push(Bind::Set(Assigner::Assign, x.clone())))
232}
233
234fn parameters(args: &[AstParameter], res: &mut Vec<Bind>, inner: &mut Vec<Bind>) {
235    for a in args {
236        let (name, typ, default) = a.split();
237        opt_type_expr(typ, res);
238        opt_expr(default, res);
239        if let Some(name) = name {
240            inner.push(Bind::Set(Assigner::Argument, name.clone()))
241        }
242    }
243}
244
245fn flow(res: &mut Vec<Bind>) {
246    res.push(Bind::Flow)
247}

249fn stmt(x: &AstStmt, res: &mut Vec<Bind>) {
250    match &**x {
251        Stmt::Statements(xs) => {
252            for x in xs {
253                stmt(x, res)
254            }
255        }
256        Stmt::Break | Stmt::Continue | Stmt::Return(None) => flow(res),
257        Stmt::Pass => {}
258        Stmt::Return(Some(x)) => {
259            expr(x, res);
260            flow(res)
261        }
262        Stmt::Expression(x) => expr(x, res),
263        Stmt::If(a, b) => {
264            expr(a, res);
265            flow(res);
266            stmt(b, res);
267            flow(res);
268        }
269        Stmt::IfElse(a, b_c) => {
270            let (b, c) = &**b_c;
271            expr(a, res);
272            flow(res);
273            stmt(b, res);
274            flow(res);
275            stmt(c, res);
276            flow(res);
277        }
278        Stmt::Def(DefP {
279            name,
280            params,
281            return_type,
282            body,
283            payload: _,
284        }) => {
285            opt_type_expr(return_type.as_ref().map(|x| &**x), res);
286            let mut inner = Vec::new();
287            parameters(params, res, &mut inner);
288            res.push(Bind::Set(Assigner::Assign, name.clone()));
289            stmt(body, &mut inner);
290            res.push(Bind::Scope(Scope::new(inner)));
291        }
292        Stmt::Assign(AssignP { lhs, ty, rhs }) => {
293            opt_type_expr(ty.as_ref(), res);
294            expr(rhs, res);
295            expr_lvalue(lhs, res);
296        }
297        Stmt::AssignModify(lhs, _, rhs) => {
298            // For a += b, we:
299            // 1. Evaluate all variables and expressions in a.
300            // 2. Evaluate b.
301            // 3. Assign to all variables in a.
302            lhs.visit_expr(|x| expr(x, res));
303            lhs.visit_lvalue(|x| {
304                res.push(Bind::Get(x.clone().map(
305                    |AssignIdentP {
306                         ident: s,
307                         payload: (),
308                     }| IdentP {
309                        ident: s,
310                        payload: (),
311                    },
312                )))
313            });
314            expr(rhs, res);
315            expr_lvalue(lhs, res);
316        }
317        Stmt::For(ForP { var, over, body }) => {
318            expr(over, res);
319            expr_lvalue(var, res);
320            flow(res);
321            stmt(body, res);
322            flow(res)
323        }
324        Stmt::Load(load) => {
325            for x in &load.args {
326                res.push(Bind::Set(
327                    Assigner::Load {
328                        path: load.module.clone(),
329                        name: x.their.clone(),
330                    },
331                    x.local.clone(),
332                ))
333            }
334        }
335    }
336}

338pub(crate) fn scope(module: &AstModule) -> Scope {
339    let mut res = Vec::new();
340    stmt(module.statement(), &mut res);
341    Scope::new(res)
342}

344#[cfg(test)]
345mod tests {
346    use std::iter;
347
348    use starlark::syntax::Dialect;
349    use starlark_syntax::slice_vec_ext::SliceExt;
350    use starlark_syntax::slice_vec_ext::VecExt;

353
354    #[test]
355    fn dotted_access_is_correct() -> starlark::Result<()> {
356        let contents = "x1.y\nx1.y().z\nx2().y\nx2().y.z";
357
358        let expected = vec![
359            vec!["x1", "y"],
360            vec!["x1", "y", "z"],
361            vec!["x2", "y"],
362            vec!["x2", "y", "z"],
363        ]
364        .into_map(|names| names.into_map(String::from));
365
366        let module = AstModule::parse(
367            "foo.star",
368            contents.to_owned(),
369            &Dialect::AllOptionsInternal,
370        )?;
371        let scope = scope(&module);
372
373        let found_bindings = scope
374            .inner
375            .iter()
376            .map(|b| match b {
377                Bind::GetDotted(get) => Ok(iter::once(get.variable.ident.as_str())
378                    .chain(get.attributes.map(|a| a.node.as_str()))
379                    .map(|s| s.to_owned())
380                    .collect()),
381                _ => Err(anyhow::anyhow!("Unexpected bind {:?}", b)),
382            })
383            .collect::<anyhow::Result<Vec<Vec<_>>>>()?;
384
385        assert_eq!(expected, found_bindings);
386
387        Ok(())
388    }
389
390    #[test]
391    fn dotted_contains_is_correct() -> starlark::Result<()> {
392        let contents = "x1.y1.z1\nx2.y2.z2";
393        let module = AstModule::parse(
394            "foo.star",
395            contents.to_owned(),
396            &Dialect::AllOptionsInternal,
397        )?;
398        let scope = scope(&module);
399
400        let get = scope
401            .inner
402            .iter()
403            .find_map(|b| match b {
404                Bind::GetDotted(get) => Some((*get).clone()),
405                _ => None,
406            })
407            .unwrap();
408
409        let x1_pos = Pos::new(1);
410        let y1_pos = Pos::new(4);
411        let z1_pos = Pos::new(7);
412        let x2_pos = Pos::new(10);
413
414        let x1_expected = Some((None, Span::new(Pos::new(0), Pos::new(2))));
415        let y1_expected = Some((Some(0), Span::new(Pos::new(3), Pos::new(5))));
416        let z1_expected = Some((Some(1), Span::new(Pos::new(6), Pos::new(8))));
417
418        assert_eq!(x1_expected, get.contains(x1_pos));
419        assert_eq!(y1_expected, get.contains(y1_pos));
420        assert_eq!(z1_expected, get.contains(z1_pos));
421        assert!(get.contains(x2_pos).is_none());
422
423        Ok(())
424    }
