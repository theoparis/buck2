135/// Errors that can happen when converting LspUrl and Url to/from each other.
136#[derive(thiserror::Error, Debug)]
137pub enum LspUrlError {
138    /// The path component of the URL was not absolute. This is required for all supported
139    /// URL types.
140    #[error("`{}` does not have an absolute path component", .0)]
141    NotAbsolute(Url),
142    /// For some reason the PathBuf/Url in the LspUrl could not be converted back to a URL.
143    #[error("`{}` could not be converted back to a URL", .0)]
144    Unparsable(LspUrl),
145    #[error("invalid URL for file:// schema (possibly not absolute?): `{}`", .0)]
146    InvalidFileUrl(Url),
147}
148
149/// A URL that represents the two types (plus an "Other") of URIs that are supported.
150#[derive(Clone, Debug, Hash, Eq, PartialEq, Display)]
151pub enum LspUrl {
152    /// A "file://" url with a path sent from the LSP client.
153    #[display("file://{}", _0.display())]
154    File(PathBuf),
155    /// A "starlark:" url. This is mostly used for native types that don't actually
156    /// exist on the filesystem. The path component always has a leading slash.
157    #[display("starlark:{}", _0.display())]
158    Starlark(PathBuf),
159    /// Any other type. Often should just be ignored, or return an error.
160    #[display("{}", _0)]
161    Other(Url),
162}
163
164impl Serialize for LspUrl {
165    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
166    where
167        S: Serializer,
168    {
169        match Url::try_from(self) {
170            Ok(url) => url.serialize(serializer),
171            Err(e) => Err(serde::ser::Error::custom(e.to_string())),
172        }
173    }
174}
175
176impl<'de> Deserialize<'de> for LspUrl {
177    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
178    where
179        D: Deserializer<'de>,
180    {
181        let url = Url::deserialize(deserializer)?;
182        LspUrl::try_from(url).map_err(|e| serde::de::Error::custom(e.to_string()))
183    }
184}

186impl LspUrl {
187    /// Returns the path component of the underlying URL
188    pub fn path(&self) -> &Path {
189        match self {
190            LspUrl::File(p) => p.as_path(),
191            LspUrl::Starlark(p) => p.as_path(),
192            LspUrl::Other(u) => Path::new(u.path()),
193        }
194    }
195}

197impl TryFrom<Url> for LspUrl {
198    type Error = LspUrlError;
199
200    fn try_from(url: Url) -> Result<Self, Self::Error> {
201        match url.scheme() {
202            "file" => {
203                let file_path = PathBuf::from(
204                    url.to_file_path()
205                        .map_err(|_| LspUrlError::InvalidFileUrl(url.clone()))?,
206                );
207                if file_path.is_absolute() {
208                    Ok(Self::File(file_path))
209                } else {
210                    Err(LspUrlError::NotAbsolute(url))
211                }
212            }
213            "starlark" => {
214                // Need to perform the replace to standardize on / instead of \ on windows.
215                let path = PathBuf::from(url.path().replace('\\', "/"));
216                // Use "starts with a /" because, while leading slashes are accepted on
217                // windows, they do not report "true" from `is_absolute()`.
218                if path.to_string_lossy().starts_with('/') {
219                    Ok(Self::Starlark(path))
220                } else {
221                    Err(LspUrlError::NotAbsolute(url))
222                }
223            }
224            _ => Ok(Self::Other(url)),
225        }
226    }
227}

229impl TryFrom<LspUrl> for Url {
230    type Error = LspUrlError;
231
232    fn try_from(url: LspUrl) -> Result<Self, Self::Error> {
233        Url::try_from(&url)
234    }
235}
236
237impl TryFrom<&LspUrl> for Url {
238    type Error = LspUrlError;
239
240    fn try_from(url: &LspUrl) -> Result<Self, Self::Error> {
241        match &url {
242            LspUrl::File(p) => {
243                Url::from_file_path(p).map_err(|_| LspUrlError::Unparsable(url.clone()))
244            }
245            LspUrl::Starlark(p) => Url::parse(&format!("starlark:{}", p.display()))
246                .map_err(|_| LspUrlError::Unparsable(url.clone())),
247            LspUrl::Other(u) => Ok(u.clone()),
248        }
249    }
250}

403impl<T: LspContext> Backend<T> {
404    fn server_capabilities(settings: LspServerSettings) -> ServerCapabilities {
405        let definition_provider = settings.enable_goto_definition.then_some({
406            OneOf::Right(DefinitionOptions {
407                work_done_progress_options: WorkDoneProgressOptions {
408                    work_done_progress: None,
409                },
410            })
411        });
412        ServerCapabilities {
413            text_document_sync: Some(TextDocumentSyncCapability::Kind(TextDocumentSyncKind::FULL)),
414            definition_provider,
415            completion_provider: Some(CompletionOptions::default()),
416            hover_provider: Some(HoverProviderCapability::Simple(true)),
417            ..ServerCapabilities::default()
418        }
419    }

421    fn get_ast(&self, uri: &LspUrl) -> Option<Arc<LspModule>> {
422        let last_valid_parse = self.last_valid_parse.read().unwrap();
423        last_valid_parse.get(uri).duped()
424    }

426    pub(crate) fn get_ast_or_load_from_disk(
427        &self,
428        uri: &LspUrl,
429    ) -> anyhow::Result<Option<Arc<LspModule>>> {
430        let module = match self.get_ast(uri) {
431            Some(result) => Some(result),
432            None => self
433                .context
434                .parse_file(uri)?
435                .and_then(|eval_result| eval_result.ast.map(|ast| Arc::new(LspModule::new(ast)))),
436        };
437        Ok(module)
438    }
439
440    fn validate(&self, uri: Url, version: Option<i64>, text: String) -> anyhow::Result<()> {
441        let lsp_url = uri.clone().try_into()?;
442        let eval_result = self.context.parse_file_with_contents(&lsp_url, text);
443        if let Some(ast) = eval_result.ast {
444            let module = Arc::new(LspModule::new(ast));
445            let mut last_valid_parse = self.last_valid_parse.write().unwrap();
446            last_valid_parse.insert(lsp_url, module);
447        }
448        self.publish_diagnostics(uri, eval_result.diagnostics, version);
449        Ok(())
450    }

452    fn did_open(&self, params: DidOpenTextDocumentParams) -> anyhow::Result<()> {
453        self.validate(
454            params.text_document.uri,
455            Some(params.text_document.version as i64),
456            params.text_document.text,
457        )
458    }

460    fn did_change(&self, params: DidChangeTextDocumentParams) -> anyhow::Result<()> {
461        // We asked for Sync full, so can just grab all the text from params
462        let change = params.content_changes.into_iter().next().unwrap();
463        self.validate(
464            params.text_document.uri,
465            Some(params.text_document.version as i64),
466            change.text,
467        )
468    }
469
470    fn did_close(&self, params: DidCloseTextDocumentParams) -> anyhow::Result<()> {
471        {
472            let mut last_valid_parse = self.last_valid_parse.write().unwrap();
473            last_valid_parse.remove(&params.text_document.uri.clone().try_into()?);
474        }
475        self.publish_diagnostics(params.text_document.uri, Vec::new(), None);
476        Ok(())
477    }

555    /// Find the ultimate places that an identifier is defined.
556    ///
557    /// Takes a definition location and if necessary loads other files trying
558    /// to find where the symbol was defined in a useful way. e.g. pointing to the
559    /// symbol in a "load()" statement isn't useful, but going to the file it is
560    /// loaded from and pointing at a function definition very much is.
561    fn resolve_definition_location(
562        &self,
563        definition: IdentifierDefinition,
564        source: ResolvedSpan,
565        member: Option<&str>,
566        uri: &LspUrl,
567        workspace_root: Option<&Path>,
568    ) -> anyhow::Result<Option<LocationLink>> {
569        let ret = match definition {
570            IdentifierDefinition::Location {
571                destination: target,
572                ..
573            } => Self::location_link(source, uri, target)?,
574            IdentifierDefinition::LoadedLocation {
575                destination: location,
576                path,
577                name,
578                ..
579            } => {
580                let load_uri = self.resolve_load_path(&path, uri, workspace_root)?;
581                let loaded_location =
582                    self.get_ast_or_load_from_disk(&load_uri)?
583                        .and_then(|ast| match member {
584                            Some(member) => ast.find_exported_symbol_and_member(&name, member),
585                            None => ast.find_exported_symbol_span(&name),
586                        });
587                match loaded_location {
588                    None => Self::location_link(source, uri, location)?,
589                    Some(loaded_location) => {
590                        Self::location_link(source, &load_uri, loaded_location)?
591                    }
592                }
593            }
594            IdentifierDefinition::NotFound => None,
595            IdentifierDefinition::LoadPath { path, .. } => {
596                match self.resolve_load_path(&path, uri, workspace_root) {
597                    Ok(load_uri) => Self::location_link(source, &load_uri, Range::default())?,
598                    Err(_) => None,
599                }
600            }
601            IdentifierDefinition::StringLiteral { literal, .. } => {
602                let Ok(resolved_literal) =
603                    self.context
604                        .resolve_string_literal(&literal, uri, workspace_root)
605                else {
606                    return Ok(None);
607                };
608                match resolved_literal {
609                    Some(StringLiteralResult {
610                        url,
611                        location_finder: Some(location_finder),
612                    }) => {
613                        // If there's an error loading the file to parse it, at least
614                        // try to get to the file.
615                        let result =
616                            self.get_ast_or_load_from_disk(&url)
617                                .and_then(|ast| match ast {
618                                    Some(module) => location_finder(&module.ast).map(|span| {
619                                        span.map(|span| module.ast.codemap().resolve_span(span))
620                                    }),
621                                    None => Ok(None),
622                                });
623                        let result = match result {
624                            Ok(result) => result,
625                            Err(e) => {
626                                eprintln!("Error jumping to definition: {e:#}");
627                                None
628                            }
629                        };
630                        let target_range = result.unwrap_or_default();
631                        Self::location_link(source, &url, target_range)?
632                    }
633                    Some(StringLiteralResult {
634                        url,
635                        location_finder: None,
636                    }) => Self::location_link(source, &url, Range::default())?,
637                    _ => None,
638                }
639            }
640            IdentifierDefinition::Unresolved { name, .. } => {
641                match self.context.get_url_for_global_symbol(uri, &name)? {
642                    Some(uri) => {
643                        let loaded_location =
644                            self.get_ast_or_load_from_disk(&uri)?
645                                .and_then(|ast| match member {
646                                    Some(member) => {
647                                        ast.find_exported_symbol_and_member(&name, member)
648                                    }
649                                    None => ast.find_exported_symbol_span(&name),
650                                });
651
652                        Self::location_link(source, &uri, loaded_location.unwrap_or_default())?
653                    }
654                    None => None,
655                }
656            }
657        };
658        Ok(ret)
659    }

661    fn find_definition(
662        &self,
663        params: GotoDefinitionParams,
664        initialize_params: &InitializeParams,
665    ) -> anyhow::Result<GotoDefinitionResponse> {
666        let uri = params
667            .text_document_position_params
668            .text_document
669            .uri
670            .try_into()?;
671        let line = params.text_document_position_params.position.line;
672        let character = params.text_document_position_params.position.character;
673        let workspace_root =
674            Self::get_workspace_root(initialize_params.workspace_folders.as_ref(), &uri);
675
676        let location = match self.get_ast(&uri) {
677            Some(ast) => {
678                let location = ast.find_definition_at_location(line, character);
679                let source = location.source().unwrap_or_default();
680                match location {
681                    Definition::Identifier(definition) => self.resolve_definition_location(
682                        definition,
683                        source,
684                        None,
685                        &uri,
686                        workspace_root.as_deref(),
687                    )?,
688                    // In this case we don't pass the name along in the root_definition_location,
689                    // so it's simpler to do the lookup here, rather than threading a ton of
690                    // information through.
691                    Definition::Dotted(DottedDefinition {
692                        root_definition_location: IdentifierDefinition::Location { destination, .. },
693                        segments,
694                        ..
695                    }) => {
696                        let member_location = ast
697                            .find_exported_symbol_and_member(
698                                segments.first().expect("at least one segment").as_str(),
699                                segments.get(1).expect("at least two segments").as_str(),
700                            )
701                            .unwrap_or(destination);
702                        Self::location_link(source, &uri, member_location)?
703                    }
704                    Definition::Dotted(definition) => self.resolve_definition_location(
705                        definition.root_definition_location,
706                        source,
707                        Some(
708                            definition
709                                .segments
710                                .last()
711                                .expect("to have at least one component")
712                                .as_str(),
713                        ),
714                        &uri,
715                        workspace_root.as_deref(),
716                    )?,
717                }
718            }
719            None => None,
720        };
721
722        let response = match location {
723            Some(location) => vec![location],
724            None => vec![],
725        };
726        Ok(GotoDefinitionResponse::Link(response))
727    }

817    /// Using all currently loaded documents, gather a list of known exported
818    /// symbols. This list contains both the symbols exported from the loaded
819    /// files, as well as symbols loaded in the open files. Symbols that are
820    /// loaded from modules that are open are deduplicated.
821    pub(crate) fn get_all_exported_symbols<F, S>(
822        &self,
823        except_from: Option<&LspUrl>,
824        symbols: &HashMap<String, S>,
825        workspace_root: Option<&Path>,
826        current_document: &LspUrl,
827        format_text_edit: F,
828    ) -> Vec<CompletionItem>
829    where
830        F: Fn(&str, &str) -> TextEdit,
831    {
832        let mut seen = HashSet::new();
833        let mut result = Vec::new();
834
835        let all_documents = self.last_valid_parse.read().unwrap();
836
837        for (doc_uri, doc) in all_documents
838            .iter()
839            .filter(|&(doc_uri, _)| match except_from {
840                Some(uri) => doc_uri != uri,
841                None => true,
842            })
843        {
844            let Ok(load_path) =
845                self.context
846                    .render_as_load(doc_uri, current_document, workspace_root)
847            else {
848                continue;
849            };
850
851            for symbol in doc
852                .get_exported_symbols()
853                .into_iter()
854                .filter(|symbol| !symbols.contains_key(&symbol.name))
855            {
856                seen.insert(format!("{load_path}:{}", &symbol.name));
857
858                let text_edits = Some(vec![format_text_edit(&load_path, &symbol.name)]);
859                let mut completion_item: CompletionItem = symbol.into();
860                completion_item.detail = Some(format!("Load from {load_path}"));
861                completion_item.additional_text_edits = text_edits;
862
863                result.push(completion_item)
864            }
865        }
866
867        for (doc_uri, symbol) in all_documents
868            .iter()
869            .filter(|&(doc_uri, _)| match except_from {
870                Some(uri) => doc_uri != uri,
871                None => true,
872            })
873            .flat_map(|(doc_uri, doc)| {
874                doc.get_loaded_symbols()
875                    .into_iter()
876                    .map(move |symbol| (doc_uri, symbol))
877            })
878            .filter(|(_, symbol)| !symbols.contains_key(symbol.name))
879        {
880            let Ok(url) = self
881                .context
882                .resolve_load(symbol.loaded_from, doc_uri, workspace_root)
883            else {
884                continue;
885            };
886            let Ok(load_path) = self
887                .context
888                .render_as_load(&url, current_document, workspace_root)
889            else {
890                continue;
891            };
892
893            if seen.insert(format!("{}:{}", &load_path, symbol.name)) {
894                result.push(CompletionItem {
895                    label: symbol.name.to_owned(),
896                    detail: Some(format!("Load from {}", &load_path)),
897                    kind: Some(CompletionItemKind::CONSTANT),
898                    additional_text_edits: Some(vec![format_text_edit(&load_path, symbol.name)]),
899                    ..Default::default()
900                })
901            }
902        }
903
904        result
905    }

930    pub(crate) fn get_load_text_edit<P>(
931        module: &str,
932        symbol: &str,
933        ast: &LspModule,
934        last_load: Option<ResolvedSpan>,
935        existing_load: Option<&(Vec<LoadArgP<P>>, Span)>,
936    ) -> TextEdit
937    where
938        P: AstPayload,
939    {
940        match existing_load {
941            Some((previously_loaded_symbols, load_span)) => {
942                // We're already loading a symbol from this module path, construct
943                // a text edit that amends the existing load.
944                let load_span = ast.ast.codemap().resolve_span(*load_span);
945                let mut load_args: Vec<(&str, &str)> = previously_loaded_symbols
946                    .iter()
947                    .map(|LoadArgP { local, their, .. }| {
948                        (local.ident.as_str(), their.node.as_str())
949                    })
950                    .collect();
951                load_args.push((symbol, symbol));
952                load_args.sort_by(|(_, a), (_, b)| a.cmp(b));
953
954                TextEdit::new(
955                    load_span.into(),
956                    format!(
957                        "load(\"{module}\", {})",
958                        load_args
959                            .into_iter()
960                            .map(|(assign, import)| {
961                                if assign == import {
962                                    format!("\"{import}\"")
963                                } else {
964                                    format!("{assign} = \"{import}\"")
965                                }
966                            })
967                            .join(", ")
968                    ),
969                )
970            }
971            None => {
972                // We're not yet loading from this module, construct a text edit that
973                // inserts a new load statement after the last one we found.
974                TextEdit::new(
975                    match last_load {
976                        Some(span) => Range::new(
977                            Position::new(span.end.line as u32, span.end.column as u32),
978                            Position::new(span.end.line as u32, span.end.column as u32),
979                        ),
980                        None => Range::new(Position::new(0, 0), Position::new(0, 0)),
981                    },
982                    format!(
983                        "{}load(\"{module}\", \"{symbol}\"){}",
984                        if last_load.is_some() { "\n" } else { "" },
985                        if last_load.is_some() { "" } else { "\n\n" },
986                    ),
987                )
988            }
989        }
990    }

1060    fn get_hover_for_identifier_definition(
1061        &self,
1062        identifier_definition: IdentifierDefinition,
1063        document: &LspModule,
1064        document_uri: &LspUrl,
1065        workspace_root: Option<&Path>,
1066    ) -> anyhow::Result<Option<Hover>> {
1067        Ok(match identifier_definition {
1068            IdentifierDefinition::Location {
1069                destination,
1070                name,
1071                source,
1072            } => {
1073                // TODO: This seems very inefficient. Once the document starts
1074                // holding the `Scope` including AST nodes, this indirection
1075                // should be removed.
1076                find_symbols_at_location(
1077                    document.ast.codemap(),
1078                    document.ast.statement(),
1079                    ResolvedPos {
1080                        line: destination.begin.line,
1081                        column: destination.begin.column,
1082                    },
1083                )
1084                .remove(&name)
1085                .and_then(|symbol| {
1086                    symbol
1087                        .doc
1088                        .map(|docs| Hover {
1089                            contents: HoverContents::Array(vec![MarkedString::String(
1090                                render_doc_item_no_link(&symbol.name, &docs),
1091                            )]),
1092                            range: Some(source.into()),
1093                        })
1094                        .or_else(|| {
1095                            symbol.param.map(|(starred_name, doc)| Hover {
1096                                contents: HoverContents::Array(vec![MarkedString::String(
1097                                    render_doc_param(starred_name, &doc),
1098                                )]),
1099                                range: Some(source.into()),
1100                            })
1101                        })
1102                })
1103            }
1104            IdentifierDefinition::LoadedLocation {
1105                path, name, source, ..
1106            } => {
1107                // Symbol loaded from another file. Find the file and get the definition
1108                // from there, hopefully including the docs.
1109                let load_uri = self.resolve_load_path(&path, document_uri, workspace_root)?;
1110                self.get_ast_or_load_from_disk(&load_uri)?.and_then(|ast| {
1111                    ast.find_exported_symbol(&name).and_then(|symbol| {
1112                        symbol.docs.map(|docs| Hover {
1113                            contents: HoverContents::Array(vec![MarkedString::String(
1114                                render_doc_item_no_link(&symbol.name, &docs),
1115                            )]),
1116                            range: Some(source.into()),
1117                        })
1118                    })
1119                })
1120            }
1121            IdentifierDefinition::StringLiteral { source, literal } => {
1122                let Ok(resolved_literal) =
1123                    self.context
1124                        .resolve_string_literal(&literal, document_uri, workspace_root)
1125                else {
1126                    // We might just be hovering a string that's not a file/target/etc,
1127                    // so just return nothing.
1128                    return Ok(None);
1129                };
1130                match resolved_literal {
1131                    Some(StringLiteralResult {
1132                        url,
1133                        location_finder: Some(location_finder),
1134                    }) => {
1135                        // If there's an error loading the file to parse it, at least
1136                        // try to get to the file.
1137                        let module = match self.get_ast_or_load_from_disk(&url) {
1138                            Ok(Some(ast)) => ast,
1139                            _ => {
1140                                return Ok(None);
1141                            }
1142                        };
1143                        let result = location_finder(&module.ast)?;
1144
1145                        result.map(|location| Hover {
1146                            contents: HoverContents::Array(vec![MarkedString::LanguageString(
1147                                LanguageString {
1148                                    language: "python".to_owned(),
1149                                    value: module.ast.codemap().source_span(location).to_owned(),
1150                                },
1151                            )]),
1152                            range: Some(source.into()),
1153                        })
1154                    }
1155                    _ => None,
1156                }
1157            }
1158            IdentifierDefinition::Unresolved { source, name } => {
1159                // Try to resolve as a global symbol.
1160                self.context
1161                    .get_environment(document_uri)
1162                    .members
1163                    .into_iter()
1164                    .find(|symbol| symbol.0 == name)
1165                    .map(|symbol| Hover {
1166                        contents: HoverContents::Array(vec![MarkedString::String(
1167                            render_doc_item_no_link(&symbol.0, &symbol.1),
1168                        )]),
1169                        range: Some(source.into()),
1170                    })
1171            }
1172            IdentifierDefinition::LoadPath { .. } | IdentifierDefinition::NotFound => None,
1173        })
1174    }

1218    fn main_loop(&self, initialize_params: InitializeParams) -> anyhow::Result<()> {
1219        self.log_message(MessageType::INFO, "Starlark server initialised");
1220        for msg in &self.connection.receiver {
1221            match msg {
1222                Message::Request(req) => {
1223                    // TODO(nmj): Also implement DocumentSymbols so that some logic can
1224                    //            be handled client side.
1225                    if let Some(params) = as_request::<GotoDefinition>(&req) {

1227                    } else if let Some(params) = as_request::<StarlarkFileContentsRequest>(&req) {
1228                        self.get_starlark_file_contents(req.id, params);
1229                    } else if let Some(params) = as_request::<Completion>(&req) {
1230                        self.completion(req.id, params, &initialize_params);
1231                    } else if let Some(params) = as_request::<HoverRequest>(&req) {
1232                        self.hover(req.id, params, &initialize_params);
1233                    } else if self.connection.handle_shutdown(&req)? {
1234                        return Ok(());
1235                    }
1236                    // Currently don't handle any other requests
1237                }
1238                Message::Notification(x) => {
1239                    if let Some(params) = as_notification::<DidOpenTextDocument>(&x) {
1240                        self.did_open(params)?;
1241                    } else if let Some(params) = as_notification::<DidChangeTextDocument>(&x) {
1242                        self.did_change(params)?;
1243                    } else if let Some(params) = as_notification::<DidCloseTextDocument>(&x) {
1244                        self.did_close(params)?;
1245                    }
1246                }
1247                Message::Response(_) => {
1248                    // Don't expect any of these
1249                }
1250            }
1251        }
1252        Ok(())
1253    }

1256/// Instantiate an LSP server that reads on stdin, and writes to stdout
1257pub fn stdio_server<T: LspContext>(context: T) -> anyhow::Result<()> {
1258    // Note that  we must have our logging only write out to stderr.
1259    eprintln!("Starting Rust Starlark server");
1260
1261    let (connection, io_threads) = Connection::stdio();
1262    server_with_connection(connection, context)?;
1263    // Make sure that the io threads stop properly too.
1264    io_threads.join()?;
1265
1266    eprintln!("Stopping Rust Starlark server");
1267    Ok(())
1268}

1270/// Instantiate an LSP server that reads and writes using the given connection.
1271pub fn server_with_connection<T: LspContext>(
1272    connection: Connection,
1273    context: T,
1274) -> anyhow::Result<()> {
1275    // Run the server and wait for the main thread to end (typically by trigger LSP Exit event).
1276    let (init_request_id, init_value) = connection.initialize_start()?;
1277
1278    let initialization_params: InitializeParams = serde_json::from_value(init_value)?;
1279    let server_settings = initialization_params
1280        .initialization_options
1281        .as_ref()
1282        .and_then(|opts| serde_json::from_value(opts.clone()).ok())
1283        .unwrap_or_default();
1284    let capabilities_payload = Backend::<T>::server_capabilities(server_settings);
1285    let server_capabilities = serde_json::to_value(capabilities_payload).unwrap();
1286
1287    let initialize_data = serde_json::json!({
1288            "capabilities": server_capabilities,
1289    });
1290    connection.initialize_finish(init_request_id, initialize_data)?;
1291
1292    Backend {
1293        connection,
1294        context,
1295        last_valid_parse: RwLock::default(),
1296    }
1297    .main_loop(initialization_params)?;
1298
1299    Ok(())
1300}

1302fn as_notification<T>(x: &Notification) -> Option<T::Params>
1303where
1304    T: lsp_types::notification::Notification,

1309            panic!(
1310                "Invalid notification\nMethod: {}\n error: {}",

1312            )
1313        });
1314        Some(params)
1315    } else {

1320fn as_request<T>(x: &Request) -> Option<T::Params>
1321where
1322    T: lsp_types::request::Request,
1323    T::Params: DeserializeOwned,
1324{
1325    if x.method == T::METHOD {
1326        let params = serde_json::from_value(x.params.clone()).unwrap_or_else(|err| {
1327            panic!(
1328                "Invalid request\n  method: {}\n  error: {}\n  request: {:?}\n",
1329                x.method, err, x
1330            )
1331        });
1332        Some(params)
1333    } else {
1334        None
1335    }
1336}

1338/// Create a new `Notification` object with the correct name from the given params.

1340where
1341    T: lsp_types::notification::Notification,
1342{
1343    Notification {
1344        method: T::METHOD.to_owned(),
1345        params: serde_json::to_value(&params).unwrap(),
1346    }

1349fn new_response<T>(id: RequestId, params: anyhow::Result<T>) -> Response
1350where
1351    T: serde::Serialize,
1352{
1353    match params {
1354        Ok(params) => Response {
1355            id,
1356            result: Some(serde_json::to_value(params).unwrap()),
1357            error: None,
1358        },
1359        Err(e) => Response {
1360            id,
1361            result: None,
1362            error: Some(ResponseError {
1363                code: 0,
1364                message: format!("{e:#?}"),
1365                data: None,
1366            }),
1367        },
1368    }

2306                0

2310                    .nth((p.line - 1).try_into().unwrap())
2311                    .unwrap()
2312                    .0
2313                    + 1
2314            };
2315            l + s[l..]
2316                .char_indices()
2317                .nth((p.character).try_into().unwrap())
2318                .unwrap()
2319                .0
2320        };
