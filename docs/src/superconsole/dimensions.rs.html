11use std::cmp;
12
13use crate::Lines;
14
15/// Denotes a rectangular area.
16///
17/// Passed to [`Component`](crate::Component)s to give valid drawing area.
18#[derive(Debug, Copy, Clone, Eq, PartialEq, Default)]
19pub struct Dimensions {
20    pub width: usize,
21    pub height: usize,
22}
23
24/// Denotes a particular axis (x or y).  Used for several Components which are generalized over direction.
25#[derive(Debug, Copy, Clone, Eq, PartialEq)]
26pub enum Direction {
27    Horizontal,
28    Vertical,
29}
30
31impl From<(u16, u16)> for Dimensions {
32    fn from((x, y): (u16, u16)) -> Self {
33        Self {
34            width: x.into(),
35            height: y.into(),
36        }
37    }
38}
39
40impl From<(usize, usize)> for Dimensions {
41    fn from((width, height): (usize, usize)) -> Self {
42        Self { width, height }
43    }
44}
45
46impl Dimensions {
47    pub fn new(width: usize, height: usize) -> Self {
48        Self { width, height }
49    }
50
51    pub fn dimension(self, direction: Direction) -> usize {
52        match direction {
53            Direction::Horizontal => self.width,
54            Direction::Vertical => self.height,
55        }
56    }
57
58    /// Best effort conversion - it will truncate to the nearest valid dimension.
59    pub fn multiply(self, multiplicand: f64, direction: Direction) -> Self {
60        fn mul(lhs: usize, rhs: f64) -> usize {
61            let lhs = lhs as f64; // this might truncate the size if too large
62            let result = lhs * rhs;
63            result as usize // this might not be an exact dimension.
64        }
65
66        let Self { width, height } = self;
67        match direction {
68            Direction::Horizontal => Self {
69                width: mul(width, multiplicand),
70                height,
71            },
72            Direction::Vertical => Self {
73                width,
74                height: mul(height, multiplicand),
75            },
76        }
77    }
78
79    pub fn saturating_sub(self, subtractor: usize, direction: Direction) -> Self {
80        let Self { width, height } = self;
81        match direction {
82            Direction::Horizontal => Self {
83                width: width.saturating_sub(subtractor),
84                height,
85            },
86            Direction::Vertical => Self {
87                width,
88                height: height.saturating_sub(subtractor),
89            },
90        }
91    }
92
93    /// Finds the size of a [`Component`](crate::Component)'s output in a given dimension.
94    /// Truncates the size to at most `u16::MAX`.
95    // ptr args allowed because line trait only implemented on `Vec<line>`
96    pub fn dimension_from_output_truncated(output: &Lines, direction: Direction) -> usize {
97        match direction {
98            Direction::Horizontal => output.max_line_length(),
99            Direction::Vertical => output.len(),
100        }
101    }
102
103    /// Returns the smallest bounding box that fits in both dimensions.
104    pub fn intersect(self, Self { width, height }: Self) -> Self {
105        Self {
106            width: cmp::min(self.width, width),
107            height: cmp::min(self.height, height),
108        }
109    }
110
111    /// Returns the smallest bounding box in which both Dimensions fit.
112    pub fn union(self, Self { width, height }: Self) -> Self {
113        Self {
114            width: cmp::max(self.width, width),
115            height: cmp::max(self.height, height),
116        }
117    }
118
119    /// Returns true iff the passed dimension is smaller than or equal to in both dimensions.
120    pub fn contains(self, Self { width, height }: Self) -> bool {
121        width <= self.width && height <= self.height
122    }
123}

125#[cfg(test)]
126mod tests {
127    use super::*;
128
129    #[test]
130    fn test_intersect() {
131        let lhs = Dimensions {
132            width: 5,
133            height: 10,
134        };
135        let rhs = Dimensions {
136            width: 8,
137            height: 3,
138        };
139        let intersect = lhs.intersect(rhs);
140        assert_eq!(
141            intersect,
142            Dimensions {
143                width: 5,
144                height: 3
145            }
146        );
147    }
148
149    #[test]
150    fn test_union() {
151        let lhs = Dimensions {
152            width: 5,
153            height: 10,
154        };
155        let rhs = Dimensions {
156            width: 8,
157            height: 3,
158        };
159        let union = lhs.union(rhs);
160        assert_eq!(
161            union,
162            Dimensions {
163                width: 8,
164                height: 10
165            }
166        );
167    }
168
169    #[test]
170    fn test_contains() {
171        let lhs = Dimensions {
172            width: 8,
173            height: 10,
174        };
175        let rhs = Dimensions {
176            width: 5,
177            height: 3,
178        };
179        assert!(lhs.contains(rhs));
180        assert!(!rhs.contains(lhs));
181        assert!(lhs.contains(lhs));
182    }
183}
