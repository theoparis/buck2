22/// Controls the way the splitter displays its children.
23#[derive(Clone, Debug)]
24pub enum SplitKind {
25    // Each child component takes up area proportional to its size relative to the other sizes.
26    Sized(Vec<f64>),
27    // Each child component takes up an equal % of the area long the given dimension.
28    Equal,
29    // Each child may take as much space as it would like along the given dimension.
30    // No extra padding is given between children.
31    Adaptive,
32}
33
34/// Internally, we want to alias away the Equal case
35#[derive(Clone, Debug)]
36enum InternalSplitKind {
37    SizedNormalized(Vec<f64>),
38    Adaptive,
39}
40
41impl SplitKind {
42    fn to_internal_split_kind(&self, children_len: usize) -> InternalSplitKind {
43        match self {
44            SplitKind::Sized(sizes) => {
45                assert_eq!(
46                    sizes.len(),
47                    children_len,
48                    "There must be an equal number of ratios and children."
49                );
50
51                let total: f64 = sizes.iter().sum();
52                let normalized_sizes = sizes.iter().map(|size| size / total).collect();
53
54                InternalSplitKind::SizedNormalized(normalized_sizes)
55            }
56            SplitKind::Equal => {
57                InternalSplitKind::SizedNormalized(vec![1.0 / children_len as f64; children_len])
58            }
59            SplitKind::Adaptive => InternalSplitKind::Adaptive,
60        }
61    }
62}
63
64impl InternalSplitKind {
65    pub fn draw<'a, C: Component + 'a>(
66        &self,
67        children: impl IntoIterator<Item = &'a C>,
68        direction: Direction,
69
70        dimensions: Dimensions,
71        mode: DrawMode,
72    ) -> anyhow::Result<Vec<Lines>> {
73        match self {
74            InternalSplitKind::SizedNormalized(sizes) => children
75                .into_iter()
76                .zip(sizes.iter())
77                .map(|(child, size)| -> anyhow::Result<_> {
78                    // allocate alloted size
79                    let child_dimension = dimensions.multiply(*size, direction);
80                    let mut output = child.draw(child_dimension, mode)?;
81
82                    // bound non-splitting direction, pad splitting direction
83                    match direction {
84                        Direction::Horizontal => {
85                            output.truncate_lines_bottom(child_dimension.height);
86                            output.set_lines_to_exact_width(child_dimension.width);
87                        }
88                        Direction::Vertical => {
89                            output.truncate_lines(child_dimension.width);
90                            output.set_lines_to_exact_length(child_dimension.height);
91                        }
92                    }
93
94                    Ok(output)
95                })
96                .collect(),
97            InternalSplitKind::Adaptive => {
98                let mut available = dimensions;
99                children
100                    .into_iter()
101                    .map(|child| {
102                        let mut output = child.draw(available, mode)?;
103                        output.shrink_lines_to_dimensions(dimensions);
104
105                        // decrease size by however much was just used
106                        let used = Dimensions::dimension_from_output_truncated(&output, direction);
107                        available = available.saturating_sub(used, direction);
108
109                        Ok(output)
110                    })
111                    .collect()
112            }
113        }
114    }
115}

117/// [`Splits`](SplitKind) along a given [`direction`](crate::Direction) for its child [`components`](Component).
118/// Child components are truncated to the bounds passed to them.
119pub struct Split<C = Box<dyn Component>> {
120    children: Vec<C>,
121    direction: Direction,
122    split: InternalSplitKind,
123}
124
125impl<C> Debug for Split<C> {
126    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
127        f.debug_struct("Split")
128            .field("children", &self.children.len())
129            .field("direction", &self.direction)
130            .field("split", &self.split)
131            .finish()
132    }
133}
134
135impl<C: Component> Split<C> {
136    /// Preconditions:
137    /// * At least one child.
138    /// * If Sized, then ratios must sum to approximately 1.
139    /// * If Sized, then there must be as many ratios as components
140    pub fn new(children: Vec<C>, direction: Direction, split: SplitKind) -> Self {
141        let split = split.to_internal_split_kind(children.len());
142
143        Self {
144            children,
145            direction,
146            split,
147        }
148    }
149}
150
151impl<C: Component> Component for Split<C> {
152    fn draw_unchecked(&self, dimensions: Dimensions, mode: DrawMode) -> anyhow::Result<Lines> {
153        let outputs = self
154            .split
155            .draw(&self.children, self.direction, dimensions, mode)?;
156
157        Ok(match self.direction {
158            Direction::Horizontal => Lines::join_horizontally(outputs),
159            Direction::Vertical => outputs.into_iter().flatten().collect(),
160        })
161    }
162}

194        fn test_adaptive() {
195            let kind = SplitKind::Adaptive;
196            let dimension = Direction::Horizontal;
197
198            // left rows all the same length
199            let mut left_msg = Lines(vec![
200                vec!["test"].try_into().unwrap(),
201                vec!["ok", "so"].try_into().unwrap(),
202                vec!["sync"].try_into().unwrap(),
203            ]);
204            let right_msg = Lines(vec![
205                vec!["xx"].try_into().unwrap(),
206                vec!["yyy"].try_into().unwrap(),
207                vec!["z"].try_into().unwrap(),
208            ]);
209            let splitter = Split::new(
210                vec![Echo(left_msg.clone()), Echo(right_msg.clone())],
211                dimension,
212                kind.clone(),
213            );
214            let horizontal = Lines(vec![
215                vec!["test", "xx", " "].try_into().unwrap(),
216                vec!["ok", "so", "yyy"].try_into().unwrap(),
217                vec!["sync", "z", "  "].try_into().unwrap(),
218            ]);
219
220            let output = splitter
221                .draw(Dimensions::new(10, 10), DrawMode::Normal)
222                .unwrap();
