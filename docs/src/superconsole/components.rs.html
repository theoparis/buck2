38/// Used to mark whether a draw is final.
39#[derive(Debug, Copy, Clone, Eq, PartialEq)]
40pub enum DrawMode {
41    Normal,
42    Final,
43}
44
45/// Components are pluggable drawers that output lines of formatted text.
46/// They are composable (eventually) and re-render in place at each render.
47pub trait Component {
48    /// This method is to be implemented for components to provide the `draw` method.
49    fn draw_unchecked(&self, dimensions: Dimensions, mode: DrawMode) -> anyhow::Result<Lines>;
50
51    /// Interprets the current caller state to create its drawing.
52    /// Dimensions refers to the maximum (width, height) this component may use.
53    /// The mode refers to if this is the final time the component will be drawn.
54    /// If a child component is too large to fit in the dimensions, it is truncated.
55    fn draw(&self, dimensions: Dimensions, mode: DrawMode) -> anyhow::Result<Lines> {
56        let mut res = self.draw_unchecked(dimensions, mode)?;
57        res.shrink_lines_to_dimensions(dimensions);
58        Ok(res)
59    }
60}
61
62impl Component for Box<dyn Component> {
63    fn draw_unchecked(&self, dimensions: Dimensions, mode: DrawMode) -> anyhow::Result<Lines> {
64        (**self).draw_unchecked(dimensions, mode)
65    }
66}
67
68impl Component for Box<dyn Component + Send> {
69    fn draw_unchecked(&self, dimensions: Dimensions, mode: DrawMode) -> anyhow::Result<Lines> {
70        (**self).draw_unchecked(dimensions, mode)
71    }
72}
