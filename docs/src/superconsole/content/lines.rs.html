42/// State container that's used to parse strings with ANSI color codes in them.
43#[derive(Default)]
44struct ColoredStringParser {
45    foreground_color: Option<Color>,
46    background_color: Option<Color>,
47    underline_color: Option<Color>,
48    attributes: Attributes,
49    line_buffer: String,
50    hyperlink: Option<Hyperlink>,
51    spans: Vec<Span>,
52}
53
54impl ColoredStringParser {
55    fn push_current(&mut self) {
56        let sc = StyledContent::new(
57            ContentStyle {
58                foreground_color: self.foreground_color,
59                background_color: self.background_color,
60                underline_color: self.underline_color,
61                attributes: self.attributes,
62            },
63            std::mem::take(&mut self.line_buffer),
64        );
65        let mut span = Span::new_styled_lossy(sc);
66        if self.hyperlink.is_some() {
67            span = span.with_hyperlink(self.hyperlink.clone());
68        }
69        self.spans.push(span);
70    }
71
72    fn spec_to_color(spec: ColorSpec) -> Option<Color> {
73        match spec {
74            ColorSpec::Default => None,
75            ColorSpec::PaletteIndex(idx) => Some(Color::AnsiValue(idx)),
76            ColorSpec::TrueColor(srgba) => Some(RgbColor::from(srgba).to_tuple_rgb8().into()),
77        }
78    }
79
80    /// Given a line w/ some ANSI encoded color characters, turn it into a list of spans.
81    fn parse_line(&mut self, parser: &mut termwiz::escape::parser::Parser, s: &str) -> Line {
82        // Because we only stick "printable" characters onto the buffer, and skip any other
83        // class of characters that we don't recognize, this should be a
84        // roughly sanitized string. We do one more pass when we create the Span, just to
85        // make sure, but that is a pretty fast check.
86        parser.parse(s.as_bytes(), |a| match a {
87            Action::Print(c) => {
88                self.line_buffer.push(c);
89            }
90            Action::CSI(CSI::Sgr(Sgr::Reset)) => {
91                self.push_current();
92                self.foreground_color = None;
93                self.background_color = None;
94                self.attributes = Attributes::default();
95            }
96            Action::CSI(CSI::Sgr(Sgr::Intensity(intensity))) => {
97                self.push_current();
98                self.attributes = match intensity {
99                    Intensity::Normal => Attributes::default(),
100                    Intensity::Bold => Attributes::from(Attribute::Bold),
101                    Intensity::Half => Attributes::from(Attribute::Dim),
102                };
103            }
104            Action::CSI(CSI::Sgr(Sgr::Foreground(spec))) => {
105                self.push_current();
106                self.foreground_color = Self::spec_to_color(spec);
107            }
108
109            Action::CSI(CSI::Sgr(Sgr::Background(spec))) => {
110                self.push_current();
111                self.background_color = Self::spec_to_color(spec);
112            }
113            Action::OperatingSystemCommand(cmd) => match *cmd {
114                OperatingSystemCommand::SetHyperlink(hy) => {
115                    self.push_current();
116                    self.hyperlink = hy;
117                }
118                _ => {}
119            },
120            _ => {}
121        });
122        self.push_current();
123        Line::from_iter(mem::take(&mut self.spans))
124    }
125}

134    /// Creates an instance of [`Lines`] with a style applied from a single multiline string.
135    pub fn from_multiline_string(multiline_string: &str, style: ContentStyle) -> Lines {
136        multiline_string
137            .lines()
138            .map(|line| {
139                let styled = StyledContent::new(style, line.to_owned());
140                Line::from_iter([Span::new_styled_lossy(styled)])
141            })
142            .collect()
143    }

146    /// It's specially designed for emit/emit_aux
147    pub fn from_multiline_string_raw(multiline_string: &str, style: ContentStyle) -> Lines {
148        multiline_string
149            .lines()
150            .map(|line| {
151                let styled = StyledContent::new(style, line.to_owned());
152                Line::from_iter([Span::new_styled_raw(styled)])
153            })
154            .collect()
155    }

157    /// Takes a multiline string that might contain ANSI color codes, and returns a set of lines
158    /// that include spans representing those color codes.
159    ///
160    /// Note that any other types of control characters are omitted, and certain whitespace
161    /// characters are also disallowed / replaced.
162    pub fn from_colored_multiline_string(multiline_string: &str) -> Lines {
163        let mut parser = termwiz::escape::parser::Parser::new();
164        let mut color_parser = ColoredStringParser::default();
165        multiline_string
166            .lines()
167            .map(|s| color_parser.parse_line(&mut parser, s))
168            .collect()
169    }

203    /// Returns the max column width of any line
204    pub fn max_line_length(&self) -> usize {
205        // for each line in the LHS
206        self.iter()
207            // add the length of each word in the line
208            .map(|line| line.len())
209            // pick the maximum line length
210            .max()
211            // or set it to 0 if there are no lines on the left side
212            .unwrap_or_default()
213    }

215    /// Given a set of lines, inserts the specified amount of padding to the longest line.
216    /// However, the entire block of lines will remain a rectangle afterwards.
217    /// This means that shorter lines will receive extra paddding such that it has the same length as the longest line
218    pub fn pad_lines_right(&mut self, amount: usize) {
219        if amount == 0 {
220            return;
221        }
222
223        let longest_len = self.max_line_length();
224        for line in self.iter_mut() {
225            let len = line.len();
226            line.pad_right(amount + (longest_len - len));
227        }
228    }
229
230    /// Prepends a fixed `amount` of padding to each row.
231    /// Unlike `pad_lines_right`, this method does not add extra padding to shorter lines.
232    /// The right end of the padding remains ragged.
233    pub fn pad_lines_left(&mut self, amount: usize) {
234        if amount == 0 {
235            return;
236        }
237
238        self.iter_mut().for_each(|line| {
239            line.pad_left(amount);
240        });
241    }
242
243    /// Adds padding to each line dependent on its length so that each line is the same length.
244    /// The line is left justified always.  This is because right or center aligning lines inherently justifies them.
245    /// Therefore, there is no explicit method for justifying text any other way.
246    pub fn justify(&mut self) {
247        let longest_len = self.max_line_length();
248        for line in self.iter_mut() {
249            let len = line.len();
250            line.pad_right(longest_len - len);
251        }
252    }

254    /// Given a set of lines, set them all to the exact width
255    pub fn set_lines_to_exact_width(&mut self, exact_width: usize) {
256        self.iter_mut()
257            .for_each(|line| line.to_exact_width(exact_width));
258    }

260    /// Extends the Lines list by the given length, adding empty lines at the bottom
261    pub fn pad_lines_bottom(&mut self, amount: usize) {
262        let extender = iter::repeat_n(Line::default(), amount);
263        self.extend(extender);
264    }
265
266    /// Same functionality as `pad_lines_bottom` but on the top.
267    pub fn pad_lines_top(&mut self, amount: usize) {
268        let extender = iter::repeat_n(Line::default(), amount);
269
270        self.0.splice(0..0, extender);
271    }

273    /// Truncates the line list to the given length, removing entries at the end.
274    pub fn truncate_lines_bottom(&mut self, desired_length: usize) {
275        self.0.truncate(desired_length);
276    }

278    /// Sets the line list to the given length, padding or truncating from the bottom.
279    pub fn set_lines_to_exact_length(&mut self, desired_length: usize) {
280        match self.len().cmp(&desired_length) {
281            Ordering::Less => {
282                self.pad_lines_bottom(desired_length - self.len());
283            }
284            Ordering::Equal => {}
285            Ordering::Greater => {
286                self.truncate_lines_bottom(desired_length);
287            }
288        }
289    }

291    /// Truncates columns and rows that do not fit within a bounding box
292    pub fn shrink_lines_to_dimensions(&mut self, dimensions: Dimensions) {
293        self.iter_mut()
294            .for_each(|line| line.truncate_line(dimensions.width));
295        self.truncate_lines_bottom(dimensions.height);
296    }

298    /// Like `render`, but with a limit.
299    /// If a limit is specified, no more than that amount will be drained.
300    /// The limit is on the number of *lines*, **NOT** the number of *bytes*.
301    /// Care should be taken with calling a limit of 0 - this will cause no lines to render and the buffer to never be drained.
302    ///
303    /// Returns the remain limit after rendering.  If the limit is None, means no limit
304    pub(crate) fn render_with_limit(
305        &mut self,
306        writer: &mut Vec<u8>,
307        limit: Option<usize>,
308    ) -> anyhow::Result<Option<usize>> {
309        let output_limit = limit.unwrap_or(self.len());
310        let amt = cmp::min(output_limit, self.len());
311        for line in self.0.drain(..amt) {
312            line.render_with_clear_and_nl(writer)?;
313        }
314        if limit.is_some() {
315            Ok(Some(output_limit - amt))
316        } else {
317            // if the original limit was None, it means no limit, so just return None meaning no limit
318            Ok(None)
319        }
320    }

322    /// Formats and renders all lines to `buffer`.
323    /// Notably, this *queues* the lines for rendering.  You must flush the buffer.
324    pub(crate) fn render_from_line(
325        &self,
326        writer: &mut Vec<u8>,
327        start: usize,
328    ) -> anyhow::Result<()> {
329        for line in self.0.iter().skip(start) {
330            line.render_with_clear_and_nl(writer)?;
331        }
332        Ok(())
333    }

335    /// Render the lines without an escape sequence to clear the line.
336    /// It will clear the lines at the end.
337    pub(crate) fn render_raw(&mut self, writer: &mut Vec<u8>) -> anyhow::Result<()> {
338        for line in self.0.iter() {
339            writeln!(writer, "{}", line.render())?;
340        }
341        self.0.clear();
342        Ok(())
343    }

345    pub(crate) fn lines_equal(&self, other: &Self) -> usize {
346        self.0
347            .iter()
348            .zip(other.0.iter())
349            .take_while(|(l1, l2)| l1 == l2)
350            .count()
351    }

353    /// Returns the maximum line width and the number of lines.
354    /// This corresponds to how much space a justified version of the output would take.
355    pub fn dimensions(&self) -> anyhow::Result<Dimensions> {
356        let x = self.max_line_length();
357        let y = self.len();

359        Ok((x, y).into())

368    /// Join blocks horizontally, i.e. side by side.
369    pub fn join_horizontally(blocks: Vec<Lines>) -> Lines {
370        if blocks.is_empty() {
371            return Lines::new();
372        }
373
374        // unwrap ok because guaranteed > 0 children from constructor
375        let longest = blocks.iter().map(|output| output.len()).max().unwrap();
376
377        // pad all other outputs to be the same length.  Then, justify them to be uniform blocks.
378        let padded = blocks.into_iter().update(|output| {
379            output.set_lines_to_exact_length(longest);
380            output.justify();
381        });
382
383        // can't do arbitrary zip, so this'll have to do
384        padded
385            .reduce(|mut all, output| {
386                for (all_line, output_line) in all.iter_mut().zip(output) {
387                    all_line.extend(output_line);
388                }
389
390                all
391            })
392            // safe to unwrap because at least one child component required
393            .unwrap()
394    }

396    pub fn fmt_for_test(&self) -> impl Display + '_ {
397        struct Impl<'a>(&'a Lines);
398        impl Display for Impl<'_> {
399            fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
400                for line in self.0.iter() {
401                    writeln!(f, "{}", line.fmt_for_test())?;
402                }
403                Ok(())
404            }
405        }
406        Impl(self)
407    }
