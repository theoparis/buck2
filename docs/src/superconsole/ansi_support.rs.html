11#[cfg(not(windows))]
12pub(crate) fn enable_ansi_support() -> anyhow::Result<()> {
13    Ok(())
14}
15
16#[cfg(windows)]
17pub(crate) fn enable_ansi_support() -> anyhow::Result<()> {
18    #[derive(Debug, thiserror::Error)]
19    #[error("Terminal does not support ANSI escapes")]
20    struct AnsiIsNotSupported;
21
22    // This functions has side effect: it enables ANSI support in terminals on Windows.
23    // https://github.com/crossterm-rs/crossterm/blob/769b18151c6c177d43c752cc964d2b3d058bcf14/src/ansi_support.rs#L39
24    // Note if this function returns false, we cannot buffer ANSI escape sequences.
25    if !crossterm::ansi_support::supports_ansi() {
26        return Err(AnsiIsNotSupported.into());
27    }
28
29    Ok(())
30}
