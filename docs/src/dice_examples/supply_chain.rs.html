11//!
12//! A supply chain simulation built on top of DICE.
13//! * Each company has a name and manufactures a list of resources.
14//! * Each resource has a recipe associated with it.
15//!
16//! When a company is asked to produce a resource:
17//! * It looks up the recipe of sub-resources.
18//! * For each sub-resource, it finds the cheapest manufacturer.
19//! * It then return a quote of the cost.
20//!
21//! The cost of a resource is the sum of each item cost + a company specific flat fee.
22

44#[derive(Display, Debug, Hash, Eq, Clone, PartialEq, Dupe, Allocative)]
45pub enum Resource {
46    Wood,
47    Plank,
48    Stick,
49    Cobblestone,
50    Pickaxe,
51    CraftingTable,
52}
53
54pub struct Recipe {
55    /// The number of items this recipe produces (currently unused)
56    pub produces: u8,
57    /// A list of the resources (and how many) are required.
58    pub ingredients: &'static [(u8, Resource)],
59}
60
61impl Recipe {
62    /// Singleton for a primitive resource.
63    const PRIMITIVE: Self = Self {
64        produces: 1,
65        ingredients: &[],
66    };
67    pub const WOOD: Self = Self::PRIMITIVE;
68    pub const COBBLESTONE: Self = Self::PRIMITIVE;
69    pub const PLANK: Self = Self {
70        produces: 4,
71        ingredients: &[(1, Resource::Wood)],
72    };
73    pub const STICK: Self = Self {
74        produces: 4,
75        ingredients: &[(2, Resource::Plank)],
76    };
77    pub const PICKAXE: Self = Self {
78        produces: 1,
79        ingredients: &[(3, Resource::Cobblestone), (2, Resource::Stick)],
80    };
81    pub const CRAFTING_TABLE: Self = Self {
82        produces: 1,
83        ingredients: &[(4, Resource::Plank)],
84    };
85}
86
87impl Resource {
88    pub const RESOURCES: [Resource; 6] = [
89        Self::Wood,
90        Self::Plank,
91        Self::Stick,
92        Self::Cobblestone,
93        Self::Pickaxe,
94        Self::CraftingTable,
95    ];
96    pub fn recipe(&self) -> Recipe {
97        match self {
98            Self::Wood => Recipe::WOOD,
99            Self::Plank => Recipe::PLANK,
100            Self::Stick => Recipe::STICK,
101            Self::Cobblestone => Recipe::COBBLESTONE,
102            Self::Pickaxe => Recipe::PICKAXE,
103            Self::CraftingTable => Recipe::CRAFTING_TABLE,
104        }
105    }
106}

108/// Each company manufactures a set of resources for a given upcharge
109#[derive(Eq, Clone, PartialEq, Allocative)]
110pub struct Company {
111    pub name: Arc<String>,
112    /// A mapping between resource the company makes and its flat upcharge
113    pub makes: HashMap<Resource, u16>,
114}
115
116#[async_trait]
117pub trait Setup {
118    /// Must be called before any companies are added.
119    /// Sets the state of the resource map.
120    fn init_state(&mut self) -> anyhow::Result<()>;
121    /// Adds a list of companies and maps them to their resources.
122    async fn add_companies(&mut self, companies: Vec<Company>) -> anyhow::Result<()>;
123}
124
125#[async_trait]
126impl Setup for DiceTransactionUpdater {
127    fn init_state(&mut self) -> anyhow::Result<()> {
128        self.changed_to(
129            Resource::RESOURCES
130                .iter()
131                .map(|resource| (LookupResource(resource.dupe()), Arc::new(vec![]))),
132        )?;
133
134        Ok(())
135    }
136
137    async fn add_companies(&mut self, companies: Vec<Company>) -> anyhow::Result<()> {
138        let mut resource_to_company_local = HashMap::new();
139
140        // convert company to insertion ready format
141        let insertion_ready_companies = companies.into_map(|company| {
142            let lookup = LookupCompany(company.name.dupe());
143
144            // construct a resource => company lookups mapping across all companies
145            for resource in company.makes.keys() {
146                resource_to_company_local
147                    .entry(resource.dupe())
148                    .or_insert_with(Vec::new)
149                    .push(lookup.clone());
150            }
151
152            (lookup, Arc::new(company))
153        });
154
155        self.changed_to(insertion_ready_companies)?;
156
157        // split iterators to talk about resources and companies separately
158        let (resources, companies): (Vec<_>, Vec<_>) = resource_to_company_local
159            .into_iter()
160            .map(|(k, v)| (LookupResource(k), v))
161            .unzip();
162
163        // get the remote resources => company mapping
164        let mut state = self.existing_state().await;
165        let remote_resources = join_all(state.compute_many(resources.iter().map(|res| {
166            DiceComputations::declare_closure(
167                |ctx: &mut DiceComputations<'_>| -> BoxFuture<DiceResult<Arc<Vec<LookupCompany>>>> {
168                    ctx.compute(res).boxed()
169                },
170            )
171        })))
172        .await;
173
174        // combine remote company list with local company list for reach resource
175        let joined: Vec<_> = resources
176            .into_iter()
177            .zip(companies)
178            .zip(remote_resources)
179            .map(|((resource, mut local_companies), remote_companies)| {
180                local_companies.append(&mut (*remote_companies?).clone());
181
182                Ok((resource, Arc::new(local_companies)))
183            })
184            .collect::<anyhow::Result<_>>()?;
185
186        self.changed_to(joined)?;
187
188        Ok(())
189    }
190}

192#[async_trait]
193pub trait Cost {

196        &mut self,
197        resource: &Resource,
198    ) -> Result<Option<u16>, Arc<anyhow::Error>>;
199}

201#[async_trait]
202pub trait CostUpdater {
203    /// Change the upcharge of a company for a resource.
204    async fn change_company_resource_cost(
205        &mut self,
206        company: &str,
207        resource: &Resource,
208        new_price: u16,
209    ) -> anyhow::Result<()>;
210}
211
212async fn lookup_company_resource_cost(
213    ctx: &mut DiceComputations<'_>,
214    company: &LookupCompany,
215    resource: &Resource,
216) -> Result<Option<u16>, Arc<anyhow::Error>> {
217    #[derive(Display, Debug, Hash, Eq, Clone, Dupe, PartialEq, Allocative)]
218    #[display("{:?}", self)]
219    struct LookupCompanyResourceCost(LookupCompany, Resource);
220    #[async_trait]
221    impl Key for LookupCompanyResourceCost {
222        type Value = Result<Option<u16>, Arc<anyhow::Error>>;
223
224        async fn compute(
225            &self,
226            ctx: &mut DiceComputations,
227            _cancellations: &CancellationContext,
228        ) -> Self::Value {
229            let company = ctx
230                .compute(&self.0)
231                .await
232                .map_err(|e| Arc::new(anyhow::anyhow!(e)))?;
233            let recipe = self.1.recipe();
234
235            let upcharge = company.makes.get(&self.1);
236            if upcharge.is_none() {
237                return Ok(None);
238            }
239
240            // get the unit cost for each resource needed to make item
241            let mut futs : FuturesUnordered<_> =
242                ctx.compute_many(recipe.ingredients.iter().map(|(required, resource)| {
243                    DiceComputations::declare_closure(|ctx: &mut DiceComputations<'_>| -> BoxFuture<Result<Option<u16>, Arc<anyhow::Error>>> {
244                            ctx.resource_cost(resource).map(|res| {
245                                Ok::<_, Arc<anyhow::Error>>(res?.map(|x| x * *required as u16))
246                            }).boxed()
247                    })
248                })).into_iter().collect();
249
250            let mut sum = 0;
251            while let Some(x) = futs.next().await {
252                if let Some(x) = x? {
253                    sum += x;
254                } else {
255                    return Ok(None);
256                }
257            }
258
259            Ok(Some(sum + upcharge.unwrap()))
260        }
261
262        fn equality(x: &Self::Value, y: &Self::Value) -> bool {
263            match (x, y) {
264                (Ok(x), Ok(y)) => x == y,
265                _ => false,
266            }
267        }
268    }
269    let key = LookupCompanyResourceCost(company.clone(), resource.dupe());
270
271    ctx.compute(&key)
272        .await
273        .map_err(|e| Arc::new(anyhow::anyhow!(e)))?
274}

276#[async_trait]
277impl Cost for DiceComputations<'_> {
278    async fn resource_cost(
279        &mut self,
280        resource: &Resource,
281    ) -> Result<Option<u16>, Arc<anyhow::Error>> {
282        #[derive(Display, Debug, Hash, Eq, Dupe, Clone, PartialEq, RefCast, Allocative)]
283        #[repr(transparent)]
284        struct LookupResourceCost(Resource);
285        #[async_trait]
286        impl Key for LookupResourceCost {
287            type Value = Result<Option<u16>, Arc<anyhow::Error>>;
288
289            async fn compute(
290                &self,
291                ctx: &mut DiceComputations,
292                _cancellations: &CancellationContext,
293            ) -> Self::Value {
294                let companies = ctx
295                    .compute(LookupResource::ref_cast(&self.0))
296                    .await
297                    .map_err(|e| Arc::new(anyhow::anyhow!(e)))?;
298
299                let costs = join_all(ctx
300                    .compute_many(companies.iter().map(|company| {
301                        DiceComputations::declare_closure(
302                            |ctx: &mut DiceComputations<'_>| -> BoxFuture<Result<Option<u16>, Arc<anyhow::Error>>> {
303                                lookup_company_resource_cost(ctx, company, &self.0).boxed()
304                            }
305                        )
306                    })))
307                    .await;
308
309                Ok(costs
310                    .into_iter()
311                    .collect::<Result<Vec<_>, Arc<_>>>()?
312                    .into_iter()
313                    .min()
314                    .flatten())
315            }
316
317            fn equality(x: &Self::Value, y: &Self::Value) -> bool {
318                match (x, y) {
319                    (Ok(x), Ok(y)) => x == y,
320                    _ => false,
321                }
322            }
323        }
324
325        self.compute(LookupResourceCost::ref_cast(resource))
326            .await
327            .map_err(|e| Arc::new(anyhow::anyhow!(e)))?
328    }
329}

331#[async_trait]
332impl CostUpdater for DiceTransactionUpdater {
333    async fn change_company_resource_cost(
334        &mut self,
335        company: &str,
336        resource: &Resource,
337        new_price: u16,
338    ) -> anyhow::Result<()> {
339        let company_lookup = LookupCompany(Arc::new(company.to_owned()));
340        let old_company = self.existing_state().await.compute(&company_lookup).await?;
341        let mut new_company = (*old_company).clone();
342        let old_price = new_company.makes.get_mut(resource).ok_or_else(|| {
343            anyhow::anyhow!("Tried to update cost for a resource company does not make")
344        })?;
345        *old_price = new_price;
346
347        self.changed_to(vec![(company_lookup, Arc::new(new_company))])?;
348
349        Ok(())
350    }
351}

353#[derive(Display, Debug, Hash, Eq, Clone, Dupe, PartialEq, RefCast, Allocative)]
354#[repr(transparent)]
355pub struct LookupCompany(pub Arc<String>);
356impl InjectedKey for LookupCompany {
357    type Value = Arc<Company>;
358
359    fn equality(x: &Self::Value, y: &Self::Value) -> bool {
360        x == y
361    }
362}
363
364#[derive(Display, Debug, Hash, Eq, Dupe, Clone, PartialEq, RefCast, Allocative)]
365#[repr(transparent)]
366pub struct LookupResource(pub Resource);
367impl InjectedKey for LookupResource {
368    type Value = Arc<Vec<LookupCompany>>;
369
370    fn equality(x: &Self::Value, y: &Self::Value) -> bool {
371        x == y
372    }
