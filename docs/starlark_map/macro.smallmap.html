Create a SmallMap from a list of key-value pairs.

ยงExample

use starlark_map::smallmap;

let map = smallmap! {
    "a" => 1,
    "b" => 2,
};
assert_eq!(map.get("a"), Some(&1));
assert_eq!(map.get("b"), Some(&2));
assert_eq!(map.get("c"), None);
