Struct StyledContent

pub struct StyledContent<D>
where
    D: Display,
{ /* private fields */ }
Expand description

The style with the content to be styled.

§Examples

use crossterm::style::{style, Color, Attribute, Stylize};

let styled = "Hello there"
    .with(Color::Yellow)
    .on(Color::Blue)
    .attribute(Attribute::Bold);

println!("{}", styled);
